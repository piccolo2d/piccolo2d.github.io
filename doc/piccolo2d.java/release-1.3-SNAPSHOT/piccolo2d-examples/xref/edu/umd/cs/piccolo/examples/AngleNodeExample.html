21  /**
22   * This shows how to create a simple node that has two handles and can be used
23   * for specifying angles. The nodes UI desing isn't very exciting, but the
24   * example shows one way to create a custom node with custom handles.
25   */
26  public class AngleNodeExample extends PFrame {
27  
28      public AngleNodeExample() {
29          this(null);
30      }
31  
32      public AngleNodeExample(PCanvas aCanvas) {
33          super("AngleNodeExample", false, aCanvas);
34      }
35  
36      public void initialize() {
37          PCanvas c = getCanvas();
38          PLayer l = c.getLayer();
39          l.addChild(new AngleNode());
40      }
41  
42      public static void main(String[] args) {
43          new AngleNodeExample();
44      }

46      // the angle node class
47      public static class AngleNode extends PNode {
48          protected Point2D.Double pointOne;
49          protected Point2D.Double pointTwo;
50          protected Stroke stroke;
51  
52          public AngleNode() {
53              pointOne = new Point2D.Double(100, 0);
54              pointTwo = new Point2D.Double(0, 100);
55              stroke = new BasicStroke(1, BasicStroke.CAP_ROUND, BasicStroke.JOIN_ROUND);
56              setPaint(Color.BLACK);
57              updateBounds();
58              addHandles();
59          }
60  
61          public void addHandles() {
62              // point one
63              PLocator l = new PLocator() {
64                  public double locateX() {
65                      return pointOne.getX();
66                  }
67  
68                  public double locateY() {
69                      return pointOne.getY();
70                  }
71              };
72              PHandle h = new PHandle(l) {
73                  public void dragHandle(PDimension aLocalDimension, PInputEvent aEvent) {
74                      localToParent(aLocalDimension);
75                      pointOne.setLocation(pointOne.getX() + aLocalDimension.getWidth(), pointOne.getY()
76                              + aLocalDimension.getHeight());
77                      updateBounds();
78                      relocateHandle();
79                  }
80              };
81              addChild(h);
82  
83              // point two
84              l = new PLocator() {
85                  public double locateX() {
86                      return pointTwo.getX();
87                  }
88  
89                  public double locateY() {
90                      return pointTwo.getY();
91                  }
92              };
93              h = new PHandle(l) {
94                  public void dragHandle(PDimension aLocalDimension, PInputEvent aEvent) {
95                      localToParent(aLocalDimension);
96                      pointTwo.setLocation(pointTwo.getX() + aLocalDimension.getWidth(), pointTwo.getY()
97                              + aLocalDimension.getHeight());
98                      updateBounds();
99                      relocateHandle();
100                 }
101             };
102             addChild(h);
103         }
104 
105         protected void paint(PPaintContext paintContext) {
106             Graphics2D g2 = paintContext.getGraphics();
107             g2.setStroke(stroke);
108             g2.setPaint(getPaint());
109             g2.draw(getAnglePath());
110         }
111 
112         protected void updateBounds() {
113             GeneralPath p = getAnglePath();
114             Rectangle2D b = stroke.createStrokedShape(p).getBounds2D();
115             super.setBounds(b.getX(), b.getY(), b.getWidth(), b.getHeight());
116         }
117 
118         public GeneralPath getAnglePath() {
119             GeneralPath p = new GeneralPath();
120             p.moveTo((float) pointOne.getX(), (float) pointOne.getY());
121             p.lineTo(0, 0);
122             p.lineTo((float) pointTwo.getX(), (float) pointTwo.getY());
123             return p;
124         }
125 
126         public boolean setBounds(double x, double y, double width, double height) {
127             return false; // bounds can be set externally
128         }
129     }
