261     /**
262      * Stop this activity immediately, and remove it from the activity
263      * scheduler. The default termination behavior is call activityFinished if
264      * the activity is currently stepping. Use terminate(terminationBehavior)
265      * use a different termination behavior.
266      */
267     public void terminate() {
268         terminate(TERMINATE_AND_FINISH_IF_STEPPING);
269     }
270 
271     /**
272      * Stop this activity immediately, and remove it from the activity
273      * scheduler. The termination behavior determines when and if
274      * activityStarted and activityFinished get called. The possible termination
275      * behaviors are as follow:
276      * 
277      * TERMINATE_WITHOUT_FINISHING - The method activityFinished will never get
278      * called and so the activity will be terminated midway.
279      * TERMINATE_AND_FINISH - The method activityFinished will always get
280      * called. And so the activity will always end in it's completed state. If
281      * the activity has not yet started the method activityStarted will also be
282      * called. TERMINATE_AND_FINISH_IF_STEPPING - The method activityFinished
283      * will only be called if the activity has previously started.
284      */
285     public void terminate(int terminationBehavior) {
286         if (scheduler != null) {
287             scheduler.removeActivity(this);
288         }
289 
290         switch (terminationBehavior) {
291             case TERMINATE_WITHOUT_FINISHING:
292                 stepping = false;
293                 break;
294 
295             case TERMINATE_AND_FINISH:
296                 if (stepping) {
297                     stepping = false;
298                     activityFinished();
299                 }
300                 else {
301                     activityStarted();
302                     activityFinished();
303                 }
304 
305                 break;
306 
307             case TERMINATE_AND_FINISH_IF_STEPPING:
308                 if (stepping) {
309                     stepping = false;
310                     activityFinished();
311                 }
312                 break;
313         }
314     }

316     /**
317      * The activity scheduler calls this method and it is here that the activity
318      * decides if it should do a step or not for the given time.
319      */
320     public long processStep(long currentTime) {
321         // if before start time
322         if (currentTime < startTime) {
323             return startTime - currentTime;
324         }
325 
326         // if past stop time
327         if (currentTime > getStopTime()) {
328             if (stepping) {
329                 stepping = false;
330                 scheduler.removeActivity(this);
331                 activityFinished();
332             }
333             else {
334                 activityStarted();
335                 scheduler.removeActivity(this);
336                 activityFinished();
337             }
338             return -1;
339         }
340 
341         // else should be stepping
342         if (!stepping) {
343             activityStarted();
344             stepping = true;
345         }
346 
347         if (currentTime >= nextStepTime) {
348             activityStep(currentTime - startTime);
349             nextStepTime = currentTime + stepRate;
350         }
351 
352         return stepRate;
353     }
354 
355     /**
356      * Return the time when this activity should finish running. At this time
357      * (or soon after) the stoppedStepping method will be called
358      */
359     public long getStopTime() {
360         if (duration == -1) {
361             return Long.MAX_VALUE;
362         }
363         return startTime + duration;
364     }
