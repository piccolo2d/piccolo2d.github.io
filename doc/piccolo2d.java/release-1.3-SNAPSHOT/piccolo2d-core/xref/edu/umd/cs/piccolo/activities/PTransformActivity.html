30  package edu.umd.cs.piccolo.activities;

32  import java.awt.geom.AffineTransform;
33  
34  import edu.umd.cs.piccolo.util.PAffineTransform;
35  
36  /**
37   * <b>PTransformActivity</b> interpolates between two transforms setting its
38   * target's transform as it goes. See PNode. animate*() for an example of this
39   * activity in used. The source transform is retrieved from the target just
40   * before the animation is scheduled to start.
41   * <P>
42   * @version 1.0
43   * @author Jesse Grosjean
44   */
45  public class PTransformActivity extends PInterpolatingActivity {
46  
47  	private static PAffineTransform STATIC_TRANSFORM = new PAffineTransform();
48  
49  	private double[] source;
50  	private double[] destination;
51  	private Target target;
52  
53  	/**
54  	 * <b>Target</b> Objects that want to get transformed by the transform 
55  	 * activity must implement this interface. See PNode.animateToTransform() 
56  	 * for one way to do this.
57  	 */
58  	public interface Target {
59  		
60  		/**
61  		 * This will be called by the transform activity for each new transform
62  		 * that it computes while it is stepping.
63  		 */
64  		public void setTransform(AffineTransform aTransform);
65  		
66  		/**
67  		 * This method is called right before the transform activity starts. That
68  		 * way an object is always animated from its current position.
69  		 */
70  		public void getSourceMatrix(double[] aSource);
71  	}
72  
73  	public PTransformActivity(long duration, long stepRate, Target aTarget) {
74  		this(duration, stepRate, aTarget, null);
75  	}
76  	
77  	public PTransformActivity(long duration, long stepRate, Target aTarget, AffineTransform aDestination) {
78  		this(duration, stepRate, 1, PInterpolatingActivity.SOURCE_TO_DESTINATION, aTarget, aDestination);
79  	}
80  
81  	/**
82  	 * Create a new PTransformActivity.
83  	 * <P>
84  	 * @param duration the length of one loop of the activity
85  	 * @param stepRate the amount of time between steps of the activity
86  	 * @param loopCount number of times the activity should reschedule itself
87  	 * @param mode defines how the activity interpolates between states
88  	 * @param aTarget the object that the activity will be applied to and where
89  	 * the source state will be taken from.
90  	 * @param aDestination the destination color state
91  	 */
92  	public PTransformActivity(long duration, long stepRate, int loopCount, int mode, Target aTarget, AffineTransform aDestination) {
93  		super(duration, stepRate, loopCount, mode);
94  		source = new double[6];
95  		destination = new double[6];
96  		target = aTarget;
97  		if (aDestination != null) aDestination.getMatrix(destination);
98  	}
99  	
100 	protected boolean isAnimation() {
101 		return true;
102 	}
103 
104 	/**
105 	 * Return the final transform that will be set on the transform activities
106 	 * target when the transform activity stops stepping.
107 	 */
108 	public double[] getDestinationTransform() {
109 		return destination;
110 	}
111 
112 	/**
113 	 * Set the final transform that will be set on the transform activities
114 	 * target when the transform activity stops stepping.
115 	 */
116 	public void setDestinationTransform(double[] newDestination) {
117 		destination = newDestination;
118 	}
119 	
120 	protected void activityStarted() { 
121 		if (getFirstLoop()) target.getSourceMatrix(source);
122 		super.activityStarted();
123 	}
124 	
125 	public void setRelativeTargetValue(float zeroToOne) {
126 		super.setRelativeTargetValue(zeroToOne);
127 
128 		STATIC_TRANSFORM.setTransform(source[0] + (zeroToOne * (destination[0] - source[0])),
129 									  source[1] + (zeroToOne * (destination[1] - source[1])),
130 									  source[2] + (zeroToOne * (destination[2] - source[2])),
131 									  source[3] + (zeroToOne * (destination[3] - source[3])),
132 									  source[4] + (zeroToOne * (destination[4] - source[4])),
133 									  source[5] + (zeroToOne * (destination[5] - source[5])));
134 									  
135 		target.setTransform(STATIC_TRANSFORM);
136 	}
137 	
138 	//****************************************************************
139 	// Debugging - methods for debugging
140 	//****************************************************************
141 
142 	/**
143 	 * Returns a string representing the state of this activity. This method is
144 	 * intended to be used only for debugging purposes, and the content and
145 	 * format of the returned string may vary between implementations. The
146 	 * returned string may be empty but may not be <code>null</code>.
147 	 *
148 	 * @return  a string representation of this activity's state
149 	 */
150 	protected String paramString() {
151 		StringBuffer result = new StringBuffer();
152 
153 		result.append("source=" + (source == null ? "null" : source.toString()));
154 		result.append(",destination=" + (destination == null ? "null" : destination.toString()));
155 		result.append(',');
156 		result.append(super.paramString());
157 		
158 		return result.toString();
159 	}
