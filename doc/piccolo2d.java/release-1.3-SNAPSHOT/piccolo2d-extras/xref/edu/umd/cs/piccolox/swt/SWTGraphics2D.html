91          Rectangle aRect = new Rectangle(rect.x, rect.y, rect.width, rect.height);
92          try {
93              SWTShapeManager.transform(aRect, transform.createInverse());
94          }
95          catch (Exception e) {
96              e.printStackTrace();
97          }
98          return aRect;

235     public Font getFont() {
236         if (curFont != null) {
237             int style = Font.PLAIN;
238 
239             FontData[] fd = curFont.getFontData();
240             if (fd.length > 0) {
241                 if ((fd[0].getStyle() & SWT.BOLD) != 0) {
242                     style = style | Font.BOLD;
243                 }
244                 if ((fd[0].getStyle() & SWT.ITALIC) != 0) {
245                     style = style | SWT.ITALIC;
246                 }
247 
248                 return new Font(fd[0].getName(), style, fd[0].height);
249             }
250             return null;
251         }
252         else {
253             return null;
254         }
255     }

268     public org.eclipse.swt.graphics.Font getFont(String fontString) {
269         org.eclipse.swt.graphics.Font cachedFont = (org.eclipse.swt.graphics.Font) FONT_CACHE.get(fontString);
270         if (cachedFont == null) {
271             int style = 0;
272             if (fontString.indexOf("bold=true") != -1) {
273                 style = style | SWT.BOLD;
274             }
275             if (fontString.indexOf("italic=true") != -1) {
276                 style = style | SWT.ITALIC;
277             }
278 
279             String name = fontString.substring(0, fontString.indexOf(";"));
280             String size = fontString.substring(fontString.lastIndexOf(";") + 1, fontString.length());
281             int sizeInt = 12;
282             try {
283                 sizeInt = Integer.parseInt(size.substring(size.indexOf("=") + 1, size.length()));
284             }
285             catch (Exception e) {
286                 e.printStackTrace();
287             }
288 
289             cachedFont = new org.eclipse.swt.graphics.Font(device,
290                     name.substring(name.indexOf("=") + 1, name.length()), sizeInt, style);
291             FONT_CACHE.put(fontString, cachedFont);
292         }
293         return cachedFont;
294     }

296     protected org.eclipse.swt.graphics.Font getTransformedFont() {
297         if (curFont != null) {
298             FontData fontData = curFont.getFontData()[0];
299             int height = fontData.getHeight();
300             RECT.setRect(0, 0, height, height);
301             SWTShapeManager.transform(RECT, transform);
302             height = (int) (RECT.getHeight() + 0.5);
303 
304             String fontString = "name=" + fontData.getName() + ";bold=" + ((fontData.getStyle() & SWT.BOLD) != 0)
305                     + ";italic=" + ((fontData.getStyle() & SWT.ITALIC) != 0) + ";size=" + height;
306             return getFont(fontString);
307         }
308         return null;
309     }

359     public void draw(Shape s) {
360         if (s instanceof Rectangle2D) {
361             Rectangle2D r2 = (Rectangle2D) s;
362             drawRect(r2.getX(), r2.getY(), r2.getWidth(), r2.getHeight());
363         }
364         else if (s instanceof Ellipse2D) {
365             Ellipse2D e2 = (Ellipse2D) s;
366             drawOval(e2.getX(), e2.getY(), e2.getWidth(), e2.getHeight());
367         }
368         else if (s instanceof RoundRectangle2D) {
369             RoundRectangle2D r2 = (RoundRectangle2D) s;
370             drawRoundRect(r2.getX(), r2.getY(), r2.getWidth(), r2.getHeight(), r2.getArcWidth(), r2.getArcHeight());
371         }
372         else if (s instanceof Arc2D) {
373             Arc2D a2 = (Arc2D) s;
374             drawArc(a2.getX(), a2.getY(), a2.getWidth(), a2.getHeight(), a2.getAngleStart(), a2.getAngleExtent());
375         }
376         else {
377             double[] pts = (double[]) SHAPE_CACHE.get(s);
378 
379             if (pts == null) {
380                 pts = SWTShapeManager.shapeToPolyline(s);
381                 SHAPE_CACHE.put(s, pts);
382             }
383 
384             drawPolyline(pts);
385         }
386     }

417     public void drawPolyline(int[] xPoints, int[] yPoints, int nPoints) {
418         int[] ptArray = new int[2 * nPoints];
419         for (int i = 0; i < nPoints; i++) {
420             PT.setLocation(xPoints[i], yPoints[i]);
421             transform.transform(PT, PT);
422             ptArray[2 * i] = xPoints[i];
423             ptArray[2 * i + 1] = yPoints[i];
424         }
425 
426         gc.setLineWidth(getTransformedLineWidth());
427         gc.drawPolyline(ptArray);
428     }
