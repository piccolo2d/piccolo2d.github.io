50  	// This method demonstrates the kinds of things that can be done with any node.
51  	public void nodeDemo() {
52  		PLayer layer = getCanvas().getLayer();
53  		PNode aNode = PPath.createRectangle(0, 0, 100, 80);
54  		
55  		// A node needs to be a descendent of the root to be displayed on the screen.
56  		layer.addChild(aNode);
57  		
58  		// The default color for a node is blue, but you can change that with 
59  		// the setPaint method.
60  		aNode.setPaint(Color.red);
61  				
62  		// A node can have children nodes added to it.
63  		aNode.addChild(PPath.createRectangle(0, 0, 100, 80));
64  				
65  		// The base bounds of a node is easy to change. Note that changing the base 
66  		// bounds of a node will not change it's children.
67  		aNode.setBounds(-10, -10, 200, 110);
68  		
69  		// Each node has a transform that can be used to transform the node, and 
70  		// all its children on the screen.
71  		aNode.translate(100, 100);
72  		aNode.scale(1.5);
73  		aNode.rotate(45);
74  		
75  		// The transparency of any node can be set, this transparency will be 
76  		// applied to any of the nodes children as well.
77  		aNode.setTransparency(0.75f);
78  		
79  		// Its easy to copy nodes.
80  		PNode aCopy = (PNode) aNode.clone();
81  
82  		// Make is so that the copies children are not pickable. For this example
83  		// that means you will not be able to grab the child and remove it from
84  		// its parent.
85  		aNode.setChildrenPickable(false);
86  
87  		// Change the look of the copy
88  		aNode.setPaint(Color.GREEN);
89  		aNode.setTransparency(1.0f);
90  		
91  		// Let's add the copy to the root, and translate it so that it does not 
92  		// cover the original node.
93  		layer.addChild(aCopy);		
94  		aCopy.setOffset(0, 0);		
95  		aCopy.rotate(-45);
96  	}

113 	// Another way to create nodes is to customize other nodes that already
114 	// exist. Here we create an ellipse, except when you press the mouse on
115 	// this ellipse it turns into a square, when you release the mouse it
116 	// goes back to being an ellipse.
117 	public void subclassExistingClasses() {
118 		final PNode n = new PPath(new Ellipse2D.Float(0, 0, 100, 80)) {
119 						
120 			public void paint(PPaintContext aPaintContext) {
121 				if (fIsPressed) {
122 					// if mouse is pressed draw self as a square.
123 					Graphics2D g2 = aPaintContext.getGraphics();
124 					g2.setPaint(getPaint());
125 					g2.fill(getBoundsReference());
126 				} else {
127 					// if mouse is not pressed draw self normally.
128 					super.paint(aPaintContext);
129 				}
130 			}
131 		};		
132 		
133 		n.addInputEventListener(new PBasicInputEventHandler() {
134 			public void mousePressed(PInputEvent aEvent) {
135 				super.mousePressed(aEvent);
136 				fIsPressed = true;
137 				n.invalidatePaint(); // this tells the framework that the node needs to be redisplayed.
138 			}
139 			
140 			public void mouseReleased(PInputEvent aEvent) {
141 				super.mousePressed(aEvent);
142 				fIsPressed = false;
143 				n.invalidatePaint(); // this tells the framework that the node needs to be redisplayed.
144 			}
145 		});
146 		
147 		n.setPaint(Color.ORANGE);
148 		getCanvas().getLayer().addChild(n);
149 	}

151 	// Here a new "face" node is created. But instead of drawing the face directly
152 	// using Graphics2D we compose the face from other nodes.
153 	public void composeOtherNodes() {
154 		PNode myCompositeFace = PPath.createRectangle(0, 0, 100, 80);
155 		
156 		// create parts for the face.
157 		PNode eye1 = PPath.createEllipse(0, 0, 20, 20);
158 		eye1.setPaint(Color.YELLOW);
159 		PNode eye2 = (PNode) eye1.clone();
160 		PNode mouth = PPath.createRectangle(0, 0, 40, 20);
161 		mouth.setPaint(Color.BLACK);
162 		
163 		// add the face parts
164 		myCompositeFace.addChild(eye1);
165 		myCompositeFace.addChild(eye2);
166 		myCompositeFace.addChild(mouth);
167 		
168 		// don't want anyone grabbing out our eye's.
169 		myCompositeFace.setChildrenPickable(false);
170 		
171 		// position the face parts.
172 		eye2.translate(25, 0);
173 		mouth.translate(0, 30);
174 		
175 		// set the face bounds so that it neatly contains the face parts.
176 		PBounds b = myCompositeFace.getUnionOfChildrenBounds(null);
177 		myCompositeFace.setBounds(b.inset(-5, -5));
178 		
179 		// opps it to small, so scale it up.
180 		myCompositeFace.scale(1.5);
181 		
182 		getCanvas().getLayer().addChild(myCompositeFace);
183 	}

185 	// Here a completely new kind of node, a grid node" is created. We do
186 	// all the drawing ourselves here instead of passing the work off to 
187 	// other parts of the framework.
188 	public void createCustomNode() {
189 		PNode n = new PNode() {
190 			public void paint(PPaintContext aPaintContext) {
191 				double bx = getX();
192 				double by = getY();
193 				double rightBorder = bx + getWidth();
194 				double bottomBorder = by + getHeight();
195 				
196 				Line2D line = new Line2D.Double();
197 				Graphics2D g2 = aPaintContext.getGraphics();
198 				
199 				g2.setStroke(new BasicStroke(0));
200 				g2.setPaint(getPaint());
201 				
202 				// draw vertical lines
203 				for (double x = bx; x < rightBorder; x += 5) {
204 					line.setLine(x, by, x, bottomBorder);
205 					g2.draw(line);
206 				}
207 				
208 				for (double y = by; y < bottomBorder; y += 5) {
209 					line.setLine(bx, y, rightBorder, y);
210 					g2.draw(line);
211 				}
212 			}
213 		};
214 		n.setBounds(0, 0, 100, 80);
215 		n.setPaint(Color.black);
216 		getCanvas().getLayer().addChild(n);
217 	}
