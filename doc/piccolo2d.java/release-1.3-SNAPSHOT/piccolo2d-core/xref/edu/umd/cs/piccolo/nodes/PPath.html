52  /**
53   * <b>PPath</b> is a wrapper around a java.awt.geom.GeneralPath. The setBounds
54   * method works by scaling the path to fit into the specified bounds. This
55   * normally works well, but if the specified base bounds get too small then it
56   * is impossible to expand the path shape again since all its numbers have
57   * tended to zero, so application code may need to take this into consideration.
58   * <P>
59   * One option that applications have is to call <code>startResizeBounds</code>
60   * before starting an interaction that may make the bounds very small, and
61   * calling <code>endResizeBounds</code> when this interaction is finished. When
62   * this is done PPath will use a copy of the original path to do the resizing so
63   * the numbers in the path wont loose resolution.
64   * <P>
65   * This class also provides methods for constructing common shapes using a
66   * general path.
67   * <P>
68   * 
69   * @version 1.0
70   * @author Jesse Grosjean
71   */
72  public class PPath extends PNode {

166     /**
167      * Construct this path with the given shape and stroke. This method may be
168      * used to optimize the creation of a large number of PPaths. Normally
169      * PPaths have a default stroke of width one, but when a path has a non null
170      * stroke it takes significantly longer to compute its bounds. This method
171      * allows you to override that default stroke before the bounds are ever
172      * calculated, so if you pass in a null stroke here you won't ever have to
173      * pay that bounds calculation price if you don't need to.
174      */
175     public PPath(Shape aShape, Stroke aStroke) {
176         this();
177         stroke = aStroke;
178         if (aShape != null)
179             append(aShape, false);
180     }

221     /**
222      * Set the bounds of this path. This method works by scaling the path to fit
223      * into the specified bounds. This normally works well, but if the specified
224      * base bounds get too small then it is impossible to expand the path shape
225      * again since all its numbers have tended to zero, so application code may
226      * need to take this into consideration.
227      */
228     protected void internalUpdateBounds(double x, double y, double width, double height) {
229         if (updatingBoundsFromPath)
230             return;
231         if (path == null)
232             return;
233 
234         if (resizePath != null) {
235             path.reset();
236             path.append(resizePath, false);
237         }
238 
239         Rectangle2D pathBounds = path.getBounds2D();
240         Rectangle2D pathStrokeBounds = getPathBoundsWithStroke();
241         double strokeOutset = Math.max(pathStrokeBounds.getWidth() - pathBounds.getWidth(), pathStrokeBounds
242                 .getHeight()
243                 - pathBounds.getHeight());
244 
245         x += strokeOutset / 2;
246         y += strokeOutset / 2;
247         width -= strokeOutset;
248         height -= strokeOutset;
249 
250         double scaleX = (width == 0 || pathBounds.getWidth() == 0) ? 1 : width / pathBounds.getWidth();
251         double scaleY = (height == 0 || pathBounds.getHeight() == 0) ? 1 : height / pathBounds.getHeight();
252 
253         TEMP_TRANSFORM.setToIdentity();
254         TEMP_TRANSFORM.translate(x, y);
255         TEMP_TRANSFORM.scale(scaleX, scaleY);
256         TEMP_TRANSFORM.translate(-pathBounds.getX(), -pathBounds.getY());
257 
258         path.transform(TEMP_TRANSFORM);
259     }

261     public boolean intersects(Rectangle2D aBounds) {
262         if (super.intersects(aBounds)) {
263             if (getPaint() != null && path.intersects(aBounds)) {
264                 return true;
265             }
266             else if (stroke != null && strokePaint != null) {
267                 return stroke.createStrokedShape(path).intersects(aBounds);
268             }
269         }
270         return false;
271     }
272 
273     public Rectangle2D getPathBoundsWithStroke() {
274         if (stroke != null) {
275             return stroke.createStrokedShape(path).getBounds2D();
276         }
277         else {
278             return path.getBounds2D();
279         }
280     }

282     public void updateBoundsFromPath() {
283         updatingBoundsFromPath = true;
284         if (path == null) {
285             resetBounds();
286         }
287         else {
288             Rectangle2D b = getPathBoundsWithStroke();
289             setBounds(b.getX(), b.getY(), b.getWidth(), b.getHeight());
290         }
291         updatingBoundsFromPath = false;
292     }

298     protected void paint(PPaintContext paintContext) {
299         Paint p = getPaint();
300         Graphics2D g2 = paintContext.getGraphics();
301 
302         if (p != null) {
303             g2.setPaint(p);
304             g2.fill(path);
305         }
306 
307         if (stroke != null && strokePaint != null) {
308             g2.setPaint(strokePaint);
309             g2.setStroke(stroke);
310             g2.draw(path);
311         }
312     }
