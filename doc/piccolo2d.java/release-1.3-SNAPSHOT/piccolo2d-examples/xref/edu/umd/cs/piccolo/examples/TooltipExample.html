14  /**
15   * Simple example of one way to add tooltips
16   * 
17   * @author jesse
18   */
19  public class TooltipExample extends PFrame {
20  	
21  	public TooltipExample() {
22  		this(null);
23  	}
24  
25  	public TooltipExample(PCanvas aCanvas) {
26  		super("TooltipExample", false, aCanvas);
27  	}
28  
29  	public void initialize() {
30  		PNode n1 = PPath.createEllipse(0, 0, 100, 100);
31  		PNode n2 = PPath.createRectangle(300, 200, 100, 100);
32  		
33  		n1.addAttribute("tooltip", "node 1");
34  		n2.addAttribute("tooltip", "node 2");
35  		getCanvas().getLayer().addChild(n1);
36  		getCanvas().getLayer().addChild(n2);
37  		
38  		final PCamera camera = getCanvas().getCamera();
39  		final PText tooltipNode = new PText();
40  		
41  		tooltipNode.setPickable(false);
42  		camera.addChild(tooltipNode);
43  		
44  		camera.addInputEventListener(new PBasicInputEventHandler() {	
45  			public void mouseMoved(PInputEvent event) {
46  				updateToolTip(event);
47  			}
48  
49  			public void mouseDragged(PInputEvent event) {
50  				updateToolTip(event);
51  			}
52  
53  			public void updateToolTip(PInputEvent event) {
54  				PNode n = event.getInputManager().getMouseOver().getPickedNode();
55  				String tooltipString = (String) n.getAttribute("tooltip");
56  				Point2D p = event.getCanvasPosition();
57  				
58  				event.getPath().canvasToLocal(p, camera);
59  				
60  				tooltipNode.setText(tooltipString);
61  				tooltipNode.setOffset(p.getX() + 8, 
62  									  p.getY() - 8);
63  			}
64  		});
65  	}
66  	
67  	public static void main(String[] argv) {
68  		new TooltipExample();
69  	}
