70      /**
71       * Sets whether this event handler can fire events.
72       *
73       * @param active
74       */
75      void setActive( boolean active ) {
76          if( this.active && !active ) {
77              if( listenNode != null ) {
78                  this.active = false;
79                  listenNode.removeInputEventListener( this );
80              }
81          }
82          else if( !this.active && active ) {
83              if( listenNode != null ) {
84                  this.active = true;
85                  listenNode.addInputEventListener( this );
86              }
87          }
88      }

99      /**
100      * Finds the component at the specified location (must be showing).
101      *
102      * @param c
103      * @param x
104      * @param y
105      * @return the component at the specified location.
106      */
107     private Component findShowingComponentAt( Component c, int x, int y ) {
108         if( !c.contains( x, y ) ) {
109             return null;
110         }
111 
112         if( c instanceof Container ) {
113             Container contain = ( (Container)c );
114             int ncomponents = contain.getComponentCount();
115             Component component[] = contain.getComponents();
116 
117             for( int i = 0; i < ncomponents; i++ ) {
118                 Component comp = component[i];
119                 if( comp != null ) {
120                     Point p = comp.getLocation();
121                     if( comp instanceof Container ) {
122                         comp = findShowingComponentAt( comp, x - (int)p.getX(), y - (int)p.getY() );
123                     }
124                     else {
125                         comp = comp.getComponentAt( x - (int)p.getX(), y - (int)p.getY() );
126                     }
127                     if( comp != null && comp.isShowing() ) {
128                         return comp;
129                     }
130                 }
131             }
132         }
133         return c;
134     }

136     /**
137      * Determines if any Swing components in Piccolo
138      * should receive the given MouseEvent and
139      * forwards the event to that component.
140      * However, mouseEntered and mouseExited are independent of the buttons.
141      * Also, notice the notes on mouseEntered and mouseExited.
142      *
143      * @param pSwingMouseEvent
144      * @param aEvent
145      */
146     void dispatchEvent( PSwingMouseEvent pSwingMouseEvent, PInputEvent aEvent ) {
147         Component comp = null;
148         Point2D pt = null;
149         PNode pickedNode = pSwingMouseEvent.getPath().getPickedNode();
150 
151         // The offsets to put the event in the correct context
152         int offX = 0;
153         int offY = 0;
154 
155         PNode currentNode = pSwingMouseEvent.getCurrentNode();
156 
157         if( currentNode instanceof PSwing ) {
158 
159             PSwing swing = (PSwing)currentNode;
160             PNode grabNode = pickedNode;
161 
162             if( grabNode.isDescendentOf( canvas.getRoot() ) ) {
163                 pt = new Point2D.Double( pSwingMouseEvent.getX(), pSwingMouseEvent.getY() );
164                 cameraToLocal( pSwingMouseEvent.getPath().getTopCamera(), pt, grabNode );
165                 prevPoint = new Point2D.Double( pt.getX(), pt.getY() );
166 
167                 // This is only partially fixed to find the deepest
168                 // component at pt.  It needs to do something like
169                 // package private method:
170                 // Container.getMouseEventTarget(int,int,boolean)
171                 comp = findShowingComponentAt( swing.getComponent(), (int)pt.getX(), (int)pt.getY() );
172 
173                 // We found the right component - but we need to
174                 // get the offset to put the event in the component's
175                 // coordinates
176                 if( comp != null && comp != swing.getComponent() ) {
177                     for( Component c = comp; c != swing.getComponent(); c = c.getParent() ) {
178                         offX += c.getLocation().getX();
179                         offY += c.getLocation().getY();
180                     }
181                 }
182 
183                 // Mouse Pressed gives focus - effects Mouse Drags and
184                 // Mouse Releases
185                 if( comp != null && pSwingMouseEvent.getID() == MouseEvent.MOUSE_PRESSED ) {
186                     if( SwingUtilities.isLeftMouseButton( pSwingMouseEvent ) ) {
187                         leftButtonData.setState( swing, pickedNode, comp, offX, offY );
188                     }
189                     else if( SwingUtilities.isMiddleMouseButton( pSwingMouseEvent ) ) {
190                         middleButtonData.setState( swing, pickedNode, comp, offX, offY );
191                     }
192                     else if( SwingUtilities.isRightMouseButton( pSwingMouseEvent ) ) {
193                         rightButtonData.setState( swing, pickedNode, comp, offX, offY );
194                     }
195                 }
196             }
197         }
198 
199         // This first case we don't want to give events to just
200         // any Swing component - but to the one that got the
201         // original mousePressed
202         if( pSwingMouseEvent.getID() == MouseEvent.MOUSE_DRAGGED ||
203             pSwingMouseEvent.getID() == MouseEvent.MOUSE_RELEASED ) {
204 
205             // LEFT MOUSE BUTTON
206             if( SwingUtilities.isLeftMouseButton( pSwingMouseEvent ) && leftButtonData.getFocusedComponent() != null ) {
207                 handleButton( pSwingMouseEvent, aEvent, leftButtonData );
208             }
209 
210             // MIDDLE MOUSE BUTTON
211             if( SwingUtilities.isMiddleMouseButton( pSwingMouseEvent ) && middleButtonData.getFocusedComponent() != null )
212             {
213                 handleButton( pSwingMouseEvent, aEvent, middleButtonData );
214             }
215 
216             // RIGHT MOUSE BUTTON
217             if( SwingUtilities.isRightMouseButton( pSwingMouseEvent ) && rightButtonData.getFocusedComponent() != null )
218             {
219                 handleButton( pSwingMouseEvent, aEvent, rightButtonData );
220             }
221         }
222         // This case covers the cases mousePressed, mouseClicked,
223         // and mouseMoved events
224         else if( ( pSwingMouseEvent.getID() == MouseEvent.MOUSE_PRESSED ||
225                    pSwingMouseEvent.getID() == MouseEvent.MOUSE_CLICKED ||
226                    pSwingMouseEvent.getID() == MouseEvent.MOUSE_MOVED ) &&
227                                                                         ( comp != null ) ) {
228 
229             MouseEvent e_temp = new MouseEvent( comp,
230                                                 pSwingMouseEvent.getID(),
231                                                 pSwingMouseEvent.getWhen(),
232                                                 pSwingMouseEvent.getModifiers(),
233                                                 (int)pt.getX() - offX,
234                                                 (int)pt.getY() - offY,
235                                                 pSwingMouseEvent.getClickCount(),
236                                                 pSwingMouseEvent.isPopupTrigger() );
237 
238             PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent( e_temp.getID(), e_temp, aEvent );
239             dispatchEvent( comp, e2 );
240             pSwingMouseEvent.consume();
241         }
242 
243         // Now we need to check if an exit or enter event needs to
244         // be dispatched - this code is independent of the mouseButtons.
245         // I tested in normal Swing to see the correct behavior.
246         if( prevComponent != null ) {
247             // This means mouseExited
248 
249             // This shouldn't happen - since we're only getting node events
250             if( comp == null || pSwingMouseEvent.getID() == MouseEvent.MOUSE_EXITED ) {
251                 MouseEvent e_temp = createExitEvent( pSwingMouseEvent );
252 
253                 PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent( e_temp.getID(), e_temp, aEvent );
254 
255                 dispatchEvent( prevComponent, e2 );
256                 prevComponent = null;
257 
258                 if( pSwingMouseEvent.getID() == MouseEvent.MOUSE_EXITED ) {
259                     pSwingMouseEvent.consume();
260                 }
261             }
262 
263             // This means mouseExited prevComponent and mouseEntered comp
264             else if( prevComponent != comp ) {
265                 MouseEvent e_temp = createExitEvent( pSwingMouseEvent );
266                 PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent( e_temp.getID(), e_temp, aEvent );
267                 dispatchEvent( prevComponent, e2 );
268 
269                 e_temp = createEnterEvent( comp, pSwingMouseEvent, offX, offY );
270                 e2 = PSwingMouseEvent.createMouseEvent( e_temp.getID(), e_temp, aEvent );
271                 comp.dispatchEvent( e2 );
272             }
273         }
274         else {
275             // This means mouseEntered
276             if( comp != null ) {
277                 MouseEvent e_temp = createEnterEvent( comp, pSwingMouseEvent, offX, offY );
278                 PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent( e_temp.getID(), e_temp, aEvent );
279                 dispatchEvent( comp, e2 );
280             }
281         }
282 
283         //todo add cursors
284 //        // We have to manager our own Cursors since this is normally
285 //        // done on the native side
286 //        if( comp != cursorComponent &&
287 //            focusNodeLeft == null &&
288 //            focusNodeMiddle == null &&
289 //            focusNodeRight == null ) {
290 //            if( comp != null ) {
291 //                cursorComponent = comp;
292 //                canvas.setCursor( comp.getCursor(), false );
293 //            }
294 //            else {
295 //                cursorComponent = null;
296 //                canvas.resetCursor();
297 //            }
298 //        }
299 
300         // Set the previous variables for next time
301         prevComponent = comp;
302 
303         if( comp != null ) {
304             prevOff = new Point2D.Double( offX, offY );
305         }
306     }

308     private MouseEvent createEnterEvent( Component comp, PSwingMouseEvent e1, int offX, int offY ) {
309         return new MouseEvent( comp, MouseEvent.MOUSE_ENTERED,
310                                e1.getWhen(), 0,
311                                (int)prevPoint.getX() - offX, (int)prevPoint.getY() - offY,
312                                e1.getClickCount(), e1.isPopupTrigger() );
313     }

315     private MouseEvent createExitEvent( PSwingMouseEvent e1 ) {
316         return new MouseEvent( prevComponent, MouseEvent.MOUSE_EXITED,
317                                e1.getWhen(), 0,
318                                (int)prevPoint.getX() - (int)prevOff.getX(), (int)prevPoint.getY() - (int)prevOff.getY(),
319                                e1.getClickCount(), e1.isPopupTrigger() );

322     private void handleButton( PSwingMouseEvent e1, PInputEvent aEvent, ButtonData buttonData ) {
323         Point2D pt;
324         if( buttonData.getPNode().isDescendentOf( canvas.getRoot() ) ) {
325             pt = new Point2D.Double( e1.getX(), e1.getY() );
326             cameraToLocal( e1.getPath().getTopCamera(), pt, buttonData.getPNode() );
327             //todo this probably won't handle viewing through multiple cameras.
328             MouseEvent e_temp = new MouseEvent( buttonData.getFocusedComponent(),
329                                                 e1.getID(), e1.getWhen(), e1.getModifiers(),
330                                                 (int)pt.getX() - buttonData.getOffsetX(),
331                                                 (int)pt.getY() - buttonData.getOffsetY(),
332                                                 e1.getClickCount(), e1.isPopupTrigger() );
333 
334             PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent( e_temp.getID(), e_temp, aEvent );
335             dispatchEvent( buttonData.getFocusedComponent(), e2 );
336         }
337         else {
338             dispatchEvent( buttonData.getFocusedComponent(), e1 );
339         }
340         //buttonData.getPSwing().repaint();  //Experiment with SliderExample (from Martin) suggests this line is unnecessary, and a serious problem in performance.
341         e1.consume();
342         if( e1.getID() == MouseEvent.MOUSE_RELEASED ) {
343             buttonData.mouseReleased();
344         }
345     }

355     private void cameraToLocal( PCamera topCamera, Point2D pt, PNode node ) {
356         AffineTransform inverse = null;
357         try {
358             inverse = topCamera.getViewTransform().createInverse();
359         }
360         catch( NoninvertibleTransformException e ) {
361             e.printStackTrace();
362         }
363         
364         /* Only apply the camera's view transform when this node is a descendant of PLayer */
365         PNode searchNode = node;
366         do {
367              searchNode = searchNode.getParent();
368              if (searchNode instanceof PLayer) {
369                  inverse.transform( pt, pt );
370                  break;
371              }
372         } while(searchNode != null);
373         
374         if( node != null ) {
375             node.globalToLocal( pt );
376         }
377         return;
378     }
