31  	public void initialize() {
32  		int numNodes = 50;
33  		int numEdges = 50;
34  
35  		// Initialize, and create a layer for the edges (always underneath the nodes)
36  		PLayer nodeLayer = getCanvas().getLayer();
37  		PLayer edgeLayer = new PLayer();
38  		getCanvas().getCamera().addLayer(0, edgeLayer);
39  		Random rnd = new Random();
40  		ArrayList tmp;
41  		for (int i = 0; i < numNodes; i++) {
42  			float x = (float) (300. * rnd.nextDouble());
43  			float y = (float) (400. * rnd.nextDouble());
44  			PPath path = PPath.createEllipse(x, y, 20, 20);
45  			tmp = new ArrayList();
46  			path.addAttribute("edges", tmp);
47  			nodeLayer.addChild(path);
48  		}
49  
50  		// Create some random edges
51  		// Each edge's Tag has an ArrayList used to store associated nodes
52  		for (int i = 0; i < numEdges; i++) {
53  			int n1 = rnd.nextInt(numNodes);
54  			int n2 = rnd.nextInt(numNodes);
55  			PNode node1 = nodeLayer.getChild(n1);
56  			PNode node2 = nodeLayer.getChild(n2);
57  
58  			Point2D.Double bound1 = (Point2D.Double) node1.getBounds().getCenter2D();
59  			Point2D.Double bound2 = (Point2D.Double) node2.getBounds().getCenter2D();
60  
61  			PPath edge = new PPath();
62  			edge.moveTo((float) bound1.getX(), (float) bound1.getY());
63  			edge.lineTo((float) bound2.getX(), (float) bound2.getY());
64  
65  			tmp = (ArrayList) node1.getAttribute("edges");
66  			tmp.add(edge);
67  			tmp = (ArrayList) node2.getAttribute("edges");
68  			tmp.add(edge);
69  
70  			tmp = new ArrayList();
71  			tmp.add(node1);
72  			tmp.add(node2);
73  			edge.addAttribute("nodes", tmp);
74  
75  			edgeLayer.addChild(edge);
76  		}
77  
78  		// Create event handler to move nodes and update edges
79  		nodeLayer.addInputEventListener(new NodeDragHandler());
80  	}

86  	/// <summary>
87  	/// Simple event handler which applies the following actions to every node it is called on:
88  	///   * Turn node red when the mouse goes over the node
89  	///   * Turn node white when the mouse exits the node
90  	///   * Drag the node, and associated edges on mousedrag
91  	/// It assumes that the node's Tag references an ArrayList with a list of associated
92  	/// edges where each edge is a PPath which each have a Tag that references an ArrayList
93  	/// with a list of associated nodes.
94  	/// </summary>
95  	class NodeDragHandler extends PDragSequenceEventHandler {
96  		public NodeDragHandler() {
97  			getEventFilter().setMarksAcceptedEventsAsHandled(true);
98  		}
99  		public void mouseEntered(PInputEvent e) {
100 			if (e.getButton() == 0) {
101 				e.getPickedNode().setPaint(Color.red);
102 			}
103 		}
104 
105 		public void mouseExited(PInputEvent e) {
106 			if (e.getButton() == 0) {
107 				e.getPickedNode().setPaint(Color.white);
108 			}
109 		}
110 
111 		public void drag(PInputEvent e) {
112 			PNode node = e.getPickedNode();
113 			node.translate(e.getDelta().width, e.getDelta().height);
114 
115 			ArrayList edges = (ArrayList) e.getPickedNode().getAttribute("edges");
116 
117 			int i;
118 			for (i = 0; i < edges.size(); i++) {
119 				PPath edge = (PPath) edges.get(i);
120 				ArrayList nodes = (ArrayList) edge.getAttribute("nodes");
121 				PNode node1 = (PNode) nodes.get(0);
122 				PNode node2 = (PNode) nodes.get(1);
123 
124 				edge.reset();
125 				// Note that the node's "FullBounds" must be used (instead of just the "Bound") 
126 				// because the nodes have non-identity transforms which must be included when
127 				// determining their position.
128 				Point2D.Double bound1 = (Point2D.Double) node1.getFullBounds().getCenter2D();
129 				Point2D.Double bound2 = (Point2D.Double) node2.getFullBounds().getCenter2D();
130 
131 				edge.moveTo((float) bound1.getX(), (float) bound1.getY());
132 				edge.lineTo((float) bound2.getX(), (float) bound2.getY());
133 			}
134 		}
135 	}
