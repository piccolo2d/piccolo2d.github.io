52  /**
53   * <b>PFixedWidthStroke</b> is the same as {@link java.awt.BasicStroke} except
54   * that PFixedWidthStroke has a fixed width on the screen so that even when the
55   * canvas view is zooming its width stays the same in canvas coordinates. Note
56   * that this stroke draws in the inside of the stroked shape, instead of the
57   * normal draw on center behavior.
58   * <P>

166     public PFixedWidthStroke(float width, int cap, int join, float miterlimit, float dash[], float dash_phase) {
167         if (width < 0.0f) {
168             throw new IllegalArgumentException("negative width");
169         }
170         if (cap != CAP_BUTT && cap != CAP_ROUND && cap != CAP_SQUARE) {
171             throw new IllegalArgumentException("illegal end cap value");
172         }
173         if (join == JOIN_MITER) {
174             if (miterlimit < 1.0f) {
175                 throw new IllegalArgumentException("miter limit < 1");
176             }
177         }
178         else if (join != JOIN_ROUND && join != JOIN_BEVEL) {
179             throw new IllegalArgumentException("illegal line join value");
180         }
181         if (dash != null) {
182             if (dash_phase < 0.0f) {
183                 throw new IllegalArgumentException("negative dash phase");
184             }
185             boolean allzero = true;
186             for (int i = 0; i < dash.length; i++) {
187                 float d = dash[i];
188                 if (d > 0.0) {
189                     allzero = false;
190                 }
191                 else if (d < 0.0) {
192                     throw new IllegalArgumentException("negative dash length");
193                 }
194             }
195 
196             if (allzero) {
197                 throw new IllegalArgumentException("dash lengths all zero");
198             }
199         }
200         this.width = width;
201         this.cap = cap;
202         this.join = join;
203         this.miterlimit = miterlimit;
204         if (dash != null) {
205             this.dash = (float[]) dash.clone();
206         }
207         this.dash_phase = dash_phase;
208     }

220     public Shape createStrokedShape(Shape s) {
221         FillAdapter filler = new FillAdapter();
222         PathStroker stroker = new PathStroker(filler);
223         PathConsumer consumer;
224 
225         // Fixed Width Additions, always stroke path inside shape.
226         // Also keeps dashes fixed when zooming (thanks to Shawn Castrianni -
227         // Dec 2006)
228         float fixedScale = 1.0f;
229 
230         if (PDebug.getProcessingOutput()) {
231             if (PPaintContext.CURRENT_PAINT_CONTEXT != null) {
232                 fixedScale = 1.0f / (float) PPaintContext.CURRENT_PAINT_CONTEXT.getScale();
233             }
234         }
235         else {
236             if (PPickPath.CURRENT_PICK_PATH != null) {
237                 fixedScale = 1.0f / (float) PPickPath.CURRENT_PICK_PATH.getScale();
238             }
239         }
240         float fixedWidth = width * fixedScale;
241 
242         Rectangle2D bounds = s.getBounds2D();
243         double scale = 1.0;
244 
245         if (bounds.getWidth() > bounds.getHeight()) {
246             if (bounds.getWidth() != 0) {
247                 scale = (bounds.getWidth() - fixedWidth) / bounds.getWidth();
248             }
249         }
250         else {
251             if (bounds.getHeight() != 0) {
252                 scale = (bounds.getHeight() - fixedWidth) / bounds.getHeight();
253             }
254         }
255 
256         TEMP_TRANSFORM.setToIdentity();
257         TEMP_TRANSFORM.scaleAboutPoint(scale, bounds.getCenterX(), bounds.getCenterY());
258         stroker.setPenDiameter(fixedWidth);
259         PathIterator pi = s.getPathIterator(TEMP_TRANSFORM);
260 
261         stroker.setPenT4(null);
262         stroker.setCaps(RasterizerCaps[cap]);
263         stroker.setCorners(RasterizerCorners[join], miterlimit);
264         if (dash != null) {
265             // Fixed Width Additions
266             float fixedDash[] = new float[dash.length];
267             for (int i = 0; i < dash.length; i++) {
268                 fixedDash[i] = dash[i] * fixedScale;
269             }
270             float fixedDashPhase = dash_phase * fixedScale;
271             PathDasher dasher = new PathDasher(stroker);
272             dasher.setDash(fixedDash, fixedDashPhase);
273             dasher.setDashT4(null);
274             consumer = dasher;
275         }
276         else {
277             consumer = stroker;
278         }
279 
280         try {
281             consumer.beginPath();
282             boolean pathClosed = false;
283             float mx = 0.0f;
284             float my = 0.0f;
285             float point[] = new float[6];
286 
287             while (!pi.isDone()) {
288                 int type = pi.currentSegment(point);
289                 if (pathClosed == true) {
290                     pathClosed = false;
291                     if (type != PathIterator.SEG_MOVETO) {
292                         // Force current point back to last moveto point
293                         consumer.beginSubpath(mx, my);
294                     }
295                 }
296                 switch (type) {
297                     case PathIterator.SEG_MOVETO:
298                         mx = point[0];
299                         my = point[1];
300                         consumer.beginSubpath(point[0], point[1]);
301                         break;
302                     case PathIterator.SEG_LINETO:
303                         consumer.appendLine(point[0], point[1]);
304                         break;
305                     case PathIterator.SEG_QUADTO:
306                         // Quadratic curves take two points
307                         consumer.appendQuadratic(point[0], point[1], point[2], point[3]);
308                         break;
309                     case PathIterator.SEG_CUBICTO:
310                         // Cubic curves take three points
311                         consumer.appendCubic(point[0], point[1], point[2], point[3], point[4], point[5]);
312                         break;
313                     case PathIterator.SEG_CLOSE:
314                         consumer.closedSubpath();
315                         pathClosed = true;
316                         break;
317                 }
318                 pi.next();
319             }
320 
321             consumer.endPath();
322 
323             consumer.dispose(); // hack to fix memory leak, shouldn't be
324                                 // neccessary but is.
325         }
326         catch (PathException e) {
327             throw new InternalError("Unable to Stroke shape (" + e.getMessage() + ")");
328         }
329 
330         return filler.getShape();
331     }

333     public boolean equals(Object obj) {
334         if (!(obj instanceof PFixedWidthStroke)) {
335             return false;
336         }
337 
338         PFixedWidthStroke bs = (PFixedWidthStroke) obj;
339         if (width != bs.width) {
340             return false;
341         }
342 
343         if (join != bs.join) {
344             return false;
345         }
346 
347         if (cap != bs.cap) {
348             return false;
349         }
350 
351         if (miterlimit != bs.miterlimit) {
352             return false;
353         }
354 
355         if (dash != null) {
356             if (dash_phase != bs.dash_phase) {
357                 return false;
358             }
359 
360             if (!java.util.Arrays.equals(dash, bs.dash)) {
361                 return false;
362             }
363         }
364         else if (bs.dash != null) {
365             return false;
366         }
367 
368         return true;
369     }

399     public int hashCode() {
400         int hash = Float.floatToIntBits(width);
401         hash = hash * 31 + join;
402         hash = hash * 31 + cap;
403         hash = hash * 31 + Float.floatToIntBits(miterlimit);
404         if (dash != null) {
405             hash = hash * 31 + Float.floatToIntBits(dash_phase);
406             for (int i = 0; i < dash.length; i++) {
407                 hash = hash * 31 + Float.floatToIntBits(dash[i]);
408             }
409         }
410         return hash;
