48  /**
49   * <b>PInputEvent</b> is used to notify PInputEventListeners of keyboard and
50   * mouse input. It has methods for normal event properties such as event
51   * modifier keys and event canvas location.
52   * <P>
53   * In addition is has methods to get the mouse position and delta in a variety
54   * of coordinate systems.
55   * <P>
56   * Last of all it provides access to the dispatch manager that can be queried to
57   * find the current mouse over, mouse focus, and keyboard focus.
58   * <P>
59   * 
60   * @version 1.0
61   * @author Jesse Grosjean
62   */
63  public class PInputEvent {
64  
65      private InputEvent inputEvent;
66      private PPickPath pickPath;
67      private PInputManager inputManager;
68      private boolean handled;
69  
70      public PInputEvent(PInputManager inputManager, InputEvent event) {
71          super();
72          inputEvent = event;
73          this.inputManager = inputManager;
74      }

323     public boolean isMouseEnteredOrMouseExited() {
324         if (isMouseEvent()) {
325             return inputEvent.getID() == MouseEvent.MOUSE_ENTERED || inputEvent.getID() == MouseEvent.MOUSE_EXITED;

330     /**
331      * Returns whether or not this event is a popup menu trigger event for the
332      * platform. Must not be called if this event isn't a mouse event.
333      * <p>
334      * <b>Note</b>: Popup menus are triggered differently on different systems.
335      * Therefore, <code>isPopupTrigger</code> should be checked in both
336      * <code>mousePressed</code> and <code>mouseReleased</code> for proper
337      * cross-platform functionality.
338      * 
339      * @return boolean, true if this event triggers a popup menu for this
340      *         platform
341      * @throws IllegalStateException if this event is not a mouse event
342      */
343     public boolean isPopupTrigger() {
344         if (isMouseEvent()) {
345             return ((MouseEvent) inputEvent).isPopupTrigger();
346         }
347         throw new IllegalStateException("Can't get clickcount from key event");
348     }

356     /**
357      * Return the mouse position in PCanvas coordinates.
358      */
359     public Point2D getCanvasPosition() {
360         return (Point2D) inputManager.getCurrentCanvasPosition().clone();
361     }
362 
363     /**
364      * Return the delta between the last and current mouse position in PCanvas
365      * coordinates.
366      */
367     public PDimension getCanvasDelta() {
368         Point2D last = inputManager.getLastCanvasPosition();
369         Point2D current = inputManager.getCurrentCanvasPosition();
370         return new PDimension(current.getX() - last.getX(), current.getY() - last.getY());
371     }
372 
373     /**
374      * Return the mouse position relative to a given node on the pick path.
375      */
376     public Point2D getPositionRelativeTo(PNode nodeOnPath) {
377         Point2D r = getCanvasPosition();
378         return pickPath.canvasToLocal(r, nodeOnPath);
379     }
380 
381     /**
382      * Return the delta between the last and current mouse positions relative to
383      * a given node on the pick path.
384      */
385     public PDimension getDeltaRelativeTo(PNode nodeOnPath) {
386         PDimension r = getCanvasDelta();
387         return (PDimension) pickPath.canvasToLocal(r, nodeOnPath);
388     }
389 
390     /**
391      * Return the mouse position transformed through the view transform of the
392      * bottom camera.
393      */
394     public Point2D getPosition() {
395         Point2D r = getCanvasPosition();
396         pickPath.canvasToLocal(r, getCamera());
397         return getCamera().localToView(r);
398     }
399 
400     /**
401      * Return the delta between the last and current mouse positions transformed
402      * through the view transform of the bottom camera.
403      */
404     public PDimension getDelta() {
405         PDimension r = getCanvasDelta();
406         pickPath.canvasToLocal(r, getCamera());
407         return (PDimension) getCamera().localToView(r);
408     }
