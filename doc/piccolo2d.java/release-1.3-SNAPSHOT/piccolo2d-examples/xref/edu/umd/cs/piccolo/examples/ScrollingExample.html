27  /**
28   * This creates a simple scene and allows switching between
29   * traditional scrolling where the scrollbars control the view
30   * and alternate scrolling where the scrollbars control the
31   * document.  In laymans terms - in traditional window scrolling the stuff
32   * in the window moves in the opposite direction of the scroll bars and in
33   * document scrolling the stuff moves in the same direction as the scrollbars.
34   *
35   * Toggle buttons are provided to toggle between these two types
36   * of scrolling.
37   *
38   * @author Lance Good
39   * @author Ben Bederson
40   */
41  public class ScrollingExample extends PFrame {

109 	/**
110 	 * A modified scroll director that performs document based scroling
111 	 * rather than window based scrolling (ie. the scrollbars act in
112 	 * the inverse direction as normal)
113 	 */
114 	public class DocumentScrollDirector extends PDefaultScrollDirector {
115 
116 		/**
117 		 * Get the View position given the specified camera bounds - modified
118 		 * such that:
119 		 *
120 		 * Rather than finding the distance from the upper left corner
121 		 * of the window to the upper left corner of the document -
122 		 * we instead find the distance from the lower right corner
123 		 * of the window to the upper left corner of the document
124 		 * THEN we subtract that value from total document width so
125 		 * that the position is inverted
126 		 *
127 		 * @param viewBounds The bounds for which the view position will be computed
128 		 * @return The view position
129 		 */
130 		public Point getViewPosition(Rectangle2D viewBounds) {
131 			Point pos = new Point();
132 			if (camera != null) {
133 				// First we compute the union of all the layers
134 				PBounds layerBounds = new PBounds();
135 				java.util.List layers = camera.getLayersReference();
136 				for (Iterator i = layers.iterator(); i.hasNext();) {
137 					PLayer layer = (PLayer) i.next();
138 					layerBounds.add(layer.getFullBoundsReference());
139 				}
140 
141 				// Then we put the bounds into camera coordinates and
142 				// union the camera bounds
143 				camera.viewToLocal(layerBounds);
144 				layerBounds.add(viewBounds);
145 
146 				// Rather than finding the distance from the upper left corner
147 				// of the window to the upper left corner of the document -
148 				// we instead find the distance from the lower right corner
149 				// of the window to the upper left corner of the document
150 				// THEN we measure the offset from the lower right corner
151 				// of the document
152 				pos.setLocation(
153 					(int) (layerBounds.getWidth() - (viewBounds.getX() + viewBounds.getWidth() - layerBounds.getX()) + 0.5),
154 					(int) (layerBounds.getHeight() - (viewBounds.getY() + viewBounds.getHeight() - layerBounds.getY()) + 0.5));
155 			}
156 
157 			return pos;
158 
159 		}
160 
161 		/**
162 		 * We do the same thing we did in getViewPosition above to flip the
163 		 * document-window position relationship
164 		 *
165 		 * @param x The new x position
166 		 * @param y The new y position
167 		 */
168 		public void setViewPosition(double x, double y) {
169 			if (camera != null) {
170 				// If a scroll is in progress - we ignore new scrolls -
171 				// if we didn't, since the scrollbars depend on the camera location
172 				// we can end up with an infinite loop
173 				if (!scrollInProgress) {
174 					scrollInProgress = true;
175 
176 					// Get the union of all the layers' bounds
177 					PBounds layerBounds = new PBounds();
178 					java.util.List layers = camera.getLayersReference();
179 					for (Iterator i = layers.iterator(); i.hasNext();) {
180 						PLayer layer = (PLayer) i.next();
181 						layerBounds.add(layer.getFullBoundsReference());
182 					}
183 
184 					PAffineTransform at = camera.getViewTransform();
185 					at.transform(layerBounds,layerBounds);
186 
187 					// Union the camera view bounds
188 					PBounds viewBounds = camera.getBoundsReference();
189 					layerBounds.add(viewBounds);
190 
191 					// Now find the new view position in view coordinates -
192 					// This is basically the distance from the lower right
193 					// corner of the window to the upper left corner of the
194 					// document 
195 					// We then measure the offset from the lower right corner
196 					// of the document
197 					Point2D newPoint =
198 						new Point2D.Double(
199 							layerBounds.getX() + layerBounds.getWidth() - (x + viewBounds.getWidth()),
200 							layerBounds.getY() + layerBounds.getHeight() - (y + viewBounds.getHeight()));
201 
202 					// Now transform the new view position into global coords
203 					camera.localToView(newPoint);
204 
205 					// Compute the new matrix values to put the camera at the
206 					// correct location
207 					double newX = - (at.getScaleX() * newPoint.getX() + at.getShearX() * newPoint.getY());
208 					double newY = - (at.getShearY() * newPoint.getX() + at.getScaleY() * newPoint.getY());
209 
210 					at.setTransform(at.getScaleX(), at.getShearY(), at.getShearX(), at.getScaleY(), newX, newY);
211 
212 					// Now actually set the camera's transform
213 					camera.setViewTransform(at);
214 					scrollInProgress = false;
215 				}
216 			}
217 		}
218 	}
