22  /**
23   * An extension to PCamera that provides a fast image based animationToCenterBounds method

25   * @author Lance Good

27  public class PCacheCamera extends PCamera {

30  	private boolean imageAnimate;
31  	private PBounds imageAnimateBounds;

33  	/**
34  	 * Get the buffer used to provide fast image based animation 
35  	 */
36  	protected BufferedImage getPaintBuffer() {
37  	    PBounds fRef = getFullBoundsReference();
38  	    if (paintBuffer == null || paintBuffer.getWidth() < fRef.getWidth() || paintBuffer.getHeight() < fRef.getHeight()) {
39  	        paintBuffer = GraphicsEnvironment.getLocalGraphicsEnvironment().getDefaultScreenDevice().getDefaultConfiguration().createCompatibleImage((int)Math.ceil(fRef.getWidth()),(int)Math.ceil(fRef.getHeight()));
40  	    }
41  	    return paintBuffer;
42  	}

44  	/**
45  	 * Caches the information necessary to animate from the current view bounds to the
46  	 * specified centerBounds 
47  	 */
48  	private AffineTransform cacheViewBounds(Rectangle2D centerBounds, boolean scaleToFit) {
49  	    PBounds viewBounds = getViewBounds();	    
50  	    
51  	    // Initialize the image to the union of the current and destination bounds
52  	    PBounds imageBounds = new PBounds(viewBounds); 
53  	    imageBounds.add(centerBounds);
54  
55  	    animateViewToCenterBounds(imageBounds,scaleToFit,0);
56  
57          imageAnimateBounds = getViewBounds();
58  
59          // Now create the actual cache image that we will use to animate fast 
60          
61  		BufferedImage buffer = getPaintBuffer();
62  		Paint fPaint = Color.white;
63  		if (getPaint() != null) {
64  		    fPaint = getPaint();
65  		}
66  		toImage(buffer,fPaint);
67  
68  		// Do this after the painting above!
69  		imageAnimate = true;
70  		
71  		// Return the bounds to the previous viewbounds
72  		animateViewToCenterBounds(viewBounds,scaleToFit,0);
73  		
74  		// The code below is just copied from animateViewToCenterBounds to create the
75  		// correct transform to center the specified bounds
76  		
77  		PDimension delta = viewBounds.deltaRequiredToCenter(centerBounds);
78  		PAffineTransform newTransform = getViewTransform();
79  		newTransform.translate(delta.width, delta.height);
80  		
81  		if (scaleToFit) {
82  			double s = Math.min(viewBounds.getWidth() / centerBounds.getWidth(), viewBounds.getHeight() / centerBounds.getHeight());
83  			newTransform.scaleAboutPoint(s, centerBounds.getCenterX(), centerBounds.getCenterY());
84  		}
85  		
86  		return newTransform;
87  	}

89  	/**
90  	 * Turns off the fast image animation and does any other applicable cleanup
91  	 */
92  	private void clearViewCache() {
93  	    imageAnimate = false;
94  	    imageAnimateBounds = null;
95  	}

97  	/**
98  	 *  Mimics the standard animateViewToCenterBounds but uses a cached image for performance
99  	 *  rather than re-rendering the scene at each step 
100 	 */
101 	public PTransformActivity animateStaticViewToCenterBoundsFast(Rectangle2D centerBounds, boolean shouldScaleToFit, long duration) {
102 	    if (duration == 0) {
103 	        return animateViewToCenterBounds(centerBounds,shouldScaleToFit,duration);	        
104 	    }
105 	    
106 		AffineTransform newViewTransform = cacheViewBounds(centerBounds,shouldScaleToFit);		
107 
108 		return animateStaticViewToTransformFast(newViewTransform, duration);
109 	}

111 	/**
112 	 * This copies the behavior of the standard animateViewToTransform but clears the cache
113 	 * when it is done
114 	 */
115 	protected PTransformActivity animateStaticViewToTransformFast(AffineTransform destination, long duration) {
116 		if (duration == 0) {
117 			setViewTransform(destination);
118 			return null;
119 		}
120 		
121 		PTransformActivity.Target t = new PTransformActivity.Target() {
122 			public void setTransform(AffineTransform aTransform) {
123 				PCacheCamera.this.setViewTransform(aTransform);
124 			}
125 			public void getSourceMatrix(double[] aSource) {
126 				getViewTransformReference().getMatrix(aSource);
127 			}
128 		};
129 		
130 		PTransformActivity ta = new PTransformActivity(duration, PUtil.DEFAULT_ACTIVITY_STEP_RATE, t, destination) {
131             protected void activityFinished() {
132                 clearViewCache();
133                 repaint();
134                 super.activityFinished();
135             }
136 		};
137 		
138 		PRoot r = getRoot();
139 		if (r != null) {
140 			r.getActivityScheduler().addActivity(ta);
141 		}
142 		
143 		return ta;
144 	}

146 	/**
147 	 * Overrides the camera's full paint method to do the fast rendering when possible
148 	 */
149     public void fullPaint(PPaintContext paintContext) {
150 		if (imageAnimate) {
151 		    PBounds fRef = getFullBoundsReference();
152 		    PBounds viewBounds = getViewBounds();
153 		    double scale = getFullBoundsReference().getWidth()/imageAnimateBounds.getWidth();
154 		    double xOffset = (viewBounds.getX()-imageAnimateBounds.getX())*scale;
155 		    double yOffset = (viewBounds.getY()-imageAnimateBounds.getY())*scale;
156 		    double scaleW = viewBounds.getWidth()*scale;
157 		    double scaleH = viewBounds.getHeight()*scale;
158 		    paintContext.getGraphics().drawImage(paintBuffer,0,0,(int)Math.ceil(fRef.getWidth()),(int)Math.ceil(fRef.getHeight()),
159 		            (int)Math.floor(xOffset),(int)Math.floor(yOffset),(int)Math.ceil(xOffset+scaleW),(int)Math.ceil(yOffset+scaleH),null);
160 		}
161 		else {
162 		    super.fullPaint(paintContext);
163 		}
164     }
