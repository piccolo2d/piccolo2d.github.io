12  /**
13   * This example shows how a node can get the keyboard focus.
14   */
15  public class KeyEventFocusExample extends PFrame {
16  
17      public KeyEventFocusExample() {
18          this(null);
19      }
20  
21      public KeyEventFocusExample(PCanvas aCanvas) {
22          super("KeyEventFocusExample", false, aCanvas);
23      }
24  
25      public void initialize() {
26          // Create a green and red node and add them to canvas layer.
27          PCanvas canvas = getCanvas();
28          PNode nodeGreen = PPath.createRectangle(0, 0, 100, 100);
29          PNode nodeRed = PPath.createRectangle(0, 0, 100, 100);
30          nodeRed.translate(200, 0);
31          nodeGreen.setPaint(Color.green);
32          nodeRed.setPaint(Color.red);
33          canvas.getLayer().addChild(nodeGreen);
34          canvas.getLayer().addChild(nodeRed);
35  
36          // Add an event handler to the green node the prints
37          // "green mousepressed"
38          // when the mouse is pressed on the green node, and "green keypressed"
39          // when
40          // the key is pressed and the event listener has keyboard focus.
41          nodeGreen.addInputEventListener(new PBasicInputEventHandler() {
42              public void keyPressed(PInputEvent event) {
43                  System.out.println("green keypressed");
44              }
45  
46              // Key board focus is managed by the PInputManager, accessible from
47              // the root object, or from an incoming PInputEvent. In this case
48              // when
49              // the mouse is pressed in the green node, then the event handler
50              // associated
51              // with it will set the keyfocus to itself. Now it will receive key
52              // events
53              // until someone else gets the focus.
54              public void mousePressed(PInputEvent event) {
55                  event.getInputManager().setKeyboardFocus(event.getPath());
56                  System.out.println("green mousepressed");
57              }
58  
59              public void keyboardFocusGained(PInputEvent event) {
60                  System.out.println("green focus gained");
61              }
62  
63              public void keyboardFocusLost(PInputEvent event) {
64                  System.out.println("green focus lost");
65              }
66          });
67  
68          // do the same thing with the red node.
69          nodeRed.addInputEventListener(new PBasicInputEventHandler() {
70              public void keyPressed(PInputEvent event) {
71                  System.out.println("red keypressed");
72              }
73  
74              public void mousePressed(PInputEvent event) {
75                  event.getInputManager().setKeyboardFocus(event.getPath());
76                  System.out.println("red mousepressed");
77              }
78  
79              public void keyboardFocusGained(PInputEvent event) {
80                  System.out.println("red focus gained");
81              }
82  
83              public void keyboardFocusLost(PInputEvent event) {
84                  System.out.println("red focus lost");
85              }
86          });
87      }

90          new KeyEventFocusExample();
91      }
