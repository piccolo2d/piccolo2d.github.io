69      /**
70       * Sets whether this event handler can fire events.
71       * 
72       * @param active
73       */
74      void setActive(boolean active) {
75          if (this.active && !active) {
76              if (listenNode != null) {
77                  this.active = false;
78                  listenNode.removeInputEventListener(this);
79              }
80          }
81          else if (!this.active && active) {
82              if (listenNode != null) {
83                  this.active = true;
84                  listenNode.addInputEventListener(this);
85              }
86          }
87      }

98      /**
99       * Finds the component at the specified location (must be showing).
100      * 
101      * @param c
102      * @param x
103      * @param y
104      * @return the component at the specified location.
105      */
106     private Component findShowingComponentAt(Component c, int x, int y) {
107         if (!c.contains(x, y)) {
108             return null;
109         }
110 
111         if (c instanceof Container) {
112             Container contain = ((Container) c);
113             int ncomponents = contain.getComponentCount();
114             Component component[] = contain.getComponents();
115 
116             for (int i = 0; i < ncomponents; i++) {
117                 Component comp = component[i];
118                 if (comp != null) {
119                     Point p = comp.getLocation();
120                     if (comp instanceof Container) {
121                         comp = findShowingComponentAt(comp, x - (int) p.getX(), y - (int) p.getY());
122                     }
123                     else {
124                         comp = comp.getComponentAt(x - (int) p.getX(), y - (int) p.getY());
125                     }
126                     if (comp != null && comp.isShowing()) {
127                         return comp;
128                     }
129                 }
130             }
131         }
132         return c;
133     }

135     /**
136      * Determines if any Swing components in Piccolo should receive the given
137      * MouseEvent and forwards the event to that component. However,
138      * mouseEntered and mouseExited are independent of the buttons. Also, notice
139      * the notes on mouseEntered and mouseExited.
140      * 
141      * @param pSwingMouseEvent
142      * @param aEvent
143      */
144     void dispatchEvent(PSwingMouseEvent pSwingMouseEvent, PInputEvent aEvent) {
145         Component comp = null;
146         Point2D pt = null;
147         PNode pickedNode = pSwingMouseEvent.getPath().getPickedNode();
148 
149         // The offsets to put the event in the correct context
150         int offX = 0;
151         int offY = 0;
152 
153         PNode currentNode = pSwingMouseEvent.getCurrentNode();
154 
155         if (currentNode instanceof PSwing) {
156 
157             PSwing swing = (PSwing) currentNode;
158             PNode grabNode = pickedNode;
159 
160             if (grabNode.isDescendentOf(canvas.getRoot())) {
161                 pt = new Point2D.Double(pSwingMouseEvent.getX(), pSwingMouseEvent.getY());
162                 cameraToLocal(pSwingMouseEvent.getPath().getTopCamera(), pt, grabNode);
163                 prevPoint = new Point2D.Double(pt.getX(), pt.getY());
164 
165                 // This is only partially fixed to find the deepest
166                 // component at pt. It needs to do something like
167                 // package private method:
168                 // Container.getMouseEventTarget(int,int,boolean)
169                 comp = findShowingComponentAt(swing.getComponent(), (int) pt.getX(), (int) pt.getY());
170 
171                 // We found the right component - but we need to
172                 // get the offset to put the event in the component's
173                 // coordinates
174                 if (comp != null && comp != swing.getComponent()) {
175                     for (Component c = comp; c != swing.getComponent(); c = c.getParent()) {
176                         offX += c.getLocation().getX();
177                         offY += c.getLocation().getY();
178                     }
179                 }
180 
181                 // Mouse Pressed gives focus - effects Mouse Drags and
182                 // Mouse Releases
183                 if (comp != null && pSwingMouseEvent.getID() == MouseEvent.MOUSE_PRESSED) {
184                     if (SwingUtilities.isLeftMouseButton(pSwingMouseEvent)) {
185                         leftButtonData.setState(swing, pickedNode, comp, offX, offY);
186                     }
187                     else if (SwingUtilities.isMiddleMouseButton(pSwingMouseEvent)) {
188                         middleButtonData.setState(swing, pickedNode, comp, offX, offY);
189                     }
190                     else if (SwingUtilities.isRightMouseButton(pSwingMouseEvent)) {
191                         rightButtonData.setState(swing, pickedNode, comp, offX, offY);
192                     }
193                 }
194             }
195         }
196 
197         // This first case we don't want to give events to just
198         // any Swing component - but to the one that got the
199         // original mousePressed
200         if (pSwingMouseEvent.getID() == MouseEvent.MOUSE_DRAGGED
201                 || pSwingMouseEvent.getID() == MouseEvent.MOUSE_RELEASED) {
202 
203             // LEFT MOUSE BUTTON
204             if (SwingUtilities.isLeftMouseButton(pSwingMouseEvent) && leftButtonData.getFocusedComponent() != null) {
205                 handleButton(pSwingMouseEvent, aEvent, leftButtonData);
206             }
207 
208             // MIDDLE MOUSE BUTTON
209             if (SwingUtilities.isMiddleMouseButton(pSwingMouseEvent) && middleButtonData.getFocusedComponent() != null) {
210                 handleButton(pSwingMouseEvent, aEvent, middleButtonData);
211             }
212 
213             // RIGHT MOUSE BUTTON
214             if (SwingUtilities.isRightMouseButton(pSwingMouseEvent) && rightButtonData.getFocusedComponent() != null) {
215                 handleButton(pSwingMouseEvent, aEvent, rightButtonData);
216             }
217         }
218         // This case covers the cases mousePressed, mouseClicked,
219         // and mouseMoved events
220         else if ((pSwingMouseEvent.getID() == MouseEvent.MOUSE_PRESSED
221                 || pSwingMouseEvent.getID() == MouseEvent.MOUSE_CLICKED || pSwingMouseEvent.getID() == MouseEvent.MOUSE_MOVED)
222                 && (comp != null)) {
223 
224             MouseEvent e_temp = new MouseEvent(comp, pSwingMouseEvent.getID(), pSwingMouseEvent.getWhen(),
225                     pSwingMouseEvent.getModifiers(), (int) pt.getX() - offX, (int) pt.getY() - offY, pSwingMouseEvent
226                             .getClickCount(), pSwingMouseEvent.isPopupTrigger());
227 
228             PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent(e_temp.getID(), e_temp, aEvent);
229             dispatchEvent(comp, e2);
230             pSwingMouseEvent.consume();
231         }
232 
233         // Now we need to check if an exit or enter event needs to
234         // be dispatched - this code is independent of the mouseButtons.
235         // I tested in normal Swing to see the correct behavior.
236         if (prevComponent != null) {
237             // This means mouseExited
238 
239             // This shouldn't happen - since we're only getting node events
240             if (comp == null || pSwingMouseEvent.getID() == MouseEvent.MOUSE_EXITED) {
241                 MouseEvent e_temp = createExitEvent(pSwingMouseEvent);
242 
243                 PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent(e_temp.getID(), e_temp, aEvent);
244 
245                 dispatchEvent(prevComponent, e2);
246                 prevComponent = null;
247 
248                 if (pSwingMouseEvent.getID() == MouseEvent.MOUSE_EXITED) {
249                     pSwingMouseEvent.consume();
250                 }
251             }
252 
253             // This means mouseExited prevComponent and mouseEntered comp
254             else if (prevComponent != comp) {
255                 MouseEvent e_temp = createExitEvent(pSwingMouseEvent);
256                 PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent(e_temp.getID(), e_temp, aEvent);
257                 dispatchEvent(prevComponent, e2);
258 
259                 e_temp = createEnterEvent(comp, pSwingMouseEvent, offX, offY);
260                 e2 = PSwingMouseEvent.createMouseEvent(e_temp.getID(), e_temp, aEvent);
261                 comp.dispatchEvent(e2);
262             }
263         }
264         else {
265             // This means mouseEntered
266             if (comp != null) {
267                 MouseEvent e_temp = createEnterEvent(comp, pSwingMouseEvent, offX, offY);
268                 PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent(e_temp.getID(), e_temp, aEvent);
269                 dispatchEvent(comp, e2);
270             }
271         }
272 
273         // todo add cursors
274         // // We have to manager our own Cursors since this is normally
275         // // done on the native side
276         // if( comp != cursorComponent &&
277         // focusNodeLeft == null &&
278         // focusNodeMiddle == null &&
279         // focusNodeRight == null ) {
280         // if( comp != null ) {
281         // cursorComponent = comp;
282         // canvas.setCursor( comp.getCursor(), false );
283         // }
284         // else {
285         // cursorComponent = null;
286         // canvas.resetCursor();
287         // }
288         // }
289 
290         // Set the previous variables for next time
291         prevComponent = comp;
292 
293         if (comp != null) {
294             prevOff = new Point2D.Double(offX, offY);
295         }
296     }

298     private MouseEvent createEnterEvent(Component comp, PSwingMouseEvent e1, int offX, int offY) {
299         return new MouseEvent(comp, MouseEvent.MOUSE_ENTERED, e1.getWhen(), 0, (int) prevPoint.getX() - offX,
300                 (int) prevPoint.getY() - offY, e1.getClickCount(), e1.isPopupTrigger());
301     }
302 
303     private MouseEvent createExitEvent(PSwingMouseEvent e1) {
304         return new MouseEvent(prevComponent, MouseEvent.MOUSE_EXITED, e1.getWhen(), 0, (int) prevPoint.getX()
305                 - (int) prevOff.getX(), (int) prevPoint.getY() - (int) prevOff.getY(), e1.getClickCount(), e1
306                 .isPopupTrigger());
307     }

309     private void handleButton(PSwingMouseEvent e1, PInputEvent aEvent, ButtonData buttonData) {
310         Point2D pt;
311         if (buttonData.getPNode().isDescendentOf(canvas.getRoot())) {
312             pt = new Point2D.Double(e1.getX(), e1.getY());
313             cameraToLocal(e1.getPath().getTopCamera(), pt, buttonData.getPNode());
314             // todo this probably won't handle viewing through multiple cameras.
315             MouseEvent e_temp = new MouseEvent(buttonData.getFocusedComponent(), e1.getID(), e1.getWhen(), e1
316                     .getModifiers(), (int) pt.getX() - buttonData.getOffsetX(), (int) pt.getY()
317                     - buttonData.getOffsetY(), e1.getClickCount(), e1.isPopupTrigger());
318 
319             PSwingMouseEvent e2 = PSwingMouseEvent.createMouseEvent(e_temp.getID(), e_temp, aEvent);
320             dispatchEvent(buttonData.getFocusedComponent(), e2);
321         }
322         else {
323             dispatchEvent(buttonData.getFocusedComponent(), e1);
324         }
325         // buttonData.getPSwing().repaint(); //Experiment with SliderExample
326         // (from Martin) suggests this line is unnecessary, and a serious
327         // problem in performance.
328         e1.consume();
329         if (e1.getID() == MouseEvent.MOUSE_RELEASED) {
330             buttonData.mouseReleased();
331         }
332     }

342     private void cameraToLocal(PCamera topCamera, Point2D pt, PNode node) {
343         AffineTransform inverse = null;
344         try {
345             inverse = topCamera.getViewTransform().createInverse();
346         }
347         catch (NoninvertibleTransformException e) {
348             e.printStackTrace();
349         }
350 
351         /*
352          * Only apply the camera's view transform when this node is a descendant
353          * of PLayer
354          */
355         PNode searchNode = node;
356         do {
357             searchNode = searchNode.getParent();
358             if (searchNode instanceof PLayer) {
359                 inverse.transform(pt, pt);
360                 break;
361             }
362         } while (searchNode != null);
363 
364         if (node != null) {
365             node.globalToLocal(pt);
366         }
367         return;
368     }

370     /**
371      * Process a piccolo event and (if active) dispatch the corresponding Swing
372      * event.
373      * 
374      * @param aEvent
375      * @param type
376      */
377     public void processEvent(PInputEvent aEvent, int type) {
378         if (aEvent.isMouseEvent()) {
379             InputEvent sourceSwingEvent = aEvent.getSourceSwingEvent();
380             if (sourceSwingEvent instanceof MouseEvent) {
381                 MouseEvent swingMouseEvent = (MouseEvent) sourceSwingEvent;
382                 PSwingMouseEvent pSwingMouseEvent = PSwingMouseEvent.createMouseEvent(swingMouseEvent.getID(),
383                         swingMouseEvent, aEvent);
384                 if (!recursing) {
385                     recursing = true;
386                     dispatchEvent(pSwingMouseEvent, aEvent);
387                     recursing = false;
388                 }
389             }
390             else {
391                 new Exception("PInputEvent.getSourceSwingEvent was not a MouseEvent.  Actual event: "
392                         + sourceSwingEvent + ", class=" + sourceSwingEvent.getClass().getName()).printStackTrace();
393             }
394         }
395 
396         /*
397          * if( !( EventQueue.getCurrentEvent() instanceof MouseEvent ) ) { new
398          * Exception(
399          * "EventQueue.getCurrentEvent was not a MouseEvent, consider making PInputEvent.getSourceSwingEvent public.  Actual event: "
400          * + EventQueue.getCurrentEvent() + ", class=" +
401          * EventQueue.getCurrentEvent().getClass().getName()
402          * ).printStackTrace(); } if( aEvent.isMouseEvent() &&
403          * EventQueue.getCurrentEvent() instanceof MouseEvent ) { MouseEvent
404          * sourceSwingEvent = (MouseEvent)EventQueue.getCurrentEvent();
405          * PSwingMouseEvent pSwingMouseEvent =
406          * PSwingMouseEvent.createMouseEvent( sourceSwingEvent.getID(),
407          * sourceSwingEvent, aEvent ); if( !recursing ) { recursing = true;
408          * dispatchEvent( pSwingMouseEvent, aEvent ); recursing = false; } }
409          */
