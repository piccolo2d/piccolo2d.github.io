45          JPanel tabPanel = new JPanel( false );

64          panel.setLocation( 350, 350 );

75          buttonPreset.setLocation( 240, 285 );
76          tabPanel.add( buttonPreset );
77          // Create a pswing object for the tab panel

94      private JPanel createNestedPanel() {
95          // A panel MUST be created with double buffering off
96          JPanel panel;
97          JLabel label;
98          panel = new JPanel( false );
99          panel.setLayout( new BorderLayout() );
100         label = new JLabel( "A Panel within a panel" );
101         label.setHorizontalAlignment( SwingConstants.CENTER );
102         label.setForeground( Color.white );
103         JLabel label2 = new JLabel( "A Panel within a panel" );
104         label2.setHorizontalAlignment( SwingConstants.CENTER );
105         JSlider slider = new JSlider();
106         JCheckBox cbox1 = new JCheckBox( "Checkbox 1" );
107         JCheckBox cbox2 = new JCheckBox( "Checkbox 2" );
108         JPanel panel3 = new JPanel( false );
109         panel3.setLayout( new BoxLayout( panel3, BoxLayout.PAGE_AXIS ) );
110         panel3.setBorder( new EmptyBorder( 3, 3, 3, 3 ) );
111         panel3.add( label2 );
112         panel3.add( slider );
113         panel3.add( cbox1 );
114         panel3.add( cbox2 );
115         JPanel panel2 = new JPanel( false );
116         panel2.setBackground( Color.blue );
117         panel.setBorder( new EmptyBorder( 1, 1, 1, 1 ) );
118         panel2.add( label );
119         panel2.add( panel3 );
120         panel.setBackground( Color.red );
121         panel.setSize( new Dimension( 250, 250 ) );
122         panel.setBorder( new EmptyBorder( 5, 5, 5, 5 ) );
123         panel.add( panel2, "Center" );
124         panel.revalidate();
125         return panel;
126     }
