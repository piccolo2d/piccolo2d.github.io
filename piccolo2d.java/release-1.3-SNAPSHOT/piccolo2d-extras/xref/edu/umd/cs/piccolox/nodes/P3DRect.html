33  import java.awt.geom.*;
34  import edu.umd.cs.piccolo.*;

38  /**
39   * This is a simple node that draws a "3D" rectangle within the bounds of the node.
40   * Drawing a 3D rectangle in a zooming environment is a little tricky because
41   * if you just use the regular (Java2D) 3D rectangle, the 3D borders get scaled,
42   * and that is ugly.  This version always draws the 3D border at fixed 2 pixel width.
43   * 
44   * @author Ben Bederson
45   */
46  public class P3DRect extends PNode {
47  
48  	private Color topLeftOuterColor;
49  	private Color topLeftInnerColor;
50  	private Color bottomRightInnerColor;
51  	private Color bottomRightOuterColor;
52  	private GeneralPath path;
53  	private Stroke stroke;
54  	private boolean raised;
55  	
56  	public P3DRect() {
57  		raised = true;
58  		stroke = new BasicStroke(0);
59  		path = new GeneralPath();
60  	}
61  	
62  	public P3DRect(Rectangle2D bounds) {
63  		this(bounds.getX(), bounds.getY(), bounds.getWidth(), bounds.getHeight());
64  	}
65  	
66  	public P3DRect(double x, double y, double width, double height) {
67  		this();
68  		setBounds(x, y, width, height);
69  	}
70  	
71  	public void setRaised(boolean raised) {
72  		this.raised = raised;
73  		setPaint(getPaint());
74  	}
75  	
76  	public boolean getRaised() {
77  		return raised;
78  	}
79  	
80  	protected void paint(PPaintContext paintContext) {
81  		Graphics2D g2 = paintContext.getGraphics();
82  				
83  		double x = getX();
84  		double y = getY();
85  		double width = getWidth();
86  		double height = getHeight();
87  		double magX = g2.getTransform().getScaleX();
88  		double magY = g2.getTransform().getScaleY();
89  		double dx = (float)(1.0 / magX);
90  		double dy = (float)(1.0 / magY);
91  		PBounds bounds = getBounds();
92  		
93  		g2.setPaint(getPaint());
94  		g2.fill(bounds);
95  		g2.setStroke(stroke);	
96  
97  		path.reset();
98  		path.moveTo((float)(x+width), (float)y);
99  		path.lineTo((float)x, (float)y);
100 		path.lineTo((float)x, (float)(y+height));
101 		g2.setPaint(topLeftOuterColor);
102 		g2.draw(path);
103 
104 		path.reset();
105 		path.moveTo((float)(x+width), (float)(y+dy));
106 		path.lineTo((float)(x+dx), (float)(y+dy));
107 		path.lineTo((float)(x+dx), (float)(y+height));
108 		g2.setPaint(topLeftInnerColor);
109 		g2.draw(path);
110 
111 		path.reset();
112 		path.moveTo((float)(x+width), (float)(y));
113 		path.lineTo((float)(x+width), (float)(y+height));
114 		path.lineTo((float)(x), (float)(y+height));
115 		g2.setPaint(bottomRightOuterColor);
116 		g2.draw(path);
117 
118 		path.reset();
119 		path.moveTo((float)(x+width-dx), (float)(y+dy));
120 		path.lineTo((float)(x+width-dx), (float)(y+height-dy));
121 		path.lineTo((float)(x), (float)(y+height-dy));
122 		g2.setPaint(bottomRightInnerColor);
123 		g2.draw(path);
124 	}	
125 
126 	public void setPaint(Paint newPaint) {
127 		super.setPaint(newPaint);
128 		
129 		if (newPaint instanceof Color) {
130 			Color color = (Color)newPaint;
131 			
132 			if (raised) {
133 				topLeftOuterColor = color.brighter();
134 				topLeftInnerColor = topLeftOuterColor.brighter();
135 				bottomRightInnerColor = color.darker();
136 				bottomRightOuterColor = bottomRightInnerColor.darker();
137 			} else {
138 				topLeftOuterColor = color.darker();
139 				topLeftInnerColor = topLeftOuterColor.darker();
140 				bottomRightInnerColor = color.brighter();
141 				bottomRightOuterColor = bottomRightInnerColor.brighter();			
142 			}
143 		} else {
144 			topLeftOuterColor = null;
145 			topLeftInnerColor = null;
146 			bottomRightInnerColor = null;
147 			bottomRightOuterColor = null;
148 		}
149 	}
150 
151 	public static void main(String[] args) {
152 		new PFrame() {
153 			public void initialize() {
154 				getCanvas().setDefaultRenderQuality(PPaintContext.LOW_QUALITY_RENDERING);
155 
156 				P3DRect rect1 = new P3DRect(50, 50, 100, 100);
157 				rect1.setPaint(new Color(239, 235, 222));
158 
159 				P3DRect rect2 = new P3DRect(50, 50, 100, 100);
160 				rect2.setPaint(new Color(239, 235, 222));
161 				rect2.translate(110, 0);
162 				rect2.setRaised(false);
163 				
164 				getCanvas().getLayer().addChild(rect1);
165 				getCanvas().getLayer().addChild(rect2);
166 			}
167 		};
168 	}
