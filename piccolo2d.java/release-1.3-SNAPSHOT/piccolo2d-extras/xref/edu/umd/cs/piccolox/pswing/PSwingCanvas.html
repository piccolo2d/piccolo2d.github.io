11  package edu.umd.cs.piccolox.pswing;

13  import edu.umd.cs.piccolo.PCanvas;

15  import javax.swing.*;
16  import java.awt.*;

18  /**
19   * The <b>PSwingCanvas</b> is a PCanvas that can display Swing components with the PSwing adapter.
20   *
21   * @author Benjamin B. Bederson
22   * @author Sam R. Reid
23   * @author Lance E. Good
24   */
25  
26  public class PSwingCanvas extends PCanvas {
27      public static final String SWING_WRAPPER_KEY = "Swing Wrapper";
28      private static PSwingRepaintManager pSwingRepaintManager = new PSwingRepaintManager();
29  
30      private SwingWrapper swingWrapper;
31      private PSwingEventHandler swingEventHandler;
32  
33      /**
34       * Construct a new PSwingCanvas.
35       */
36      public PSwingCanvas() {
37          swingWrapper = new SwingWrapper( this );
38          add( swingWrapper );
39          RepaintManager.setCurrentManager( pSwingRepaintManager );
40          pSwingRepaintManager.addPSwingCanvas( this );
41  
42          swingEventHandler = new PSwingEventHandler( this, getCamera() );//todo or maybe getCameraLayer() or getRoot()?
43          swingEventHandler.setActive( true );
44      }
45  
46      JComponent getSwingWrapper() {
47          return swingWrapper;
48      }
49  
50      public void addPSwing( PSwing pSwing ) {
51          swingWrapper.add( pSwing.getComponent() );
52      }
53  
54      public void removePSwing( PSwing pSwing ) {
55          swingWrapper.remove( pSwing.getComponent() );
56      }
57  
58      private static class SwingWrapper extends JComponent {
59          private PSwingCanvas pSwingCanvas;
60  
61          public SwingWrapper( PSwingCanvas pSwingCanvas ) {
62              this.pSwingCanvas = pSwingCanvas;
63              setSize( new Dimension( 0, 0 ) );
64              setPreferredSize( new Dimension( 0, 0 ) );
65              putClientProperty( SWING_WRAPPER_KEY, SWING_WRAPPER_KEY );
66          }
67  
68          public PSwingCanvas getpSwingCanvas() {
69              return pSwingCanvas;
70          }
71      }
72  
73  }
