13  import edu.umd.cs.piccolo.util.PBounds;

15  import javax.swing.*;
16  import java.awt.*;
17  import java.util.ArrayList;
18  import java.util.Vector;

20  /**
21   * This RepaintManager replaces the default Swing implementation, and is used to
22   * intercept and repaint dirty regions of PSwing components.
23   * <p>
24   * This is an internal class used by Piccolo to support Swing components in
25   * Piccolo. This should not be instantiated, though all the public methods of
26   * javax.swing.RepaintManager may still be called and perform in the expected
27   * manner.
28   * </p>
29   * <p>
30   * PBasicRepaint Manager is an extension of RepaintManager that traps those
31   * repaints called by the Swing components that have been added to the PCanvas
32   * and passes these repaints to the SwingVisualComponent rather than up the
33   * component hierarchy as usually happens.
34   * </p>
35   * <p>
36   * Also traps revalidate calls made by the Swing components added to the PCanvas
37   * to reshape the applicable Visual Component.
38   * </p>
39   * <p>
40   * Also keeps a list of PSwings that are painting. This disables repaint until
41   * the component has finished painting. This is to address a problem introduced
42   * by Swing's CellRendererPane which is itself a work-around. The problem is
43   * that JTable's, JTree's, and JList's cell renderers need to be validated
44   * before repaint. Since we have to repaint the entire Swing component hierarchy
45   * (in the case of a Swing component group used as a Piccolo visual component).
46   * This causes an infinite loop. So we introduce the restriction that no
47   * repaints can be triggered by a call to paint.
48   * </p>
49   * 
50   * @author Benjamin B. Bederson
51   * @author Lance E. Good
52   * @author Sam R. Reid
53   */
54  public class PSwingRepaintManager extends RepaintManager {
55      private ArrayList swingWrappers = new ArrayList();
56  
57      // The components that are currently painting
58      // This needs to be a vector for thread safety
59      private Vector paintingComponents = new Vector();
60  
61      /**
62       * Locks repaint for a particular (Swing) component displayed by PCanvas
63       * 
64       * @param c The component for which the repaint is to be locked
65       */
66      public void lockRepaint(JComponent c) {
67          paintingComponents.addElement(c);
68      }
69  
70      /**
71       * Unlocks repaint for a particular (Swing) component displayed by PCanvas
72       * 
73       * @param c The component for which the repaint is to be unlocked
74       */
75      public void unlockRepaint(JComponent c) {
76          synchronized (paintingComponents) {
77              paintingComponents.removeElementAt(paintingComponents.lastIndexOf(c));
78          }
79      }
80  
81      /**
82       * Returns true if repaint is currently locked for a component and false
83       * otherwise
84       * 
85       * @param c The component for which the repaint status is desired
86       * @return Whether the component is currently painting
87       */
88      public boolean isPainting(JComponent c) {
89          return paintingComponents.contains(c);
90      }
91  
92      /**
93       * This is the method "repaint" now calls in the Swing components.
94       * Overridden to capture repaint calls from those Swing components which are
95       * being used as Piccolo visual components and to call the Piccolo repaint
96       * mechanism rather than the traditional Component hierarchy repaint
97       * mechanism. Otherwise, behaves like the superclass.
98       * 
99       * @param c Component to be repainted
100      * @param x X coordinate of the dirty region in the component
101      * @param y Y coordinate of the dirty region in the component
102      * @param w Width of the dirty region in the component
103      * @param h Height of the dirty region in the component
104      */
105     public synchronized void addDirtyRegion(JComponent c, int x, int y, final int w, final int h) {
106         boolean captureRepaint = false;
107         JComponent capturedComponent = null;
108         int captureX = x, captureY = y;
109 
110         // We have to check to see if the PCanvas
111         // (ie. the SwingWrapper) is in the components ancestry. If so,
112         // we will want to capture that repaint. However, we also will
113         // need to translate the repaint request since the component may
114         // be offset inside another component.
115         for (Component comp = c; comp != null && comp.isLightweight() && !captureRepaint; comp = comp.getParent()) {
116             if (swingWrappers.contains(comp.getParent())) {
117                 if (comp instanceof JComponent) {
118                     captureRepaint = true;
119                     capturedComponent = (JComponent) comp;
120                 }
121             }
122             else {
123                 // Adds to the offset since the component is nested
124                 captureX += comp.getLocation().getX();
125                 captureY += comp.getLocation().getY();
126             }
127         }
128 
129         // Now we check to see if we should capture the repaint and act
130         // accordingly
131         if (captureRepaint) {
132             if (!isPainting(capturedComponent)) {
133                 final PSwing vis = (PSwing) capturedComponent.getClientProperty(PSwing.PSWING_PROPERTY);
134                 if (vis != null) {
135                     final int repaintX = captureX;
136                     final int repaintY = captureY;
137                     Runnable repainter = new Runnable() {
138                         public void run() {
139                             vis.repaint(new PBounds((double) repaintX, (double) repaintY, (double) w, (double) h));
140                         }
141                     };
142                     SwingUtilities.invokeLater(repainter);
143                 }
144             }
145         }
146         else {
147             super.addDirtyRegion(c, x, y, w, h);
148         }
149     }
150 
151     /**
152      * This is the method "revalidate" calls in the Swing components. Overridden
153      * to capture revalidate calls from those Swing components being used as
154      * Piccolo visual components and to update Piccolo's visual component
155      * wrapper bounds (these are stored separately from the Swing component).
156      * Otherwise, behaves like the superclass.
157      * 
158      * @param invalidComponent The Swing component that needs validation
159      */
160     public synchronized void addInvalidComponent(JComponent invalidComponent) {
161         final JComponent capturedComponent = invalidComponent;
162 
163         if (capturedComponent.getParent() != null
164                 && capturedComponent.getParent() instanceof JComponent
165                 && ((JComponent) capturedComponent.getParent()).getClientProperty(PSwingCanvas.SWING_WRAPPER_KEY) != null) {
166 
167             Runnable validater = new Runnable() {
168                 public void run() {
169                     capturedComponent.validate();
170                     PSwing swing = (PSwing) capturedComponent.getClientProperty(PSwing.PSWING_PROPERTY);
171                     swing.reshape();
172                 }
173             };
174             SwingUtilities.invokeLater(validater);
175         }
176         else {
177             super.addInvalidComponent(invalidComponent);
178         }
179     }
180 
181     void addPSwingCanvas(PSwingCanvas swingWrapper) {
182         swingWrappers.add(swingWrapper.getSwingWrapper());
183     }
