32  import java.awt.Graphics2D;
33  import java.awt.Image;
34  import java.awt.geom.Dimension2D;

36  import edu.umd.cs.piccolo.PNode;
37  import edu.umd.cs.piccolo.util.PBounds;
38  import edu.umd.cs.piccolo.util.PDimension;
39  import edu.umd.cs.piccolo.util.PPaintContext;
40  import edu.umd.cs.piccolo.util.PPickPath;

42  /**
43   * <b>PNodeCache</b> caches a visual representation of it's children 
44   * into an image and uses this cached image for painting instead of
45   * painting it's children directly. This  is intended to be used in 
46   * two ways.
47   * <P>
48   * First it can be used as a simple optimization technique. If a node 
49   * has many descendents it may be faster to paint the cached image 
50   * representation instead of painting each node.
51   * <P>
52   * Second PNodeCache provides a place where "image" effects such as
53   * blurring and drop shadows can be added to the Piccolo scene graph.
54   * This can be done by overriding the method createImageCache and
55   * returing an image with the desired effect applied.
56   * <P>
57   * @version 1.0
58   * @author Jesse Grosjean
59   */
60  public class PNodeCache extends PNode {
61  	
62  	private transient Image imageCache;
63  	private boolean validatingCache;
64  	
65  	/**
66  	 * Override this method to customize the image cache creation process. For
67  	 * example if you want to create a shadow effect you would do that here. Fill
68  	 * in the cacheOffsetRef if needed to make your image cache line up with the
69  	 * nodes children.
70  	 */
71  	public Image createImageCache(Dimension2D cacheOffsetRef) {
72  		return toImage();		
73  	}
74  	
75  	public Image getImageCache() {
76  		if (imageCache == null) {			
77  			PDimension cacheOffsetRef = new PDimension();
78  			validatingCache = true;
79  			resetBounds();
80  			imageCache = createImageCache(cacheOffsetRef);
81  			PBounds b = getFullBoundsReference();
82  			setBounds(b.getX() + cacheOffsetRef.getWidth(),
83  					  b.getY() + cacheOffsetRef.getHeight(),
84  					  imageCache.getWidth(null), 
85  					  imageCache.getHeight(null));
86  			validatingCache = false;
87  		}
88  		return imageCache;
89  	}
90  	
91  	public void invalidateCache() {
92  		imageCache = null;
93  	}	
94  	
95  	public void invalidatePaint() {
96  		if (!validatingCache) {
97  			super.invalidatePaint();
98  		}
99  	}
100 	
101 	public void repaintFrom(PBounds localBounds, PNode childOrThis) {
102 		if (!validatingCache) {
103 			super.repaintFrom(localBounds, childOrThis);
104 			invalidateCache();
105 		}
106 	}	
107 	
108 	public void fullPaint(PPaintContext paintContext) {		
109 		if (validatingCache) {
110 			super.fullPaint(paintContext);
111 		} else {
112 			Graphics2D g2 = paintContext.getGraphics();
113 			g2.drawImage(getImageCache(), (int) getX(), (int) getY(), null);
114 		}
115 	}
116 	
117 	protected boolean pickAfterChildren(PPickPath pickPath) {
118 		return false;
119 	}
120 }
