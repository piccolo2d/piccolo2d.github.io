52  /**
53   * <b>PPath</b> is a wrapper around a java.awt.geom.GeneralPath. The
54   * setBounds method works by scaling the path to fit into the specified
55   * bounds. This normally works well, but if the specified base bounds
56   * get too small then it is impossible to expand the path shape again since
57   * all its numbers have tended to zero, so application code may need to take
58   * this into consideration. 
59   * <P>
60   * One option that applications have is to call <code>startResizeBounds</code> before
61   * starting an interaction that may make the bounds very small, and calling 
62   * <code>endResizeBounds</code> when this interaction is finished. When this is done
63   * PPath will use a copy of the original path to do the resizing so the numbers
64   * in the path wont loose resolution.
65   * <P>
66   * This class also provides methods for constructing common shapes using a 
67   * general path.
68   * <P>
69   * @version 1.0
70   * @author Jesse Grosjean
71   */
72  public class PPath extends PNode {

165 	/**
166 	 * Construct this path with the given shape and stroke.
167 	 * This method may be used to optimize the creation of a large number of
168 	 * PPaths. Normally PPaths have a default stroke of width one, but when a
169 	 * path has a non null stroke it takes significantly longer to compute its
170 	 * bounds. This method allows you to override that default stroke before the
171 	 * bounds are ever calculated, so if you pass in a null stroke here you
172 	 * won't ever have to pay that bounds calculation price if you don't need
173 	 * to.
174 	 */
175 	public PPath(Shape aShape, Stroke aStroke) {
176 		this();
177 		stroke = aStroke;
178 		if (aShape != null) append(aShape, false);
179 	}

220 	/**
221 	 * Set the bounds of this path. This method works by scaling the path 
222 	 * to fit into the specified bounds. This normally works well, but if 
223 	 * the specified base bounds get too small then it is impossible to 
224 	 * expand the path shape again since all its numbers have tended to zero, 
225 	 * so application code may need to take this into consideration.
226 	 */
227 	protected void internalUpdateBounds(double x, double y, double width, double height) {
228 		if (updatingBoundsFromPath) return;
229 		if (path == null) return;
230 		
231 		if (resizePath != null) {
232 			path.reset();
233 			path.append(resizePath, false);
234 		}
235 
236 		Rectangle2D pathBounds = path.getBounds2D();
237 		Rectangle2D pathStrokeBounds = getPathBoundsWithStroke();
238 		double strokeOutset = Math.max(pathStrokeBounds.getWidth() - pathBounds.getWidth(), 
239 										pathStrokeBounds.getHeight() - pathBounds.getHeight());
240 		
241 		x += strokeOutset / 2;
242 		y += strokeOutset / 2;
243 		width -= strokeOutset;
244 		height -= strokeOutset;
245 		
246 		double scaleX = (width == 0 || pathBounds.getWidth() == 0) ? 1 : width / pathBounds.getWidth();
247 		double scaleY = (height == 0 || pathBounds.getHeight() == 0) ? 1 : height / pathBounds.getHeight();
248 		
249 		TEMP_TRANSFORM.setToIdentity();
250 		TEMP_TRANSFORM.translate(x, y);
251 		TEMP_TRANSFORM.scale(scaleX, scaleY);
252 		TEMP_TRANSFORM.translate(-pathBounds.getX(), -pathBounds.getY());
253 		
254 		path.transform(TEMP_TRANSFORM);
255 	}

257 	public boolean intersects(Rectangle2D aBounds) {
258 		if (super.intersects(aBounds)) {
259 			if (getPaint() != null && path.intersects(aBounds)) {
260 				return true;
261 			} else if (stroke != null && strokePaint != null) {
262 				return stroke.createStrokedShape(path).intersects(aBounds);
263 			}
264 		}
265 		return false;
266 	}

268 	public Rectangle2D getPathBoundsWithStroke() {
269 		if (stroke != null) {
270 			return stroke.createStrokedShape(path).getBounds2D();
271 		} else {
272 			return path.getBounds2D();
273 		}
274 	}

276 	public void updateBoundsFromPath() {
277 		updatingBoundsFromPath = true;
278 		if (path == null) {
279 			resetBounds();
280 		} else {
281 			Rectangle2D b = getPathBoundsWithStroke();
282 			setBounds(b.getX(), b.getY(), b.getWidth(), b.getHeight());
283 		}
284 		updatingBoundsFromPath = false;
285 	}

291 	protected void paint(PPaintContext paintContext) {
292 		Paint p = getPaint();
293 		Graphics2D g2 = paintContext.getGraphics();
294 		
295 		if (p != null) {
296 			g2.setPaint(p);
297 			g2.fill(path);
298 		}
299 
300 		if (stroke != null && strokePaint != null) {
301 			g2.setPaint(strokePaint);
302 			g2.setStroke(stroke);
303 			g2.draw(path);
304 		}		
305 	}
