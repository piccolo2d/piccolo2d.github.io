44          JPanel tabPanel = new JPanel(false);
45          tabPanel.setLayout(null);

47          // Populate the tab panel with four instances of nested panel.

62          panel.setSize(new Dimension(250, 250));
63          panel.setLocation(350, 350);
64          tabPanel.add(panel);
65          // Add the default zoom button
66          JButton buttonPreset = new JButton("Zoom = 100%");
67          buttonPreset.addActionListener(new ActionListener() {

73          buttonPreset.setSize(new Dimension(120, 25));
74          buttonPreset.setLocation(240, 285);
75          tabPanel.add(buttonPreset);
76          // Create a pswing object for the tab panel
77          swing = new PSwing(tabPanel);
78          swing.translate(0, 0);

93      private JPanel createNestedPanel() {
94          // A panel MUST be created with double buffering off
95          JPanel panel;
96          JLabel label;
97          panel = new JPanel(false);
98          panel.setLayout(new BorderLayout());
99          label = new JLabel("A Panel within a panel");
100         label.setHorizontalAlignment(SwingConstants.CENTER);
101         label.setForeground(Color.white);
102         JLabel label2 = new JLabel("A Panel within a panel");
103         label2.setHorizontalAlignment(SwingConstants.CENTER);
104         JSlider slider = new JSlider();
105         JCheckBox cbox1 = new JCheckBox("Checkbox 1");
106         JCheckBox cbox2 = new JCheckBox("Checkbox 2");
107         JPanel panel3 = new JPanel(false);
108         panel3.setLayout(new BoxLayout(panel3, BoxLayout.PAGE_AXIS));
109         panel3.setBorder(new EmptyBorder(3, 3, 3, 3));
110         panel3.add(label2);
111         panel3.add(slider);
112         panel3.add(cbox1);
113         panel3.add(cbox2);
114         JPanel panel2 = new JPanel(false);
115         panel2.setBackground(Color.blue);
116         panel.setBorder(new EmptyBorder(1, 1, 1, 1));
117         panel2.add(label);
118         panel2.add(panel3);
119         panel.setBackground(Color.red);
120         panel.setSize(new Dimension(250, 250));
121         panel.setBorder(new EmptyBorder(5, 5, 5, 5));
122         panel.add(panel2, "Center");
123         panel.revalidate();
124         return panel;
125     }
