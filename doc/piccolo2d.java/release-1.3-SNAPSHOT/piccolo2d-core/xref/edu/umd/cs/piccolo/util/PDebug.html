38  /**
39   * <b>PDebug</b> is used to set framework wide debugging flags.
40   * <P>
41   * 
42   * @version 1.0
43   * @author Jesse Grosjean
44   */
45  public class PDebug {
46  
47      public static boolean debugRegionManagement = false;
48      public static boolean debugPaintCalls = false;
49      public static boolean debugPrintFrameRate = false;
50      public static boolean debugPrintUsedMemory = false;
51      public static boolean debugBounds = false;
52      public static boolean debugFullBounds = false;
53      public static boolean debugThreads = false;
54      public static int printResultsFrameRate = 10;
55  
56      private static int debugPaintColor;
57  
58      private static long framesProcessed;
59      private static long startProcessingOutputTime;
60      private static long startProcessingInputTime;
61      private static long processOutputTime;
62      private static long processInputTime;
63      private static boolean processingOutput;
64  
65      private PDebug() {
66          super();
67      }
68  
69      public static Color getDebugPaintColor() {
70          int color = 100 + (debugPaintColor++ % 10) * 10;
71          return new Color(color, color, color, 150);
72      }
73  
74      // called when scene graph needs update.
75      public static void scheduleProcessInputs() {
76          if (debugThreads && !SwingUtilities.isEventDispatchThread()) {
77              System.out.println("scene graph manipulated on wrong thread");
78          }
79      }
80  
81      public static void processRepaint() {
82          if (processingOutput && debugPaintCalls) {
83              System.err
84                      .println("Got repaint while painting scene. This can result in a recursive process that degrades performance.");
85          }
86  
87          if (debugThreads && !SwingUtilities.isEventDispatchThread()) {
88              System.out.println("repaint called on wrong thread");
89          }
90      }
91  
92      public static boolean getProcessingOutput() {
93          return processingOutput;
94      }
95  
96      public static void startProcessingOutput() {
97          processingOutput = true;
98          startProcessingOutputTime = System.currentTimeMillis();
99      }
100 
101     public static void endProcessingOutput(Graphics g) {
102         processOutputTime += (System.currentTimeMillis() - startProcessingOutputTime);
103         framesProcessed++;
104 
105         if (PDebug.debugPrintFrameRate) {
106             if (framesProcessed % printResultsFrameRate == 0) {
107                 System.out.println("Process output frame rate: " + getOutputFPS() + " fps");
108                 System.out.println("Process input frame rate: " + getInputFPS() + " fps");
109                 System.out.println("Total frame rate: " + getTotalFPS() + " fps");
110                 System.out.println();
111                 resetFPSTiming();
112             }
113         }
114 
115         if (PDebug.debugPrintUsedMemory) {
116             if (framesProcessed % printResultsFrameRate == 0) {
117                 System.out.println("Approximate used memory: " + getApproximateUsedMemory() / 1024 + " k");
118             }
119         }
120 
121         if (PDebug.debugRegionManagement) {
122             Graphics2D g2 = (Graphics2D) g;
123             g.setColor(PDebug.getDebugPaintColor());
124             g2.fill(g.getClipBounds().getBounds2D());
125         }
126 
127         processingOutput = false;
128     }
129 
130     public static void startProcessingInput() {
131         startProcessingInputTime = System.currentTimeMillis();
132     }
133 
134     public static void endProcessingInput() {
135         processInputTime += (System.currentTimeMillis() - startProcessingInputTime);
136     }
137 
138     /**
139      * Return how many frames are processed and painted per second. Note that
140      * since piccolo doesn't paint continuously this rate will be slow unless
141      * you are interacting with the system or have activities scheduled.
142      */
143     public static double getTotalFPS() {
144         if ((framesProcessed > 0)) {
145             return 1000.0 / ((processInputTime + processOutputTime) / (double) framesProcessed);
146         }
147         else {
148             return 0;
149         }
150     }
151 
152     /**
153      * Return the frames per second used to process input events and activities.
154      */
155     public static double getInputFPS() {
156         if ((processInputTime > 0) && (framesProcessed > 0)) {
157             return 1000.0 / (processInputTime / (double) framesProcessed);
158         }
159         else {
160             return 0;
161         }
162     }
163 
164     /**
165      * Return the frames per seconds used to paint graphics to the screen.
166      */
167     public static double getOutputFPS() {
168         if ((processOutputTime > 0) && (framesProcessed > 0)) {
169             return 1000.0 / (processOutputTime / (double) framesProcessed);
170         }
171         else {
172             return 0;
173         }
174     }
175 
176     /**
177      * Return the number of frames that have been processed since the last time
178      * resetFPSTiming was called.
179      */
180     public long getFramesProcessed() {
181         return framesProcessed;
182     }
183 
184     /**
185      * Reset the variables used to track FPS. If you reset seldom they you will
186      * get good average FPS values, if you reset more often only the frames
187      * recorded after the last reset will be taken into consideration.
188      */
189     public static void resetFPSTiming() {
190         framesProcessed = 0;
191         processInputTime = 0;
192         processOutputTime = 0;
193     }
194 
195     public static long getApproximateUsedMemory() {
196         System.gc();
197         System.runFinalization();
198         long totalMemory = Runtime.getRuntime().totalMemory();
199         long free = Runtime.getRuntime().freeMemory();
200         return totalMemory - free;
201     }
