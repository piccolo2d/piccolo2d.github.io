58   * <code>PSelectionEventHandler</code> provides standard interaction for selection.  Clicking
59   * selects the object under the cursor.  Shift-clicking allows multiple objects to be
60   * selected.  Dragging offers marquee selection.  Pressing the delete key deletes
61   * the selection by default.

132 	public void select(Collection items) {
133 		boolean changes = false;
134 		Iterator itemIt = items.iterator();
135 		while (itemIt.hasNext()) {
136 			PNode node = (PNode)itemIt.next();
137 			changes |= internalSelect(node);
138 		}
139 		if (changes) {
140 			postSelectionChanged();
141 		}
142 	}

144 	public void select(Map items) {
145 		select( items.keySet() );
146 	}

148 	private boolean internalSelect( PNode node ) {
149 		if (isSelected(node)) {
150 			return false;
151 		}
152 
153 		selection.put(node, Boolean.TRUE);
154 		decorateSelectedNode(node);
155 		return true;
156 	}

159 	{
160 		PNotificationCenter.defaultCenter().postNotification(SELECTION_CHANGED_NOTIFICATION, this);
161 	}

164 		if (internalSelect(node)) {

174 		boolean changes = false;
175 		Iterator itemIt = items.iterator();
176 		while (itemIt.hasNext()) {
177 			PNode node = (PNode)itemIt.next();

237 	/**
238 	 * Determine if the specified node is selectable (i.e., if it is a child
239 	 * of the one the list of selectable parents.
240 	 */
241 	protected boolean isSelectable(PNode node) {
242 		boolean selectable = false;
243 
244 		Iterator parentsIt = selectableParents.iterator();
245 		while (parentsIt.hasNext()) {
246 			PNode parent = (PNode)parentsIt.next();
247 			if (parent.getChildrenReference().contains(node)) {
248 				selectable = true;
249 				break;
250 			}
251 			else if (parent instanceof PCamera) {
252 				for(int i=0; i<((PCamera)parent).getLayerCount(); i++) {
253 					PLayer layer = ((PCamera)parent).getLayer(i);	
254 					if (layer.getChildrenReference().contains(node)) {
255 						selectable = true;
256 						break;	
257 					}
258 				}
259 			}
260 		}
261 		
262 		return selectable;
263 	}

316 		}

320 		super.drag(e);
321 
322 		if (isMarqueeSelection(e)) {
323 			updateMarquee(e);	
324

326 				computeMarqueeSelection(e);
327 			}
328 			else {
329 				computeOptionMarqueeSelection(e);
330 			}
331 		} else {
332 			dragStandardSelection(e);
333 		}
334 	}

387 							// Option indicator not down - clear selection, and start fresh
388 		if (!isSelected(pressNode)) {

398 							// Option indicator is down, toggle selection
399 		if (isSelectable(pressNode)) {

408 	protected void updateMarquee(PInputEvent pie) {
409 		PBounds b = new PBounds();
410 
411 		if (marqueeParent instanceof PCamera) {
412 			b.add(canvasPressPt);
413 			b.add(pie.getCanvasPosition());
414 		}
415 		else {
416 			b.add(presspt);
417 			b.add(pie.getPosition());
418 		}
419 
420 		marquee.globalToLocal(b);
421 		marquee.setPathToRectangle((float) b.x, (float) b.y, (float) b.width, (float) b.height);				
422 		b.reset();
423 		b.add(presspt);
424 		b.add(pie.getPosition());
425 
426 		allItems.clear();
427 		PNodeFilter filter = createNodeFilter(b);
428 		Iterator parentsIt = selectableParents.iterator();
429 		while (parentsIt.hasNext()) {
430 			PNode parent = (PNode) parentsIt.next();
431 			
432 			Collection items;
433 			if (parent instanceof PCamera) {
434 				items = new ArrayList();
435 				for(int i=0; i<((PCamera)parent).getLayerCount(); i++) {
436 					((PCamera)parent).getLayer(i).getAllNodes(filter,items);	
437 				}
438 			}
439 			else {
440 				items = parent.getAllNodes(filter, null);
441 			}
442 			
443 			Iterator itemsIt = items.iterator();
444 			while (itemsIt.hasNext()) {
445 				allItems.put(itemsIt.next(), Boolean.TRUE);
446 			}
447 		}
448 	}

450 	protected void computeMarqueeSelection(PInputEvent pie) {
451 		unselectList.clear();
452 		// Make just the items in the list selected
453 		// Do this efficiently by first unselecting things not in the list
454 		Iterator selectionEn = selection.keySet().iterator();
455 		while (selectionEn.hasNext()) {
456 			PNode node = (PNode) selectionEn.next();
457 			if (!allItems.containsKey(node)) {
458 				unselectList.add(node);
459 			}
460 		}
461 		unselect(unselectList);
462 		
463 		// Then select the rest
464 		selectionEn = allItems.keySet().iterator();
465 		while (selectionEn.hasNext()) {
466 			PNode node = (PNode) selectionEn.next();
467 			if (!selection.containsKey(node) && !marqueeMap.containsKey(node) && isSelectable(node)) {
468 				marqueeMap.put(node,Boolean.TRUE);
469 			}
470 			else if (!isSelectable(node)) {
471 				selectionEn.remove();
472 			}
473 		}
474 		
475 		select(allItems);		
476 	}

478 	protected void computeOptionMarqueeSelection(PInputEvent pie) {
479 		unselectList.clear();
480 		Iterator selectionEn = selection.keySet().iterator();
481 		while (selectionEn.hasNext()) {
482 			PNode node = (PNode) selectionEn.next();
483 			if (!allItems.containsKey(node) && marqueeMap.containsKey(node)) {
484 				marqueeMap.remove(node);
485 				unselectList.add(node);
486 			}
487 		}
488 		unselect(unselectList);
489 		
490 
491 		// Then select the rest
492 		selectionEn = allItems.keySet().iterator();
493 		while (selectionEn.hasNext()) {
494 			PNode node = (PNode) selectionEn.next();
495 			if (!selection.containsKey(node) && !marqueeMap.containsKey(node) && isSelectable(node)) {
496 				marqueeMap.put(node,Boolean.TRUE);
497 			}
498 			else if (!isSelectable(node)) {
499 				selectionEn.remove();
500 			}
501 		}
502 
503 		select(allItems);	
504 	}

517 	protected void dragStandardSelection(PInputEvent e) {
518 		// There was a press node, so drag selection
519 	    PDimension d = e.getCanvasDelta();
520 	    e.getTopCamera().localToView(d);
521 
522 	    PDimension gDist = new PDimension();
523 		Iterator selectionEn = getSelection().iterator();
524 		while (selectionEn.hasNext()) {
525 			PNode node = (PNode) selectionEn.next();
526 
527 			gDist.setSize(d);
528 			node.getParent().globalToLocal(gDist);
529 			node.offset(gDist.getWidth(), gDist.getHeight());
530 		}		
531 	}

548 	protected void dragActivityStep(PInputEvent aEvent) {
549 		if (marquee != null) {
550 			float origStrokeNum = strokeNum;
551 			strokeNum = (strokeNum + 0.5f) % NUM_STROKES;	// Increment by partial steps to slow down animation
552 			if ((int)strokeNum != (int)origStrokeNum) {
553 				marquee.setStroke(strokes[(int)strokeNum]);
554 			}

594 	protected class BoundsFilter implements PNodeFilter {
595 		PBounds localBounds = new PBounds();
596 		PBounds bounds;
597 		
598 		protected BoundsFilter(PBounds bounds) {
599 			this.bounds = bounds;
600 		}
601 
602 		public boolean accept(PNode node) {
603 			localBounds.setRect(bounds);
604 			node.globalToLocal(localBounds);
605 			
606 			boolean boundsIntersects = node.intersects(localBounds);
607 			boolean isMarquee = (node == marquee);
608 			return (node.getPickable() && boundsIntersects && !isMarquee && !selectableParents.contains(node) && !isCameraLayer(node));
609 		}
610 
611 		public boolean acceptChildrenOf(PNode node) {
612 			return selectableParents.contains(node) || isCameraLayer(node);
613 		}
614 		
615 		public boolean isCameraLayer(PNode node) {
616 			if (node instanceof PLayer) {
617 				for(Iterator i=selectableParents.iterator(); i.hasNext();) {
618 					PNode parent = (PNode)i.next();
619 					if (parent instanceof PCamera) {
620 						if (((PCamera)parent).indexOfLayer((PLayer)node) != -1) {
621 							return true;	
622 						}
623 					}
624 				}	
625 			}
626 			return false;
627 		}
628 	}
