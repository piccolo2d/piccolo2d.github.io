18  /**
19   * <b>PMouseEvent</b> is an event which indicates that a mouse action occurred in a node.
20   * <p/>
21   * This low-level event is generated by a node object for:
22   * <ul>
23   * <li>Mouse Events
24   * <ul>
25   * <li>a mouse button is pressed
26   * <li>a mouse button is released
27   * <li>a mouse button is clicked (pressed and released)
28   * <li>the mouse cursor enters a node
29   * <li>the mouse cursor exits a node
30   * </ul>
31   * <p/>
32   * A PMouseEvent object is passed to every <code>PMouseListener</code>
33   * or <code>PMouseAdapter</code> object which registered to receive
34   * the "interesting" mouse events using the component's
35   * <code>addMouseListener</code> method.
36   * (<code>PMouseAdapter</code> objects implement the
37   * <code>PMouseListener</code> interface.) Each such listener object
38   * gets a <code>PMouseEvent</code> containing the mouse event.
39   * <p/>
40   * <b>Warning:</b> Serialized objects of this class will not be
41   * compatible with future Piccolo releases. The current serialization support is
42   * appropriate for short term storage or RMI between applications running the
43   * same version of Piccolo. A future release of Piccolo will provide support for long
44   * term persistence.
45   *
46   * @author Benjamin B. Bederson
47   * @author Sam R. Reid
48   * @author Lance E. Good
49   */
50  public class PSwingMouseEvent extends MouseEvent implements Serializable {
51      private int id;
52      private PInputEvent event;

54      /**
55       * Constructs a new PMouse event from a Java MouseEvent.
56       *
57       * @param id The event type (MOUSE_PRESSED, MOUSE_RELEASED, MOUSE_CLICKED, MOUSE_ENTERED, MOUSE_EXITED)
58       * @param e  The original Java mouse event
59       *           when in MOUSE_RELEASED events.
60       */
61      protected PSwingMouseEvent( int id, MouseEvent e, PInputEvent event ) {
62          super( (Component)e.getSource(), e.getID(), e.getWhen(), e.getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger() );
63          this.id = id;
64          this.event = event;
65      }

67      /**
68       * Creates and returns a new PMouse event from a Java MouseEvent.
69       *
70       * @param id The event type (MOUSE_PRESSED, MOUSE_RELEASED, MOUSE_CLICKED, MOUSE_ENTERED, MOUSE_EXITED, MOUSE_MOVED, MOUSE_DRAGGED)
71       * @param e  The original Java mouse event
72       *           when in MOUSE_DRAGGED and MOUSE_RELEASED events.
73       */
74      public static PSwingMouseEvent createMouseEvent( int id,
75                                                       MouseEvent e,
76                                                       PInputEvent pEvent ) {
77          if( id == PSwingMouseEvent.MOUSE_MOVED ||
78              id == PSwingMouseEvent.MOUSE_DRAGGED ) {
79              return new PSwingMouseMotionEvent( id, e, pEvent );
80          }
81          else {
82              return new PSwingMouseEvent( id, e, pEvent );
83          }
84      }

191     /**
192      * Calls appropriate method on the listener based on this events ID.
193      *
194      * @param listener the MouseListener or MouseMotionListener to dispatch to.
195      */
196     public void dispatchTo( Object listener ) {
197         if( listener instanceof MouseListener ) {
198             MouseListener mouseListener = (MouseListener)listener;
199             switch( getID() ) {
200                 case PSwingMouseEvent.MOUSE_CLICKED:
201                     mouseListener.mouseClicked( this );
202                     break;
203                 case PSwingMouseEvent.MOUSE_ENTERED:
204                     mouseListener.mouseEntered( this );
205                     break;
206                 case PSwingMouseEvent.MOUSE_EXITED:
207                     mouseListener.mouseExited( this );
208                     break;
209                 case PSwingMouseEvent.MOUSE_PRESSED:
210                     mouseListener.mousePressed( this );
211                     break;
212                 case PSwingMouseEvent.MOUSE_RELEASED:
213                     mouseListener.mouseReleased( this );
214                     break;
215                 default:
216                     throw new RuntimeException( "PMouseEvent with bad ID" );
217             }
218         }
219         else {
220             MouseMotionListener mouseMotionListener = (MouseMotionListener)listener;
221             switch( getID() ) {
222                 case PSwingMouseEvent.MOUSE_DRAGGED:
223                     mouseMotionListener.mouseDragged( this );
224                     break;
225                 case PSwingMouseEvent.MOUSE_MOVED:
226                     mouseMotionListener.mouseMoved( this );
227                     break;
228                 default:
229                     throw new RuntimeException( "PMouseMotionEvent with bad ID" );
230             }
231         }
232     }
