42  /**
43   * <b>PImage</b> is a wrapper around a java.awt.Image. If this node is
44   * copied or serialized that image will be converted into a BufferedImage if
45   * it is not already one.
46   * <P>
47   * @version 1.0
48   * @author Jesse Grosjean
49   */
50  public class PSWTImage extends PNode {
51  	
52  	private transient PSWTCanvas canvas;
53  
54  	private transient Image image;
55  
56  	public PSWTImage(PSWTCanvas canvas) {
57  		super();
58  		
59  		this.canvas = canvas;
60  		canvas.addDisposeListener(new DisposeListener() {
61  			public void widgetDisposed(DisposeEvent de) {
62  				if (image != null) {
63  					image.dispose();
64  				}	
65  			}
66  		});
67  	}
68  	
69  	public PSWTImage(PSWTCanvas canvas, Image newImage) {
70  		this(canvas);
71  		setImage(newImage);
72  	}
73  
74  	public PSWTImage(PSWTCanvas canvas, String fileName) {
75  		this(canvas);
76  		setImage(fileName);	
77  	}

79  	/**
80  	 * Returns the image that is shown by this node.
81  	 * @return the image that is shown by this node
82  	 */ 
83  	public Image getImage() {
84  		return image;
85  	}
86  
87  	/**
88  	 * Set the image that is wrapped by this PImage node. This method will also load
89  	 * the image using a MediaTracker before returning. And if the this PImage is
90  	 * accelerated that I'm will be copied into an accelerated image if needed. Note
91  	 * that this may cause undesired results with images that have transparent regions,
92  	 * for those cases you may want to set the PImage to be not accelerated.
93  	 */
94  	public void setImage(String fileName) {
95  		setImage(new Image(canvas.getDisplay(),fileName));
96  	}
97  
98  	/**
99  	 * Set the image that is wrapped by this PImage node. This method will also load
100 	 * the image using a MediaTracker before returning. And if the this PImage is
101 	 * accelerated that I'm will be copied into an accelerated image if needed. Note
102 	 * that this may cause undesired results with images that have transparent regions,
103 	 * for those cases you may want to set the PImage to be not accelerated.
104 	 */
105 	public void setImage(Image newImage) {
106 		Image old = image;
107 		image = newImage;
108 				
109 		if (image != null) {								
110 			Rectangle bounds = getImage().getBounds();
111 			setBounds(0, 0, bounds.width, bounds.height);
112 			invalidatePaint();
113 		} else {
114 			image = null;
115 		}
116 		
117 		firePropertyChange(PImage.PROPERTY_CODE_IMAGE, PImage.PROPERTY_IMAGE, old, image);
118 	}
119 
120 	protected void paint(PPaintContext paintContext) {
121 		if (getImage() != null) {
122 			Rectangle r = image.getBounds();
123 			double iw = r.width;
124 			double ih = r.height;
125 			PBounds b = getBoundsReference();
126 			SWTGraphics2D g2 = (SWTGraphics2D)paintContext.getGraphics();
127 
128 			if (b.x != 0 || b.y != 0 || b.width != iw || b.height != ih) {
129 				g2.translate(b.x, b.y);
130 				g2.scale(b.width / iw, b.height / ih);
131 				g2.drawImage(image, 0, 0);
132 				g2.scale(iw / b.width, ih / b.height);
133 				g2.translate(-b.x, -b.y);
134 			} else {
135 				g2.drawImage(image, 0, 0);
136 			}
137 		}
138 	}
