32  /**
33   * <b>PInterpolatingActivity</b> interpolates between two states (source and
34   * destination) over the duration of the activity. The interpolation can be
35   * either linear or slow- in, slow-out.
36   * <P>
37   * The mode determines how the activity interpolates between the two states. The
38   * default mode interpolates from source to destination, but you can also go
39   * from destination to source, and from source to destination to source.
40   * <P>
41   * A loopCount of greater then one will make the activity reschedule itself when
42   * it has finished. This makes the activity loop between the two states.
43   * <P>
44   * @version 1.0
45   * @author Jesse Grosjean
46   */
47  public class PInterpolatingActivity extends PActivity {
48  
49  	public static final int SOURCE_TO_DESTINATION = 1;
50  	public static final int DESTINATION_TO_SOURCE = 2;
51  	public static final int SOURCE_TO_DESTINATION_TO_SOURCE = 3;
52  
53  	private int mode;
54  	private boolean slowInSlowOut;
55  	private int loopCount;
56  	private boolean firstLoop;
57  	
58  	public PInterpolatingActivity(long duration, long stepRate) {
59  		this(duration, stepRate, 1, PInterpolatingActivity.SOURCE_TO_DESTINATION);
60  	}
61  
62  	public PInterpolatingActivity(long duration, long stepRate, int loopCount, int mode) {
63  		this(duration, stepRate, System.currentTimeMillis(), loopCount, mode);
64  	}
65  
66  	/**
67  	 * Create a new PInterpolatingActivity.
68  	 * <P>
69  	 * @param duration the length of one loop of the activity
70  	 * @param stepRate the amount of time between steps of the activity
71  	 * @param startTime the time (relative to System.currentTimeMillis()) that
72  	 * this activity should start.
73  	 * @param loopCount number of times the activity should reschedule itself
74  	 * @param mode defines how the activity interpolates between states
75  	 */
76  	public PInterpolatingActivity(long duration, long stepRate, long startTime, int loopCount, int mode) {
77  		super(duration, stepRate, startTime);
78  		this.loopCount = loopCount;
79  		this.mode = mode;
80  		slowInSlowOut = true;
81  		firstLoop = true;
82  	}

84  	/**
85  	 * Set the amount of time that this activity should take to complete,
86  	 * after the startStepping method is called. The duration must be greater
87  	 * then zero so that the interpolation value can be computed.
88  	 */
89  	public void setDuration(long aDuration) {
90  		if (aDuration <= 0) 
91  			throw new IllegalArgumentException("Duration for PInterpolatingActivity must be greater then 0");

93  		super.setDuration(aDuration);
94  	}

156 	//****************************************************************
157 	// Stepping - Instead of overriding the step methods subclasses
158 	// of this activity will normally override setRelativeTargetValue().
159 	// This method will be called for every step of the activity with
160 	// a value ranging from 0,0 (for the first step) to 1.0 (for the 
161 	// final step). See PTransformActivity for an example.
162 	//****************************************************************
163 	
164 	protected void activityStarted() {
165 		super.activityStarted();
166 		setRelativeTargetValueAdjustingForMode(0);
167 	}
168 		
169 	protected void activityStep(long elapsedTime) { 
170 		super.activityStep(elapsedTime);
171 
172 		float t = elapsedTime / (float) getDuration();
173 				
174 		t = Math.min(1, t);
175 		t = Math.max(0, t);
176 		
177 		if (getSlowInSlowOut()) {
178 			t = computeSlowInSlowOut(t);
179 		}
180 		
181 		setRelativeTargetValueAdjustingForMode(t);		
182 	}
183 	
184 	protected void activityFinished() {
185 		setRelativeTargetValueAdjustingForMode(1);
186 		super.activityFinished();
187 		
188 		PActivityScheduler scheduler = getActivityScheduler();
189 		if (loopCount > 1) {
190 			if (loopCount != Integer.MAX_VALUE) loopCount--;
191 			firstLoop = false;
192 			setStartTime(scheduler.getRoot().getGlobalTime());
193 			scheduler.addActivity(this);
194 		}
195 	}
196 	
197 	/**
198 	 * Stop this activity immediately, and remove it from the activity
199 	 * scheduler. If this activity is currently running then stoppedStepping
200 	 * will be called after it has been removed from the activity scheduler. 
201 	 */
202 	public void terminate() {
203 		loopCount = 0; // set to zero so that we don't reschedule self.
204 		super.terminate();
205 	}

214 	public float computeSlowInSlowOut(float zeroToOne) {
215 		if (zeroToOne < 0.5) {
216 			return 2.0f * zeroToOne * zeroToOne;
217 		} else {
218 			float complement = 1.0f - zeroToOne;
219 			return 1.0f - (2.0f * complement * complement);
220 		}
221 	}
222 	
223 	protected void setRelativeTargetValueAdjustingForMode(float zeroToOne) {
224 		switch (mode) {
225 			case SOURCE_TO_DESTINATION:
226 				break;
227 
228 			case DESTINATION_TO_SOURCE:
229 				zeroToOne = 1 - zeroToOne;
230 				break;
231 
232 			case SOURCE_TO_DESTINATION_TO_SOURCE:
233 				if (zeroToOne <= 0.5) {
234 					zeroToOne *= 2;
235 				} else {
236 					zeroToOne = 1 - ((zeroToOne - 0.5f) * 2);
237 				}
238 				break;
239 		}
240 
241 		setRelativeTargetValue(zeroToOne);
242 	}
