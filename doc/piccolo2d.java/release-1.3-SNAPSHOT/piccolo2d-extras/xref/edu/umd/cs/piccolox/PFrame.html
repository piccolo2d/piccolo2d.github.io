50  /**
51   * <b>PFrame</b> is meant to be subclassed by applications that just need a PCanvas in a JFrame.
52   * It also includes full screen mode functionality when run in JDK 1.4. These
53   * subclasses should override the initialize method and start adding their own
54   * code there. Look in the examples package to see lots of uses of PFrame.
55   *
56   * @version 1.0
57   * @author Jesse Grosjean
58   */
59  public class PFrame extends JFrame {

73  	public PFrame(String title, GraphicsDevice aDevice, final boolean fullScreenMode, final PCanvas aCanvas) {
74  		super(title, aDevice.getDefaultConfiguration());
75  		
76  		graphicsDevice = aDevice;
77  		
78  		setBackground(null);
79  		setBounds(getDefaultFrameBounds());
80  		
81  		try {
82  			setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
83  		} catch (SecurityException e) {} // expected from applets
84  		
85  		if (aCanvas == null) {
86  			canvas = new PCanvas();
87  		} else {
88  			canvas = aCanvas;
89  		}
90  						
91  		setContentPane(canvas);
92  		validate(); 	
93  		setFullScreenMode(fullScreenMode);
94  		canvas.requestFocus();
95  		beforeInitialize();
96  
97  		// Manipulation of Piccolo's scene graph should be done from Swings
98  		// event dispatch thread since Piccolo is not thread safe. This code calls
99  		// initialize() from that thread once the PFrame is initialized, so you are 
100 		// safe to start working with Piccolo in the initialize() method.
101 		SwingUtilities.invokeLater(new Runnable() {
102 			public void run() {
103 				PFrame.this.initialize();
104 				repaint();
105 			}
106 		});
107 	}

121 	public boolean isFullScreenMode() {
122 		return graphicsDevice.getFullScreenWindow() != null;
123 	}

125 	public void setFullScreenMode(boolean fullScreenMode) {
126 		if (fullScreenMode) {
127 			addEscapeFullScreenModeListener();
128 			
129 			if (isDisplayable()) {
130 				dispose();
131 			}
132 			
133 			setUndecorated(true);
134 			setResizable(false);
135 			graphicsDevice.setFullScreenWindow(this);			 
136 			
137 			if (graphicsDevice.isDisplayChangeSupported()) {
138 				chooseBestDisplayMode(graphicsDevice);
139 			}		 
140 			validate();
141 		} else {
142 			removeEscapeFullScreenModeListener();
143 			
144 			if (isDisplayable()) {
145 				dispose();
146 			}
147 			
148 			setUndecorated(false);
149 			setResizable(true);
150 			graphicsDevice.setFullScreenWindow(null);					 
151 			validate();
152 			setVisible(true);
153 		}		
154 	}

156 	protected void chooseBestDisplayMode(GraphicsDevice device) {
157 		DisplayMode best = getBestDisplayMode(device);
158 		if (best != null) {
159 			device.setDisplayMode(best);
160 		}
161 	}
162 	
163 	protected DisplayMode getBestDisplayMode(GraphicsDevice device) {
164 		Iterator itr = getPreferredDisplayModes(device).iterator();
165 		while (itr.hasNext()) {
166 			DisplayMode each = (DisplayMode) itr.next();
167 			DisplayMode[] modes = device.getDisplayModes();
168 			for (int i = 0; i < modes.length; i++) {
169 				if (modes[i].getWidth() == each.getWidth() && 
170 					modes[i].getHeight() == each.getHeight() && 
171 					modes[i].getBitDepth() == each.getBitDepth()) {
172 						return each;
173 				}
174 			}			
175 		}
176 		
177 		return null;
178 	}
179 	
180 	/**
181 	 * By default return the current display mode. Subclasses may override this method
182 	 * to return other modes in the collection.
183 	 */
184 	protected Collection getPreferredDisplayModes(GraphicsDevice device) {
185 		ArrayList result = new ArrayList();
186 		
187 		result.add(device.getDisplayMode());
188 		/*result.add(new DisplayMode(640, 480, 32, 0));
189 		result.add(new DisplayMode(640, 480, 16, 0));
190 		result.add(new DisplayMode(640, 480, 8, 0));*/
191 		
192 		return result;
193 	}

195 	/**
196 	 * This method adds a key listener that will take this PFrame out of full
197 	 * screen mode when the escape key is pressed. This is called for you
198 	 * automatically when the frame enters full screen mode.
199 	 */
200 	public void addEscapeFullScreenModeListener() {
201 		removeEscapeFullScreenModeListener();
202 		escapeFullScreenModeListener = new KeyAdapter() {
203 			public void keyPressed(KeyEvent aEvent) {
204 				if (aEvent.getKeyCode() == KeyEvent.VK_ESCAPE) {
205 					setFullScreenMode(false);
206 				}
207 			}
208 		};	
209 		canvas.addKeyListener((KeyListener)escapeFullScreenModeListener);
210 	}
211 	
212 	/**
213 	 * This method removes the escape full screen mode key listener. It will be
214 	 * called for you automatically when full screen mode exits, but the method
215 	 * has been made public for applications that wish to use other methods for
216 	 * exiting full screen mode.
217 	 */
218 	public void removeEscapeFullScreenModeListener() {
219 		if (escapeFullScreenModeListener != null) {
220 			canvas.removeKeyListener((KeyListener)escapeFullScreenModeListener);
221 			escapeFullScreenModeListener = null;
222 		}
223 	}
