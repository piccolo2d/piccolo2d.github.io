72      /**
73       * Disable or enable key actions on this PScrollPane
74       * 
75       * @param disable true disables key actions, false enables key actions

77      public void setKeyActionsDisabled(boolean disable) {
78          if (disable && this.disableKeyActions != disable) {
79              this.disableKeyActions = disable;
80              disableKeyActions();
81          }
82          else if (!disable && this.disableKeyActions != disable) {
83              this.disableKeyActions = disable;
84              installCustomKeyActions();

102     /**
103      * Install custom key actions (in place of the Swing defaults) to correctly
104      * scroll the view
105      */
106     protected void installCustomKeyActions() {
107         ActionMap map = getActionMap();
108 
109         map.put("scrollUp", new PScrollAction("scrollUp", SwingConstants.VERTICAL, -1, true));
110         map.put("scrollDown", new PScrollAction("scrollDown", SwingConstants.VERTICAL, 1, true));
111         map.put("scrollLeft", new PScrollAction("scrollLeft", SwingConstants.HORIZONTAL, -1, true));
112 
113         map.put("scrollRight", new PScrollAction("ScrollRight", SwingConstants.HORIZONTAL, 1, true));
114         map.put("unitScrollRight", new PScrollAction("UnitScrollRight", SwingConstants.HORIZONTAL, 1, false));
115         map.put("unitScrollLeft", new PScrollAction("UnitScrollLeft", SwingConstants.HORIZONTAL, -1, false));
116         map.put("unitScrollUp", new PScrollAction("UnitScrollUp", SwingConstants.VERTICAL, -1, false));
117         map.put("unitScrollDown", new PScrollAction("UnitScrollDown", SwingConstants.VERTICAL, 1, false));
118 
119         map.put("scrollEnd", new PScrollEndAction("ScrollEnd"));
120         map.put("scrollHome", new PScrollHomeAction("ScrollHome"));
121     }

154     /**
155      * Action to scroll left/right/up/down. Modified from
156      * javax.swing.plaf.basic.BasicScrollPaneUI.ScrollAction
157      * 
158      * Gets the view parameters (position and size) from the Viewport rather
159      * than directly from the view - also only performs its actions when the
160      * relevant scrollbar is visible
161      */
162     protected static class PScrollAction extends AbstractAction {
163         /** Direction to scroll. */
164         protected int orientation;
165         /** 1 indicates scroll down, -1 up. */
166         protected int direction;
167         /** True indicates a block scroll, otherwise a unit scroll. */
168         private boolean block;
169 
170         protected PScrollAction(String name, int orientation, int direction, boolean block) {
171             super(name);
172             this.orientation = orientation;
173             this.direction = direction;
174             this.block = block;
175         }
176 
177         public void actionPerformed(ActionEvent e) {
178             JScrollPane scrollpane = (JScrollPane) e.getSource();
179             // LEG: Modification to only perform these actions if the relevant
180             // scrollbar is actually showing
181             if ((orientation == SwingConstants.VERTICAL && scrollpane.getVerticalScrollBar().isShowing())
182                     || (orientation == SwingConstants.HORIZONTAL && scrollpane.getHorizontalScrollBar().isShowing())) {
183 
184                 JViewport vp = scrollpane.getViewport();
185                 Component view;
186                 if (vp != null && (view = vp.getView()) != null) {
187                     Rectangle visRect = vp.getViewRect();
188                     // LEG: Modification to query the viewport for the
189                     // view size rather than going directly to the view
190                     Dimension vSize = vp.getViewSize();
191                     int amount;
192 
193                     if (view instanceof Scrollable) {
194                         if (block) {
195                             amount = ((Scrollable) view).getScrollableBlockIncrement(visRect, orientation, direction);
196                         }
197                         else {
198                             amount = ((Scrollable) view).getScrollableUnitIncrement(visRect, orientation, direction);
199                         }
200                     }
201                     else {
202                         if (block) {
203                             if (orientation == SwingConstants.VERTICAL) {
204                                 amount = visRect.height;
205                             }
206                             else {
207                                 amount = visRect.width;
208                             }
209                         }
210                         else {
211                             amount = 10;
212                         }
213                     }
214                     if (orientation == SwingConstants.VERTICAL) {
215                         visRect.y += (amount * direction);
216                         if ((visRect.y + visRect.height) > vSize.height) {
217                             visRect.y = Math.max(0, vSize.height - visRect.height);
218                         }
219                         else if (visRect.y < 0) {
220                             visRect.y = 0;
221                         }
222                     }
223                     else {
224                         visRect.x += (amount * direction);
225                         if ((visRect.x + visRect.width) > vSize.width) {
226                             visRect.x = Math.max(0, vSize.width - visRect.width);
227                         }
228                         else if (visRect.x < 0) {
229                             visRect.x = 0;
230                         }
231                     }
232                     vp.setViewPosition(visRect.getLocation());
233                 }
234             }
235         }
236     }

238     /**
239      * Action to scroll to x,y location of 0,0. Modified from
240      * javax.swing.plaf.basic.BasicScrollPaneUI.ScrollEndAction
241      * 
242      * Only performs the event if a scrollbar is visible
243      */
244     private static class PScrollHomeAction extends AbstractAction {
245         protected PScrollHomeAction(String name) {
246             super(name);
247         }
248 
249         public void actionPerformed(ActionEvent e) {
250             JScrollPane scrollpane = (JScrollPane) e.getSource();
251             // LEG: Modification to only perform these actions if one of the
252             // scrollbars is actually showing
253             if (scrollpane.getVerticalScrollBar().isShowing() || scrollpane.getHorizontalScrollBar().isShowing()) {
254                 JViewport vp = scrollpane.getViewport();
255                 if (vp != null && vp.getView() != null) {
256                     vp.setViewPosition(new Point(0, 0));
257                 }
258             }
259         }
260     }

262     /**
263      * Action to scroll to last visible location. Modified from
264      * javax.swing.plaf.basic.BasicScrollPaneUI.ScrollEndAction
265      * 
266      * Gets the view size from the viewport rather than directly from the view -
267      * also only performs the event if a scrollbar is visible
268      */
269     protected static class PScrollEndAction extends AbstractAction {
270         protected PScrollEndAction(String name) {
271             super(name);
272         }
273 
274         public void actionPerformed(ActionEvent e) {
275             JScrollPane scrollpane = (JScrollPane) e.getSource();
276             // LEG: Modification to only perform these actions if one of the
277             // scrollbars is actually showing
278             if (scrollpane.getVerticalScrollBar().isShowing() || scrollpane.getHorizontalScrollBar().isShowing()) {
279 
280                 JViewport vp = scrollpane.getViewport();
281                 if (vp != null && vp.getView() != null) {
282 
283                     Rectangle visRect = vp.getViewRect();
284                     // LEG: Modification to query the viewport for the
285                     // view size rather than going directly to the view
286                     Dimension size = vp.getViewSize();
287                     vp.setViewPosition(new Point(size.width - visRect.width, size.height - visRect.height));
288                 }
289             }
290         }
