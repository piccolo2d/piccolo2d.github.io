15  /**
16   * This example show how to add the default handles to a node, and also how
17   * to create your own custom handles.
18   */
19  public class HandleExample extends PFrame {
20  	
21  	public HandleExample() {
22  		this(null);
23  	}
24  	
25  	public HandleExample(PCanvas aCanvas) {
26  		super("HandleExample", false, aCanvas);
27  	}
28  	
29  	public void initialize() {
30  		PPath n = PPath.createRectangle(0, 0, 100, 80);
31  		
32  		// add another node the the root as a reference point so that we can 
33  		// tell that our node is getting dragged, as opposed the the canvas 
34  		// view being panned.
35  		getCanvas().getLayer().addChild(PPath.createRectangle(0, 0, 100, 80));
36  		
37  		getCanvas().getLayer().addChild(n);
38  		
39  		// tell the node to show its default handles.
40  		PBoundsHandle.addBoundsHandlesTo(n);
41  		
42  		// The default PBoundsHandle implementation doesn't work well with PPaths that have strokes. The reason
43  		// for this is that the default PBoundsHandle modifies the bounds of an PNode, but when adding handles to
44  		// a PPath we really want it to be modifying the underlying geometry of the PPath, the shape without the
45  		// stroke. The solution is that we need to create handles specific to PPaths that locate themselves on the
46  		// paths internal geometry, not the external bounds geometry...
47  		
48  		n.setStroke(new BasicStroke(10));		
49  		n.setPaint(Color.green);
50  		
51  		// Here we create our own custom handle. This handle is located in the center of its parent
52  		// node and you can use it to drag the parent around. This handle also updates its color when
53  		// the is pressed/released in it. 
54  		final PHandle h = new PHandle(new PNodeLocator(n)) { // the default locator locates the center of a node.						
55  			public void dragHandle(PDimension aLocalDimension, PInputEvent aEvent) {
56  				localToParent(aLocalDimension);
57  				getParent().translate(aLocalDimension.getWidth(), aLocalDimension.getHeight());
58  			}			
59  		};
60  		
61  		h.addInputEventListener(new PBasicInputEventHandler() {
62  			public void mousePressed(PInputEvent aEvent) {
63  				h.setPaint(Color.YELLOW);
64  			}
65  			
66  			public void mouseReleased(PInputEvent aEvent) {
67  				h.setPaint(Color.RED);
68  			}
69  		});
70  		
71  		// make this handle appear a bit different then the default handle appearance.
72  		h.setPaint(Color.RED);
73  		h.setBounds(-10, -10, 20, 20);
74  		
75  		// also add our new custom handle to the node.
76  		n.addChild(h);
77  	}
78  
79  	public static void main(String[] args) {
80  		new HandleExample();
81  	}
