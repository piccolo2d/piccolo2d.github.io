33  /**
34   * <b>PNodeLocator</b> provides an abstraction for locating points on a node.
35   * Points are located in the local corrdinate system of the node. The default
36   * behavior is to locate the center point of the nodes bounds. The node where
37   * the point is located is stored internal to this locator (as an instance
38   * varriable). If you want to use the same locator to locate center points on
39   * many different nodes you will need to call setNode() before asking for each
40   * location.
41   * <P>
42   * @version 1.0
43   * @author Jesse Grosjean
44   */
45  public class PNodeLocator extends PLocator {
46  	
47  	protected PNode node;
48  
49  	public PNodeLocator(PNode node) {
50  		setNode(node);
51  	}
52  
53  	public PNode getNode() {
54  		return node;
55  	}
56  	
57  	public void setNode(PNode node) {
58  		this.node = node;
59  	}
60  
61  	public double locateX() {
62  		return node.getBoundsReference().getCenterX();
63  	}
64  
65  	public double locateY() {
66  		return node.getBoundsReference().getCenterY();
67  	}
68  }
