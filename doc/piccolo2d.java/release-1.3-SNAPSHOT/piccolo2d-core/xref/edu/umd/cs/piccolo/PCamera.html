57  /**
58   * <b>PCamera</b> represents a viewport onto a list of layer nodes.
59   * Each camera maintains a view transform through which it views these
60   * layers. Translating and scaling this view transform is how zooming
61   * and panning are implemented.
62   * <p>
63   * Cameras are also the point through which all PInputEvents enter Piccolo. The
64   * canvas coordinate system, and the local coordinate system of the topmost camera
65   * should always be the same.
66   * <p>
67   * @see PLayer
68   * @version 1.0
69   * @author Jesse Grosjean
70   */
71  public class PCamera extends PNode {
72

75  	 * layers (see {@link #getLayer getLayer}, {@link #getLayerCount
76  	 * getLayerCount}, {@link #getLayersReference getLayersReference}). A
77  	 * property change event's new value will be a reference to the list of this
78  	 * nodes layers, but old value will always be null.
79  	 */

88  	 * value will always be null.

91      public static final int PROPERTY_CODE_VIEW_TRANSFORM = 1 << 12;

129 	/**
130 	 * Repaint this camera, and forward the repaint request to the camera's
131 	 * canvas if it is not null.
132 	 */
133 	public void repaintFrom(PBounds localBounds, PNode descendentOrThis) {
134 		if (getParent() != null) {
135 			if (descendentOrThis != this) {
136 				localToParent(localBounds);
137 			}
138 			
139 			if (component != null) {
140 				component.repaint(localBounds);
141 			}
142 			
143 			getParent().repaintFrom(localBounds, this);
144 		}

147 	private static PBounds TEMP_REPAINT_RECT = new PBounds();
148 		
149 	/**
150 	 * Repaint from one of the cameras layers. The repaint region needs to be
151 	 * transformed from view to local in this case. Unlike most repaint
152 	 * methods in piccolo this one must not modify the viewBounds parameter.
153 	 */
154 	public void repaintFromLayer(PBounds viewBounds, PNode repaintedLayer) {
155 		TEMP_REPAINT_RECT.setRect(viewBounds);
156 		
157 		viewToLocal(TEMP_REPAINT_RECT);
158 		if (getBoundsReference().intersects(TEMP_REPAINT_RECT)) {
159 			PBounds.intersect(TEMP_REPAINT_RECT, getBoundsReference(), TEMP_REPAINT_RECT);
160 			repaintFrom(TEMP_REPAINT_RECT, repaintedLayer);
161 		}
162 	}

245 	/**
246 	 * Paint this camera (default background color is white) and then paint
247 	 * the cameras view through the view transform.
248 	 */
249 	protected void paint(PPaintContext paintContext) {
250 		super.paint(paintContext);
251 		
252 		paintContext.pushClip(getBoundsReference());			
253 		paintContext.pushTransform(viewTransform);
254 
255 		paintCameraView(paintContext);		
256 		paintDebugInfo(paintContext);
257 						
258 		paintContext.popTransform(viewTransform);
259 		paintContext.popClip(getBoundsReference());
260 	}
261 	
262 	/**
263 	 * Paint all the layers that the camera is looking at, this method is
264 	 * only called when the cameras view transform and clip are applied
265 	 * to the paintContext.
266 	 */
267 	protected void paintCameraView(PPaintContext paintContext) {
268 		int count = getLayerCount();			
269 		for (int i = 0; i < count; i++) {
270 			PLayer each = (PLayer) layers.get(i);
271 			each.fullPaint(paintContext);
272 		}			
273 	}
274 
275 	protected void paintDebugInfo(PPaintContext paintContext) {
276 		if (PDebug.debugBounds || PDebug.debugFullBounds) {
277 			Graphics2D g2 = paintContext.getGraphics();
278 			paintContext.setRenderQuality(PPaintContext.LOW_QUALITY_RENDERING);
279 			g2.setStroke(new BasicStroke(0));
280 			ArrayList nodes = new ArrayList();
281 			PBounds nodeBounds = new PBounds();
282 		
283 			Color boundsColor = Color.red;
284 			Color fullBoundsColor = new Color(1.0f, 0f, 0f, 0.2f);
285 		
286 			for (int i = 0; i < getLayerCount(); i++) {
287 				getLayer(i).getAllNodes(null, nodes);
288 			}				
289 			
290 			Iterator i = getAllNodes(null, nodes).iterator();
291 			
292 			while (i.hasNext()) {
293 				PNode each = (PNode) i.next();
294 		
295 				if (PDebug.debugBounds) {
296 					g2.setPaint(boundsColor);
297 					nodeBounds.setRect(each.getBoundsReference());
298 					
299 					if (!nodeBounds.isEmpty()) {
300 						each.localToGlobal(nodeBounds);
301 						globalToLocal(nodeBounds);
302 						if (each == this || each.isDescendentOf(this)) {
303 							localToView(nodeBounds);
304 						}
305 						g2.draw(nodeBounds);
306 					}
307 				}
308 		
309 				if (PDebug.debugFullBounds) {
310 					g2.setPaint(fullBoundsColor);
311 					nodeBounds.setRect(each.getFullBoundsReference());
312 
313 					if (!nodeBounds.isEmpty()) {
314 						if (each.getParent() != null) {
315 							each.getParent().localToGlobal(nodeBounds);
316 						}
317 						globalToLocal(nodeBounds);		
318 						if (each == this || each.isDescendentOf(this)) {
319 							localToView(nodeBounds);
320 						}	
321 						g2.fill(nodeBounds);				
322 					}
323 				}
324 			}
325 		}
326 	}

342 	/**
343 	 * Generate and return a PPickPath for the point x,y specified in the local
344 	 * coord system of this camera. Picking is done with a rectangle, halo
345 	 * specifies how large that rectangle will be.
346 	 */
347 	public PPickPath pick(double x, double y, double halo) {		
348 		PBounds b = new PBounds(new Point2D.Double(x, y), -halo, -halo);
349 		PPickPath result = new PPickPath(this, b);
350 		
351 		fullPick(result);
352 			
353 		// make sure this camera is pushed.
354 		if (result.getNodeStackReference().size() == 0) {
355 			result.pushNode(this);
356 			result.pushTransform(getTransformReference(false));
357 		}
358 		
359 		return result;
360 	}
361 	
362 	/**
363 	 * After the direct children of the camera have been given a chance to be
364 	 * picked objects viewed by the camera are given a chance to be picked.
365 	 */
366 	protected boolean pickAfterChildren(PPickPath pickPath) {
367 		if (intersects(pickPath.getPickBounds())) { 		
368 			pickPath.pushTransform(viewTransform);
369 			
370 			if (pickCameraView(pickPath)) {
371 				return true;	
372 			}
373 				
374 			pickPath.popTransform(viewTransform);			
375 			return true;
376 		}
377 		return false;
378 	}
379 	
380 	/**
381 	 * Pick all the layers that the camera is looking at, this method is
382 	 * only called when the cameras view transform and clip are applied
383 	 * to the pickPath.
384 	 */
385 	protected boolean pickCameraView(PPickPath pickPath) {
386 		int count = getLayerCount();
387 		for (int i = count - 1; i >= 0; i--) {
388 			PLayer each = (PLayer) layers.get(i);
389 			if (each.fullPick(pickPath)) {
390 				return true;
391 			}
392 		}
393 		return false;
394 	}

505 	/**
506 	 * Animate the camera's view from its current transform when the activity
507 	 * starts to a new transform that centers the given bounds in the camera
508 	 * layers coordinate system into the cameras view bounds. If the duration is
509 	 * 0 then the view will be transformed immediately, and null will be
510 	 * returned. Else a new PTransformActivity will get returned that is set to
511 	 * animate the camera's view transform to the new bounds. If shouldScale
512 	 * is true, then the camera will also scale its view so that the given
513 	 * bounds fit fully within the cameras view bounds, else the camera will
514 	 * maintain its original scale.
515 	 */
516 	public PTransformActivity animateViewToCenterBounds(Rectangle2D centerBounds, boolean shouldScaleToFit, long duration) {		
517 		PBounds viewBounds = getViewBounds();
518 		PDimension delta = viewBounds.deltaRequiredToCenter(centerBounds);
519 		PAffineTransform newTransform = getViewTransform();
520 		newTransform.translate(delta.width, delta.height);
521 		
522 		if (shouldScaleToFit) {
523 			double s = Math.min(viewBounds.getWidth() / centerBounds.getWidth(), viewBounds.getHeight() / centerBounds.getHeight());
524 			if (s != Double.POSITIVE_INFINITY && s != 0) {
525 				newTransform.scaleAboutPoint(s, centerBounds.getCenterX(), centerBounds.getCenterY());
526 			}
527 		}
528 
529 		return animateViewToTransform(newTransform, duration);
530 	}
531 
532 	/**
533 	 * Pan the camera's view from its current transform when the activity starts
534 	 * to a new transform so that the view bounds will contain (if possible, intersect 
535 	 * if not possible) the new bounds in the camera layers coordinate system. 
536 	 * If the duration is 0 then the view will be transformed immediately, and null 
537 	 * will be returned. Else a new PTransformActivity will get returned that is set 
538 	 * to animate the camera's view transform to the new bounds.
539 	 */
540 	public PTransformActivity animateViewToPanToBounds(Rectangle2D panToBounds, long duration) {
541 		PBounds viewBounds = getViewBounds();
542 		PDimension delta = viewBounds.deltaRequiredToContain(panToBounds);
543 		
544 		if (delta.width != 0 || delta.height != 0) {
545 			if (duration == 0) {
546 				translateView(-delta.width, -delta.height);
547 			} else {
548 				AffineTransform at = getViewTransform();
549 				at.translate(-delta.width, -delta.height);
550 				return animateViewToTransform(at, duration);
551 			}
552 		}
553 
554 		return null;
555 	}

562 	}

564 	/**
565 	 * Animate the cameras view transform from its current value when the
566 	 * activity starts to the new destination transform value.
567 	 */
568 	public PTransformActivity animateViewToTransform(AffineTransform destination, long duration) {
569 		if (duration == 0) {
570 			setViewTransform(destination);
571 			return null;
572 		}
573 		
574 		PTransformActivity.Target t = new PTransformActivity.Target() {
575 			public void setTransform(AffineTransform aTransform) {
576 				PCamera.this.setViewTransform(aTransform);
577 			}
578 			public void getSourceMatrix(double[] aSource) {
579 				PCamera.this.viewTransform.getMatrix(aSource);
580 			}
581 		};
582 		
583 		PTransformActivity ta = new PTransformActivity(duration, PUtil.DEFAULT_ACTIVITY_STEP_RATE, t, destination);
584 		
585 		PRoot r = getRoot();
586 		if (r != null) {
587 			r.getActivityScheduler().addActivity(ta);
588 		}
589 		
590 		return ta;
591 	}

594 	// View Transform Constraints - Methods for setting and applying
595 	// constraints to the view transform.
596 	//****************************************************************
597 	
598 	public int getViewConstraint() {
599 		return viewConstraint;
600 	}
601 	public void setViewConstraint(int constraint) {
602 		viewConstraint = constraint;
603 		applyViewConstraints();
604 	}
605 	
606 	protected void applyViewConstraints() {
607 		if (viewConstraint == VIEW_CONSTRAINT_NONE)
608 			return;
609 
610 		PBounds viewBounds = getViewBounds();
611 		PBounds layerBounds = (PBounds) globalToLocal(getUnionOfLayerFullBounds());
612 		PDimension constraintDelta = null;
613 					
614 		switch (viewConstraint) {
615 			case VIEW_CONSTRAINT_ALL:
616 				constraintDelta = viewBounds.deltaRequiredToContain(layerBounds);
617 				break;
618 
619 			case VIEW_CONSTRAINT_CENTER:
620 				layerBounds.setRect(layerBounds.getCenterX(), layerBounds.getCenterY(), 0, 0);
621 				constraintDelta = viewBounds.deltaRequiredToContain(layerBounds);
622 				break;
623 		}
624 		
625 		viewTransform.translate(-constraintDelta.width, -constraintDelta.height);	
626 	}

660 	/**
661 	 * Convert the point from the camera's local coordinate system to the 
662 	 * camera's view coordinate system. The given point is modified by this method.
663 	 */
664 	public Point2D localToView(Point2D localPoint) {
665 		try {
666 			return viewTransform.inverseTransform(localPoint, localPoint);
667 		} catch (NoninvertibleTransformException e) {
