93       * camera, and layer. Event handlers for zooming and panning are

106         setBackground(Color.WHITE);
107     }

246     /**
247      * Set if this canvas is interacting. If so the canvas will normally render
248      * at a lower quality that is faster. Also repaints the canvas if the render
249      * quality should change.
250      */
251     public void setInteracting(boolean isInteracting) {
252         boolean wasInteracting = getInteracting();
253 
254         if (isInteracting) {
255             interacting++;
256         }
257         else {
258             interacting--;
259         }
260 
261         if (!getInteracting()) { // determine next render quality and repaint if
262             // it's greater then the old
263             // interacting render quality.
264             int nextRenderQuality = defaultRenderQuality;
265             if (getAnimating())
266                 nextRenderQuality = animatingRenderQuality;
267             if (nextRenderQuality > interactingRenderQuality) {
268                 repaint();
269             }
270         }
271 
272         isInteracting = getInteracting();
273 
274         if (wasInteracting != isInteracting) {
275             firePropertyChange(INTERATING_CHANGED_NOTIFICATION, wasInteracting, isInteracting);
276         }
277     }

365     protected void installInputSources() {
366         if (mouseListener == null) {
367             mouseListener = new MouseListener() {
368                 public void mouseClicked(MouseEvent e) {
369                     sendInputEventToInputManager(e, MouseEvent.MOUSE_CLICKED);
370                 }

372                 public void mouseEntered(MouseEvent e) {
373                     MouseEvent simulated = null;
374 
375                     if ((e.getModifiersEx() & (InputEvent.BUTTON1_DOWN_MASK | InputEvent.BUTTON2_DOWN_MASK | InputEvent.BUTTON3_DOWN_MASK)) != 0) {
376                         simulated = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_DRAGGED, e.getWhen(), e
377                                 .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(), e
378                                 .getButton());
379                     }
380                     else {
381                         simulated = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_MOVED, e.getWhen(), e
382                                 .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(), e
383                                 .getButton());
384                     }
385 
386                     sendInputEventToInputManager(e, MouseEvent.MOUSE_ENTERED);
387                     sendInputEventToInputManager(simulated, simulated.getID());
388                 }
389 
390                 public void mouseExited(MouseEvent e) {
391                     MouseEvent simulated = null;
392 
393                     if ((e.getModifiersEx() & (InputEvent.BUTTON1_DOWN_MASK | InputEvent.BUTTON2_DOWN_MASK | InputEvent.BUTTON3_DOWN_MASK)) != 0) {
394                         simulated = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_DRAGGED, e.getWhen(), e
395                                 .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(), e
396                                 .getButton());
397                     }
398                     else {
399                         simulated = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_MOVED, e.getWhen(), e
400                                 .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(), e

408                 public void mousePressed(MouseEvent e) {
409                     requestFocus();
410 
411                     boolean shouldBalanceEvent = false;
412 
413                     if (e.getButton() == MouseEvent.NOBUTTON) {
414                         if ((e.getModifiers() & MouseEvent.BUTTON1_MASK) == MouseEvent.BUTTON1_MASK) {
415                             e = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_PRESSED, e.getWhen(), e
416                                     .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(),
417                                     MouseEvent.BUTTON1);
418                         }
419                         else if ((e.getModifiers() & MouseEvent.BUTTON2_MASK) == MouseEvent.BUTTON2_MASK) {
420                             e = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_PRESSED, e.getWhen(), e
421                                     .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(),
422                                     MouseEvent.BUTTON2);
423                         }
424                         else if ((e.getModifiers() & MouseEvent.BUTTON3_MASK) == MouseEvent.BUTTON3_MASK) {
425                             e = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_PRESSED, e.getWhen(), e
426                                     .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(),
427                                     MouseEvent.BUTTON3);
428                         }
429                     }
430 
431                     switch (e.getButton()) {
432                         case MouseEvent.BUTTON1:
433                             if (isButton1Pressed) {
434                                 shouldBalanceEvent = true;
435                             }
436                             isButton1Pressed = true;
437                             break;
438 
439                         case MouseEvent.BUTTON2:
440                             if (isButton2Pressed) {
441                                 shouldBalanceEvent = true;
442                             }
443                             isButton2Pressed = true;
444                             break;
445 
446                         case MouseEvent.BUTTON3:
447                             if (isButton3Pressed) {
448                                 shouldBalanceEvent = true;
449                             }
450                             isButton3Pressed = true;
451                             break;
452                     }
453 
454                     if (shouldBalanceEvent) {
455                         MouseEvent balanceEvent = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_RELEASED,
456                                 e.getWhen(), e.getModifiers(), e.getX(), e.getY(), e.getClickCount(), e
457                                         .isPopupTrigger(), e.getButton());
458                         sendInputEventToInputManager(balanceEvent, MouseEvent.MOUSE_RELEASED);
459                     }
460 
461                     sendInputEventToInputManager(e, MouseEvent.MOUSE_PRESSED);
462                 }
463 
464                 public void mouseReleased(MouseEvent e) {
465                     boolean shouldBalanceEvent = false;
466 
467                     if (e.getButton() == MouseEvent.NOBUTTON) {
468                         if ((e.getModifiers() & MouseEvent.BUTTON1_MASK) == MouseEvent.BUTTON1_MASK) {
469                             e = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_RELEASED, e.getWhen(), e
470                                     .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(),
471                                     MouseEvent.BUTTON1);
472                         }
473                         else if ((e.getModifiers() & MouseEvent.BUTTON2_MASK) == MouseEvent.BUTTON2_MASK) {
474                             e = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_RELEASED, e.getWhen(), e
475                                     .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(),
476                                     MouseEvent.BUTTON2);
477                         }
478                         else if ((e.getModifiers() & MouseEvent.BUTTON3_MASK) == MouseEvent.BUTTON3_MASK) {
479                             e = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_RELEASED, e.getWhen(), e
480                                     .getModifiers(), e.getX(), e.getY(), e.getClickCount(), e.isPopupTrigger(),
481                                     MouseEvent.BUTTON3);
482                         }
483                     }
484 
485                     switch (e.getButton()) {
486                         case MouseEvent.BUTTON1:
487                             if (!isButton1Pressed) {
488                                 shouldBalanceEvent = true;
489                             }
490                             isButton1Pressed = false;
491                             break;
492 
493                         case MouseEvent.BUTTON2:
494                             if (!isButton2Pressed) {
495                                 shouldBalanceEvent = true;
496                             }
497                             isButton2Pressed = false;
498                             break;
499 
500                         case MouseEvent.BUTTON3:
501                             if (!isButton3Pressed) {
502                                 shouldBalanceEvent = true;
503                             }
504                             isButton3Pressed = false;
505                             break;
506                     }
507 
508                     if (shouldBalanceEvent) {
509                         MouseEvent balanceEvent = new MouseEvent((Component) e.getSource(), MouseEvent.MOUSE_PRESSED, e
510                                 .getWhen(), e.getModifiers(), e.getX(), e.getY(), e.getClickCount(),
511                                 e.isPopupTrigger(), e.getButton());
512                         sendInputEventToInputManager(balanceEvent, MouseEvent.MOUSE_PRESSED);
513                     }
514 
515                     sendInputEventToInputManager(e, MouseEvent.MOUSE_RELEASED);
516                 }
517             };
518             addMouseListener(mouseListener);
519         }
520 
521         if (mouseMotionListener == null) {
522             mouseMotionListener = new MouseMotionListener() {
523                 public void mouseDragged(MouseEvent e) {
524                     sendInputEventToInputManager(e, MouseEvent.MOUSE_DRAGGED);
525                 }
526 
527                 public void mouseMoved(MouseEvent e) {
528                     sendInputEventToInputManager(e, MouseEvent.MOUSE_MOVED);
529                 }
530             };
531             addMouseMotionListener(mouseMotionListener);
532         }
533 
534         if (mouseWheelListener == null) {
535             mouseWheelListener = new MouseWheelListener() {
536                 public void mouseWheelMoved(MouseWheelEvent e) {
537                     sendInputEventToInputManager(e, e.getScrollType());
538                     if (!e.isConsumed() && getParent() != null) {
539                         getParent().dispatchEvent(e);
540                     }
541                 }
542             };
543             addMouseWheelListener(mouseWheelListener);
544         }
545 
546         if (keyListener == null) {
547             keyListener = new KeyListener() {
548                 public void keyPressed(KeyEvent e) {
549                     sendInputEventToInputManager(e, KeyEvent.KEY_PRESSED);
550                 }
551 
552                 public void keyReleased(KeyEvent e) {
553                     sendInputEventToInputManager(e, KeyEvent.KEY_RELEASED);
554                 }
555 
556                 public void keyTyped(KeyEvent e) {
557                     sendInputEventToInputManager(e, KeyEvent.KEY_TYPED);
558                 }
559             };
560             addKeyListener(keyListener);
561         }

602     public void paintComponent(Graphics g) {
603         PDebug.startProcessingOutput();
604 
605         Graphics2D g2 = (Graphics2D) g.create();
606         g2.setColor(getBackground());
607         g2.fillRect(0, 0, getWidth(), getHeight());
608 
609         // create new paint context and set render quality to lowest common
610         // denominator render quality.
611         PPaintContext paintContext = new PPaintContext(g2);
612         if (getInteracting() || getAnimating()) {
613             if (interactingRenderQuality < animatingRenderQuality) {
614                 paintContext.setRenderQuality(interactingRenderQuality);
615             }
616             else {
617                 paintContext.setRenderQuality(animatingRenderQuality);
618             }
619         }
620         else {
621             paintContext.setRenderQuality(defaultRenderQuality);
622         }
623 
624         // paint piccolo
625         camera.fullPaint(paintContext);
626 
627         // if switched state from animating to not animating invalidate the
628         // entire
629         // screen so that it will be drawn with the default instead of animating
630         // render quality.
631         if (!getAnimating() && animatingOnLastPaint) {
632             repaint();
633         }
634         animatingOnLastPaint = getAnimating();
635 
636         PDebug.endProcessingOutput(g2);
637     }

646         paintingImmediately = false;
