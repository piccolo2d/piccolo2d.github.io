18  /**
19   * A subclass of JViewport that talks to the scroll director to negotiate
20   * the view positions and sizes.
21   *
22   * @author Lance Good
23   */
24  public class PViewport extends JViewport {
25  
26  	/**
27  	 * Controls what happens when scrolling occurs
28  	 */
29  	PScrollDirector scrollDirector;
30  
31  	/**
32  	 * Pass constructor info to super
33  	 */
34  	public PViewport() {
35  		super();
36  
37  		setScrollDirector(createScrollDirector());
38  	}
39  
40  	/**
41  	 * Subclassers can override this to install a different
42  	 * layout manager (or <code>null</code>) in the constructor.  Returns
43  	 * a new <code>ViewportLayout</code> object.
44  	 *
45  	 * @return a <code>LayoutManager</code>
46  	 */
47  	protected LayoutManager createLayoutManager() {
48  		return new PViewportLayout();
49  	}
50  
51  	/**
52  	 * Subclassers can override this to install a different scroll director
53  	 * in the constructor.	Returns a new <code>PScrollDirector</code> object.
54  	 * @return a <code>PScrollDirector
55  	 */
56  	protected PScrollDirector createScrollDirector() {
57  		return new PDefaultScrollDirector();
58  	}
59  
60  	/**
61  	 * Set the scroll director on this viewport
62  	 * @param scrollDirector The new scroll director
63  	 */
64  	public void setScrollDirector(PScrollDirector scrollDirector) {
65  		if (this.scrollDirector != null) {
66  			this.scrollDirector.unInstall();
67  		}
68  		this.scrollDirector = scrollDirector;
69  		if (scrollDirector != null) {
70  			this.scrollDirector.install(this, (PCanvas) getView());
71  		}
72  	}
73  
74  	/**
75  	 * @return The scroll director on this viewport
76  	 */
77  	public PScrollDirector getScrollDirector() {
78  		return scrollDirector;
79  	}
80  
81  	/**
82  	 * Overridden to throw an exception if the view is not a ZCanvas
83  	 * @param view The new view - it better be a ZCanvas!
84  	 */
85  	public void setView(Component view) {
86  		if (!(view instanceof PCanvas)) {
87  			throw new UnsupportedOperationException("PViewport only supports ZCanvas");
88  		}
89  
90  		super.setView(view);
91  
92  		if (scrollDirector != null) {
93  			scrollDirector.install(this, (PCanvas) view);
94  		}
95  	}
96  
97  	/**
98  	 * Sets the view coordinates that appear in the upper left
99  	 * hand corner of the viewport, does nothing if there's no view.
100 	 *
101 	 * @param p  a <code>Point</code> object giving the upper left coordinates
102 	 */
103 	public void setViewPosition(Point p) {
104 		if (getView() == null) {
105 			return;
106 		}
107 
108 		double oldX = 0, oldY = 0, x = p.x, y = p.y;
109 
110 		Point2D vp = getViewPosition();
111 		if (vp != null) {
112 			oldX = vp.getX();
113 			oldY = vp.getY();
114 		}
115 
116 		/**
117 		 * Send the scroll director the exact view position and let it
118 		 * interpret it as needed
119 		 */
120 		double newX = x;
121 		double newY = y;
122 
123 		if ((oldX != newX) || (oldY != newY)) {
124 			scrollUnderway = true;
125 
126 			scrollDirector.setViewPosition(newX, newY);
127 
128 			fireStateChanged();
129 		}
130 	}
131 
132 	/**
133 	 * Gets the view position from the scroll director based on the current
134 	 * extent size
135 	 * @return The new view position
136 	 */
137 	public Point getViewPosition() {
138 		if (scrollDirector != null) {
139 			Dimension extent = getExtentSize();
140 			return scrollDirector.getViewPosition(new PBounds(0, 0, extent.getWidth(), extent.getHeight()));
141 		}
142 		else {
143 			return null;
144 		}
145 	}
146 
147 	/**
148 	 * Gets the view size from the scroll director based on the current
149 	 * extent size
150 	 * @return The new view size
151 	 */
152 	public Dimension getViewSize() {
153 		Dimension extent = getExtentSize();
154 		return scrollDirector.getViewSize(new PBounds(0, 0, extent.getWidth(), extent.getHeight()));
155 	}
156 
157 	/**
158 	 * Gets the view size from the scroll director based on the specified
159 	 * extent size
160 	 * @param r The extent size from which the view is computed
161 	 * @return The new view size
162 	 */
163 	public Dimension getViewSize(Rectangle2D r) {
164 		return scrollDirector.getViewSize(r);
165 	}
166 
167 	/**
168 	 * Notifies all <code>ChangeListeners</code> when the views
169 	 * size, position, or the viewports extent size has changed.
170 	 */
171 	public void fireStateChanged() {
172 		super.fireStateChanged();
173 	}
174 
175 	/**
176 	 * A simple layout manager to give the ZCanvas the same size as the
177 	 * Viewport
178 	 */
179 	public static class PViewportLayout extends ViewportLayout {
180 		/**
181 		 * Called when the specified container needs to be laid out.
182 		 *
183 		 * @param parent  the container to lay out
184 		 */
185 		public void layoutContainer(Container parent) {
186 			JViewport vp = (JViewport) parent;
187 			Component view = vp.getView();
188 
189 			if (view == null) {
190 				return;
191 			}
192 
193 			Dimension extentSize = vp.getSize();
194 
195 			vp.setViewSize(extentSize);
196 		}
197 
198 	}
