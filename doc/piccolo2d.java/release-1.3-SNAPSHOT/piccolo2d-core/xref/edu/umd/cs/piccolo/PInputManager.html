44  /**
45   * <b>PInputManager</b> is responsible for dispatching PInputEvents
46   * to node's event listeners. Events are dispatched from PRoot's processInputs
47   * method.
48   * <P>
49   * @see edu.umd.cs.piccolo.event.PInputEvent

52   * @author Jesse Grosjean

92  	public void setKeyboardFocus(PInputEventListener eventHandler) {		
93  		PInputEvent focusEvent = new PInputEvent(this, null);
94  		
95  		if (keyboardFocus != null) {
96  			dispatchEventToListener(focusEvent, FocusEvent.FOCUS_LOST, keyboardFocus);
97  		}
98  		
99  		keyboardFocus = eventHandler;
100 		
101 		if (keyboardFocus != null) {
102 			dispatchEventToListener(focusEvent, FocusEvent.FOCUS_GAINED, keyboardFocus);
103 		}
104 	}

192 	public void mousePressed(PInputEvent event) {		
193 		if (pressedCount == 0) {
194 			setMouseFocus(getMouseOver());
195 		}
196 		pressedCount++;
197 		dispatchEventToListener(event, MouseEvent.MOUSE_PRESSED, mouseFocus);
198 		if (pressedCount < 1 || pressedCount > 3) System.err.println("invalid pressedCount on mouse pressed: " + pressedCount);
199 	}
200 	
201 	public void mouseReleased(PInputEvent event) {
202 		pressedCount--;
203 		checkForMouseEnteredAndExited(event);		
204 		dispatchEventToListener(event, MouseEvent.MOUSE_RELEASED, mouseFocus);
205 		if (pressedCount == 0) {
206 			setMouseFocus(null);
207 		}
208 		if (pressedCount < 0 || pressedCount > 2) System.err.println("invalid pressedCount on mouse released: " + pressedCount);
209 	}
210 	
211 	protected void checkForMouseEnteredAndExited(PInputEvent event) {		
212 		PNode c = (mouseOver != null) ? mouseOver.getPickedNode() : null; 
213 		PNode p = (previousMouseOver != null) ? previousMouseOver.getPickedNode() : null;
214 		
215 		if (c != p) {
216 			dispatchEventToListener(event, MouseEvent.MOUSE_EXITED, previousMouseOver);
217 			dispatchEventToListener(event, MouseEvent.MOUSE_ENTERED, mouseOver);
218 			previousMouseOver = mouseOver;
219 		}
220 	}

226 	public void processInput() {
227 		if (nextInput == null) return;
228 
229 		PInputEvent e = new PInputEvent(this, nextInput);
230 		
231 		Point2D newCurrentCanvasPosition = null;
232 		Point2D newLastCanvasPosition = null;
233 		
234 		if (e.isMouseEvent()) {
235 			if (e.isMouseEnteredOrMouseExited()) {
236 				PPickPath aPickPath = nextInputSource.pick(((MouseEvent)nextInput).getX(), ((MouseEvent)nextInput).getY(), 1);
237 				setMouseOver(aPickPath);
238 				previousMouseOver = aPickPath;
239 				newCurrentCanvasPosition = (Point2D) currentCanvasPosition.clone();
240 				newLastCanvasPosition = (Point2D) lastCanvasPosition.clone();
241 			} else {
242 				lastCanvasPosition.setLocation(currentCanvasPosition);
243 				currentCanvasPosition.setLocation(((MouseEvent)nextInput).getX(), ((MouseEvent)nextInput).getY());						
244 				PPickPath aPickPath = nextInputSource.pick(currentCanvasPosition.getX(), currentCanvasPosition.getY(), 1);
245 				setMouseOver(aPickPath);			
246 			}
247 		}
248 				
249 		nextInput = null;
250 		nextInputSource = null;
251 		
252 		this.processEvent(e, nextType);
253 		
254 		if (newCurrentCanvasPosition != null && newLastCanvasPosition != null) {
255 			currentCanvasPosition.setLocation(newCurrentCanvasPosition);
256 			lastCanvasPosition.setLocation(newLastCanvasPosition);
257 		}
258 	}

260 	public void processEventFromCamera(InputEvent event, int type, PCamera camera) {
261 		// queue input
262 		nextInput = event;
263 		nextType = type;
264 		nextInputSource = camera;
265 		
266 		// tell root to process queued inputs
267 		camera.getRoot().processInputs();
268 	}
269 	
270 	private void dispatchEventToListener(PInputEvent event, int type, PInputEventListener listener) {
271 		if (listener != null) {
272 			// clear the handled bit since the same event object is used to send multiple events such as
273 			// mouseEntered/mouseExited and mouseMove.
274 			event.setHandled(false); 
275 			listener.processEvent(event, type);
276 		}		
277 	}
