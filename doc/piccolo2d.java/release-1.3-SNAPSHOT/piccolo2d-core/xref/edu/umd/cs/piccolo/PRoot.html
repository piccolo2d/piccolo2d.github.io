57       * The property name that identifies a change in the set of this root's
58       * input sources (see {@link InputSource InputSource}). In any property
59       * change event the new value will be a reference to the list of this root's
60       * input sources, but old value will always be null.

63      public static final int PROPERTY_CODE_INPUT_SOURCES = 1 << 14;

115     /**
116      * Wait for all scheduled activities to finish before returning from this
117      * method. This will freeze out user input, and so it is generally
118      * recommended that you use PActivities.setTriggerTime() to offset
119      * activities instead of using this method.
120      */
121     public void waitForActivities() {
122         PNodeFilter cameraWithCanvas = new PNodeFilter() {
123             public boolean accept(PNode aNode) {
124                 return (aNode instanceof PCamera) && (((PCamera) aNode).getComponent() != null);
125             }
126 
127             public boolean acceptChildrenOf(PNode aNode) {
128                 return true;
129             }
130         };
131 
132         while (activityScheduler.getActivitiesReference().size() > 0) {
133             processInputs();
134             Iterator i = getAllNodes(cameraWithCanvas, null).iterator();
135             while (i.hasNext()) {
136                 PCamera each = (PCamera) i.next();
137                 each.getComponent().paintImmediately();
138             }
139         }
140     }

199     /**
200      * Get the global Piccolo time. This is set to System.currentTimeMillis() at

202      * should usually use this global time instead of System.

206         return globalTime;
207     }

209     /**
210      * This is the heartbeat of the Piccolo framework. Pending input events are
211      * processed. Activities are given a chance to run, and the bounds caches
212      * and any paint damage is validated.
213      */
214     public void processInputs() {
215         PDebug.startProcessingInput();
216         processingInputs = true;
217 
218         globalTime = System.currentTimeMillis();
219         int count = inputSources == null ? 0 : inputSources.size();
220         for (int i = 0; i < count; i++) {
221             InputSource each = (InputSource) inputSources.get(i);
222             each.processInput();
223         }
224 
225         activityScheduler.processActivities(globalTime);
226         validateFullBounds();
227         validateFullPaint();
228 
229         processingInputs = false;
230         PDebug.endProcessingInput();
231     }

253     public void scheduleProcessInputsIfNeeded() {
254         /*
255          * The reason for the special case here (when not in the event dispatch
256          * thread) is that the SwingUtilitiles.invokeLater code below only
257          * invokes later with respect to the event dispatch thread, it will
258          * invoke concurrently with other threads.
259          */
260         if (!SwingUtilities.isEventDispatchThread()) {
261             /*
262              * Piccolo is not thread safe and should amost always be called from
263              * the Swing event dispatch thread. It should only reach this point
264              * when a new canvas is being created.
265              */
266             return;
267         }
268 
269         PDebug.scheduleProcessInputs();
270 
271         if (!processInputsScheduled && !processingInputs
272                 && (getFullBoundsInvalid() || getChildBoundsInvalid() || getPaintInvalid() || getChildPaintInvalid())) {
273 
274             processInputsScheduled = true;
275             SwingUtilities.invokeLater(new Runnable() {
276                 public void run() {
277                     processInputs();
278                     PRoot.this.processInputsScheduled = false;
279                 }
280             });
281         }
