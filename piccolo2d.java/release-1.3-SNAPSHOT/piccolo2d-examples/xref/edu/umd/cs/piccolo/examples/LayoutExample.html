23  	}

25  	public void initialize() {		
26  		// Create a new node and override its validateLayoutAfterChildren method so
27  		// that it lays out its children in a row from left to
28  		// right. 
29  		
30  		final PNode layoutNode = new PNode() {
31  			public void layoutChildren() {				
32  				double xOffset = 0;
33  				double yOffset = 0;
34  
35  				Iterator i = getChildrenIterator(); 							
36  				while (i.hasNext()) {
37  					PNode each = (PNode) i.next();
38  					each.setOffset(xOffset - each.getX(), yOffset);
39  					xOffset += each.getWidth();
40  				}
41  			}
42  		};
43  		
44  		layoutNode.setPaint(Color.red);
45  
46  		// add some children to the layout node.
47  		for (int i = 0; i < 1000; i++) {
48  			// create child to add to the layout node.
49  			PNode each = PPath.createRectangle(0, 0, 100, 80);
50  
51  			// add the child to the layout node.		
52  			layoutNode.addChild(each);
53  		}
54  	
55  		PBoundsHandle.addBoundsHandlesTo(layoutNode.getChild(0));
56  		
57  		// add layoutNode to the root so it will be displayed.
58  		getCanvas().getLayer().addChild(layoutNode);
59  	}
