217 	private static final TextLayout[] EMPTY_TEXT_LAYOUT_ARRAY = new TextLayout[0];

219 	/**
220 	 * Compute the bounds of the text wrapped by this node. The text layout
221 	 * is wrapped based on the bounds of this node.
222 	 */
223 	public void recomputeLayout() {
224 		ArrayList linesList = new ArrayList();
225 		double textWidth = 0;
226 		double textHeight = 0;
227 
228 		if (text != null && text.length() > 0) {		
229 			AttributedString atString = new AttributedString(text);
230 			atString.addAttribute(TextAttribute.FONT, getFont());
231 			AttributedCharacterIterator itr = atString.getIterator();
232 			LineBreakMeasurer measurer = new LineBreakMeasurer(itr, PPaintContext.RENDER_QUALITY_HIGH_FRC);
233 			float availableWidth = constrainWidthToTextWidth ? Float.MAX_VALUE : (float) getWidth();
234 			
235 			int nextLineBreakOffset = text.indexOf('\n');
236 			if (nextLineBreakOffset == -1) {
237 				nextLineBreakOffset = Integer.MAX_VALUE;
238 			} else {
239 				nextLineBreakOffset++;
240 			}
241 			
242 			while (measurer.getPosition() < itr.getEndIndex()) {
243 				TextLayout aTextLayout = computeNextLayout(measurer, availableWidth, nextLineBreakOffset);
244 
245 				if (nextLineBreakOffset == measurer.getPosition()) {
246 					nextLineBreakOffset = text.indexOf('\n', measurer.getPosition());
247 					if (nextLineBreakOffset == -1) {
248 						nextLineBreakOffset = Integer.MAX_VALUE;
249 					} else {
250 						nextLineBreakOffset++;
251 					}
252 				}
253 								
254 				linesList.add(aTextLayout);
255 				textHeight += aTextLayout.getAscent();
256 				textHeight += aTextLayout.getDescent() + aTextLayout.getLeading();
257 				textWidth = Math.max(textWidth, aTextLayout.getAdvance());
258 			}
259 		}
260 				
261 		lines = (TextLayout[]) linesList.toArray(EMPTY_TEXT_LAYOUT_ARRAY);
262 					
263 		if (constrainWidthToTextWidth || constrainHeightToTextHeight) {
264 			double newWidth = getWidth();
265 			double newHeight = getHeight();
266 			
267 			if (constrainWidthToTextWidth) {
268 				newWidth = textWidth;
269 			}
270 			
271 			if (constrainHeightToTextHeight) {
272 				newHeight = textHeight;
273 			}
274 	
275 			super.setBounds(getX(), getY(), newWidth, newHeight);
276 		}	
277 	}

280 	protected TextLayout computeNextLayout(LineBreakMeasurer measurer, float availibleWidth, int nextLineBreakOffset) {
281 		return measurer.nextLayout(availibleWidth, nextLineBreakOffset, false);

284 	protected void paint(PPaintContext paintContext) {		
285 		super.paint(paintContext);
286 		
287 		float screenFontSize = getFont().getSize() * (float) paintContext.getScale();
288 		if (textPaint != null && screenFontSize > greekThreshold) {
289 			float x = (float) getX();
290 			float y = (float) getY();
291 			float bottomY = (float) getHeight() + y;
292 			
293 			Graphics2D g2 = paintContext.getGraphics();
294 			
295 			if (lines == null) {
296 				recomputeLayout();
297 				repaint();
298 				return;
299 			}
300 
301 			g2.setPaint(textPaint);
302 			
303 			for (int i = 0; i < lines.length; i++) {
304                 TextLayout tl = lines[i];
305 				y += tl.getAscent();
306 				
307 				if (bottomY < y) {
308 					return;
309 				}
310                 
311                 float offset = (float) (getWidth() - tl.getAdvance()) * justification;
312                 tl.draw(g2, x + offset, y);
313 	
314 				y += tl.getDescent() + tl.getLeading();
315 			}
316 		}
317 	}
