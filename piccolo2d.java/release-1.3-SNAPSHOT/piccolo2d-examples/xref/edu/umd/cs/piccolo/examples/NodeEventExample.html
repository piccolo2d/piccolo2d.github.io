13  /**
14   * This example shows how to make a node handle events.
15   */
16  public class NodeEventExample extends PFrame {
17  
18  	public NodeEventExample() {
19  		this(null);
20  	}
21  
22  	public NodeEventExample(PCanvas aCanvas) {
23  		super("NodeEventExample", false, aCanvas);
24  	}
25  	
26  	public void initialize() {		
27  		PLayer layer = getCanvas().getLayer();
28  		
29  		// create a new node and override some of the event handling
30  		// methods so that the node changes to orange when the mouse (Button 1) is
31  		// pressed on the node, and changes back to green when the mouse
32  		// is released. Also when the mouse is dragged the node updates its
33  		// position so that the node is "dragged". Note that this only serves
34  		// as a simple example, most of the time dragging nodes is best done
35  		// with the PDragEventHandler, but this shows another way to do it.
36  		//
37  		// Note that each of these methods marks the event as handled. This is so that
38  		// when the node is being dragged the zoom and pan event handles
39  		// (that are installed by default) do not also operate, but they will
40  		// still respond to events that are not handled by the node. (try to uncomment
41  		// the aEvent.setHandled() calls and see what happens.
42  		final PNode aNode = new PNode();
43  		aNode.addInputEventListener(new PBasicInputEventHandler() { 		
44  			public void mousePressed(PInputEvent aEvent) {
45  				aNode.setPaint(Color.orange);
46  				printEventCoords(aEvent);
47  				aEvent.setHandled(true);
48  			}
49  			
50  			public void mouseDragged(PInputEvent aEvent) {
51  				Dimension2D delta = aEvent.getDeltaRelativeTo(aNode);
52  				aNode.translate(delta.getWidth(), delta.getHeight());
53  				printEventCoords(aEvent);
54  				aEvent.setHandled(true);
55  			}
56  			
57  			public void mouseReleased(PInputEvent aEvent) {
58  				aNode.setPaint(Color.green);
59  				printEventCoords(aEvent);
60  				aEvent.setHandled(true);
61  			}
62  			
63  			// Note this slows things down a lot, comment it out to see how the normal
64  			// speed of things is.
65  			// 
66  			// For fun the coords of each event that the node handles are printed out.
67  			// This can help to understand how coordinate systems work. Notice that when
68  			// the example first starts all the values for (canvas, global, and local) are
69  			// equal. But once you drag the node then the local coordinates become different
70  			// then the screen and global coordinates. When you pan or zoom then the screen
71  			// coordinates become different from the global coordinates.
72  			public void printEventCoords(PInputEvent aEvent) {
73  				System.out.println("Canvas Location: " + aEvent.getCanvasPosition());
74  				//System.out.println("Global Location: " + aEvent.getGlobalLocation());
75  				System.out.println("Local Location: " + aEvent.getPositionRelativeTo(aNode));
76  				System.out.println("Canvas Delta: " + aEvent.getCanvasDelta());
77  				//System.out.println("Global Delta: " + aEvent.getGlobalDelta());
78  				System.out.println("Local Delta: " + aEvent.getDeltaRelativeTo(aNode));
79  			}
80  		});
81  		aNode.setBounds(0, 0, 200, 200);
82  		aNode.setPaint(Color.green);
83  
84  		// By default the filter accepts all events, but here we constrain the kinds of
85  		// events that aNode receives to button 1 events. Comment this line out and then
86  		// you will be able to drag the node with any mouse button.
87  		//aNode.setEventFilter(new PInputEventFilter(InputEvent.BUTTON1_MASK));
88  
89  		// add another node to the canvas that does not handle events as a reference
90  		// point, so that we can make sure that our green node is getting dragged.				
91  		layer.addChild(PPath.createRectangle(0, 0, 100, 80));
92  		layer.addChild(aNode);		
93  	}
94  
95  	public static void main(String[] args) {
96  		new NodeEventExample();
97  	}
