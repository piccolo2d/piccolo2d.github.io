22  	public void initialize() {
23  		PLayer layer = getCanvas().getLayer();
24  		final PNode animatedNode = PPath.createRectangle(0, 0, 100, 80);
25  		layer.addChild(animatedNode);
26  		
27  		// create animation path
28  		GeneralPath path = new GeneralPath();
29  		path.moveTo(0, 0);
30  		path.lineTo(300, 300);
31  		path.lineTo(300, 0);
32  		path.append(new Arc2D.Float(0, 0, 300, 300, 90, -90, Arc2D.OPEN), true);
33  		path.closePath();
34  
35  		// create node to display animation path
36  		PPath ppath = new PPath(path);
37  		layer.addChild(ppath);
38  
39  		// create activity to run animation.
40  		PPositionPathActivity positionPathActivity = new PPositionPathActivity(5000, 0, new PPositionPathActivity.Target() {
41  			public void setPosition(double x, double y) {
42  				animatedNode.setOffset(x, y);
43  			}
44  		});
45  //		positionPathActivity.setSlowInSlowOut(false);
46  		positionPathActivity.setPositions(path);
47  		positionPathActivity.setLoopCount(Integer.MAX_VALUE);
48  		
49  		// add the activity.
50  		animatedNode.addActivity(positionPathActivity);
51  	}
