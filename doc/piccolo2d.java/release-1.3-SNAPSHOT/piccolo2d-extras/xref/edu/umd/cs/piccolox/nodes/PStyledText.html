134     public void syncWithDocument() {
135         // The paragraph start and end indices
136         ArrayList pEnds = null;
137 
138         // The current position in the specified range
139         int pos = 0;
140 
141         // First get the actual text and stick it in an Attributed String
142         try {
143 
144             stringContents = new ArrayList();
145             pEnds = new ArrayList();
146 
147             String s = document.getText(0, document.getLength());
148             StringTokenizer tokenizer = new StringTokenizer(s, "\n", true);
149 
150             // lastNewLine is used to detect the case when two newlines follow
151             // in direct succession
152             // & lastNewLine should be true to start in case the first character
153             // is a newline
154             boolean lastNewLine = true;
155             for (int i = 0; tokenizer.hasMoreTokens(); i++) {
156                 String token = tokenizer.nextToken();
157 
158                 // If the token
159                 if (token.equals("\n")) {
160                     if (lastNewLine) {
161                         stringContents.add(new AttributedString(" "));
162                         pEnds.add(new RunInfo(pos, pos + 1));
163 
164                         pos = pos + 1;
165 
166                         lastNewLine = true;
167                     }
168                     else {
169                         pos = pos + 1;
170 
171                         lastNewLine = true;
172                     }
173                 }
174                 // If the token is empty - create an attributed string with a
175                 // single space
176                 // since LineBreakMeasurers don't work with an empty string
177                 // - note that this case should only arise if the document is
178                 // empty
179                 else if (token.equals("")) {
180                     stringContents.add(new AttributedString(" "));
181                     pEnds.add(new RunInfo(pos, pos));
182 
183                     lastNewLine = false;
184                 }
185                 // This is the normal case - where we have some text
186                 else {
187                     stringContents.add(new AttributedString(token));
188                     pEnds.add(new RunInfo(pos, pos + token.length()));
189 
190                     // Increment the position
191                     pos = pos + token.length();
192 
193                     lastNewLine = false;
194                 }
195             }
196 
197             // Add one more newline if the last character was a newline
198             if (lastNewLine) {
199                 stringContents.add(new AttributedString(" "));
200                 pEnds.add(new RunInfo(pos, pos + 1));
201 
202                 lastNewLine = false;
203             }
204         }
205         catch (Exception e) {
206             e.printStackTrace();
207         }
208 
209         // The default style context - which will be reused
210         StyleContext style = StyleContext.getDefaultStyleContext();
211 
212         RunInfo pEnd = null;
213         for (int i = 0; i < stringContents.size(); i++) {
214             pEnd = (RunInfo) pEnds.get(i);
215             pos = pEnd.runStart;
216 
217             // The current element will be used as a temp variable while
218             // searching
219             // for the leaf element at the current position
220             Element curElement = null;
221 
222             // Small assumption here that there is one root element - can fix
223             // for more general support later
224             Element rootElement = document.getDefaultRootElement();
225 
226             // If the string is length 0 then we just need to add the attributes
227             // once
228             if (pEnd.runStart != pEnd.runLimit) {
229                 // OK, now we loop until we find all the leaf elements in the
230                 // range
231                 while (pos < pEnd.runLimit) {
232 
233                     // Before each pass, start at the root
234                     curElement = rootElement;
235 
236                     // Now we descend the hierarchy until we get to a leaf
237                     while (!curElement.isLeaf()) {
238                         curElement = curElement.getElement(curElement.getElementIndex(pos));
239                     }
240 
241                     // These are the mandatory attributes
242 
243                     AttributeSet attributes = curElement.getAttributes();
244                     Color foreground = style.getForeground(attributes);
245 
246                     ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.FOREGROUND, foreground,
247                             (int) Math.max(0, curElement.getStartOffset() - pEnd.runStart), (int) Math.min(
248                                     pEnd.runLimit - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
249 
250                     Font font = (attributes.isDefined(StyleConstants.FontSize) || attributes
251                             .isDefined(StyleConstants.FontFamily)) ? style.getFont(attributes) : null;
252                     if (font == null) {
253                         if (document instanceof DefaultStyledDocument) {
254                             font = style.getFont(((DefaultStyledDocument) document).getCharacterElement(pos)
255                                     .getAttributes());
256                             if (font == null) {
257                                 font = style.getFont(((DefaultStyledDocument) document).getParagraphElement(pos)
258                                         .getAttributes());
259                             }
260                             if (font == null) {
261                                 font = style.getFont(rootElement.getAttributes());
262                             }
263                         }
264                         else {
265                             font = style.getFont(rootElement.getAttributes());
266                         }
267                     }
268                     if (font != null) {
269                         ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.FONT, font, (int) Math
270                                 .max(0, curElement.getStartOffset() - pEnd.runStart), (int) Math.min(pEnd.runLimit
271                                 - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
272                     }
273 
274                     // These are the optional attributes
275 
276                     Color background = (attributes.isDefined(StyleConstants.Background)) ? style
277                             .getBackground(attributes) : null;
278                     if (background != null) {
279                         ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.BACKGROUND, background,
280                                 (int) Math.max(0, curElement.getStartOffset() - pEnd.runStart), (int) Math.min(
281                                         pEnd.runLimit - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
282                     }
283 
284                     boolean underline = StyleConstants.isUnderline(attributes);
285                     if (underline) {
286                         ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.UNDERLINE, Boolean.TRUE,
287                                 (int) Math.max(0, curElement.getStartOffset() - pEnd.runStart), (int) Math.min(
288                                         pEnd.runLimit - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
289                     }
290 
291                     boolean strikethrough = StyleConstants.isStrikeThrough(attributes);
292                     if (strikethrough) {
293                         ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.STRIKETHROUGH,
294                                 Boolean.TRUE, (int) Math.max(0, curElement.getStartOffset() - pEnd.runStart),
295                                 (int) Math
296                                         .min(pEnd.runLimit - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
297                     }
298 
299                     // And set the position to the end of the given attribute
300                     pos = curElement.getEndOffset();
301                 }
302             }
303             else {
304                 // Before each pass, start at the root
305                 curElement = rootElement;
306 
307                 // Now we descend the hierarchy until we get to a leaf
308                 while (!curElement.isLeaf()) {
309                     curElement = curElement.getElement(curElement.getElementIndex(pos));
310                 }
311 
312                 // These are the mandatory attributes
313 
314                 AttributeSet attributes = curElement.getAttributes();
315                 Color foreground = style.getForeground(attributes);
316 
317                 ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.FOREGROUND, foreground,
318                         (int) Math.max(0, curElement.getStartOffset() - pEnd.runStart), (int) Math.min(pEnd.runLimit
319                                 - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
320 
321                 // These are the optional attributes
322 
323                 Font font = (attributes.isDefined(StyleConstants.FontSize) || attributes
324                         .isDefined(StyleConstants.FontFamily)) ? style.getFont(attributes) : null;
325                 if (font == null) {
326                     if (document instanceof DefaultStyledDocument) {
327                         font = style.getFont(((DefaultStyledDocument) document).getCharacterElement(pos)
328                                 .getAttributes());
329                         if (font == null) {
330                             font = style.getFont(((DefaultStyledDocument) document).getParagraphElement(pos)
331                                     .getAttributes());
332                         }
333                         if (font == null) {
334                             font = style.getFont(rootElement.getAttributes());
335                         }
336                     }
337                     else {
338                         font = style.getFont(rootElement.getAttributes());
339                     }
340                 }
341                 if (font != null) {
342                     ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.FONT, font, (int) Math.max(0,
343                             curElement.getStartOffset() - pEnd.runStart), (int) Math.min(pEnd.runLimit - pEnd.runStart,
344                             curElement.getEndOffset() - pEnd.runStart));
345                 }
346 
347                 Color background = (attributes.isDefined(StyleConstants.Background)) ? style.getBackground(attributes)
348                         : null;
349                 if (background != null) {
350                     ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.BACKGROUND, background,
351                             (int) Math.max(0, curElement.getStartOffset() - pEnd.runStart), (int) Math.min(
352                                     pEnd.runLimit - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
353                 }
354 
355                 boolean underline = StyleConstants.isUnderline(attributes);
356                 if (underline) {
357                     ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.UNDERLINE, Boolean.TRUE,
358                             (int) Math.max(0, curElement.getStartOffset() - pEnd.runStart), (int) Math.min(
359                                     pEnd.runLimit - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
360                 }
361 
362                 boolean strikethrough = StyleConstants.isStrikeThrough(attributes);
363                 if (strikethrough) {
364                     ((AttributedString) stringContents.get(i)).addAttribute(TextAttribute.STRIKETHROUGH, Boolean.TRUE,
365                             (int) Math.max(0, curElement.getStartOffset() - pEnd.runStart), (int) Math.min(
366                                     pEnd.runLimit - pEnd.runStart, curElement.getEndOffset() - pEnd.runStart));
367                 }
368             }
369         }
370 
371         recomputeLayout();
372     }

374     /**
375      * Compute the bounds of the text wrapped by this node. The text layout is
376      * wrapped based on the bounds of this node. If the shrinkBoundsToFit
377      * parameter is true then after the text has been laid out the bounds of
378      * this node are shrunk to fit around those text bounds.
379      */
380     public void recomputeLayout() {
381         if (stringContents == null)
382             return;
383 
384         ArrayList linesList = new ArrayList();
385 
386         double textWidth = 0;
387         double textHeight = 0;
388 
389         for (int i = 0; i < stringContents.size(); i++) {
390 
391             AttributedString ats = (AttributedString) stringContents.get(i);
392             AttributedCharacterIterator itr = ats.getIterator();
393 
394             LineBreakMeasurer measurer;
395             ArrayList breakList = null;
396 
397             // First we have to do an initial pass with a LineBreakMeasurer to
398             // find out where Swing is going to break the lines - i.e.
399             // because it doesn't use fractional metrics
400 
401             measurer = new LineBreakMeasurer(itr, SWING_FRC);
402             breakList = new ArrayList();
403             while (measurer.getPosition() < itr.getEndIndex()) {
404                 if (constrainWidthToTextWidth) {
405                     measurer.nextLayout(Float.MAX_VALUE);
406                 }
407                 else {
408                     measurer.nextLayout((float) Math.ceil(getWidth() - insets.left - insets.right));
409                 }
410 
411                 breakList.add(new Integer(measurer.getPosition()));
412             }
413 
414             measurer = new LineBreakMeasurer(itr, PPaintContext.RENDER_QUALITY_HIGH_FRC);
415 
416             // Need to change the lineinfo data structure to know about multiple
417             // text layouts per line
418 
419             LineInfo lineInfo = null;
420             boolean newLine = true;
421             double lineWidth = 0;
422             while (measurer.getPosition() < itr.getEndIndex()) {
423                 TextLayout aTextLayout = null;
424 
425                 if (newLine) {
426                     newLine = false;
427 
428                     // Add in the old line dimensions
429                     double lineHeight = (lineInfo == null) ? 0 : lineInfo.maxAscent + lineInfo.maxDescent
430                             + lineInfo.leading;
431                     textHeight = textHeight + lineHeight;
432                     textWidth = Math.max(textWidth, lineWidth);
433 
434                     // Now create a new line
435                     lineInfo = new LineInfo();
436                     linesList.add(lineInfo);
437                 }
438 
439                 int lineEnd = ((Integer) breakList.get(0)).intValue();
440                 if (lineEnd <= itr.getRunLimit()) {
441                     breakList.remove(0);
442                     newLine = true;
443                 }
444 
445                 aTextLayout = measurer.nextLayout(Float.MAX_VALUE, Math.min(lineEnd, itr.getRunLimit()), false);
446 
447                 SegmentInfo sInfo = new SegmentInfo();
448                 sInfo.font = (Font) itr.getAttribute(TextAttribute.FONT);
449                 sInfo.foreground = (Color) itr.getAttribute(TextAttribute.FOREGROUND);
450                 sInfo.background = (Color) itr.getAttribute(TextAttribute.BACKGROUND);
451                 sInfo.underline = (Boolean) itr.getAttribute(TextAttribute.UNDERLINE);
452                 sInfo.layout = aTextLayout;
453 
454                 FontMetrics metrics = StyleContext.getDefaultStyleContext().getFontMetrics(
455                         (Font) itr.getAttribute(TextAttribute.FONT));
456                 lineInfo.maxAscent = Math.max(lineInfo.maxAscent, metrics.getMaxAscent());
457                 lineInfo.maxDescent = Math.max(lineInfo.maxDescent, metrics.getMaxDescent());
458                 lineInfo.leading = Math.max(lineInfo.leading, metrics.getLeading());
459 
460                 lineInfo.segments.add(sInfo);
461 
462                 itr.setIndex(measurer.getPosition());
463                 lineWidth = lineWidth + aTextLayout.getAdvance();
464             }
465 
466             double lineHeight = (lineInfo == null) ? 0 : lineInfo.maxAscent + lineInfo.maxDescent + lineInfo.leading;
467             textHeight = textHeight + lineHeight;
468             textWidth = Math.max(textWidth, lineWidth);
469         }
470 
471         lines = (LineInfo[]) linesList.toArray(new LineInfo[0]);
472 
473         if (constrainWidthToTextWidth || constrainHeightToTextHeight) {
474             double newWidth = getWidth();
475             double newHeight = getHeight();
476 
477             if (constrainWidthToTextWidth) {
478                 newWidth = textWidth + insets.left + insets.right;
479             }
480 
481             if (constrainHeightToTextHeight) {
482                 newHeight = Math.max(textHeight, getInitialFontHeight()) + insets.top + insets.bottom;
483             }
484 
485             super.setBounds(getX(), getY(), newWidth, newHeight);
486         }
487     }

489     /**
490      * Get the height of the font at the beginning of the document
491      */
492     public double getInitialFontHeight() {
493 
494         // Small assumption here that there is one root element - can fix
495         // for more general support later
496         Element rootElement = document.getDefaultRootElement();
497 
498         // The current element will be used as a temp variable while searching
499         // for the leaf element at the current position
500         Element curElement = rootElement;
501 
502         // Now we descend the hierarchy until we get to a leaf
503         while (!curElement.isLeaf()) {
504             curElement = curElement.getElement(curElement.getElementIndex(0));
505         }
506 
507         StyleContext context = StyleContext.getDefaultStyleContext();
508         Font font = context.getFont(curElement.getAttributes());
509 
510         FontMetrics curFM = context.getFontMetrics(font);
511 
512         return curFM.getMaxAscent() + curFM.getMaxDescent() + curFM.getLeading();
513     }

515     protected void paint(PPaintContext paintContext) {
516         float x = (float) (getX() + insets.left);
517         float y = (float) (getY() + insets.top);
518         float bottomY = (float) (getY() + getHeight() - insets.bottom);
519 
520         if (lines == null || lines.length == 0) {
521             return;
522         }
523 
524         Graphics2D g2 = paintContext.getGraphics();
525 
526         if (getPaint() != null) {
527             g2.setPaint(getPaint());
528             g2.fill(getBoundsReference());
529         }
530 
531         float curX;
532         for (int i = 0; i < lines.length; i++) {
533             y += lines[i].maxAscent;
534             curX = x;
535 
536             if (bottomY < y) {
537                 return;
538             }
539 
540             for (int j = 0; j < lines[i].segments.size(); j++) {
541                 SegmentInfo sInfo = (SegmentInfo) lines[i].segments.get(j);
542                 float width = sInfo.layout.getAdvance();
543 
544                 if (sInfo.background != null) {
545                     g2.setPaint(sInfo.background);
546                     g2.fill(new Rectangle2D.Double(curX, y - lines[i].maxAscent, width, lines[i].maxAscent
547                             + lines[i].maxDescent + lines[i].leading));
548                 }
549 
550                 if (sInfo.font != null) {
551                     g2.setFont(sInfo.font);
552                 }
553 
554                 // Manually set the paint - this is specified in the
555                 // AttributedString but seems to be
556                 // ignored by the TextLayout. To handle multiple colors we
557                 // should be breaking up the lines
558                 // but that functionality can be added later as needed
559                 g2.setPaint(sInfo.foreground);
560                 sInfo.layout.draw(g2, curX, y);
561 
562                 // Draw the underline and the strikethrough after the text
563                 if (sInfo.underline != null) {
564                     paintLine.setLine(x, y + lines[i].maxDescent / 2, x + width, y + lines[i].maxDescent / 2);
565                     g2.draw(paintLine);
566                 }
567 
568                 curX = curX + width;
569             }
570 
571             y += lines[i].maxDescent + lines[i].leading;
572         }
573     }
