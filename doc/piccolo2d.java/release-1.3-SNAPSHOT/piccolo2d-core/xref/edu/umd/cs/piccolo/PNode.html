258     // so if you need more control over your animation look at the
259     // activities package. Each animate method creates an animation that
260     // will animate the node from its current state to the new state
261     // specified over the given duration. These methods will try to

267     /**
268      * Animate this node's bounds from their current location when the activity
269      * starts to the specified bounds. If this node descends from the root then
270      * the activity will be scheduled, else the returned activity should be
271      * scheduled manually. If two different transform activities are scheduled
272      * for the same node at the same time, they will both be applied to the
273      * node, but the last one scheduled will be applied last on each frame, so
274      * it will appear to have replaced the original. Generally you will not want
275      * to do that. Note this method animates the node's bounds, but does not
276      * change the node's transform. Use animateTransformToBounds() to animate
277      * the node's transform instead.
278      * 
279      * @param duration amount of time that the animation should take
280      * @return the newly scheduled activity
281      */
282     public PInterpolatingActivity animateToBounds(double x, double y, double width, double height, long duration) {
283         if (duration == 0) {
284             setBounds(x, y, width, height);
285             return null;
286         }
287         else {
288             final PBounds dst = new PBounds(x, y, width, height);
289 
290             PInterpolatingActivity ta = new PInterpolatingActivity(duration, PUtil.DEFAULT_ACTIVITY_STEP_RATE) {
291                 private PBounds src;
292 
293                 protected void activityStarted() {
294                     src = getBounds();
295                     startResizeBounds();
296                     super.activityStarted();
297                 }
298 
299                 public void setRelativeTargetValue(float zeroToOne) {
300                     PNode.this.setBounds(src.x + (zeroToOne * (dst.x - src.x)), src.y + (zeroToOne * (dst.y - src.y)),
301                             src.width + (zeroToOne * (dst.width - src.width)), src.height
302                                     + (zeroToOne * (dst.height - src.height)));
303                 }
304 
305                 protected void activityFinished() {
306                     super.activityFinished();
307                     endResizeBounds();
308                 }
309             };
310 
311             addActivity(ta);
312             return ta;
313         }
314     }

316     /**
317      * Animate this node from it's current transform when the activity starts a
318      * new transform that will fit the node into the given bounds. If this node
319      * descends from the root then the activity will be scheduled, else the
320      * returned activity should be scheduled manually. If two different
321      * transform activities are scheduled for the same node at the same time,
322      * they will both be applied to the node, but the last one scheduled will be
323      * applied last on each frame, so it will appear to have replaced the
324      * original. Generally you will not want to do that. Note this method
325      * animates the node's transform, but does not directly change the node's
326      * bounds rectangle. Use animateToBounds() to animate the node's bounds
327      * rectangle instead.
328      * 
329      * @param duration amount of time that the animation should take
330      * @return the newly scheduled activity
331      */
332     public PTransformActivity animateTransformToBounds(double x, double y, double width, double height, long duration) {
333         PAffineTransform t = new PAffineTransform();
334         t.setToScale(width / getWidth(), height / getHeight());
335         double scale = t.getScale();
336         t.setOffset(x - (getX() * scale), y - (getY() * scale));
337         return animateToTransform(t, duration);
338     }

527     /**
528      * Add an arbitrary key/value to this node.
529      * <p>
530      * The <code>get/add attribute<code> methods provide access to
531      * a small per-instance attribute set. Callers can use get/add attribute
532      * to annotate nodes that were created by another module.
533      * <p>
534      * If value is null this method will remove the attribute.
535      */
536     public void addAttribute(Object key, Object value) {
537         if (value == null && clientProperties == null)
538             return;
539 
540         Object oldValue = getAttribute(key);
541 
542         if (value != oldValue) {
543             if (clientProperties == null) {
544                 clientProperties = new SimpleAttributeSet();
545             }
546 
547             if (value == null) {
548                 clientProperties.removeAttribute(key);
549             }
550             else {
551                 clientProperties.addAttribute(key, value);
552             }
553 
554             if (clientProperties.getAttributeCount() == 0 && clientProperties.getResolveParent() == null) {
555                 clientProperties = null;
556             }
557 
558             firePropertyChange(PROPERTY_CODE_CLIENT_PROPERTIES, PROPERTY_CLIENT_PROPERTIES, null, clientProperties);
559             firePropertyChange(PROPERTY_CODE_CLIENT_PROPERTIES, key.toString(), oldValue, value);
560         }
561     }

660     // ****************************************************************
661     // Coordinate System Conversions - Methods for converting
662     // geometry between this nodes local coordinates and the other
663     // major coordinate systems.
664     //
665     // Each nodes has an affine transform that it uses to define its
666     // own coordinate system. For example if you create a new node and
667     // add it to the canvas it will appear in the upper right corner. Its
668     // coordinate system matches the coordinate system of its parent
669     // (the root node) at this point. But if you move this node by calling
670     // node.translate() the nodes affine transform will be modified and the
671     // node will appear at a different location on the screen. The node
672     // coordinate system no longer matches the coordinate system of its
673     // parent.
674     //

1545      * Invalidates the full bounds of this node, and sets the child bounds
1546      * invalid flag on each of this node's ancestors.
1547      */
1548     public void invalidateFullBounds() {
1549         setFullBoundsInvalid(true);
1550 
1551         PNode n = parent;
1552         while (n != null && !n.getChildBoundsInvalid()) {
1553             n.setChildBoundsInvalid(true);
1554             n = n.parent;
1555         }
1556 
1557         if (SCENE_GRAPH_DELEGATE != null)
1558             SCENE_GRAPH_DELEGATE.nodeFullBoundsInvalidated(this);

1561     /**
1562      * This method is called to validate the bounds of this node and all of its
1563      * descendents. It returns true if this nodes bounds or the bounds of any of
1564      * its descendents are marked as volatile.
1565      * 
1566      * @return true if this node or any of its descendents have volatile bounds
1567      */
1568     protected boolean validateFullBounds() {
1569         boolean boundsVolatile = getBoundsVolatile();
1570 
1571         // 1. Only compute new bounds if invalid flags are set.
1572         if (fullBoundsInvalid || childBoundsInvalid || boundsVolatile || childBoundsVolatile) {
1573 
1574             // 2. If my bounds are volatile and they have not been changed then
1575             // signal a change.
1576             //
1577             // For most cases this will do nothing, but if a nodes bounds depend
1578             // on its model, then
1579             // validate bounds has the responsibility of making the bounds match
1580             // the models value.
1581             // For example PPaths validateBounds method makes sure that the
1582             // bounds are equal to the
1583             // bounds of the GeneralPath model.
1584             if (boundsVolatile && !boundsChanged) {
1585                 signalBoundsChanged();
1586             }
1587 
1588             // 3. If the bounds of on of my decendents are invalidate then
1589             // validate the bounds of all of my children.
1590             if (childBoundsInvalid || childBoundsVolatile) {
1591                 childBoundsVolatile = false;
1592                 int count = getChildrenCount();
1593                 for (int i = 0; i < count; i++) {
1594                     PNode each = (PNode) children.get(i);
1595                     childBoundsVolatile |= each.validateFullBounds();
1596                 }
1597             }
1598 
1599             // 4. Now that my children's bounds are valid and my own bounds are
1600             // valid run any layout algorithm here. Note that if you try to
1601             // layout volatile
1602             // children piccolo will most likely start a "soft" infinite loop.
1603             // It won't freeze
1604             // your program, but it will make an infinite number of calls to
1605             // SwingUtilities
1606             // invoke later. You don't want to do that.
1607             layoutChildren();
1608 
1609             // 5. If the full bounds cache is invalid then recompute the full
1610             // bounds cache here after our own bounds and the children's bounds
1611             // have been computed above.
1612             if (fullBoundsInvalid) {
1613                 double oldX = fullBoundsCache.x;
1614                 double oldY = fullBoundsCache.y;
1615                 double oldWidth = fullBoundsCache.width;
1616                 double oldHeight = fullBoundsCache.height;
1617                 boolean oldEmpty = fullBoundsCache.isEmpty();
1618 
1619                 // 6. This will call getFullBoundsReference on all of the
1620                 // children. So if the above
1621                 // layoutChildren method changed the bounds of any of the
1622                 // children they will be
1623                 // validated again here.
1624                 fullBoundsCache = computeFullBounds(fullBoundsCache);
1625 
1626                 boolean fullBoundsChanged = fullBoundsCache.x != oldX || fullBoundsCache.y != oldY
1627                         || fullBoundsCache.width != oldWidth || fullBoundsCache.height != oldHeight
1628                         || fullBoundsCache.isEmpty() != oldEmpty;
1629 
1630                 // 7. If the new full bounds cache differs from the previous
1631                 // cache then
1632                 // tell our parent to invalidate their full bounds. This is how
1633                 // bounds changes
1634                 // deep in the tree percolate up.
1635                 if (fullBoundsChanged) {
1636                     if (parent != null)
1637                         parent.invalidateFullBounds();
1638                     firePropertyChange(PROPERTY_CODE_FULL_BOUNDS, PROPERTY_FULL_BOUNDS, null, fullBoundsCache);
1639 
1640                     // 8. If our paint was invalid make sure to repaint our old
1641                     // full bounds. The
1642                     // new bounds will be computed later in the validatePaint
1643                     // pass.
1644                     if (paintInvalid && !oldEmpty) {
1645                         TEMP_REPAINT_BOUNDS.setRect(oldX, oldY, oldWidth, oldHeight);
1646                         repaintFrom(TEMP_REPAINT_BOUNDS, this);
1647                     }
1648                 }
1649             }
1650 
1651             // 9. Clear the invalid bounds flags.
1652             boundsChanged = false;
1653             fullBoundsInvalid = false;
1654             childBoundsInvalid = false;
1655         }
1656 
1657         return boundsVolatile || childBoundsVolatile;
1658     }

1994     /**
1995      * This will calculate the necessary transform in order to make this node
1996      * appear at a particular position relative to the specified bounding box.
1997      * The source point specifies a point in the unit square (0, 0) - (1, 1)
1998      * that represents an anchor point on the corresponding node to this
1999      * transform. The destination point specifies an anchor point on the
2000      * reference node. The position method then computes the transform that
2001      * results in transforming this node so that the source anchor point
2002      * coincides with the reference anchor point. This can be useful for layout
2003      * algorithms as it is straightforward to position one object relative to
2004      * another.
2005      * <p>
2006      * For example, If you have two nodes, A and B, and you call
2007      * 
2008      * <PRE>
2009      * Point2D srcPt = new Point2D.Double(1.0, 0.0);
2010      * Point2D destPt = new Point2D.Double(0.0, 0.0);
2011      * A.position(srcPt, destPt, B.getGlobalBounds(), 750, null);
2012      * </PRE>
2013      * 
2014      * The result is that A will move so that its upper-right corner is at the
2015      * same place as the upper-left corner of B, and the transition will be
2016      * smoothly animated over a period of 750 milliseconds.
2017      * 
2018      * @param srcPt The anchor point on this transform's node (normalized to a
2019      *            unit square)
2020      * @param destPt The anchor point on destination bounds (normalized to a
2021      *            unit square)
2022      * @param destBounds The bounds (in global coordinates) used to calculate
2023      *            this transform's node
2024      * @param millis Number of milliseconds over which to perform the animation
2025      */
2026     public void position(Point2D srcPt, Point2D destPt, Rectangle2D destBounds, int millis) {
2027         double srcx, srcy;
2028         double destx, desty;
2029         double dx, dy;
2030         Point2D pt1, pt2;
2031 
2032         if (parent != null) {
2033             // First compute translation amount in global coordinates
2034             Rectangle2D srcBounds = getGlobalFullBounds();
2035             srcx = lerp(srcPt.getX(), srcBounds.getX(), srcBounds.getX() + srcBounds.getWidth());
2036             srcy = lerp(srcPt.getY(), srcBounds.getY(), srcBounds.getY() + srcBounds.getHeight());
2037             destx = lerp(destPt.getX(), destBounds.getX(), destBounds.getX() + destBounds.getWidth());
2038             desty = lerp(destPt.getY(), destBounds.getY(), destBounds.getY() + destBounds.getHeight());
2039 
2040             // Convert vector to local coordinates
2041             pt1 = new Point2D.Double(srcx, srcy);
2042             globalToLocal(pt1);
2043             pt2 = new Point2D.Double(destx, desty);
2044             globalToLocal(pt2);
2045             dx = (pt2.getX() - pt1.getX());
2046             dy = (pt2.getY() - pt1.getY());
2047 
2048             // Finally, animate change
2049             PAffineTransform at = new PAffineTransform(getTransformReference(true));
2050             at.translate(dx, dy);
2051             animateToTransform(at, millis);
2052         }
2053     }
