66   * <b>PCanvas</b> is a simple Swing component that can be used to embed
67   * Piccolo into a Java Swing application. Canvas's view the Piccolo scene
68   * graph through a camera. The canvas manages screen updates coming from
69   * this camera, and forwards swing mouse and keyboard events to the camera.
70   * <P>
71   * @version 1.0
72   * @author Jesse Grosjean
73   */

95  	 * root, camera, and layer. Event handlers for zooming and panning
96  	 * are automatically installed.

98  	public PSWTCanvas(Composite parent, int style) {
99  		super(parent,style | SWT.NO_BACKGROUND | SWT.NO_REDRAW_RESIZE);
100 		
101 		CURRENT_CANVAS = this;
102 		cursorStack = new PStack();
103 		setCamera(createBasicSceneGraph());
104 		installInputSources();		
105 		setDefaultRenderQuality(PPaintContext.HIGH_QUALITY_RENDERING);
106 		setAnimatingRenderQuality(PPaintContext.LOW_QUALITY_RENDERING);
107 		setInteractingRenderQuality(PPaintContext.LOW_QUALITY_RENDERING);
108 		panEventHandler = new PPanEventHandler();
109 		zoomEventHandler = new PZoomEventHandler(); 			
110 		addInputEventListener(panEventHandler);
111 		addInputEventListener(zoomEventHandler);
112 		
113 		// Add a paint listener to call paint
114 		addPaintListener(new PaintListener() {
115 			public void paintControl(PaintEvent pe) {
116 				paintComponent(pe.gc,pe.x,pe.y,pe.width,pe.height);
117 			}
118 		});
119 		
120 		// Keep track of the references so we can dispose of the Fonts and Colors
121 		SWTGraphics2D.incrementGCCount();
122 		addDisposeListener(new DisposeListener() {
123 			public void widgetDisposed(DisposeEvent de) {
124 				getRoot().getActivityScheduler().removeAllActivities();
125 				SWTGraphics2D.decrementGCCount();	
126 			}
127 		});
128 	}

306 	 * Set the canvas cursor, and remember the previous cursor on the
307 	 * cursor stack.
308 	 */ 
309 	public void pushCursor(java.awt.Cursor cursor) {
310 	    Cursor aCursor = null;
311 	    if (cursor.getType() == java.awt.Cursor.N_RESIZE_CURSOR) {
312 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZEN);
313 	    }
314 	    else if (cursor.getType() == java.awt.Cursor.NE_RESIZE_CURSOR) {
315 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZENE);
316 	    }
317 	    else if (cursor.getType() == java.awt.Cursor.NW_RESIZE_CURSOR) {
318 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZENW);
319 	    }
320 	    else if (cursor.getType() == java.awt.Cursor.S_RESIZE_CURSOR) {
321 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZES);
322 	    }
323 	    else if (cursor.getType() == java.awt.Cursor.SE_RESIZE_CURSOR) {
324 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZESE);
325 	    }
326 	    else if (cursor.getType() == java.awt.Cursor.SW_RESIZE_CURSOR) {
327 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZESW);
328 	    }
329 	    else if (cursor.getType() == java.awt.Cursor.E_RESIZE_CURSOR) {
330 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZEE);
331 	    }
332 	    else if (cursor.getType() == java.awt.Cursor.W_RESIZE_CURSOR) {
333 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZEW);
334 	    }
335 	    else if (cursor.getType() == java.awt.Cursor.TEXT_CURSOR) {
336 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_IBEAM);
337 	    }
338 	    else if (cursor.getType() == java.awt.Cursor.HAND_CURSOR) {
339 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_HAND);
340 	    }
341 	    else if (cursor.getType() == java.awt.Cursor.MOVE_CURSOR) {
342 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_SIZEALL);
343 	    }
344 	    else if (cursor.getType() == java.awt.Cursor.CROSSHAIR_CURSOR) {
345 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_CROSS);
346 	    }
347 	    else if (cursor.getType() == java.awt.Cursor.WAIT_CURSOR) {
348 	        aCursor = new Cursor(this.getDisplay(),SWT.CURSOR_WAIT);
349 	    }
350 	    
351 	    if (aCursor != null) {
352 	        if (curCursor != null) {
353 	            cursorStack.push(curCursor);
354 	        }
355 	        curCursor = aCursor;
356 	        setCursor(aCursor);
357 	    }
358 	}

360 	/**
361 	 * Pop the cursor on top of the cursorStack and set it as the 
362 	 * canvas cursor.
363 	 */ 
364 	public void popCursor() {
365 	    if (curCursor != null) {
366 	        // We must manually dispose of cursors under SWT
367 	        curCursor.dispose();
368 	    }
369 	    
370 	    if (!cursorStack.isEmpty()) {
371 	        curCursor = (Cursor)cursorStack.pop();
372 	    }
373 	    else {
374 	        curCursor = null;
375 	    }
376 	    
377 	    // This sets the cursor back to default
378 	    setCursor(curCursor);
379 	}

382 	// Code to manage connection to Swing. There appears to be a bug in
383 	// swing where it will occasionally send to many mouse pressed or mouse
384 	// released events. Below we attempt to filter out those cases before
385 	// they get delivered to the Piccolo framework.
386 	//****************************************************************	
387 	
388 	private boolean isButton1Pressed;
389 	private boolean isButton2Pressed;
390 	private boolean isButton3Pressed;	
391 	
392 	/**
393 	 * This method installs mouse and key listeners on the canvas that forward
394 	 * those events to piccolo.
395 	 */
396 	protected void installInputSources() {
397 		this.addMouseListener(new MouseListener() {
398 			public void mouseDown(MouseEvent me) {
399 				boolean shouldBalanceEvent = false;
400 				
401 				switch (me.button) {
402 					case 1:
403 						if (isButton1Pressed) {
404 							shouldBalanceEvent = true;	
405 						}
406 						isButton1Pressed = true;
407 						break;
408 					case 2:
409 						if (isButton2Pressed) {
410 							shouldBalanceEvent = true;	
411 						}
412 						isButton2Pressed = true;
413 						break;
414 					case 3:
415 						if (isButton3Pressed) {
416 							shouldBalanceEvent = true;	
417 						}
418 						isButton3Pressed = true;
419 						break;						
420 				}
421 				
422 				if (shouldBalanceEvent) {
423 					java.awt.event.MouseEvent balanceEvent = new PSWTMouseEvent(me,java.awt.event.MouseEvent.MOUSE_RELEASED,1);
424 					sendInputEventToInputManager(balanceEvent, java.awt.event.MouseEvent.MOUSE_RELEASED);
425 				}
426 				
427 				java.awt.event.MouseEvent balanceEvent = new PSWTMouseEvent(me,java.awt.event.MouseEvent.MOUSE_PRESSED,1);
428 				sendInputEventToInputManager(balanceEvent, java.awt.event.MouseEvent.MOUSE_PRESSED);
429 			}
430 			
431 			public void mouseUp(MouseEvent me) {
432 				boolean shouldBalanceEvent = false;
433 				
434 				switch (me.button) {
435 					case 1:
436 						if (!isButton1Pressed) {
437 							shouldBalanceEvent = true;	
438 						}
439 						isButton1Pressed = false;
440 						break;
441 					case 2:
442 						if (!isButton2Pressed) {
443 							shouldBalanceEvent = true;	
444 						}
445 						isButton2Pressed = false;
446 						break;
447 					case 3:
448 						if (!isButton3Pressed) {
449 							shouldBalanceEvent = true;	
450 						}
451 						isButton3Pressed = false;
452 						break;						
453 				}
454 				
455 				if (shouldBalanceEvent) {
456 					java.awt.event.MouseEvent balanceEvent = new PSWTMouseEvent(me,java.awt.event.MouseEvent.MOUSE_PRESSED,1);
457 					sendInputEventToInputManager(balanceEvent, java.awt.event.MouseEvent.MOUSE_PRESSED);
458 				}
459 				
460 				java.awt.event.MouseEvent balanceEvent = new PSWTMouseEvent(me,java.awt.event.MouseEvent.MOUSE_RELEASED,1);
461 				sendInputEventToInputManager(balanceEvent, java.awt.event.MouseEvent.MOUSE_RELEASED);
462 			}
463 			
464 			public void mouseDoubleClick(final MouseEvent me) {
465 			    // This doesn't work with click event types for some reason - it has to do with how
466 			    // the click and release events are ordered, I think			    
467 			    java.awt.event.MouseEvent inputEvent = new PSWTMouseEvent(me,java.awt.event.MouseEvent.MOUSE_PRESSED,2);
468 				sendInputEventToInputManager(inputEvent, java.awt.event.MouseEvent.MOUSE_PRESSED);												
469 				inputEvent = new PSWTMouseEvent(me,java.awt.event.MouseEvent.MOUSE_RELEASED,2);
470 				sendInputEventToInputManager(inputEvent, java.awt.event.MouseEvent.MOUSE_RELEASED);			    
471 			}
472 		});
473 
474 		this.addMouseMoveListener(new MouseMoveListener() {
475 			public void mouseMove(MouseEvent me) {
476 				if (isButton1Pressed || isButton2Pressed || isButton3Pressed) {
477 					java.awt.event.MouseEvent inputEvent = new PSWTMouseEvent(me,java.awt.event.MouseEvent.MOUSE_DRAGGED,1);
478 					sendInputEventToInputManager(inputEvent, java.awt.event.MouseEvent.MOUSE_DRAGGED);	
479 				}
480 				else {
481 					java.awt.event.MouseEvent inputEvent = new PSWTMouseEvent(me,java.awt.event.MouseEvent.MOUSE_MOVED,1);
482 					sendInputEventToInputManager(inputEvent, java.awt.event.MouseEvent.MOUSE_MOVED);					
483 				}
484 			}	
485 		});
486 		
487 		this.addKeyListener(new KeyListener() {
488 			public void keyPressed(KeyEvent ke) {
489 				java.awt.event.KeyEvent inputEvent = new PSWTKeyEvent(ke,java.awt.event.KeyEvent.KEY_PRESSED);
490 				sendInputEventToInputManager(inputEvent, java.awt.event.KeyEvent.KEY_PRESSED);					
491 			}
492 			
493 			public void keyReleased(KeyEvent ke) {
494 				java.awt.event.KeyEvent inputEvent = new PSWTKeyEvent(ke,java.awt.event.KeyEvent.KEY_RELEASED);
495 				sendInputEventToInputManager(inputEvent, java.awt.event.KeyEvent.KEY_RELEASED);									
496 			}
497 		});

530 	public void paintComponent(GC gc, int x, int y, int w, int h) {
531 		PDebug.startProcessingOutput();
532 
533 		GC imageGC = null;
534 		Graphics2D g2 = null;
535 		if (doubleBuffered) {
536 			imageGC = new GC(backBuffer);
537 			g2 = new SWTGraphics2D(imageGC,getDisplay());
538 		}
539 		else {
540 			g2 = new SWTGraphics2D(gc,getDisplay());
541 		}
542 
543 	    g2.setColor(Color.white);
544 	    g2.setBackground(Color.white);
545 	    
546 	    Rectangle rect = getBounds();
547 	    g2.fillRect(0,0,rect.width,rect.height);
548 
549 	    // This fixes a problem with standard debugging of region management in SWT
550 		if (PDebug.debugRegionManagement) {
551 		    Rectangle r = gc.getClipping();
552 		    Rectangle2D r2 = new Rectangle2D.Double(r.x,r.y,r.width,r.height);		    
553 			g2.setBackground(PDebug.getDebugPaintColor());
554 			g2.fill(r2);
555 		}
556 			    
557 		// create new paint context and set render quality
558 		PPaintContext paintContext = new PPaintContext(g2);
559 		if (getInteracting() || getAnimating()) {
560 			if (interactingRenderQuality > animatingRenderQuality) {
561 				paintContext.setRenderQuality(interactingRenderQuality);
562 			} else {
563 				paintContext.setRenderQuality(animatingRenderQuality);
564 			}
565 		} else {
566 			paintContext.setRenderQuality(defaultRenderQuality);
567 		}
568 		
569 		// paint piccolo
570 		camera.fullPaint(paintContext);
571 		
572 		// if switched state from animating to not animating invalidate the entire
573 		// screen so that it will be drawn with the default instead of animating 
574 		// render quality.
575 		if (!getAnimating() && animatingOnLastPaint) {
576 			repaint();
577 		}
578 		animatingOnLastPaint = getAnimating();
579 
580 		boolean region = PDebug.debugRegionManagement;
581 		PDebug.debugRegionManagement = false;
582 		PDebug.endProcessingOutput(g2);
583 		PDebug.debugRegionManagement = region;
584 		
585 		if (doubleBuffered) {
586 			gc.drawImage(backBuffer,0,0);
587 		
588 			// Dispose of the allocated image gc
589 			imageGC.dispose();		
590 		}
591 	}
