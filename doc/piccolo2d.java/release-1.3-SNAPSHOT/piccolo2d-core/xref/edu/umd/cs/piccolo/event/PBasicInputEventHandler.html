37  /**
38   * <b>PBasicInputEventHandler</b> is the standard class in Piccolo that is used
39   * to register for mouse and keyboard events on a PNode. Note the events that
40   * you get depends on the node that you have registered with. For example you
41   * will only get mouse moved events when the mouse is over the node that you
42   * have registered with, not when the mouse is over some other node.
43   * <P>
44   * 
45   * @version 1.0
46   * @author Jesse Grosjean
47   */
48  public class PBasicInputEventHandler implements PInputEventListener {
49  
50      private PInputEventFilter eventFilter;
51  
52      public PBasicInputEventHandler() {
53          super();
54          eventFilter = new PInputEventFilter();
55      }
56  
57      public void processEvent(PInputEvent event, int type) {
58          if (!acceptsEvent(event, type))
59              return;
60  
61          switch (type) {
62              case KeyEvent.KEY_PRESSED:
63                  keyPressed(event);
64                  break;
65  
66              case KeyEvent.KEY_RELEASED:
67                  keyReleased(event);
68                  break;
69  
70              case KeyEvent.KEY_TYPED:
71                  keyTyped(event);
72                  break;
73  
74              case MouseEvent.MOUSE_CLICKED:
75                  mouseClicked(event);
76                  break;
77  
78              case MouseEvent.MOUSE_DRAGGED:
79                  mouseDragged(event);
80                  break;
81  
82              case MouseEvent.MOUSE_ENTERED:
83                  mouseEntered(event);
84                  break;
85  
86              case MouseEvent.MOUSE_EXITED:
87                  mouseExited(event);
88                  break;
89  
90              case MouseEvent.MOUSE_MOVED:
91                  mouseMoved(event);
92                  break;
93  
94              case MouseEvent.MOUSE_PRESSED:
95                  mousePressed(event);
96                  break;
97  
98              case MouseEvent.MOUSE_RELEASED:
99                  mouseReleased(event);
100                 break;
101 
102             case MouseWheelEvent.WHEEL_UNIT_SCROLL:
103                 mouseWheelRotated(event);
104                 break;
105 
106             case MouseWheelEvent.WHEEL_BLOCK_SCROLL:
107                 mouseWheelRotatedByBlock(event);
108                 break;
109 
110             case FocusEvent.FOCUS_GAINED:
111                 keyboardFocusGained(event);
112                 break;
113 
114             case FocusEvent.FOCUS_LOST:
115                 keyboardFocusLost(event);
116                 break;
117 
118             default:
119                 throw new RuntimeException("Bad Event Type");
120         }
121     }
122 
123     // ****************************************************************
124     // Event Filter - All this event listener can be associated with a event
125     // filter. The filter accepts and rejects events based on their modifier
126     // flags and type. If the filter is null (the
127     // default case) then it accepts all events.
128     // ****************************************************************
129 
130     public boolean acceptsEvent(PInputEvent event, int type) {
131         return eventFilter.acceptsEvent(event, type);
132     }
133 
134     public PInputEventFilter getEventFilter() {
135         return eventFilter;
136     }
137 
138     public void setEventFilter(PInputEventFilter newEventFilter) {
139         eventFilter = newEventFilter;
140     }
