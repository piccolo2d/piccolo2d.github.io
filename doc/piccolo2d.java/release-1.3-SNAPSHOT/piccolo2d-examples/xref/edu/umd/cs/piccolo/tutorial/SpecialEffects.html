10  public class SpecialEffects extends PFrame {
11      public void initialize() {
12          // Create the Target for our Activities.
13  
14          // Create a new node that we will apply different activities to, and
15          // place that node at location 200, 200.
16          final PNode aNode = PPath.createRectangle(0, 0, 100, 80);
17          PLayer layer = getCanvas().getLayer();
18          layer.addChild(aNode);
19          aNode.setOffset(200, 200);
20  
21          // Extend PActivity.
22  
23          // Store the current time in milliseconds for use below.
24          long currentTime = System.currentTimeMillis();
25  
26          // Create a new custom "flash" activity. This activity will start
27          // running in five seconds, and while it runs it will flash aNode's
28          // paint between red and green every half second.
29          PActivity flash = new PActivity(-1, 500, currentTime + 5000) {
30              boolean fRed = true;
31  
32              protected void activityStep(long elapsedTime) {
33                  super.activityStep(elapsedTime);
34  
35                  // Toggle the target node's brush color between red and green
36                  // each time the activity steps.
37                  if (fRed) {
38                      aNode.setPaint(Color.red);
39                  }
40                  else {
41                      aNode.setPaint(Color.green);
42                  }
43  
44                  fRed = !fRed;
45              }
46          };
47  
48          // Schedule the activity.
49          getCanvas().getRoot().addActivity(flash);
50  
51          // Create three activities that animate the node's position. Since our
52          // node already descends from the root node the animate methods will
53          // automatically schedule these activities for us.
54          PActivity a1 = aNode.animateToPositionScaleRotation(0, 0, 0.5, 0, 5000);
55          PActivity a2 = aNode.animateToPositionScaleRotation(100, 0, 1.5, Math.toRadians(110), 5000);
56          PActivity a3 = aNode.animateToPositionScaleRotation(200, 100, 1, 0, 5000);
57  
58          // The animate activities will start immediately (in the next call to
59          // PRoot.processInputs) by default. Here we set their start times (in
60          // PRoot global time) so that they start when the previous one has
61          // finished.
62          a1.setStartTime(currentTime);
63          a2.startAfter(a1);
64          a3.startAfter(a2);
65  
66          a1.setDelegate(new PActivity.PActivityDelegate() {
67              public void activityStarted(PActivity activity) {
68                  System.out.println("a1 started");
69              }
70  
71              public void activityStepped(PActivity activity) {
72              }
73  
74              public void activityFinished(PActivity activity) {
75                  System.out.println("a1 finished");
76              }
77          });
78      }
