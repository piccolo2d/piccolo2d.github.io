18  /**
19   * The <b>PComboBox</b> is used instead of a JComboBox in a Piccolo scene graph.
20   * This PComboBox won't work properly if it is located in an abnormal hierarchy of Cameras.
21   * Support is provided for only one (or zero) view transforms.
22   * <p/>
23   * A ComboBox for use in Piccolo.  This still has an associated JPopupMenu
24   * (which is always potentially heavyweight depending on component location
25   * relative to containing window borders.)  However, this ComboBox places
26   * the PopupMenu component of the ComboBox in the appropriate position
27   * relative to the permanent part of the ComboBox.  The PopupMenu is never
28   * transformed.
29   * <p/>
30   * This class was not designed for subclassing.  If different behavior
31   * is required, it seems more appropriate to subclass JComboBox directly
32   * using this class as a model.
33   * <p/>
34   * NOTE: There is currently a known bug, namely, if the ComboBox receives
35   * focus through 'tab' focus traversal and the keyboard is used to interact
36   * with the ComboBox, there may be unexpected results.
37   * <p/>
38   * <p/>
39   * <b>Warning:</b> Serialized objects of this class will not be
40   * compatible with future Piccolo releases. The current serialization support is
41   * appropriate for short term storage or RMI between applications running the
42   * same version of Piccolo. A future release of Piccolo will provide support for long
43   * term persistence.
44   *
45   * @author Lance Good
46   */
47  public class PComboBox extends JComboBox implements Serializable {

104         this.pSwing = pSwing;

106     }

108     /**
109      * The substitute look and feel - used to capture the mouse
110      * events on the arrowButton and the component itself and
111      * to create our PopupMenu rather than the default
112      */
113     protected class PBasicComboBoxUI extends BasicComboBoxUI {
114 
115         /**
116          * Create our Popup instead of theirs
117          */
118         protected ComboPopup createPopup() {
119             PBasicComboPopup popup = new PBasicComboPopup( comboBox );
120             popup.getAccessibleContext().setAccessibleParent( comboBox );
121             return popup;
122         }
123     }

125     /**
126      * The substitute ComboPopupMenu that places itself correctly
127      * in Piccolo.
128      */
129     protected class PBasicComboPopup extends BasicComboPopup {
130 
131         /**
132          * @param combo The parent ComboBox
133          */
134         public PBasicComboPopup( JComboBox combo ) {
135             super( combo );
136         }
137 
138 
139         /**
140          * Computes the bounds for the Popup in Piccolo if a
141          * PMouseEvent has been received.  Otherwise, it uses the
142          * default algorithm for placing the popup.
143          *
144          * @param px corresponds to the x coordinate of the popup
145          * @param py corresponds to the y coordinate of the popup
146          * @param pw corresponds to the width of the popup
147          * @param ph corresponds to the height of the popup
148          * @return The bounds for the PopupMenu
149          */
150         protected Rectangle computePopupBounds( int px, int py, int pw, int ph ) {
151             Rectangle2D r = getNodeBoundsInCanvas();
152             Rectangle sup = super.computePopupBounds( px, py, pw, ph );
153             return new Rectangle( (int)r.getX(), (int)r.getMaxY(), (int)sup.getWidth(), (int)sup.getHeight() );
154         }
155     }
156 
157     private Rectangle2D getNodeBoundsInCanvas() {
158         if( pSwing == null || canvas == null ) {
159             throw new RuntimeException( "PComboBox.setEnvironment( swing, pCanvas );//has to be done manually at present" );
160         }
161         Rectangle2D r1c = pSwing.getBounds();
162         pSwing.localToGlobal( r1c );
163         canvas.getCamera().globalToLocal( r1c );
164         r1c = canvas.getCamera().getViewTransform().createTransformedShape( r1c ).getBounds2D();
165         return r1c;
166     }
