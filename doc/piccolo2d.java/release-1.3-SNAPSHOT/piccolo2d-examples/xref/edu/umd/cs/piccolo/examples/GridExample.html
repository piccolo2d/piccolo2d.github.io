23  /**
24   * Example of drawing an infinite grid, and providing support for snap to grid.
25   */
26  public class GridExample extends PFrame {
27  
28  	static protected Line2D gridLine = new Line2D.Double();
29  	static protected Stroke gridStroke = new BasicStroke(1);
30  	static protected Color gridPaint = Color.BLACK;
31  	static protected double gridSpacing = 20;
32  
33  	public GridExample() {
34  		this(null);
35  	}
36  
37  	public GridExample(PCanvas aCanvas) {
38  		super("GridExample", false, aCanvas);
39  	}
40  
41  	public void initialize() {		
42  		PRoot root = getCanvas().getRoot();
43  		final PCamera camera = getCanvas().getCamera();
44  		final PLayer gridLayer = new PLayer() {
45  			protected void paint(PPaintContext paintContext) {
46  				// make sure grid gets drawn on snap to grid boundaries. And 
47  				// expand a little to make sure that entire view is filled.
48  				double bx = (getX() - (getX() % gridSpacing)) - gridSpacing;
49  				double by = (getY() - (getY() % gridSpacing)) - gridSpacing;
50  				double rightBorder = getX() + getWidth() + gridSpacing;
51  				double bottomBorder = getY() + getHeight() + gridSpacing;
52  
53  				Graphics2D g2 = paintContext.getGraphics();
54  				Rectangle2D clip = paintContext.getLocalClip();
55  
56  				g2.setStroke(gridStroke);
57  				g2.setPaint(gridPaint);
58  
59  				for (double x = bx; x < rightBorder; x += gridSpacing) {
60  					gridLine.setLine(x, by, x, bottomBorder);
61  					if (clip.intersectsLine(gridLine)) {
62  						g2.draw(gridLine);
63  					}
64  				}
65  
66  				for (double y = by; y < bottomBorder; y += gridSpacing) {
67  					gridLine.setLine(bx, y, rightBorder, y);
68  					if (clip.intersectsLine(gridLine)) {
69  						g2.draw(gridLine);
70  					}
71  				}
72  			}
73  		};
74  		
75  		// replace standar layer with grid layer.
76  		root.removeChild(camera.getLayer(0));
77  		camera.removeLayer(0);
78  		root.addChild(gridLayer);
79  		camera.addLayer(gridLayer);
80  
81  		// add constrains so that grid layers bounds always match cameras view bounds. This makes 
82  		// it look like an infinite grid.
83  		camera.addPropertyChangeListener(PNode.PROPERTY_BOUNDS, new PropertyChangeListener() {
84  			public void propertyChange(PropertyChangeEvent evt) {
85  				gridLayer.setBounds(camera.getViewBounds());
86  			}
87  		});
88  
89  		camera.addPropertyChangeListener(PCamera.PROPERTY_VIEW_TRANSFORM, new PropertyChangeListener() {
90  			public void propertyChange(PropertyChangeEvent evt) {
91  				gridLayer.setBounds(camera.getViewBounds());
92  			}
93  		});
94  
95  		gridLayer.setBounds(camera.getViewBounds());
96  
97  		PNode n = new PNode();
98  		n.setPaint(Color.BLUE);
99  		n.setBounds(0, 0, 100, 80);
100 		
101 		getCanvas().getLayer().addChild(n);
102 		getCanvas().removeInputEventListener(getCanvas().getPanEventHandler());
103 		
104 		// add a drag event handler that supports snap to grid.
105 		getCanvas().addInputEventListener(new PDragSequenceEventHandler() {
106 			
107 			protected PNode draggedNode;
108 			protected Point2D nodeStartPosition;
109 			
110 			protected boolean shouldStartDragInteraction(PInputEvent event) {
111 				if (super.shouldStartDragInteraction(event)) {
112 					return event.getPickedNode() != event.getTopCamera() && !(event.getPickedNode() instanceof PLayer);
113 				}
114 				return false;
115 			}
116 
117 			protected void startDrag(PInputEvent event) {
118 				super.startDrag(event);
119 				draggedNode = event.getPickedNode();
120 				draggedNode.moveToFront();
121 				nodeStartPosition = draggedNode.getOffset();
122 			}
123 
124 			protected void drag(PInputEvent event) {
125 				super.drag(event);
126 				
127 				Point2D start = getCanvas().getCamera().localToView((Point2D)getMousePressedCanvasPoint().clone());
128 				Point2D current = event.getPositionRelativeTo(getCanvas().getLayer());
129 				Point2D dest = new Point2D.Double();
130 
131 				dest.setLocation(nodeStartPosition.getX() + (current.getX() - start.getX()),
132 								 nodeStartPosition.getY() + (current.getY() - start.getY()));
133 				
134 				dest.setLocation(dest.getX() - (dest.getX() % gridSpacing),
135 								 dest.getY() - (dest.getY() % gridSpacing));
136 
137 				draggedNode.setOffset(dest.getX(), dest.getY());
138 			}
139 		});
140 	}

144 	}
