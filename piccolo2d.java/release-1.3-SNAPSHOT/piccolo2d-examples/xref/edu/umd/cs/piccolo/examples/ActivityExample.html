11  /**
12   * This example shows how create and schedule activities.
13   */
14  public class ActivityExample extends PFrame {
15  
16  	public ActivityExample() {
17  		this(null);
18  	}
19  	
20  	public ActivityExample(PCanvas aCanvas) {
21  		super("ActivityExample", false, aCanvas);
22  	}
23  	
24  	public void initialize() {
25  		long currentTime = System.currentTimeMillis();
26  		
27  		// Create a new node that we will apply different activities to, and
28  		// place that node at location 200, 200.
29  		final PNode aNode = PPath.createRectangle(0, 0, 100, 80);
30  		PLayer layer = getCanvas().getLayer();
31  		layer.addChild(aNode);
32  		aNode.setOffset(200, 200);
33  		
34  		// Create a new custom "flash" activity. This activity will start running in
35  		// five seconds, and while it runs it will flash aNode's paint between
36  		// red and green every half second.
37  		PActivity flash = new PActivity(-1, 500, currentTime + 5000) {
38  			boolean fRed = true;
39  			
40  			protected void activityStep(long elapsedTime) {
41  				super.activityStep(elapsedTime);
42  				
43  				if (fRed) {
44  					aNode.setPaint(Color.red);
45  				} else {
46  					aNode.setPaint(Color.green);
47  				}		
48  				
49  				fRed = !fRed;
50  			}
51  		};
52  		
53  		// An activity will not run unless it is scheduled with the root. Once
54  		// it has been scheduled it will be given a chance to run during the next
55  		// PRoot.processInputs() call.
56  		getCanvas().getRoot().addActivity(flash);
57  		
58  		// Use the PNode animate methods to create three activities that animate
59  		// the node's position. Since our node already descends from the root node the
60  		// animate methods will automatically schedule these activities for us.
61  		PActivity a1 = aNode.animateToPositionScaleRotation(0, 0, 0.5, 0, 5000);
62  		PActivity a2 = aNode.animateToPositionScaleRotation(100, 0, 1.5, Math.toRadians(110), 5000);
63  		PActivity a3 = aNode.animateToPositionScaleRotation(200, 100, 1, 0, 5000);
64  		PActivity a4 = aNode.animateToTransparency(0.25f, 3000);
65  
66  		// the animate activities will start immediately (in the next call to PRoot.processInputs)
67  		// by default. Here we set their start times (in PRoot global time) so that they start 
68  		// when the previous one has finished.
69  		a1.setStartTime(currentTime);
70  		
71  		a2.startAfter(a1);
72  		a3.startAfter(a2);
73  		a4.startAfter(a3);
74  		
75  		// or the previous three lines could be replaced with these lines for the same effect.
76  		//a2.setStartTime(currentTime + 5000);
77  		//a3.setStartTime(currentTime + 10000);
78  		//a4.setStartTime(currentTime + 15000);
79  	}
