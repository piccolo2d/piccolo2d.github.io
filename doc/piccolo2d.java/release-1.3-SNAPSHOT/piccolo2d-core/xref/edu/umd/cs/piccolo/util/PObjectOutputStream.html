32  import java.io.ByteArrayOutputStream;
33  import java.io.IOException;
34  import java.io.ObjectOutputStream;
35  import java.io.OutputStream;
36  import java.util.HashMap;

38  /** 
39   * <b>PObjectOutputStream</b> is an extension of ObjectOutputStream to handle optional 
40   * elements. This is similar to the concept of Java's "weak references", but applied to
41   * object serialization rather than garbage collection. Here, PObjectOutputStream
42   * provides a method, <code>writeConditionalObject</code>, which only
43   * serializes the specified object to the stream if there is a strong reference (if it
44   * has been written somewhere else using writeObject()) to that object elsewhere in the 
45   * stream.
46   * <p>
47   * To discover strong references to objects, PObjectOutputStream uses a two-phase writing 
48   * process. First, a "discovery" phase is used to find out what objects are about to 
49   * be serialized. This works by effectively serializing the object graph to /dev/null, recording 
50   * which objects are unconditionally written using the standard writeObject method. Then,
51   * in the second "write" phase, ObjectOutputStream actually serializes the data
52   * to the output stream. During this phase, calls to writeConditionalObject() will
53   * only write the specified object if the object was found to be serialized during the
54   * discovery stage. If the object was not recorded during the discovery stage, a an optional null 
55   * (the default) is unconditionally written in place of the object. To skip writting out the
56   * null use <code>writeConditionalObject(object, false)</code>
57   * <p>
58   * By careful implementation of readObject and writeObject methods, streams serialized using
59   * PObjectOutputStream can be deserialized using the standard ObjectInputStream.
60   * <p>
61   * @version 1.0
62   * @author Jon Meyer
63   * @author Jesse Grosjean
64   */
65  public class PObjectOutputStream extends ObjectOutputStream {
66  
67  	private boolean writingRoot;
68  	private HashMap unconditionallyWritten;
69  
70  	public static byte[] toByteArray(Object aRoot) throws IOException {
71  		ByteArrayOutputStream out = new ByteArrayOutputStream();
72  		PObjectOutputStream zout = new PObjectOutputStream(out);
73  		zout.writeObjectTree(aRoot);
74  		return out.toByteArray();
75  	}
76  	
77  	public PObjectOutputStream(OutputStream out) throws IOException {
78  		super(out);
79  		unconditionallyWritten = new HashMap();
80  	}
81  
82  	public void writeObjectTree(Object aRoot) throws IOException {
83  		writingRoot = true;
84  		recordUnconditionallyWritten(aRoot); // record pass
85  		writeObject(aRoot); 				 // write pass
86  		writingRoot = false;
87  	}
88  
89  	public void writeConditionalObject(Object object) throws IOException {
90  		if (!writingRoot) {
91  			throw new RuntimeException("writeConditionalObject() may only be called when a root object has been written.");
92  		}
93  		
94  		if (unconditionallyWritten.containsKey(object)) {
95  			writeObject(object);
96  		} else {
97  			writeObject(null);
98  		}
99  	}
100 
101 	public void reset() throws IOException {
102 		super.reset();
103 		unconditionallyWritten.clear();
104 	}
105 
106 	protected void recordUnconditionallyWritten(Object aRoot) throws IOException {
107 		class ZMarkObjectOutputStream extends PObjectOutputStream {
108 			public ZMarkObjectOutputStream() throws IOException {
109 				super(PUtil.NULL_OUTPUT_STREAM);
110 				enableReplaceObject(true);
111 			}
112 			public Object replaceObject(Object object) {
113 				PObjectOutputStream.this.unconditionallyWritten.put(object, Boolean.TRUE);
114 				return object;
115 			}	
116 			public void writeConditionalObject(Object object) throws IOException {
117 			}				
118 		}
119 		new ZMarkObjectOutputStream().writeObject(aRoot);
120 	}
121 }
