94      public static double[] initPoints(double[] points, int n, double[] old) {
95          if (points == null || n * 2 > points.length) {

98          if (old != null && points != old) {

125     public void addPoints(int pos, Points pts, int start, int end) {
126         if (end < 0) {
127             end = pts.getPointCount() + end + 1;
128         }
129         int n = numPoints + end - start;
130         points = initPoints(points, n, points);
131         int pos1 = pos * 2, pos2 = (pos + end - start) * 2, len = (numPoints - pos) * 2;
132         System.arraycopy(points, pos1, points, pos2, len);
133         numPoints = n;
134         if (pts == null) {
135             return;
136         }
137         for (int count = 0; start < end; count++, start++) {
138             setPoint(pos + count, pts.getX(start), pts.getY(start));
139         }
140     }

167     public void removePoints(int pos, int num) {
168         num = Math.min(num, numPoints - pos);
169         if (num <= 0)
