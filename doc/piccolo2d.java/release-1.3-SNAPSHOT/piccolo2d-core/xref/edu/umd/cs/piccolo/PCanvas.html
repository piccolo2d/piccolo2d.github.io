241 	/**
242 	 * Set if this canvas is interacting. If so the canvas will normally 
243 	 * render at a lower quality that is faster. Also repaints the canvas if the
244 	 * render quality should change.
245 	 */
246 	public void setInteracting(boolean isInteracting) {
247 		boolean wasInteracting = getInteracting();
248 		
249 		if (isInteracting) {
250 			interacting++;
251 		} else {
252 			interacting--;
253 		}
254 				
255 		if (!getInteracting()) { // determine next render quality and repaint if it's greater then the old
256 								 // interacting render quality.
257 			int nextRenderQuality = defaultRenderQuality;
258 			if (getAnimating()) nextRenderQuality = animatingRenderQuality;
259 			if (nextRenderQuality > interactingRenderQuality) {
260 				repaint();
261 			}
262 		}
263 		
264 		isInteracting = getInteracting();
265 		
266 		if (wasInteracting != isInteracting) {
267 			firePropertyChange(INTERATING_CHANGED_NOTIFICATION, wasInteracting, isInteracting);
268 		}
269 	}

383 				public void mousePressed(MouseEvent e) {
384 					requestFocus();
385 					
386 					boolean shouldBalanceEvent = false;
387 	
388 					if (e.getButton() == MouseEvent.NOBUTTON) {
389 						if ((e.getModifiers() & MouseEvent.BUTTON1_MASK) == MouseEvent.BUTTON1_MASK) {
390 							e = new MouseEvent((Component)e.getSource(),MouseEvent.MOUSE_PRESSED,e.getWhen(),e.getModifiers(),e.getX(),e.getY(),e.getClickCount(),e.isPopupTrigger(),MouseEvent.BUTTON1);
391 						}
392 						else if ((e.getModifiers() & MouseEvent.BUTTON2_MASK) == MouseEvent.BUTTON2_MASK) {
393 							e = new MouseEvent((Component)e.getSource(),MouseEvent.MOUSE_PRESSED,e.getWhen(),e.getModifiers(),e.getX(),e.getY(),e.getClickCount(),e.isPopupTrigger(),MouseEvent.BUTTON2);
394 						}
395 						else if ((e.getModifiers() & MouseEvent.BUTTON3_MASK) == MouseEvent.BUTTON3_MASK) {
396 							e = new MouseEvent((Component)e.getSource(),MouseEvent.MOUSE_PRESSED,e.getWhen(),e.getModifiers(),e.getX(),e.getY(),e.getClickCount(),e.isPopupTrigger(),MouseEvent.BUTTON3);
397 						}					
398 					}
399 	
400 					switch (e.getButton()) {
401 						case MouseEvent.BUTTON1:
402 							if (isButton1Pressed) {
403 								shouldBalanceEvent = true;
404 							}
405 							isButton1Pressed = true;
406 							break;
407 							
408 						case MouseEvent.BUTTON2:
409 							if (isButton2Pressed) {
410 								shouldBalanceEvent = true;
411 							}
412 							isButton2Pressed = true;
413 							break;
414 							
415 						case MouseEvent.BUTTON3:
416 							if (isButton3Pressed) {
417 								shouldBalanceEvent = true;
418 							}
419 							isButton3Pressed = true;
420 							break;						
421 					}
422 					
423 					if (shouldBalanceEvent) {
424 						MouseEvent balanceEvent = new MouseEvent((Component)e.getSource(),MouseEvent.MOUSE_RELEASED,e.getWhen(),e.getModifiers(),e.getX(),e.getY(),e.getClickCount(),e.isPopupTrigger(),e.getButton());
425 						sendInputEventToInputManager(balanceEvent, MouseEvent.MOUSE_RELEASED);
426 					}
427 					
428 					sendInputEventToInputManager(e, MouseEvent.MOUSE_PRESSED);
429 				}

557 	public void paintComponent(Graphics g) {
558 		PDebug.startProcessingOutput();
559 
560 		Graphics2D g2 = (Graphics2D) g.create();		
561 	    g2.setColor(getBackground());
562 	    g2.fillRect(0, 0, getWidth(), getHeight());
563 		
564 		// create new paint context and set render quality to lowest common 
565 		// denominator render quality.
566 		PPaintContext paintContext = new PPaintContext(g2);
567 		if (getInteracting() || getAnimating()) {
568 			if (interactingRenderQuality < animatingRenderQuality) {
569 				paintContext.setRenderQuality(interactingRenderQuality);
570 			} else {
571 				paintContext.setRenderQuality(animatingRenderQuality);
572 			}
573 		} else {
574 			paintContext.setRenderQuality(defaultRenderQuality);
575 		}
576 		
577 		// paint piccolo
578 		camera.fullPaint(paintContext);
579 		
580 		// if switched state from animating to not animating invalidate the entire
581 		// screen so that it will be drawn with the default instead of animating 
582 		// render quality.
583 		if (!getAnimating() && animatingOnLastPaint) {
584 			repaint();
585 		}
586 		animatingOnLastPaint = getAnimating();
587 		
588 		PDebug.endProcessingOutput(g2);				
589 	}
