46  /**
47   * Modified to use SWT paths instead of normal paths 
48   * @version 1.0
49   * @author Lance Good
50   */ 
51  public class PSWTSelectionEventHandler extends PSelectionEventHandler {
52  
53      PSWTPath marquee;
54      PNode marqueeParent;
55      Point2D pressPt;
56      Point2D canvasPressPt;
57      
58      /**
59  	 * Creates a selection event handler.
60  	 * @param marqueeParent The node to which the event handler dynamically adds a marquee
61  	 * (temporarily) to represent the area being selected.
62  	 * @param selectableParent The node whose children will be selected
63  	 * by this event handler.
64  	 */
65  	public PSWTSelectionEventHandler(PNode marqueeParent, PNode selectableParent) {
66  	    super(new PNode(),selectableParent);
67  	    this.marqueeParent = marqueeParent;
68  	}
69  
70  	/**
71  	 * Creates a selection event handler.
72  	 * @param marqueeParent The node to which the event handler dynamically adds a marquee
73  	 * (temporarily) to represent the area being selected.
74  	 * @param selectableParents A list of nodes whose children will be selected
75  	 * by this event handler.
76  	 */
77  	public PSWTSelectionEventHandler(PNode marqueeParent, List selectableParents) {
78  	    super(new PNode(),selectableParents);
79  	    this.marqueeParent = marqueeParent;
80  	}
81  
82  	public void decorateSelectedNode(PNode node) {	    
83  		PSWTBoundsHandle.addBoundsHandlesTo(node);
84  	}
85  
86  	public void undecorateSelectedNode(PNode node) {
87  		PSWTBoundsHandle.removeBoundsHandlesFrom(node);
88  	}
89  
90  	protected void initializeSelection(PInputEvent pie) {
91  	    super.initializeSelection(pie);
92  	    pressPt = pie.getPosition();
93  		canvasPressPt = pie.getCanvasPosition();
94  	}
95  	
96  	protected void initializeMarquee(PInputEvent e) {
97  	    super.initializeMarquee(e);
98  	    
99  	    marquee = new PSWTPath(new Rectangle2D.Float((float)pressPt.getX(), (float)pressPt.getY(), 0, 0)) {
100             protected void paint(PPaintContext paintContext) {
101                 SWTGraphics2D s2g = (SWTGraphics2D)paintContext.getGraphics();
102                 s2g.gc.setLineStyle(SWT.LINE_DASH);
103                 super.paint(paintContext);
104                 s2g.gc.setLineStyle(SWT.LINE_SOLID);
105             }
106 	    };
107 		marquee.setStrokeColor(Color.black);
108 		marquee.setPaint(null);
109 		marqueeParent.addChild(marquee);
110 	}
111 
112 	protected void updateMarquee(PInputEvent pie) {
113 		super.updateMarquee(pie);
114 	    
115 	    PBounds b = new PBounds();
116 
117 		if (marqueeParent instanceof PCamera) {
118 			b.add(canvasPressPt);
119 			b.add(pie.getCanvasPosition());
120 		}
121 		else {
122 			b.add(pressPt);
123 			b.add(pie.getPosition());
124 		}
125 
126 		marquee.setPathToRectangle((float) b.x, (float) b.y, (float) b.width, (float) b.height);				
127 		b.reset();
128 		b.add(pressPt);
129 		b.add(pie.getPosition());
130 	}
131 
132 	protected PBounds getMarqueeBounds() {
133 		if (marquee != null) {
134 			return marquee.getBounds();
135 		}	
136 		return new PBounds();
137 	}
138 
139 	protected void endMarqueeSelection(PInputEvent e) {
140 	    super.endMarqueeSelection(e);
141 	    
142 	    // Remove marquee
143 		marquee.removeFromParent();
144 		marquee = null; 				
145 	}
146

148 	 * This gets called continuously during the drag, and is used to animate the marquee
149 	 */
150 	protected void dragActivityStep(PInputEvent aEvent) {
151 	}
