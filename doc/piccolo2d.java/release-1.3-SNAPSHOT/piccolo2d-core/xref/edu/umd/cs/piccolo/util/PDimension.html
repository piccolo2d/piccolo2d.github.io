30  package edu.umd.cs.piccolo.util;
31  
32  import java.awt.geom.Dimension2D;
33  import java.awt.geom.Point2D;
34  import java.io.Serializable;
35  
36  /**
37   * <b>PDimension</b> this class should be removed once a concrete Dimension2D
38   * that supports doubles is added to java.
39   * <P>
40   * 
41   * @version 1.0
42   * @author Jesse Grosjean
43   */
44  public class PDimension extends Dimension2D implements Serializable {
45  
46      public double width;
47      public double height;
48  
49      public PDimension() {
50          super();
51      }
52  
53      public PDimension(Dimension2D aDimension) {
54          this(aDimension.getWidth(), aDimension.getHeight());
55      }
56  
57      public PDimension(double aWidth, double aHeight) {
58          super();
59          width = aWidth;
60          height = aHeight;
61      }
62  
63      public PDimension(Point2D p1, Point2D p2) {
64          width = p2.getX() - p1.getX();
65          height = p2.getY() - p1.getY();
66      }
67  
68      public double getHeight() {
69          return height;
70      }
71  
72      public double getWidth() {
73          return width;
74      }
75  
76      public void setSize(double aWidth, double aHeight) {
77          width = aWidth;
78          height = aHeight;
79      }
80  
81      public String toString() {
82          StringBuffer result = new StringBuffer();
83  
84          result.append(super.toString().replaceAll(".*\\.", ""));
85          result.append('[');
86          result.append("width=");
87          result.append(width);
88          result.append(",height=");
89          result.append(height);
90          result.append(']');
91  
92          return result.toString();
93      }
94  }
