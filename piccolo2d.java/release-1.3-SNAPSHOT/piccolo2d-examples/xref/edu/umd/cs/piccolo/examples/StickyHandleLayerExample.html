16  /**
17  * This example shows another way to create sticky handles. These handles are
18   * not added as children to the object that they manipulate. Instead they are
19   * added to the camera the views that objects. This means that they will not be
20   * affected by the cameras view transform, and so will stay the same size when
21   * the view is zoomed. They will also be drawn on top of all other objects, even
22   * if those objects overlap the object that they manipulate. For this setup we
23   * need to add and updateHandles activity that makes sure to relocate the handle
24   * after any change. Another way to do this would be to add change listeners to
25   * the camera and the node that they manipulate and only update them then. But
26   * this method is easier and should be plenty efficient for normal use.
27   * 
28   * @author jesse
29   */
30  public class StickyHandleLayerExample extends PFrame {
31  	
32  	public StickyHandleLayerExample() {
33  		this(null);
34  	}
35  
36  	public StickyHandleLayerExample(PCanvas aCanvas) {
37  		super("StickyHandleLayerExample", false, aCanvas);
38  	}
39  
40  	public void initialize() {
41  		PCanvas c = getCanvas();
42  		
43  		PActivity updateHandles = new PActivity(-1, 0) {
44  			protected void activityStep(long elapsedTime) {
45  				super.activityStep(elapsedTime);
46  				
47  				PRoot root = getActivityScheduler().getRoot();
48  
49  				if (root.getPaintInvalid() || root.getChildPaintInvalid()) {	
50  					Iterator i = getCanvas().getCamera().getChildrenIterator();
51  					while (i.hasNext()) {
52  						PNode each = (PNode) i.next();
53  						if (each instanceof PHandle) {
54  							PHandle handle = (PHandle) each;
55  							handle.relocateHandle();
56  						}
57  					}
58  				}
59  			}
60  		};
61  		
62  		PPath rect = PPath.createRectangle(0, 0, 100, 100);
63  		rect.setPaint(Color.RED);
64  		c.getLayer().addChild(rect);
65  
66  		c.getCamera().addChild(new PBoundsHandle(PBoundsLocator.createNorthEastLocator(rect)));
67  		c.getCamera().addChild(new PBoundsHandle(PBoundsLocator.createNorthWestLocator(rect)));
68  		c.getCamera().addChild(new PBoundsHandle(PBoundsLocator.createSouthEastLocator(rect)));
69  		c.getCamera().addChild(new PBoundsHandle(PBoundsLocator.createSouthWestLocator(rect)));
70  		
71  		c.getRoot().getActivityScheduler().addActivity(updateHandles, true);
72  	}
73  	
74  	public static void main(String[] args) {
75  		new StickyHandleLayerExample();
76  	}
