44  /**
45   * <b>PPaintContext</b> is used by piccolo nodes to paint themselves on the
46   * screen. PPaintContext wraps a Graphics2D to implement painting.
47   * <P>
48   * 
49   * @version 1.0
50   * @author Jesse Grosjean
51   */
52  public class PPaintContext {
53  
54      public static final int LOW_QUALITY_RENDERING = 0;
55      public static final int HIGH_QUALITY_RENDERING = 1;
56  
57      public static FontRenderContext RENDER_QUALITY_LOW_FRC = new FontRenderContext(null, false, true);
58      public static FontRenderContext RENDER_QUALITY_HIGH_FRC = new FontRenderContext(null, true, true);
59      public static PPaintContext CURRENT_PAINT_CONTEXT;
60  
61      private static double[] PTS = new double[4];
62  
63      private Graphics2D graphics;
64      protected PStack compositeStack;
65      protected PStack clipStack;
66      protected PStack localClipStack;
67      protected PStack cameraStack;
68      protected PStack transformStack;
69      protected int renderQuality;
70  
71      public PPaintContext(Graphics2D aGraphics) {
72          super();
73          graphics = aGraphics;
74          compositeStack = new PStack();
75          clipStack = new PStack();
76          localClipStack = new PStack();
77          cameraStack = new PStack();
78          transformStack = new PStack();
79          renderQuality = HIGH_QUALITY_RENDERING;
80  
81          Shape clip = aGraphics.getClip();
82          if (clip == null) {
83              clip = new PBounds(-Integer.MAX_VALUE / 2, -Integer.MAX_VALUE / 2, Integer.MAX_VALUE, Integer.MAX_VALUE);
84              aGraphics.setClip(clip);
85          }
86  
87          localClipStack.push(clip.getBounds2D());
88  
89          CURRENT_PAINT_CONTEXT = this;
90      }
91  
92      public Graphics2D getGraphics() {
93          return graphics;
94      }
95  
96      // ****************************************************************
97      // Context Attributes.
98      // ****************************************************************
99  
100     public Rectangle2D getLocalClip() {
101         return (Rectangle2D) localClipStack.peek();
102     }
103 
104     public double getScale() {
105         PTS[0] = 0;// x1
106         PTS[1] = 0;// y1
107         PTS[2] = 1;// x2
108         PTS[3] = 0;// y2
109         graphics.getTransform().transform(PTS, 0, PTS, 0, 2);
110         return Point2D.distance(PTS[0], PTS[1], PTS[2], PTS[3]);
111     }
112 
113     // ****************************************************************
114     // Context Attribute Stacks. attributes that can be pushed and
115     // popped.
116     // ****************************************************************
117 
118     public void pushCamera(PCamera aCamera) {
119         cameraStack.push(aCamera);
120     }
121 
122     public void popCamera(PCamera aCamera) {
123         cameraStack.pop();
124     }
125 
126     public PCamera getCamera() {
127         return (PCamera) cameraStack.peek();
128     }
129 
130     public void pushClip(Shape aClip) {
131         Shape currentClip = graphics.getClip();
132         clipStack.push(currentClip);
133         graphics.clip(aClip);
134         Rectangle2D newLocalClip = aClip.getBounds2D();
135         Rectangle2D.intersect(getLocalClip(), newLocalClip, newLocalClip);
136         localClipStack.push(newLocalClip);
137     }
138 
139     public void popClip(Shape aClip) {
140         Shape newClip = (Shape) clipStack.pop();
141         graphics.setClip(newClip);
142         localClipStack.pop();
143     }
144 
145     public void pushTransparency(float transparency) {
146         if (transparency == 1) {
147             return;
148         }
149         Composite current = graphics.getComposite();
150         float currentAlaph = 1.0f;
151         compositeStack.push(current);
152 
153         if (current instanceof AlphaComposite) {
154             currentAlaph = ((AlphaComposite) current).getAlpha();
155         }
156         AlphaComposite newComposite = AlphaComposite.getInstance(AlphaComposite.SRC_OVER, currentAlaph * transparency);
157         graphics.setComposite(newComposite);
158     }
159 
160     public void popTransparency(float transparency) {
161         if (transparency == 1) {
162             return;
163         }
164         Composite c = (Composite) compositeStack.pop();
165         graphics.setComposite(c);
166     }
167 
168     public void pushTransform(PAffineTransform aTransform) {
169         if (aTransform == null)
170             return;
171         Rectangle2D newLocalClip = (Rectangle2D) getLocalClip().clone();
172         aTransform.inverseTransform(newLocalClip, newLocalClip);
173         transformStack.push(graphics.getTransform());
174         localClipStack.push(newLocalClip);
175         graphics.transform(aTransform);
176     }
177 
178     public void popTransform(PAffineTransform aTransform) {
179         if (aTransform == null)
180             return;
181         graphics.setTransform((AffineTransform) transformStack.pop());
182         localClipStack.pop();
183     }

196     /**
197      * Set the rendering hints for this paint context. The render quality is
198      * most often set by the rendering PCanvas. Use PCanvas.setRenderQuality()
199      * and PCanvas.setInteractingRenderQuality() to set these values.
200      * 
201      * @param requestedQuality supports PPaintContext.HIGH_QUALITY_RENDERING or
202      *            PPaintContext.LOW_QUALITY_RENDERING
203      */
204     public void setRenderQuality(int requestedQuality) {
205         renderQuality = requestedQuality;
206 
207         switch (renderQuality) {
208             case HIGH_QUALITY_RENDERING:
209                 graphics.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
210                 graphics.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
211                 graphics.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
212                 graphics.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
213                         RenderingHints.VALUE_FRACTIONALMETRICS_ON);
214                 break;
215 
216             case LOW_QUALITY_RENDERING:
217                 graphics.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
218                 graphics
219                         .setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_TEXT_ANTIALIAS_OFF);
220                 graphics.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_SPEED);
221                 graphics.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
222                         RenderingHints.VALUE_FRACTIONALMETRICS_ON);
223                 break;
224         }
225     }
