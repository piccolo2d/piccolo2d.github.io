106 	protected JTextComponent createDefaultEditor() {
107 		JTextPane tComp = new JTextPane() {
108 
109 			/**
110 			 * Set some rendering hints - if we don't then the rendering can be inconsistent.  Also,
111 			 * Swing doesn't work correctly with fractional metrics.
112 			 */
113 			public void paint(Graphics g) {
114 				Graphics2D g2 = (Graphics2D)g;
115 				
116 				g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
117 				g2.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
118 				g2.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
119 				g2.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS, RenderingHints.VALUE_FRACTIONALMETRICS_OFF);				
120 				
121 				super.paint(g);
122 			}
123 		};	
124 		tComp.setBorder(new CompoundBorder(new LineBorder(Color.black),new EmptyBorder(3,3,3,3)));
125 		return tComp;
126 	}

144 	public PStyledText createText() {
145 		PStyledText newText = new PStyledText();
146 		
147 		Document doc = editor.getUI().getEditorKit(editor).createDefaultDocument();
148 		if (doc instanceof StyledDocument) {
149 			if (!doc.getDefaultRootElement().getAttributes().isDefined(StyleConstants.FontFamily)
150 				|| !doc.getDefaultRootElement().getAttributes().isDefined(StyleConstants.FontSize)) {
151 
152 				Font eFont = editor.getFont();
153 				SimpleAttributeSet sas = new SimpleAttributeSet();
154 				sas.addAttribute(StyleConstants.FontFamily, eFont.getFamily());
155 				sas.addAttribute(StyleConstants.FontSize, new Integer(eFont.getSize()));
156 
157 				((StyledDocument) doc).setParagraphAttributes(0, doc.getLength(), sas, false);
158 			}
159 		}
160 		newText.setDocument(doc);
161 		
162 		return newText;
163 	}

165 	public void mousePressed(PInputEvent inputEvent) {
166 		PNode pickedNode = inputEvent.getPickedNode();
167 				
168 		stopEditing();
169 		
170 		if (pickedNode instanceof PStyledText) {
171 			startEditing(inputEvent,(PStyledText)pickedNode);
172 		}
173 		else if (pickedNode instanceof PCamera) {
174 			PStyledText newText = createText();
175 			Insets pInsets = newText.getInsets();
176 			canvas.getLayer().addChild(newText);
177 			newText.translate(inputEvent.getPosition().getX()-pInsets.left,inputEvent.getPosition().getY()-pInsets.top);
178 			startEditing(inputEvent, newText);
179 		}
180 	}

182 	public void startEditing(PInputEvent event, PStyledText text) {
183 		// Get the node's top right hand corner
184 		Insets pInsets = text.getInsets();
185 		Point2D nodePt = new Point2D.Double(text.getX()+pInsets.left,text.getY()+pInsets.top);
186 		text.localToGlobal(nodePt);
187 		event.getTopCamera().viewToLocal(nodePt);
188 
189 		// Update the editor to edit the specified node
190 		editor.setDocument(text.getDocument());
191 		editor.setVisible(true);
192 
193 		Insets bInsets = editor.getBorder().getBorderInsets(editor);
194 		editor.setLocation((int)nodePt.getX()-bInsets.left,(int)nodePt.getY()-bInsets.top);
195 		reshapeEditorLater();
196 
197 		dispatchEventToEditor(event);
198 		canvas.repaint();
199 
200 		text.setEditing(true);
201 		text.getDocument().addDocumentListener(docListener);
202 		editedText = text;					
203 	}
204 	
205 	public void stopEditing() {
206 		if (editedText != null) {
207 			editedText.getDocument().removeDocumentListener(docListener);
208 			editedText.setEditing(false);
209 
210 			if (editedText.getDocument().getLength() == 0) {
211 				editedText.removeFromParent();		
212 			}
213 			else {
214 				editedText.syncWithDocument();
215 			}
216 
217 			editor.setVisible(false);
218 			canvas.repaint();
219 		
220 			editedText = null;
221 		}
222 	}

224 	public void dispatchEventToEditor(final PInputEvent e) {
225 		// We have to nest the mouse press in two invoke laters so that it is 
226 		// fired so that the component has been completely validated at the new size
227 		// and the mouse event has the correct offset
228 		SwingUtilities.invokeLater(new Runnable() {
229 			public void run() {
230 				SwingUtilities.invokeLater(new Runnable() {
231 					public void run() { 	
232 						MouseEvent me =
233 							new MouseEvent(
234 								editor,
235 								MouseEvent.MOUSE_PRESSED,
236 								e.getWhen(),
237 								e.getModifiers() | InputEvent.BUTTON1_MASK,
238 								(int) (e.getCanvasPosition().getX() - editor.getX()),
239 								(int) (e.getCanvasPosition().getY() - editor.getY()),
240 								1,
241 								false);
242 						editor.dispatchEvent(me);
243 					}
244 				});
245 			}
246 		});
247 	}

250 	public void reshapeEditor() {
251 		if (editedText != null) {
252 			// Update the size to fit the new document - note that it is a 2 stage process
253 			Dimension prefSize = editor.getPreferredSize();
254 	
255 			Insets pInsets = editedText.getInsets();
256 			Insets jInsets = editor.getInsets();
257 			
258 			int width = (editedText.getConstrainWidthToTextWidth()) ? (int)prefSize.getWidth() : (int)(editedText.getWidth()-pInsets.left-pInsets.right+jInsets.left+jInsets.right+3.0);
259 			prefSize.setSize(width,prefSize.getHeight());
260 			editor.setSize(prefSize);
261 
262 			prefSize = editor.getPreferredSize();
263 			int height = (editedText.getConstrainHeightToTextHeight()) ? (int)prefSize.getHeight() : (int)(editedText.getHeight()-pInsets.top-pInsets.bottom+jInsets.top+jInsets.bottom+3.0);			
264 			prefSize.setSize(width,height); 		
265 			editor.setSize(prefSize);
266 		}
267 	}
