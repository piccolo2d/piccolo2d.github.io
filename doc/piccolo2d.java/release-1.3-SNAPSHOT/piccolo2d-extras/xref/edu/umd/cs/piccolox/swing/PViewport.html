18  /**
19   * A subclass of JViewport that talks to the scroll director to negotiate the
20   * view positions and sizes.
21   * 
22   * @author Lance Good
23   */
24  public class PViewport extends JViewport {
25  
26      /**
27       * Controls what happens when scrolling occurs
28       */
29      PScrollDirector scrollDirector;
30  
31      /**
32       * Pass constructor info to super
33       */
34      public PViewport() {
35          super();
36  
37          setScrollDirector(createScrollDirector());
38      }
39  
40      /**
41       * Subclassers can override this to install a different layout manager (or
42       * <code>null</code>) in the constructor. Returns a new
43       * <code>ViewportLayout</code> object.
44       * 
45       * @return a <code>LayoutManager</code>
46       */
47      protected LayoutManager createLayoutManager() {
48          return new PViewportLayout();
49      }
50  
51      /**
52       * Subclassers can override this to install a different scroll director in
53       * the constructor. Returns a new <code>PScrollDirector</code> object.
54       * 
55       * @return a <code>PScrollDirector
56       */
57      protected PScrollDirector createScrollDirector() {
58          return new PDefaultScrollDirector();
59      }
60  
61      /**
62       * Set the scroll director on this viewport
63       * 
64       * @param scrollDirector The new scroll director
65       */
66      public void setScrollDirector(PScrollDirector scrollDirector) {
67          if (this.scrollDirector != null) {
68              this.scrollDirector.unInstall();
69          }
70          this.scrollDirector = scrollDirector;
71          if (scrollDirector != null) {
72              this.scrollDirector.install(this, (PCanvas) getView());
73          }
74      }
75  
76      /**
77       * @return The scroll director on this viewport
78       */
79      public PScrollDirector getScrollDirector() {
80          return scrollDirector;
81      }
82  
83      /**
84       * Overridden to throw an exception if the view is not a ZCanvas
85       * 
86       * @param view The new view - it better be a ZCanvas!
87       */
88      public void setView(Component view) {
89          if (!(view instanceof PCanvas)) {
90              throw new UnsupportedOperationException("PViewport only supports ZCanvas");
91          }
92  
93          super.setView(view);
94  
95          if (scrollDirector != null) {
96              scrollDirector.install(this, (PCanvas) view);
97          }
98      }
99  
100     /**
101      * Sets the view coordinates that appear in the upper left hand corner of
102      * the viewport, does nothing if there's no view.
103      * 
104      * @param p a <code>Point</code> object giving the upper left coordinates
105      */
106     public void setViewPosition(Point p) {
107         if (getView() == null) {
108             return;
109         }
110 
111         double oldX = 0, oldY = 0, x = p.x, y = p.y;
112 
113         Point2D vp = getViewPosition();
114         if (vp != null) {
115             oldX = vp.getX();
116             oldY = vp.getY();
117         }
118 
119         /**
120          * Send the scroll director the exact view position and let it interpret
121          * it as needed
122          */
123         double newX = x;
124         double newY = y;
125 
126         if ((oldX != newX) || (oldY != newY)) {
127             scrollUnderway = true;
128 
129             scrollDirector.setViewPosition(newX, newY);
130 
131             fireStateChanged();
132         }
133     }
134 
135     /**
136      * Gets the view position from the scroll director based on the current
137      * extent size
138      * 
139      * @return The new view position
140      */
141     public Point getViewPosition() {
142         if (scrollDirector != null) {
143             Dimension extent = getExtentSize();
144             return scrollDirector.getViewPosition(new PBounds(0, 0, extent.getWidth(), extent.getHeight()));
145         }
146         else {
147             return null;
148         }
149     }
150 
151     /**
152      * Gets the view size from the scroll director based on the current extent
153      * size
154      * 
155      * @return The new view size
156      */
157     public Dimension getViewSize() {
158         Dimension extent = getExtentSize();
159         return scrollDirector.getViewSize(new PBounds(0, 0, extent.getWidth(), extent.getHeight()));
160     }
161 
162     /**
163      * Gets the view size from the scroll director based on the specified extent
164      * size
165      * 
166      * @param r The extent size from which the view is computed
167      * @return The new view size
168      */
169     public Dimension getViewSize(Rectangle2D r) {
170         return scrollDirector.getViewSize(r);
171     }
172 
173     /**
174      * Notifies all <code>ChangeListeners</code> when the views size, position,
175      * or the viewports extent size has changed.
176      */
177     public void fireStateChanged() {
178         super.fireStateChanged();
179     }
180 
181     /**
182      * A simple layout manager to give the ZCanvas the same size as the Viewport
183      */
184     public static class PViewportLayout extends ViewportLayout {
185         /**
186          * Called when the specified container needs to be laid out.
187          * 
188          * @param parent the container to lay out
189          */
190         public void layoutContainer(Container parent) {
191             JViewport vp = (JViewport) parent;
192             Component view = vp.getView();
193 
194             if (view == null) {
195                 return;
196             }
197 
198             Dimension extentSize = vp.getSize();
199 
200             vp.setViewSize(extentSize);
201         }
202     }
