51  /**
52   * <b>PPath</b> is a wrapper around a java.awt.geom.GeneralPath. The setBounds
53   * method works by scaling the path to fit into the specified bounds. This
54   * normally works well, but if the specified base bounds get too small then it
55   * is impossible to expand the path shape again since all its numbers have
56   * tended to zero, so application code may need to take this into consideration.
57   * <P>
58   * One option that applications have is to call <code>startResizeBounds</code>
59   * before starting an interaction that may make the bounds very small, and
60   * calling <code>endResizeBounds</code> when this interaction is finished. When
61   * this is done PPath will use a copy of the original path to do the resizing so
62   * the numbers in the path wont loose resolution.
63   * <P>
64   * This class also provides methods for constructing common shapes using a
65   * general path.
66   * <P>
67   * 
68   * @version 1.0
69   * @author Jesse Grosjean
70   */
71  public class PSWTPath extends PNode {

159     /**
160      * Set the bounds of this path. This method works by scaling the path to fit
161      * into the specified bounds. This normally works well, but if the specified
162      * base bounds get too small then it is impossible to expand the path shape
163      * again since all its numbers have tended to zero, so application code may
164      * need to take this into consideration.
165      */
166     protected void internalUpdateBounds(double x, double y, double width, double height) {
167         if (updatingBoundsFromPath)
168             return;
169         if (origShape == null)
170             return;
171 
172         Rectangle2D pathBounds = origShape.getBounds2D();
173 
174         if (Math.abs(x - pathBounds.getX()) / x < BOUNDS_TOLERANCE
175                 && Math.abs(y - pathBounds.getY()) / y < BOUNDS_TOLERANCE
176                 && Math.abs(width - pathBounds.getWidth()) / width < BOUNDS_TOLERANCE
177                 && Math.abs(height - pathBounds.getHeight()) / height < BOUNDS_TOLERANCE) {
178             return;
179         }
180 
181         if (internalXForm == null) {
182             internalXForm = new PAffineTransform();
183         }
184         internalXForm.setToIdentity();
185         internalXForm.translate(x, y);
186         internalXForm.scale(width / pathBounds.getWidth(), height / pathBounds.getHeight());
187         internalXForm.translate(-pathBounds.getX(), -pathBounds.getY());
188 
189         try {
190             inverseXForm = internalXForm.createInverse();
191         }
192         catch (Exception e) {
193         }
194     }

196     public boolean intersects(Rectangle2D aBounds) {
197         if (super.intersects(aBounds)) {
198 
199             if (internalXForm != null) {
200                 aBounds = new PBounds(aBounds);
201                 internalXForm.inverseTransform(aBounds, aBounds);
202             }
203 
204             if (getPaint() != null && shape.intersects(aBounds)) {
205                 return true;
206             }
207             else if (strokePaint != null) {
208                 return BASIC_STROKE.createStrokedShape(shape).intersects(aBounds);
209             }
210         }
211         return false;
212     }

214     public void updateBoundsFromPath() {
215         updatingBoundsFromPath = true;
216 
217         if (origShape == null) {
218             resetBounds();
219         }
220         else {
221             Rectangle2D b = origShape.getBounds2D();
222 
223             // Note that this pen width code does not really work for SWT since
224             // it assumes
225             // that the pen width scales - in actuality it does not. However,
226             // the fix would
227             // be to have volatile bounds for all shapes which isn't a nice
228             // alternative
229             super.setBounds(b.getX() - PEN_WIDTH, b.getY() - PEN_WIDTH, b.getWidth() + 2 * PEN_WIDTH, b.getHeight() + 2
230                     * PEN_WIDTH);
231         }
232         updatingBoundsFromPath = false;
233     }

239     protected void paint(PPaintContext paintContext) {
240         Paint p = getPaint();
241         SWTGraphics2D g2 = (SWTGraphics2D) paintContext.getGraphics();
242 
243         if (internalXForm != null) {
244             g2.transform(internalXForm);
245         }
246 
247         if (p != null) {
248             g2.setBackground((Color) p);
249 
250             double lineWidth = g2.getTransformedLineWidth();
251             if (shape instanceof Rectangle2D) {
252                 g2.fillRect(shapePts[0] + lineWidth / 2, shapePts[1] + lineWidth / 2, shapePts[2] - lineWidth,
253                         shapePts[3] - lineWidth);
254             }
255             else if (shape instanceof Ellipse2D) {
256                 g2.fillOval(shapePts[0] + lineWidth / 2, shapePts[1] + lineWidth / 2, shapePts[2] - lineWidth,
257                         shapePts[3] - lineWidth);
258             }
259             else if (shape instanceof Arc2D) {
260                 g2.fillArc(shapePts[0] + lineWidth / 2, shapePts[1] + lineWidth / 2, shapePts[2] - lineWidth,
261                         shapePts[3] - lineWidth, shapePts[4], shapePts[5]);
262             }
263             else if (shape instanceof RoundRectangle2D) {
264                 g2.fillRoundRect(shapePts[0] + lineWidth / 2, shapePts[1] + lineWidth / 2, shapePts[2] - lineWidth,
265                         shapePts[3] - lineWidth, shapePts[4], shapePts[5]);
266             }
267             else {
268                 g2.fillPolygon(shapePts);
269             }
270         }
271 
272         if (strokePaint != null) {
273             g2.setColor((Color) strokePaint);
274 
275             double lineWidth = g2.getTransformedLineWidth();
276             if (shape instanceof Rectangle2D) {
277                 g2.drawRect(shapePts[0] + lineWidth / 2, shapePts[1] + lineWidth / 2, shapePts[2] - lineWidth,
278                         shapePts[3] - lineWidth);
279             }
280             else if (shape instanceof Ellipse2D) {
281                 g2.drawOval(shapePts[0] + lineWidth / 2, shapePts[1] + lineWidth / 2, shapePts[2] - lineWidth,
282                         shapePts[3] - lineWidth);
283             }
284             else if (shape instanceof Arc2D) {
285                 g2.drawArc(shapePts[0] + lineWidth / 2, shapePts[1] + lineWidth / 2, shapePts[2] - lineWidth,
286                         shapePts[3] - lineWidth, shapePts[4], shapePts[5]);
287             }
288             else if (shape instanceof RoundRectangle2D) {
289                 g2.drawRoundRect(shapePts[0] + lineWidth / 2, shapePts[1] + lineWidth / 2, shapePts[2] - lineWidth,
290                         shapePts[3] - lineWidth, shapePts[4], shapePts[5]);
291             }
292             else {
293                 // TODO The bounds may be incorrect for polylines at the moment
294                 // - resulting in graphics turds at some scales
295                 g2.drawPolyline(shapePts);
296             }
297         }
298 
299         if (inverseXForm != null) {
300             g2.transform(inverseXForm);
301         }
302     }

304     public void setShape(Shape aShape) {
305         this.shape = cloneShape(aShape);
306         this.origShape = shape;
307         updateShapePoints(aShape);
308 
309         firePropertyChange(PPath.PROPERTY_CODE_PATH, PPath.PROPERTY_PATH, null, shape);
310         updateBoundsFromPath();
311         invalidatePaint();
312     }

314     public void updateShapePoints(Shape aShape) {
315         if (aShape instanceof Rectangle2D) {
316             if (shapePts == null || shapePts.length < 4) {
317                 shapePts = new double[4];
318             }
319 
320             shapePts[0] = ((Rectangle2D) shape).getX();
321             shapePts[1] = ((Rectangle2D) shape).getY();
322             shapePts[2] = ((Rectangle2D) shape).getWidth();
323             shapePts[3] = ((Rectangle2D) shape).getHeight();
324         }
325         else if (aShape instanceof Ellipse2D) {
326             if (shapePts == null || shapePts.length < 4) {
327                 shapePts = new double[4];
328             }
329 
330             shapePts[0] = ((Ellipse2D) shape).getX();
331             shapePts[1] = ((Ellipse2D) shape).getY();
332             shapePts[2] = ((Ellipse2D) shape).getWidth();
333             shapePts[3] = ((Ellipse2D) shape).getHeight();
334         }
335         else if (aShape instanceof Arc2D) {
336             if (shapePts == null || shapePts.length < 6) {
337                 shapePts = new double[6];
338             }
339 
340             shapePts[0] = ((Arc2D) shape).getX();
341             shapePts[1] = ((Arc2D) shape).getY();
342             shapePts[2] = ((Arc2D) shape).getWidth();
343             shapePts[3] = ((Arc2D) shape).getHeight();
344             shapePts[4] = ((Arc2D) shape).getAngleStart();
345             shapePts[5] = ((Arc2D) shape).getAngleExtent();
346         }
347         else if (aShape instanceof RoundRectangle2D) {
348             if (shapePts == null || shapePts.length < 6) {
349                 shapePts = new double[6];
350             }
351 
352             shapePts[0] = ((RoundRectangle2D) shape).getX();
353             shapePts[1] = ((RoundRectangle2D) shape).getY();
354             shapePts[2] = ((RoundRectangle2D) shape).getWidth();
355             shapePts[3] = ((RoundRectangle2D) shape).getHeight();
356             shapePts[4] = ((RoundRectangle2D) shape).getArcWidth();
357             shapePts[5] = ((RoundRectangle2D) shape).getArcHeight();
358         }
359         else {
360             shapePts = SWTShapeManager.shapeToPolyline(shape);
361         }
362     }

364     public Shape cloneShape(Shape aShape) {
365         if (aShape instanceof Rectangle2D) {
366             return new PBounds((Rectangle2D) aShape);
367         }
368         else if (aShape instanceof Ellipse2D) {
369             Ellipse2D e2 = (Ellipse2D) aShape;
370             return new Ellipse2D.Double(e2.getX(), e2.getY(), e2.getWidth(), e2.getHeight());
371         }
372         else if (aShape instanceof Arc2D) {
373             Arc2D a2 = (Arc2D) aShape;
374             return new Arc2D.Double(a2.getX(), a2.getY(), a2.getWidth(), a2.getHeight(), a2.getAngleStart(), a2
375                     .getAngleExtent(), a2.getArcType());
376         }
377         else if (aShape instanceof RoundRectangle2D) {
378             RoundRectangle2D r2 = (RoundRectangle2D) aShape;
379             return new RoundRectangle2D.Double(r2.getX(), r2.getY(), r2.getWidth(), r2.getHeight(), r2.getArcWidth(),
380                     r2.getArcHeight());
381         }
382         else if (aShape instanceof Line2D) {
383             Line2D l2 = (Line2D) aShape;
384             return new Line2D.Double(l2.getP1(), l2.getP2());
385         }
386         else {
387             new Exception().printStackTrace();
388             GeneralPath aPath = new GeneralPath();
389             aPath.append(aShape, false);
390             return aPath;
391         }
392     }

409     public void setPathToPolyline(Point2D[] points) {
410         GeneralPath path = new GeneralPath();
411         path.reset();
412         path.moveTo((float) points[0].getX(), (float) points[0].getY());
413         for (int i = 1; i < points.length; i++) {
414             path.lineTo((float) points[i].getX(), (float) points[i].getY());
415         }
416         setShape(path);
417     }
