84      public boolean setBounds(double x, double y, double width, double height) {
85          if (line == null || !super.setBounds(x, y, width, height)) {
86              return false;
87          }
88  
89          Rectangle2D lineBounds = line.getBounds2D();
90          Rectangle2D lineStrokeBounds = getLineBoundsWithStroke();
91          double strokeOutset = Math.max(lineStrokeBounds.getWidth() - lineBounds.getWidth(), lineStrokeBounds
92                  .getHeight()
93                  - lineBounds.getHeight());
94  
95          x += strokeOutset / 2;
96          y += strokeOutset / 2;
97          width -= strokeOutset;
98          height -= strokeOutset;
99  
100         TEMP_TRANSFORM.setToIdentity();
101         TEMP_TRANSFORM.translate(x, y);
102         TEMP_TRANSFORM.scale(width / lineBounds.getWidth(), height / lineBounds.getHeight());
103         TEMP_TRANSFORM.translate(-lineBounds.getX(), -lineBounds.getY());
104         line.transformPoints(TEMP_TRANSFORM);
105 
106         return true;
107     }

109     public boolean intersects(Rectangle2D aBounds) {
110         if (super.intersects(aBounds)) {
111             if (line.intersects(aBounds)) {
112                 return true;
113             }
114             else if (stroke != null && strokePaint != null) {
115                 return stroke.createStrokedShape(line).intersects(aBounds);
116             }
117         }
118         return false;
119     }
