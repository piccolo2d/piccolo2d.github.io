30  package edu.umd.cs.piccolo.event;
31  
32  import java.awt.event.InputEvent;
33  
34  import edu.umd.cs.piccolo.PNode;
35  import edu.umd.cs.piccolo.util.PDimension;
36  
37  /**
38   * <b>PDragEventHandler</b> is a simple event handler for dragging a
39   * node on the canvas.
40   * <P>
41   * @version 1.0
42   * @author Jesse Grosjean
43   */
44  public class PDragEventHandler extends PDragSequenceEventHandler {
45  
46  	private PNode draggedNode;
47  	private boolean moveToFrontOnPress = false;
48  	
49  	public PDragEventHandler() {
50  		super();
51  		setEventFilter(new PInputEventFilter(InputEvent.BUTTON1_MASK));
52  	}
53  	
54  	protected PNode getDraggedNode() {
55  		return draggedNode;
56  	}
57  	
58  	protected void setDraggedNode(PNode draggedNode) {
59  		this.draggedNode = draggedNode;
60  	}
61  		
62  	protected boolean shouldStartDragInteraction(PInputEvent event) {
63  		if (super.shouldStartDragInteraction(event)) {
64  			return event.getPickedNode() != event.getTopCamera();
65  		}
66  		return false;
67  	}
68  
69  	protected void startDrag(PInputEvent event) {
70  		super.startDrag(event); 	
71  		draggedNode = event.getPickedNode();
72  		if (moveToFrontOnPress) {
73  			draggedNode.moveToFront();
74  		}
75  	}
76  
77  	protected void drag(PInputEvent event) {
78  		super.drag(event);
79  		PDimension d = event.getDeltaRelativeTo(draggedNode);		
80  		draggedNode.localToParent(d);
81  		draggedNode.offset(d.getWidth(), d.getHeight());
82  	}
83  
84  	protected void endDrag(PInputEvent event) {
85  		super.endDrag(event);
86  		draggedNode = null;
87  	}	
88  
89  	public boolean getMoveToFrontOnPress() {
90  		return moveToFrontOnPress;
91  	}
92  
93  	public void setMoveToFrontOnPress(boolean moveToFrontOnPress) {
94  		this.moveToFrontOnPress = moveToFrontOnPress;
95  	}
96  	
97  	//****************************************************************
98  	// Debugging - methods for debugging
99  	//****************************************************************
100 
101 	/**
102 	 * Returns a string representing the state of this node. This method is
103 	 * intended to be used only for debugging purposes, and the content and
104 	 * format of the returned string may vary between implementations. The
105 	 * returned string may be empty but may not be <code>null</code>.
106 	 *
107 	 * @return  a string representation of this node's state
108 	 */
109 	protected String paramString() {
110 		StringBuffer result = new StringBuffer();
111 
112 		result.append("draggedNode=" + draggedNode == null ? "null" : draggedNode.toString());
113 		if (moveToFrontOnPress) result.append(",moveToFrontOnPress");
114 		result.append(',');
115 		result.append(super.paramString());
116 
117 		return result.toString();
118 	}
