33  	public void initialize() {	
34  		PCanvas canvas = getCanvas();
35  		
36  		canvas.removeInputEventListener(canvas.getPanEventHandler());
37  		canvas.addInputEventListener(new PDragEventHandler());
38  		
39  		PNode layer = canvas.getLayer();
40  		
41  		node1 = PPath.createEllipse(0, 0, 100, 100);
42  		node2 = PPath.createEllipse(0, 0, 100, 100);
43  		link = PPath.createLine(50, 50, 50, 50);
44  		link.setPickable(false);
45  		layer.addChild(node1);
46  		layer.addChild(node2);
47  		layer.addChild(link);
48  		
49  		node2.translate(200, 200);
50  		
51  		node1.addPropertyChangeListener(PNode.PROPERTY_FULL_BOUNDS, new PropertyChangeListener() {
52  			public void propertyChange(PropertyChangeEvent arg0) {
53  				updateLink();
54  			}
55  		});
56  		
57  		node2.addPropertyChangeListener(PNode.PROPERTY_FULL_BOUNDS, new PropertyChangeListener() {
58  			public void propertyChange(PropertyChangeEvent arg0) {
59  				updateLink();
60  			}
61  		});
62  	}
63  
64  	public void updateLink() {
65  		Point2D p1 = node1.getFullBoundsReference().getCenter2D();
66  		Point2D p2 = node2.getFullBoundsReference().getCenter2D();
67  		Line2D line = new Line2D.Double(p1.getX(), p1.getY(), p2.getX(), p2.getY());
68  		link.setPathTo(line);
69  	}
