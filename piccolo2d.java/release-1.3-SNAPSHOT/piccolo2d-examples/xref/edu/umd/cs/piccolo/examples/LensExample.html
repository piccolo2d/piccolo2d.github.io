21  /**
22   * This example shows one way to create and use lens's in Piccolo.
23   */
24  public class LensExample extends PFrame {
25  
26  	public LensExample() {
27  		this(null);
28  	}
29  	
30  	public LensExample(PCanvas aCanvas) {
31  		super("LensExample", false, aCanvas);
32  	}
33  	
34  	public void initialize() {
35  		PRoot root = getCanvas().getRoot();
36  		PCamera camera = getCanvas().getCamera();		
37  		PLayer mainLayer = getCanvas().getLayer();	// viewed by the PCanvas camera, the lens is added to this layer.
38  		PLayer sharedLayer = new PLayer();			// viewed by both the lens camera and the PCanvas camera
39  		final PLayer lensOnlyLayer = new PLayer();	// viewed by only the lens camera
40  		
41  		root.addChild(lensOnlyLayer);
42  		root.addChild(sharedLayer);
43  		camera.addLayer(0, sharedLayer);
44  		
45  		final PLens lens = new PLens();
46  		lens.setBounds(10, 10, 100, 130);
47  		lens.addLayer(0, lensOnlyLayer);		
48  		lens.addLayer(1, sharedLayer);
49  		mainLayer.addChild(lens);
50  		PBoundsHandle.addBoundsHandlesTo(lens);
51  		
52  		// Create an event handler that draws squiggles on the first layer of the bottom
53  		// most camera.
54  		PDragSequenceEventHandler squiggleEventHandler = new PDragSequenceEventHandler() {
55  			protected PPath squiggle;
56  			public void startDrag(PInputEvent e) {
57  				super.startDrag(e);
58  				Point2D p = e.getPosition();
59  				squiggle = new PPath();
60  				squiggle.moveTo((float)p.getX(), (float)p.getY());
61  				
62  				// add squiggles to the first layer of the bottom camera. In the case of the
63  				// lens these squiggles will be added to the layer that is only visible by the lens,
64  				// In the case of the canvas camera the squiggles will be added to the shared layer
65  				// viewed by both the canvas camera and the lens.
66  				e.getCamera().getLayer(0).addChild(squiggle);
67  			}
68  			
69  			public void drag(PInputEvent e) {
70  				super.drag(e);				
71  				updateSquiggle(e);
72  			}
73  		
74  			public void endDrag(PInputEvent e) {
75  				super.endDrag(e);
76  				updateSquiggle(e);
77  				squiggle = null;
78  			}	
79  				
80  			public void updateSquiggle(PInputEvent aEvent) {
81  				Point2D p = aEvent.getPosition();
82  				squiggle.lineTo((float)p.getX(), (float)p.getY());
83  			}
84  		};
85  		
86  		// add the squiggle event handler to both the lens and the
87  		// canvas camera.
88  		lens.getCamera().addInputEventListener(squiggleEventHandler);
89  		camera.addInputEventListener(squiggleEventHandler);
90  
91  		// make sure that the event handler consumes events so that it doesn't
92  		// conflic with other event handlers or with itself (since its added to two
93  		// event sources).
94  		squiggleEventHandler.getEventFilter().setMarksAcceptedEventsAsHandled(true);
95  		
96  		// remove default event handlers, not really nessessary since the squiggleEventHandler
97  		// consumes everything anyway, but still good to do.
98  		getCanvas().removeInputEventListener(getCanvas().getPanEventHandler());
99  		getCanvas().removeInputEventListener(getCanvas().getZoomEventHandler());
100 		
101 		// create a node that is viewed both by the main camera and by the
102 		// lens. Note that in its paint method it checks to see which camera
103 		// is painting it, and if its the lens uses a different color.
104 		PNode sharedNode = new PNode() {
105 			protected void paint(PPaintContext paintContext) {
106 				if (paintContext.getCamera() == lens.getCamera()) {
107 					Graphics2D g2 = paintContext.getGraphics();
108 					g2.setPaint(Color.RED);
109 					g2.fill(getBoundsReference());
110 				} else {
111 					super.paint(paintContext);
112 				}
113 			}
114 		};		
115 		sharedNode.setPaint(Color.GREEN);
116 		sharedNode.setBounds(0, 0, 100, 200);
117 		sharedNode.translate(200, 200);
118 		sharedLayer.addChild(sharedNode);
119 		
120 		PText label = new PText("Move the lens \n (by dragging title bar) over the green rectangle, and it will appear red. press and drag the mouse on the canvas and it will draw squiggles. press and drag the mouse over the lens and drag squiggles that are only visible through the lens.");
121 		label.setConstrainWidthToTextWidth(false);
122 		label.setConstrainHeightToTextHeight(false);
123 		label.setBounds(200, 100, 200, 200);
124 			
125 		sharedLayer.addChild(label);				
126 	}
127 		
128 	public static void main(String[] args) {
129 		new LensExample();
130 	}
