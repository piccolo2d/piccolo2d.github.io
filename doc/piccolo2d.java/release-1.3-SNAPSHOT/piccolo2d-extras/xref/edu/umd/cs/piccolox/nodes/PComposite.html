35  /**
36   * <b>PComposite</b> is a simple node that makes a group of nodes appear to be a
37   * single node when picking and interacting. There is also partial (commented
38   * out) support for resizing the child node to fit when this nodes bounds are
39   * set.
40   * 
41   * @version 1.0
42   * @author Jesse Grosjean
43   */
44  public class PComposite extends PNode {
45  
46      /*
47       * public boolean setBounds(double x, double y, double width, double height)
48       * { PBounds childBounds = getUnionOfChildrenBounds(null);
49       * 
50       * double dx = x - childBounds.x; double dy = y - childBounds.y; double sx =
51       * width / childBounds.width; double sy = height / childBounds.height;
52       * double scale = sx > sy ? sx : sy;
53       * 
54       * Iterator i = getChildrenIterator(); while (i.hasNext()) { PNode each =
55       * (PNode) i.next(); each.offset(dx, dy); each.scaleAboutPoint(scale,
56       * each.getBoundsReference().x, each.getBoundsReference().y); }
57       * 
58       * return super.setBounds(x, y, width, height); }
59       * 
60       * protected void layoutChildren() {
61       * getBoundsReference().setRect(getUnionOfChildrenBounds(null)); }
62       */
63  
64      /**
65       * Return true if this node or any pickable descendends are picked. If a
66       * pick occurs the pickPath is modified so that this node is always returned
67       * as the picked node, event if it was a decendent node that initialy
68       * reported the pick.
69       */
70      public boolean fullPick(PPickPath pickPath) {
71          if (super.fullPick(pickPath)) {
72              PNode picked = pickPath.getPickedNode();
73  
74              // this code won't work with internal cameras, because it doesn't
75              // pop the cameras view transform.
76              while (picked != this) {
77                  pickPath.popTransform(picked.getTransformReference(false));
78                  pickPath.popNode(picked);
79                  picked = pickPath.getPickedNode();
80              }
81  
82              return true;
83          }
84          return false;
85      }
