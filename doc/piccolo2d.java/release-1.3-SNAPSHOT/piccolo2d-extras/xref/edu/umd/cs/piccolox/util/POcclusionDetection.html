32  import edu.umd.cs.piccolo.PNode;
33  import edu.umd.cs.piccolo.util.PBounds;
34  import edu.umd.cs.piccolo.util.PPickPath;

36  /**
37   * Experimental class for detecting occlusions.
38   * 
39   * @author Jesse Grosjean
40   */
41  public class POcclusionDetection {
42  
43      /**
44       * Traverse from the bottom right of the scene graph (top visible node) up
45       * the tree determining which parent nodes are occluded by their children
46       * nodes. Note that this is only detecting a subset of occlusions (parent,
47       * child), others such as overlapping siblings or cousins are not detected.
48       */
49      public void detectOccusions(PNode n, PBounds parentBounds) {
50          detectOcclusions(n, new PPickPath(null, parentBounds));
51      }
52  
53      public void detectOcclusions(PNode n, PPickPath pickPath) {
54          if (n.fullIntersects(pickPath.getPickBounds())) {
55              pickPath.pushTransform(n.getTransformReference(false));
56  
57              int count = n.getChildrenCount();
58              for (int i = count - 1; i >= 0; i--) {
59                  PNode each = (PNode) n.getChild(i);
60                  if (n.getOccluded()) {
61                      // if n has been occuded by a previous decendent then
62                      // this child must also be occuded
63                      each.setOccluded(true);
64                  }
65                  else {
66                      // see if child each occludes n
67                      detectOcclusions(each, pickPath);
68                  }
69              }
70  
71              // see if n occudes it's parents
72              if (!n.getOccluded()) {
73                  if (n.intersects(pickPath.getPickBounds())) {
74                      if (n.isOpaque(pickPath.getPickBounds())) {
75                          PNode p = n.getParent();
76                          while (p != null && !p.getOccluded()) {
77                              p.setOccluded(true);
78                          }
79                      }
80                  }
81              }
82  
83              pickPath.popTransform(n.getTransformReference(false));
84          }
85      }
86  }
