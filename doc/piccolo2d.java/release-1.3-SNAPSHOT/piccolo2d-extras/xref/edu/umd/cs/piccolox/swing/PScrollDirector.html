16  /**
17   * The interface an application can implement to control scrolling in a
18   * PScrollPane->PViewport->ZCanvas component hierarchy.
19   * 
20   * @see PDefaultScrollDirector
21   * @author Lance Good
22   */
23  public interface PScrollDirector {
24  
25      /**
26       * Installs the scroll director
27       * 
28       * @param viewport The viewport on which this director directs
29       * @param view The ZCanvas that the viewport looks at
30       */
31      public void install(PViewport viewport, PCanvas view);
32  
33      /**
34       * Uninstall the scroll director
35       */
36      public void unInstall();
37  
38      /**
39       * Get the View position given the specified camera bounds
40       * 
41       * @param viewBounds The bounds for which the view position will be computed
42       * @return The view position
43       */
44      public Point getViewPosition(Rectangle2D viewBounds);
45  
46      /**
47       * Set the view position
48       * 
49       * @param x The new x position
50       * @param y The new y position
51       */
52      public void setViewPosition(double x, double y);
53  
54      /**
55       * Get the size of the view based on the specified camera bounds
56       * 
57       * @param viewBounds The view bounds for which the view size will be
58       *            computed
59       * @return The view size
60       */
61      public Dimension getViewSize(Rectangle2D viewBounds);
62  }
