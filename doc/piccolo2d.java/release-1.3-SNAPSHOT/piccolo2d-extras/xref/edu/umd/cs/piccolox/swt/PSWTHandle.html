52  /**
53   * <b>PHandle</b> is used to modify some aspect of Piccolo when it is dragged.
54   * Each handle has a PLocator that it uses to automatically position itself. See
55   * PBoundsHandle for an example of a handle that resizes the bounds of another
56   * node.
57   * <P>
58   * 
59   * @version 1.0
60   * @author Jesse Grosjean
61   */
62  public class PSWTHandle extends PSWTPath {
63  
64      public static float DEFAULT_HANDLE_SIZE = 8;
65      public static Shape DEFAULT_HANDLE_SHAPE = new Ellipse2D.Float(0f, 0f, DEFAULT_HANDLE_SIZE, DEFAULT_HANDLE_SIZE);
66      public static Color DEFAULT_COLOR = Color.white;
67  
68      private PLocator locator;
69      private PDragSequenceEventHandler handleDragger;
70  
71      /**
72       * Construct a new handle that will use the given locator to locate itself
73       * on its parent node.
74       */
75      public PSWTHandle(PLocator aLocator) {
76          super(DEFAULT_HANDLE_SHAPE);
77          locator = aLocator;
78          setPaint(DEFAULT_COLOR);
79          installHandleEventHandlers();
80      }
81  
82      protected void installHandleEventHandlers() {
83          handleDragger = new PDragSequenceEventHandler() {
84              protected void startDrag(PInputEvent event) {
85                  super.startDrag(event);
86                  startHandleDrag(event.getPositionRelativeTo(PSWTHandle.this), event);
87              }
88  
89              protected void drag(PInputEvent event) {
90                  super.drag(event);
91                  PDimension aDelta = event.getDeltaRelativeTo(PSWTHandle.this);
92                  if (aDelta.getWidth() != 0 || aDelta.getHeight() != 0) {
93                      dragHandle(aDelta, event);
94                  }
95              }
96  
97              protected void endDrag(PInputEvent event) {
98                  super.endDrag(event);
99                  endHandleDrag(event.getPositionRelativeTo(PSWTHandle.this), event);
100             }
101         };
102 
103         addPropertyChangeListener(PNode.PROPERTY_TRANSFORM, new PropertyChangeListener() {
104             public void propertyChange(PropertyChangeEvent evt) {
105                 relocateHandle();
106             }
107         });
108 
109         handleDragger.setEventFilter(new PInputEventFilter(InputEvent.BUTTON1_MASK));
110         handleDragger.getEventFilter().setMarksAcceptedEventsAsHandled(true);
111         handleDragger.getEventFilter().setAcceptsMouseEntered(false);
112         handleDragger.getEventFilter().setAcceptsMouseExited(false);
113         // no need for moved events for handle interaction,
114         handleDragger.getEventFilter().setAcceptsMouseMoved(false);
115         // so reject them so we don't consume them
116         addInputEventListener(handleDragger);
117     }
118 
119     /**
120      * Return the event handler that is responsible for the drag handle
121      * interaction.
122      */
123     public PDragSequenceEventHandler getHandleDraggerHandler() {
124         return handleDragger;
125     }

128      * Get the locator that this handle uses to position itself on its parent

170     // ****************************************************************
171     // Layout - When a handle's parent's layout changes the handle
172     // invalidates its own layout and then repositions itself on its
173     // parents bounds using its locator to determine that new

175     // ****************************************************************
176 
177     public void setParent(PNode newParent) {
178         super.setParent(newParent);
179         relocateHandle();
180     }
181 
182     public void parentBoundsChanged() {
183         relocateHandle();
184     }
185 
186     /**
187      * Force this handle to relocate itself using its locator.
188      */
189     public void relocateHandle() {
190         if (locator != null) {
191             PBounds b = getBoundsReference();
192             Point2D aPoint = locator.locatePoint(null);
193 
194             if (locator instanceof PNodeLocator) {
195                 PNode located = ((PNodeLocator) locator).getNode();
196                 PNode parent = getParent();
197 
198                 located.localToGlobal(aPoint);
199                 globalToLocal(aPoint);
200 
201                 if (parent != located && parent instanceof PCamera) {
202                     ((PCamera) parent).viewToLocal(aPoint);
203                 }
204             }
205 
206             double newCenterX = aPoint.getX();
207             double newCenterY = aPoint.getY();
208 
209             if (newCenterX != b.getCenterX() || newCenterY != b.getCenterY()) {
210                 centerBoundsOnPoint(newCenterX, newCenterY);
211             }
212         }
213     }
