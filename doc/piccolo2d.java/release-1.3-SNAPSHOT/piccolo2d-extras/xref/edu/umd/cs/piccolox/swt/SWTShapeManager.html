22      /**
23       * Apply the specified transform to the specified rectangle, modifying the
24       * rect.
25       * 
26       * @param rect The rectangle to be transformed
27       * @param at The transform to use to transform the rectangle
28       */
29      public static void transform(Rectangle2D rect, AffineTransform at) {
30          // First, transform all 4 corners of the rectangle
31          pts[0] = rect.getX(); // top left corner
32          pts[1] = rect.getY();
33          pts[2] = rect.getX() + rect.getWidth(); // top right corner
34          pts[3] = rect.getY();
35          pts[4] = rect.getX() + rect.getWidth(); // bottom right corner
36          pts[5] = rect.getY() + rect.getHeight();
37          pts[6] = rect.getX(); // bottom left corner
38          pts[7] = rect.getY() + rect.getHeight();
39          at.transform(pts, 0, pts, 0, 4);
40  
41          // Then, find the bounds of those 4 transformed points.
42          double minX = pts[0];
43          double minY = pts[1];
44          double maxX = pts[0];
45          double maxY = pts[1];
46          int i;
47          for (i = 1; i < 4; i++) {
48              if (pts[2 * i] < minX) {
49                  minX = pts[2 * i];
50              }
51              if (pts[2 * i + 1] < minY) {
52                  minY = pts[2 * i + 1];
53              }
54              if (pts[2 * i] > maxX) {
55                  maxX = pts[2 * i];
56              }
57              if (pts[2 * i + 1] > maxY) {
58                  maxY = pts[2 * i + 1];
59              }
60          }
61          rect.setRect(minX, minY, maxX - minX, maxY - minY);
62      }

71      public static double[] shapeToPolyline(Shape s) {
72          segList.clear();
73          aPoint.setLocation(0, 0);
74  
75          PathIterator pi = s.getPathIterator(IDENTITY_XFORM, 0.000000001);
76          while (!pi.isDone()) {
77              int segType = pi.currentSegment(pts);
78              switch (segType) {
79                  case PathIterator.SEG_MOVETO:
80                      aPoint.setLocation(pts[0], pts[1]);
81                      segList.add(new Point2D.Double(pts[0], pts[1]));
82                      break;
83                  case PathIterator.SEG_LINETO:
84                      segList.add(new Point2D.Double(pts[0], pts[1]));
85                      break;
86                  case PathIterator.SEG_CLOSE:
87                      segList.add(new Point2D.Double(aPoint.getX(), aPoint.getY()));
88                      break;
89              }
90              pi.next();
91          }
92  
93          double[] polyObj = new double[2 * segList.size()];
94          for (int i = 0; i < segList.size(); i++) {
95              Point2D p2 = (Point2D) segList.get(i);
96              polyObj[2 * i] = (int) (p2.getX() + 0.5);
97              polyObj[2 * i + 1] = (int) (p2.getY() + 0.5);
98          }
99  
100         return polyObj;
101     }
