15  /**
16   * This example shows how to set up interpolating activities that repeat. For
17   * example it shows how to create a rectangle whos color pulses.
18   * 
19   * @author jesse
20   */
21  public class PulseExample extends PFrame {
22  
23  	public PulseExample() {
24  		this(null);
25  	}
26  
27  	public PulseExample(PCanvas aCanvas) {
28  		super("PulseExample", false, aCanvas);
29  	}
30  
31  	public void initialize() {
32  		PRoot root = getCanvas().getRoot();
33  		PLayer layer = getCanvas().getLayer();
34  		PActivityScheduler scheduler = root.getActivityScheduler();
35  		
36  		final PNode singlePulse = PPath.createRectangle(0, 0, 100, 80);
37  		final PPath repeatePulse = PPath.createRectangle(100, 80, 100, 80);
38  		final PNode repeateReversePulse = PPath.createRectangle(200, 160, 100, 80);
39  
40  		layer.addChild(singlePulse);
41  		layer.addChild(repeatePulse);
42  		layer.addChild(repeateReversePulse);
43  		
44  		// animate from source to destination color in one second,
45  		PColorActivity singlePulseActivity = new PColorActivity(1000, 0, 1, PInterpolatingActivity.SOURCE_TO_DESTINATION, new PColorActivity.Target() {
46  			public Color getColor() {
47  				return (Color) singlePulse.getPaint();
48  			}
49  			public void setColor(Color color) {
50  				singlePulse.setPaint(color);
51  			}
52  		}, Color.ORANGE);
53  
54  		// animate from source to destination color in one second, loop 5 times
55  		PColorActivity repeatPulseActivity = new PColorActivity(1000, 0, 5, PInterpolatingActivity.SOURCE_TO_DESTINATION, new PColorActivity.Target() {
56  			public Color getColor() {
57  				return (Color) repeatePulse.getPaint();
58  			}
59  			public void setColor(Color color) {
60  				repeatePulse.setPaint(color);
61  			}
62  		}, Color.BLUE);
63  
64  		// animate from source to destination to source color in one second, loop 10 times
65  		PColorActivity repeatReversePulseActivity = new PColorActivity(500, 0, 10, PInterpolatingActivity.SOURCE_TO_DESTINATION_TO_SOURCE, new PColorActivity.Target() {
66  			public Color getColor() {
67  				return (Color) repeateReversePulse.getPaint();
68  			}
69  			public void setColor(Color color) {
70  				repeateReversePulse.setPaint(color);
71  			}
72  		}, Color.GREEN);
73  
74  		scheduler.addActivity(singlePulseActivity);
75  		scheduler.addActivity(repeatPulseActivity);
76  		scheduler.addActivity(repeatReversePulseActivity);
77  	}
