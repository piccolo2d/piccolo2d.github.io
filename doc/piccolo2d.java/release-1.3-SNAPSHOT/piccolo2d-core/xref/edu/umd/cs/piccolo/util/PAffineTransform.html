81      public double getScale() {
82          PTS1[0] = 0;// x1
83          PTS1[1] = 0;// y1
84          PTS1[2] = 1;// x2
85          PTS1[3] = 0;// y2
86          transform(PTS1, 0, PTS2, 0, 2);
87          return Point2D.distance(PTS2[0], PTS2[1], PTS2[2], PTS2[3]);
88      }

93          scaleAboutPoint(scale / getScale(), 0, 0);

100     /**
101      * Returns the rotation applied to this affine transform in radians. The
102      * value returned will be between 0 and 2pi.
103      * 
104      * @return rotation in radians
105      */
106     public double getRotation() {
107         PTS1[0] = 0;// x1
108         PTS1[1] = 0;// y1
109         PTS1[2] = 1;// x2
110         PTS1[3] = 0;// y2
111 
112         transform(PTS1, 0, PTS2, 0, 2);
113 
114         double dy = Math.abs(PTS2[3] - PTS2[1]);
115         double l = Point2D.distance(PTS2[0], PTS2[1], PTS2[2], PTS2[3]);
116         double rotation = Math.asin(dy / l);
117 
118         // correct for quadrant
119         if (PTS2[3] - PTS2[1] > 0) {
120             if (PTS2[2] - PTS2[0] < 0) {
121                 rotation = Math.PI - rotation;
122             }
123         }
124         else {
125             if (PTS2[2] - PTS2[0] > 0) {
126                 rotation = 2 * Math.PI - rotation;
127             }
128             else {
129                 rotation = rotation + Math.PI;
130             }
131         }
132 
133         return rotation;
134     }

143     public Dimension2D transform(Dimension2D dimSrc, Dimension2D dimDst) {
144         if (dimDst == null) {
145             dimDst = (Dimension2D) dimSrc.clone();
146         }
147 
148         PTS1[0] = dimSrc.getWidth();
149         PTS1[1] = dimSrc.getHeight();
150         deltaTransform(PTS1, 0, PTS2, 0, 1);
151         dimDst.setSize(PTS2[0], PTS2[1]);
152         return dimDst;
153     }

155     public Dimension2D inverseTransform(Dimension2D dimSrc, Dimension2D dimDst) {
156         if (dimDst == null) {
157             dimDst = (Dimension2D) dimSrc.clone();
158         }
159 
160         double width = dimSrc.getWidth();
161         double height = dimSrc.getHeight();
162         double m00 = getScaleX();
163         double m11 = getScaleY();
164         double m01 = getShearX();
165         double m10 = getShearY();
166         double det = m00 * m11 - m01 * m10;
167 
168         try {
169             if (Math.abs(det) <= Double.MIN_VALUE) {
170                 throw new NoninvertibleTransformException("Determinant is " + det);
171             }
172             dimDst.setSize((width * m11 - height * m01) / det, (height * m00 - width * m10) / det);
173         }
174         catch (NoninvertibleTransformException e) {
175             e.printStackTrace();
176         }
177 
178         return dimDst;
179     }

181     public Rectangle2D transform(Rectangle2D rectSrc, Rectangle2D rectDst) {
182         if (rectDst == null) {
183             rectDst = (Rectangle2D) rectSrc.clone();
184         }
185 
186         if (rectSrc.isEmpty()) {
187             rectDst.setRect(rectSrc);
188             if (rectDst instanceof PBounds) {
189                 ((PBounds) rectDst).reset();
190             }
191             return rectDst;
192         }
193 
194         double scale;
195 
196         switch (getType()) {
197             case AffineTransform.TYPE_IDENTITY:
198                 if (rectSrc != rectDst)
199                     rectDst.setRect(rectSrc);
200                 break;
201 
202             case AffineTransform.TYPE_TRANSLATION:
203                 rectDst.setRect(rectSrc.getX() + getTranslateX(), rectSrc.getY() + getTranslateY(), rectSrc.getWidth(),
204                         rectSrc.getHeight());
205                 break;
206 
207             case AffineTransform.TYPE_UNIFORM_SCALE:
208                 scale = getScaleX();
209                 rectDst.setRect(rectSrc.getX() * scale, rectSrc.getY() * scale, rectSrc.getWidth() * scale, rectSrc
210                         .getHeight()
211                         * scale);
212                 break;
213 
214             case AffineTransform.TYPE_TRANSLATION | AffineTransform.TYPE_UNIFORM_SCALE:
215                 scale = getScaleX();
216                 rectDst.setRect((rectSrc.getX() * scale) + getTranslateX(), (rectSrc.getY() * scale) + getTranslateY(),
217                         rectSrc.getWidth() * scale, rectSrc.getHeight() * scale);
218                 break;
219 
220             default:
221                 double[] pts = rectToArray(rectSrc);
222                 transform(pts, 0, pts, 0, 4);
223                 rectFromArray(rectDst, pts);
224                 break;
225         }
226 
227         return rectDst;
228     }

230     public Rectangle2D inverseTransform(Rectangle2D rectSrc, Rectangle2D rectDst) {
231         if (rectDst == null) {
232             rectDst = (Rectangle2D) rectSrc.clone();
233         }
234 
235         if (rectSrc.isEmpty()) {
236             rectDst.setRect(rectSrc);
237             if (rectDst instanceof PBounds) {
238                 ((PBounds) rectDst).reset();
239             }
240             return rectDst;
241         }
242 
243         double scale;
244 
245         switch (getType()) {
246             case AffineTransform.TYPE_IDENTITY:
247                 if (rectSrc != rectDst)
248                     rectDst.setRect(rectSrc);
249                 break;
250 
251             case AffineTransform.TYPE_TRANSLATION:
252                 rectDst.setRect(rectSrc.getX() - getTranslateX(), rectSrc.getY() - getTranslateY(), rectSrc.getWidth(),
253                         rectSrc.getHeight());
254                 break;
255 
256             case AffineTransform.TYPE_UNIFORM_SCALE:
257                 scale = 1 / getScaleX();
258                 rectDst.setRect(rectSrc.getX() * scale, rectSrc.getY() * scale, rectSrc.getWidth() * scale, rectSrc
259                         .getHeight()
260                         * scale);
261                 break;
262 
263             case AffineTransform.TYPE_TRANSLATION | AffineTransform.TYPE_UNIFORM_SCALE:
264                 scale = 1 / getScaleX();
265                 rectDst.setRect((rectSrc.getX() - getTranslateX()) * scale, (rectSrc.getY() - getTranslateY()) * scale,
266                         rectSrc.getWidth() * scale, rectSrc.getHeight() * scale);
267                 break;
268 
269             default:
270                 double[] pts = rectToArray(rectSrc);
271                 try {
272                     inverseTransform(pts, 0, pts, 0, 4);
273                 }
274                 catch (NoninvertibleTransformException e) {
275                     e.printStackTrace();
276                 }
277                 rectFromArray(rectDst, pts);
278                 break;
279         }
280 
281         return rectDst;
282     }

284     private static double[] rectToArray(Rectangle2D aRectangle) {
285         PTS1[0] = aRectangle.getX();
286         PTS1[1] = aRectangle.getY();
287         PTS1[2] = PTS1[0] + aRectangle.getWidth();
288         PTS1[3] = PTS1[1];
289         PTS1[4] = PTS1[0] + aRectangle.getWidth();
290         PTS1[5] = PTS1[1] + aRectangle.getHeight();
291         PTS1[6] = PTS1[0];
292         PTS1[7] = PTS1[1] + aRectangle.getHeight();
293         return PTS1;
294     }

296     private static void rectFromArray(Rectangle2D aRectangle, double[] pts) {
297         double minX = pts[0];
298         double minY = pts[1];
299         double maxX = pts[0];
300         double maxY = pts[1];
301 
302         double x;
303         double y;
304 
305         for (int i = 1; i < 4; i++) {
306             x = pts[2 * i];
307             y = pts[(2 * i) + 1];
308 
309             if (x < minX) {
310                 minX = x;
311             }
312             if (y < minY) {
313                 minY = y;
314             }
315             if (x > maxX) {
316                 maxX = x;
317             }
318             if (y > maxY) {
319                 maxY = y;
320             }
321         }
322         aRectangle.setRect(minX, minY, maxX - minX, maxY - minY);
323     }
