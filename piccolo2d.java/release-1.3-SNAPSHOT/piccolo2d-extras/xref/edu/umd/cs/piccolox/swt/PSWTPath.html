51  /**
52   * <b>PPath</b> is a wrapper around a java.awt.geom.GeneralPath. The
53   * setBounds method works by scaling the path to fit into the specified
54   * bounds. This normally works well, but if the specified base bounds
55   * get too small then it is impossible to expand the path shape again since
56   * all its numbers have tended to zero, so application code may need to take
57   * this into consideration. 
58   * <P>
59   * One option that applications have is to call <code>startResizeBounds</code> before
60   * starting an interaction that may make the bounds very small, and calling 
61   * <code>endResizeBounds</code> when this interaction is finished. When this is done
62   * PPath will use a copy of the original path to do the resizing so the numbers
63   * in the path wont loose resolution.
64   * <P>
65   * This class also provides methods for constructing common shapes using a 
66   * general path.
67   * <P>
68   * @version 1.0
69   * @author Jesse Grosjean
70   */

158 	/**
159 	 * Set the bounds of this path. This method works by scaling the path 
160 	 * to fit into the specified bounds. This normally works well, but if 
161 	 * the specified base bounds get too small then it is impossible to 
162 	 * expand the path shape again since all its numbers have tended to zero, 
163 	 * so application code may need to take this into consideration.
164 	 */
165 	protected void internalUpdateBounds(double x, double y, double width, double height) {
166 		if (updatingBoundsFromPath) return;
167 		if (origShape == null) return;
168 		
169 		Rectangle2D pathBounds = origShape.getBounds2D();
170 
171 		if (Math.abs(x-pathBounds.getX())/x < BOUNDS_TOLERANCE &&
172 		    Math.abs(y-pathBounds.getY())/y < BOUNDS_TOLERANCE &&
173 		    Math.abs(width-pathBounds.getWidth())/width < BOUNDS_TOLERANCE &&
174 		    Math.abs(height-pathBounds.getHeight())/height < BOUNDS_TOLERANCE) {
175 		    return;
176 		}
177 		
178 		if (internalXForm == null) {
179 		    internalXForm = new PAffineTransform();
180 		}
181 		internalXForm.setToIdentity();
182 		internalXForm.translate(x, y);
183 		internalXForm.scale(width / pathBounds.getWidth(), height / pathBounds.getHeight());
184 		internalXForm.translate(-pathBounds.getX(), -pathBounds.getY());
185 		
186 		try {
187 		    inverseXForm = internalXForm.createInverse();
188 		}
189 		catch (Exception e) {		    
190 		}
191 	}

194 	public boolean intersects(Rectangle2D aBounds) {
195 		if (super.intersects(aBounds)) {
196 		    
197 		    if (internalXForm != null) {
198 			    aBounds = new PBounds(aBounds);		 
199 			    internalXForm.inverseTransform(aBounds,aBounds);
200 		    }
201 		    
202 			if (getPaint() != null && shape.intersects(aBounds)) {
203 				return true;
204 			} else if (strokePaint != null) {
205 				return BASIC_STROKE.createStrokedShape(shape).intersects(aBounds);
206 			}
207 		}
208 		return false;
209 	}

211 	public void updateBoundsFromPath() {
212 	    updatingBoundsFromPath = true;
213 	    
214 		if (origShape == null) {
215 			resetBounds();
216 		} else {
217 			Rectangle2D b = origShape.getBounds2D();
218 			
219 			// Note that this pen width code does not really work for SWT since it assumes
220 			// that the pen width scales - in actuality it does not.  However, the fix would
221 			// be to have volatile bounds for all shapes which isn't a nice alternative
222 			super.setBounds(b.getX()-PEN_WIDTH, 
223 			        b.getY()-PEN_WIDTH, 
224 			        b.getWidth()+2*PEN_WIDTH, 
225 			        b.getHeight()+2*PEN_WIDTH);
226 		}
227 	    updatingBoundsFromPath = false;
228 	}

234 	protected void paint(PPaintContext paintContext) {
235 	    Paint p = getPaint();
236 		SWTGraphics2D g2 = (SWTGraphics2D)paintContext.getGraphics();
237 		
238 		if (internalXForm != null) {
239 		    g2.transform(internalXForm);
240 		}
241 		
242 		if (p != null) {
243 			g2.setBackground((Color)p);
244 
245 			double lineWidth = g2.getTransformedLineWidth();
246 			if (shape instanceof Rectangle2D) {
247 				g2.fillRect(shapePts[0]+lineWidth/2,shapePts[1]+lineWidth/2,shapePts[2]-lineWidth,shapePts[3]-lineWidth);
248 			}
249 			else if (shape instanceof Ellipse2D) {
250 				g2.fillOval(shapePts[0]+lineWidth/2,shapePts[1]+lineWidth/2,shapePts[2]-lineWidth,shapePts[3]-lineWidth);
251 			}
252 			else if (shape instanceof Arc2D) {
253 				g2.fillArc(shapePts[0]+lineWidth/2,shapePts[1]+lineWidth/2,shapePts[2]-lineWidth,shapePts[3]-lineWidth,shapePts[4],shapePts[5]);
254 			}
255 			else if (shape instanceof RoundRectangle2D) {
256 				g2.fillRoundRect(shapePts[0]+lineWidth/2,shapePts[1]+lineWidth/2,shapePts[2]-lineWidth,shapePts[3]-lineWidth,shapePts[4],shapePts[5]);				
257 			}
258 			else {
259 				g2.fillPolygon(shapePts);	
260 			}
261 		}
262 		
263 		if (strokePaint != null) {
264 			g2.setColor((Color)strokePaint);
265 			
266 			double lineWidth = g2.getTransformedLineWidth();			
267 			if (shape instanceof Rectangle2D) {
268 				g2.drawRect(shapePts[0]+lineWidth/2,shapePts[1]+lineWidth/2,shapePts[2]-lineWidth,shapePts[3]-lineWidth);
269 			}
270 			else if (shape instanceof Ellipse2D) {
271 				g2.drawOval(shapePts[0]+lineWidth/2,shapePts[1]+lineWidth/2,shapePts[2]-lineWidth,shapePts[3]-lineWidth);
272 			}
273 			else if (shape instanceof Arc2D) {
274 				g2.drawArc(shapePts[0]+lineWidth/2,shapePts[1]+lineWidth/2,shapePts[2]-lineWidth,shapePts[3]-lineWidth,shapePts[4],shapePts[5]);
275 			}
276 			else if (shape instanceof RoundRectangle2D) {
277 				g2.drawRoundRect(shapePts[0]+lineWidth/2,shapePts[1]+lineWidth/2,shapePts[2]-lineWidth,shapePts[3]-lineWidth,shapePts[4],shapePts[5]);				
278 			}
279 			else {
280 			    // TODO The bounds may be incorrect for polylines at the moment - resulting in graphics turds at some scales 
281 				g2.drawPolyline(shapePts);	
282 			}
283 		}
284 		
285 		if (inverseXForm != null) {		    
286 		    g2.transform(inverseXForm);
287 		}
288 	}

290 	public void setShape(Shape aShape) {	    
291 		this.shape = cloneShape(aShape);
292 		this.origShape = shape;
293 		updateShapePoints(aShape);
294 		
295 		firePropertyChange(PPath.PROPERTY_CODE_PATH, PPath.PROPERTY_PATH, null, shape);
296 		updateBoundsFromPath();
297 		invalidatePaint();		
298 	}

300 	public void updateShapePoints(Shape aShape) {
301 		if (aShape instanceof Rectangle2D) {
302 			if (shapePts == null || shapePts.length < 4) {
303 			    shapePts = new double[4];
304 			}
305 			    
306 			shapePts[0] = ((Rectangle2D)shape).getX();	
307 			shapePts[1] = ((Rectangle2D)shape).getY();	
308 			shapePts[2] = ((Rectangle2D)shape).getWidth();	
309 			shapePts[3] = ((Rectangle2D)shape).getHeight();				
310 		}
311 		else if (aShape instanceof Ellipse2D) {
312 			if (shapePts == null || shapePts.length < 4) {
313 			    shapePts = new double[4];
314 			}
315 
316 			shapePts[0] = ((Ellipse2D)shape).getX();	
317 			shapePts[1] = ((Ellipse2D)shape).getY();	
318 			shapePts[2] = ((Ellipse2D)shape).getWidth();	
319 			shapePts[3] = ((Ellipse2D)shape).getHeight();							
320 		}
321 		else if (aShape instanceof Arc2D) {
322 			if (shapePts == null || shapePts.length < 6) {
323 			    shapePts = new double[6];
324 			}
325 		    
326 			shapePts[0] = ((Arc2D)shape).getX();	
327 			shapePts[1] = ((Arc2D)shape).getY();	
328 			shapePts[2] = ((Arc2D)shape).getWidth();	
329 			shapePts[3] = ((Arc2D)shape).getHeight();
330 			shapePts[4] = ((Arc2D)shape).getAngleStart();
331 			shapePts[5] = ((Arc2D)shape).getAngleExtent();
332 		}
333 		else if (aShape instanceof RoundRectangle2D) {
334 			if (shapePts == null || shapePts.length < 6) {
335 			    shapePts = new double[6];
336 			}
337 
338 			shapePts[0] = ((RoundRectangle2D)shape).getX();	
339 			shapePts[1] = ((RoundRectangle2D)shape).getY();	
340 			shapePts[2] = ((RoundRectangle2D)shape).getWidth();	
341 			shapePts[3] = ((RoundRectangle2D)shape).getHeight();
342 			shapePts[4] = ((RoundRectangle2D)shape).getArcWidth();
343 			shapePts[5] = ((RoundRectangle2D)shape).getArcHeight();			
344 		}
345 		else {
346 			shapePts = SWTShapeManager.shapeToPolyline(shape);			
347 		}	    
348 	}

350 	public Shape cloneShape(Shape aShape) {
351 		if (aShape instanceof Rectangle2D) {
352 		    return new PBounds((Rectangle2D)aShape);
353 		}			
354 		else if (aShape instanceof Ellipse2D) {
355 		    Ellipse2D e2 = (Ellipse2D)aShape;
356 		    return new Ellipse2D.Double(e2.getX(),e2.getY(),e2.getWidth(),e2.getHeight());
357 		}
358 		else if (aShape instanceof Arc2D) {
359 		    Arc2D a2 = (Arc2D)aShape;
360 		    return new Arc2D.Double(a2.getX(),a2.getY(),a2.getWidth(),a2.getHeight(),a2.getAngleStart(),a2.getAngleExtent(),a2.getArcType());
361 		}
362 		else if (aShape instanceof RoundRectangle2D) {
363 		    RoundRectangle2D r2 = (RoundRectangle2D)aShape;
364 		    return new RoundRectangle2D.Double(r2.getX(),r2.getY(),r2.getWidth(),r2.getHeight(),r2.getArcWidth(),r2.getArcHeight());
365 		}
366 		else if (aShape instanceof Line2D) {
367 			Line2D l2 = (Line2D)aShape;
368 			return new Line2D.Double(l2.getP1(),l2.getP2());
369 		}
370 		else {
371 	        new Exception().printStackTrace();
372 	        GeneralPath aPath = new GeneralPath();
373 	        aPath.append(aShape,false);
374 	        return aPath;
375 	    }
376 	}

393 	public void setPathToPolyline(Point2D[] points) {
394 		GeneralPath path = new GeneralPath();
395 		path.reset();
396 		path.moveTo((float)points[0].getX(), (float)points[0].getY());
397 		for (int i = 1; i < points.length; i++) {
398 			path.lineTo((float)points[i].getX(), (float)points[i].getY());
399 		}
400 		setShape(path);		
401 	}

404 		GeneralPath path = new GeneralPath();
405 		path.reset();
406 		path.moveTo(xp[0], yp[0]);
407 		for (int i = 1; i < xp.length; i++) {
408 			path.lineTo(xp[i], yp[i]);
409 		}
410 		setShape(path);
