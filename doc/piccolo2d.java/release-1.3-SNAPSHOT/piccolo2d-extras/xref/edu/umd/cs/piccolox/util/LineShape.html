84      public static boolean contains(double x, double y, double x1, double y1, double x2, double y2, 
85                                     boolean min, boolean max,
86                                     double d)
87      {
88          double dx = x2 - x1, dy = y2 - y1;
89          double dx2 = dx * dx, dy2 = dy * dy;
90          double p;
91          if (dx != 0) {
92              p = (((x - x1) / dx) + ((dy * (y - y1)) / dx2)) / (1 + (dy2 / dx2));
93          }
94          else if (dy != 0) {
95              p = (((y - y1) / dy) + ((dx * (x - x1)) / dy2)) / (1 + (dx2 / dy2));
96          }
97          else {
98              return false;
99          }
100         if (max && p > 1.0) {
101             return false;
102         }
103         else if (min && p < 0.0) {
104             return false;
105         }
106         dx = (p * dx) + x1 - x; dy = (p * dy) + y1 - y;
107         double len = dx * dx + dy * dy;
108         return (len < d);
109     }

137     public static boolean intersects(double x1, double y1, double x2, double y2,
138                                      double x3, double y3, double x4, double y4,
139                                      boolean min1, boolean max1,
140                                      boolean min2, boolean max2)
141     {
142         double dx1 = x2 - x1, dy1 = y2 - y1, dx2 = x4 - x3, dy2 = y4 - y3;
143         double d, p2, p1;
144 
145         if (dy1 != 0.0) {
146             d = dx1 / dy1;
147             p2 = (x3 - x1 + (d * (y1 - y3))) / ((d * dy2) - dx2);
148             p1 = (dy2 * p2 + y3 - y1) / dy1;
149         }
150         else if (dy2 != 0.0) {
151             d = dx2 / dy2;
152             p1 = (x1 - x3 + (d * (y3 - y1))) / ((d * dy1) - dx1);
153             p2 = (dy1 * p1 + y1 - y3) / dy2;
154         }
155         else if (dx1 != 0.0) {
156             d = dy1 / dx1;
157             p2 = (y3 - y1 + (d * (x1 - x3))) / ((d * dx2) - dy2);
158             p1 = (dx2 * p2 + x3 - x1) / dx1;
159         }
160         else if (dx2 != 0.0) {
161             d = dy2 / dx2;
162             p1 = (y1 - y3 + (d * (x3 - x1))) / ((d * dx1) - dy1);
163             p2 = (dx1 * p1 + x1 - x3) / dx2;
164         }
165         else {
166             return false;
167         }
168         return (((! min1) || (p1 >= 0.0)) && ((! max1) || (p1 <= 1.0)) &&
169                 ((! min2) || (p2 >= 0.0)) && ((! max2) || (p2 <= 1.0)));
