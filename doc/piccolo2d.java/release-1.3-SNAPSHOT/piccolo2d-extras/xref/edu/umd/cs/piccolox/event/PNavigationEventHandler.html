53  /**
54   * <b>PNavigationEventHandler</b> implements simple focus based navigation. Uses
55   * mouse button one or the arrow keys to set a new focus. Animates the canvas
56   * view to keep the focus node on the screen and at 100 percent scale with
57   * minimal view movement.

59   * @version 1.0
60   * @author Jesse Grosjean

85      public void keyPressed(PInputEvent e) {
86          PNode oldLocation = focusNode;
87  
88          switch (e.getKeyCode()) {
89              case KeyEvent.VK_LEFT:
90                  moveFocusLeft(e);
91                  break;
92  
93              case KeyEvent.VK_RIGHT:
94                  moveFocusRight(e);
95                  break;
96  
97              case KeyEvent.VK_UP:
98              case KeyEvent.VK_PAGE_UP:
99                  if (e.isAltDown()) {
100                     moveFocusOut(e);
101                 }
102                 else {
103                     moveFocusUp(e);
104                 }
105                 break;
106 
107             case KeyEvent.VK_DOWN:
108             case KeyEvent.VK_PAGE_DOWN:
109                 if (e.isAltDown()) {
110                     moveFocusIn(e);
111                 }
112                 else {
113                     moveFocusDown(e);
114                 }
115                 break;
116         }
117 
118         if (focusNode != null && oldLocation != focusNode) {
119             directCameraViewToFocus(e.getCamera(), focusNode, 500);
120         }
121     }

123     public void mousePressed(PInputEvent aEvent) {
124         moveFocusToMouseOver(aEvent);

126         if (focusNode != null) {
127             directCameraViewToFocus(aEvent.getCamera(), focusNode, 500);
128             aEvent.getInputManager().setKeyboardFocus(aEvent.getPath());
129         }

195     public PNode getNeighborInDirection(int aDirection) {
196         if (focusNode == null)
197             return null;
198 
199         NODE_TO_GLOBAL_NODE_CENTER_MAPPING.clear();
200 
201         Point2D highlightCenter = focusNode.getGlobalFullBounds().getCenter2D();
202         NODE_TO_GLOBAL_NODE_CENTER_MAPPING.put(focusNode, highlightCenter);
203 
204         List l = getNeighbors();
205         sortNodesByDistanceFromPoint(l, highlightCenter);
206 
207         Iterator i = l.iterator();
208         while (i.hasNext()) {
209             PNode each = (PNode) i.next();
210             if (nodeIsNeighborInDirection(each, aDirection)) {
211                 return each;
212             }
213         }
214 
215         return null;
216     }
217 
218     public List getNeighbors() {
219         ArrayList result = new ArrayList();
220 
221         if (focusNode == null)
222             return result;
223         if (focusNode.getParent() == null)
224             return result;
225 
226         PNode focusParent = focusNode.getParent();
227 
228         Iterator i = focusParent.getChildrenIterator();
229 
230         while (i.hasNext()) {
231             PNode each = (PNode) i.next();
232             if (each != focusNode && each.getPickable()) {
233                 result.add(each);
234             }
235         }
236 
237         result.add(focusParent);
238 
239         i = focusNode.getChildrenIterator();
240         while (i.hasNext()) {
241             result.add(i.next());
242         }
243 
244         return result;
245     }

247     public boolean nodeIsNeighborInDirection(PNode aNode, int aDirection) {
248         switch (aDirection) {
249             case IN: {
250                 return aNode.isDescendentOf(focusNode);
251             }
252 
253             case OUT: {
254                 return aNode.isAncestorOf(focusNode);
255             }
256 
257             default: {
258                 if (aNode.isAncestorOf(focusNode) || aNode.isDescendentOf(focusNode)) {
259                     return false;
260                 }
261             }
262         }
263 
264         Point2D highlightCenter = (Point2D) NODE_TO_GLOBAL_NODE_CENTER_MAPPING.get(focusNode);
265         Point2D nodeCenter = (Point2D) NODE_TO_GLOBAL_NODE_CENTER_MAPPING.get(aNode);
266 
267         double ytest1 = nodeCenter.getX() - highlightCenter.getX() + highlightCenter.getY();
268         double ytest2 = -nodeCenter.getX() + highlightCenter.getX() + highlightCenter.getY();
269 
270         switch (aDirection) {
271             case NORTH: {
272                 if (nodeCenter.getY() < highlightCenter.getY()) {
273                     if (nodeCenter.getY() < ytest1 && nodeCenter.getY() < ytest2) {
274                         return true;
275                     }
276                 }
277                 break;
278             }
279 
280             case EAST: {
281                 if (nodeCenter.getX() > highlightCenter.getX()) {
282                     if (nodeCenter.getY() < ytest1 && nodeCenter.getY() > ytest2) {
283                         return true;
284                     }
285                 }
286                 break;
287             }
288 
289             case SOUTH: {
290                 if (nodeCenter.getY() > highlightCenter.getY()) {
291                     if (nodeCenter.getY() > ytest1 && nodeCenter.getY() > ytest2) {
292                         return true;
293                     }
294                 }
295                 break;
296             }
297             case WEST: {
298                 if (nodeCenter.getX() < highlightCenter.getX()) {
299                     if (nodeCenter.getY() > ytest1 && nodeCenter.getY() < ytest2) {
300                         return true;
301                     }
302                 }
303                 break;
304             }
305         }
306         return false;
307     }

309     public void sortNodesByDistanceFromPoint(List aNodesList, final Point2D aPoint) {
310         Collections.sort(aNodesList, new Comparator() {
311             public int compare(Object o1, Object o2) {
312                 PNode each1 = (PNode) o1;
313                 PNode each2 = (PNode) o2;
314                 Point2D each1Center = each1.getGlobalFullBounds().getCenter2D();
315                 Point2D each2Center = each2.getGlobalFullBounds().getCenter2D();
316 
317                 NODE_TO_GLOBAL_NODE_CENTER_MAPPING.put(each1, each1Center);
318                 NODE_TO_GLOBAL_NODE_CENTER_MAPPING.put(each2, each2Center);
319 
320                 double distance1 = aPoint.distance(each1Center);
321                 double distance2 = aPoint.distance(each2Center);
322 
323                 if (distance1 < distance2) {
324                     return -1;
325                 }
326                 else if (distance1 == distance2) {
327                     return 0;
328                 }
329                 else {
330                     return 1;
331                 }
332             }
333         });
334     }

341     protected PActivity animateCameraViewTransformTo(final PCamera aCamera, AffineTransform aTransform, int duration) {
342         boolean wasOldAnimation = false;
343 
344         // first stop any old animations.
345         if (navigationActivity != null) {
346             navigationActivity.terminate();
347             wasOldAnimation = true;
348         }
349 
350         if (duration == 0) {
351             aCamera.setViewTransform(aTransform);
352             return null;
353         }
354 
355         AffineTransform source = aCamera.getViewTransformReference();
356 
357         if (!source.equals(aTransform)) {
358             navigationActivity = aCamera.animateViewToTransform(aTransform, duration);
359             ((PTransformActivity) navigationActivity).setSlowInSlowOut(!wasOldAnimation);
360             return navigationActivity;
361         }
362 
363         return null;
364     }

366     public PActivity directCameraViewToFocus(PCamera aCamera, PNode aFocusNode, int duration) {
367         focusNode = aFocusNode;
368         AffineTransform originalViewTransform = aCamera.getViewTransform();
369 
370         // Scale the canvas to include
371         PDimension d = new PDimension(1, 0);
372         focusNode.globalToLocal(d);
373 
374         double scaleFactor = d.getWidth() / aCamera.getViewScale();
375         Point2D scalePoint = focusNode.getGlobalFullBounds().getCenter2D();
376         if (scaleFactor != 1) {
377             aCamera.scaleViewAboutPoint(scaleFactor, scalePoint.getX(), scalePoint.getY());
378         }
379 
380         // Pan the canvas to include the view bounds with minimal canvas
381         // movement.
382         aCamera.animateViewToPanToBounds(focusNode.getGlobalFullBounds(), 0);
383 
384         // Get rid of any white space. The canvas may be panned and
385         // zoomed in to do this. But make sure not stay constrained by max
386         // magnification.
387         // fillViewWhiteSpace(aCamera);
388 
389         AffineTransform resultingTransform = aCamera.getViewTransform();
390         aCamera.setViewTransform(originalViewTransform);
391 
392         // Animate the canvas so that it ends up with the given
393         // view transform.
394         return animateCameraViewTransformTo(aCamera, resultingTransform, duration);
395     }

397     protected void fillViewWhiteSpace(PCamera aCamera) {
398         PBounds rootBounds = aCamera.getRoot().getFullBoundsReference();
399         PBounds viewBounds = aCamera.getViewBounds();
400 
401         if (!rootBounds.contains(aCamera.getViewBounds())) {
402             aCamera.animateViewToPanToBounds(rootBounds, 0);
403             aCamera.animateViewToPanToBounds(focusNode.getGlobalFullBounds(), 0);
404 
405             // center content.
406             double dx = 0;
407             double dy = 0;
408             viewBounds = aCamera.getViewBounds();
409 
410             if (viewBounds.getWidth() > rootBounds.getWidth()) {
411                 // then center along x axis.
412                 double boundsCenterX = rootBounds.getMinX() + (rootBounds.getWidth() / 2);
413                 double viewBoundsCenterX = viewBounds.getMinX() + (viewBounds.getWidth() / 2);
414                 dx = viewBoundsCenterX - boundsCenterX;
415             }
416 
417             if (viewBounds.getHeight() > rootBounds.getHeight()) {
418                 // then center along y axis.
419                 double boundsCenterY = rootBounds.getMinY() + (rootBounds.getHeight() / 2);
420                 double viewBoundsCenterY = viewBounds.getMinY() + (viewBounds.getHeight() / 2);
421                 dy = viewBoundsCenterY - boundsCenterY;
422             }
423             aCamera.translateView(dx, dy);
424         }
