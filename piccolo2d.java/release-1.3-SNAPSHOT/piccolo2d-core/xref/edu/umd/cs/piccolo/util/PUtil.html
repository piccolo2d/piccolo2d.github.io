85  	public static void writeStroke(Stroke aStroke, ObjectOutputStream out) throws IOException {
86  		if (aStroke instanceof Serializable) {
87  			out.writeBoolean(true);
88  			out.writeBoolean(true);
89  			out.writeObject(aStroke);
90  		} else if (aStroke instanceof BasicStroke) {
91  			out.writeBoolean(true);
92  			out.writeBoolean(false);
93  			BasicStroke s = (BasicStroke) aStroke;
94  			
95  			float[] dash = s.getDashArray();
96  			
97  			if (dash == null) {
98  				out.write(0);
99  			} else {
100 				out.write(dash.length);
101 				for (int i = 0; i < dash.length; i++) {
102 					out.writeFloat(dash[i]);
103 				}
104 			}
105 						
106 			out.writeFloat(s.getLineWidth());
107 			out.writeInt(s.getEndCap());
108 			out.writeInt(s.getLineJoin());
109 			out.writeFloat(s.getMiterLimit());			
110 			out.writeFloat(s.getDashPhase());
111 		} else {
112 			out.writeBoolean(false);
113 		}
114 	}

116 	public static Stroke readStroke(ObjectInputStream in) throws IOException, ClassNotFoundException {
117 		boolean wroteStroke = in.readBoolean();
118 		if (wroteStroke) {
119 			boolean serializedStroke = in.readBoolean();
120 			if (serializedStroke) {
121 				return (Stroke) in.readObject();
122 			} else {
123 				float[] dash = null;
124 				int dashLength = in.read();
125 				
126 				if (dashLength != 0) {
127 					dash = new float[dashLength];
128 					for (int i = 0; i < dashLength; i++) {
129 						dash[i] = in.readFloat();
130 					}
131 				}
132 				
133 				float lineWidth = in.readFloat();
134 				int endCap = in.readInt();
135 				int lineJoin = in.readInt();
136 				float miterLimit = in.readFloat();
137 				float dashPhase = in.readFloat();
138 				
139 				return new BasicStroke(lineWidth, endCap, lineJoin, miterLimit, dash, dashPhase);
140 			}
141 		} else {
142 			return null;
143 		}
144 	}

146 	private static final int PATH_IS_DONE = -1;
147 
148 	public static GeneralPath readPath(ObjectInputStream in) throws IOException, ClassNotFoundException {
149 		GeneralPath path = new GeneralPath();
150 	
151 		while(true) {
152 			int segType = in.readInt();
153 		
154 			switch(segType) {
155 				case PathIterator.SEG_MOVETO:
156 					path.moveTo(in.readFloat(), in.readFloat());
157 					break;
158 				
159 				case PathIterator.SEG_LINETO:
160 					path.lineTo(in.readFloat(), in.readFloat());
161 					break;
162 				
163 				case PathIterator.SEG_QUADTO:
164 					path.quadTo(in.readFloat(), in.readFloat(), in.readFloat(), in.readFloat());
165 					break;
166 				
167 				case PathIterator.SEG_CUBICTO:
168 					path.curveTo(in.readFloat(), in.readFloat(), in.readFloat(), in.readFloat(), in.readFloat(), in.readFloat());
169 					break;
170 				
171 				case PathIterator.SEG_CLOSE:
172 					path.closePath();
173 					break;
174 				
175 				case PATH_IS_DONE:
176 					return path;
177 
178 				default:
179 					throw new IOException();
180 			}
181 		}
182 	}

184 	public static void writePath(GeneralPath path, ObjectOutputStream out) throws IOException {		
185 		PathIterator i = path.getPathIterator(null);
186 		float[] data = new float[6];
187 	
188 		while(!i.isDone()) {
189 			switch(i.currentSegment(data)) {
190 				case PathIterator.SEG_MOVETO:
191 					out.writeInt(PathIterator.SEG_MOVETO);
192 					out.writeFloat(data[0]);
193 					out.writeFloat(data[1]);
194 					break;
195 	
196 				case PathIterator.SEG_LINETO:
197 					out.writeInt(PathIterator.SEG_LINETO);
198 					out.writeFloat(data[0]);
199 					out.writeFloat(data[1]);
200 					break;
201 	
202 				case PathIterator.SEG_QUADTO:
203 					out.writeInt(PathIterator.SEG_QUADTO);
204 					out.writeFloat(data[0]);
205 					out.writeFloat(data[1]);
206 					out.writeFloat(data[2]);
207 					out.writeFloat(data[3]);
208 					break;
209 						
210 				case PathIterator.SEG_CUBICTO:
211 					out.writeInt(PathIterator.SEG_CUBICTO);
212 					out.writeFloat(data[0]);
213 					out.writeFloat(data[1]);
214 					out.writeFloat(data[2]);
215 					out.writeFloat(data[3]);
216 					out.writeFloat(data[4]);
217 					out.writeFloat(data[5]);				
218 					break;
219 						
220 				case PathIterator.SEG_CLOSE :
221 					out.writeInt(PathIterator.SEG_CLOSE);					
222 					break;
223 						
224 				default :
225 					throw new IOException();
226 			}
227 	
228 			i.next();
229 		}
230 			
231 		out.writeInt(PATH_IS_DONE);
232 	}
