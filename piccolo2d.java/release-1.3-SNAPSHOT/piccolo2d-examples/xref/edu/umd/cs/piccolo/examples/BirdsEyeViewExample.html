72  	// node.
73  	public void nodeDemo() {
74  		PLayer layer = getCanvas().getLayer();
75  		PNode aNode = PPath.createRectangle(0, 0, 100, 80);
76  
77  		// A node needs to be a descendent of the root to be displayed on the
78  		// screen.
79  		layer.addChild(aNode);
80  
81  		// The default color for a node is blue, but you can change that with
82  		// the setPaint method.
83  		aNode.setPaint(Color.red);
84  
85  		// A node can have children nodes added to it.
86  		aNode.addChild(PPath.createRectangle(0, 0, 100, 80));
87  
88  		// The base bounds of a node is easy to change. Note that changing the
89  		// base
90  		// bounds of a node will not change it's children.
91  		aNode.setBounds(-10, -10, 200, 110);
92  
93  		// Each node has a transform that can be used to transform the node, and
94  		// all its children on the screen.
95  		aNode.translate(100, 100);
96  		aNode.scale(1.5);
97  		aNode.rotate(45);
98  
99  		// The transparency of any node can be set, this transparency will be
100 		// applied to any of the nodes children as well.
101 		aNode.setTransparency(0.75f);
102 
103 		// Its easy to copy nodes.
104 		PNode aCopy = (PNode) aNode.clone();
105 
106 		// Make is so that the copies children are not pickable. For this
107 		// example
108 		// that means you will not be able to grab the child and remove it from
109 		// its parent.
110 		aNode.setChildrenPickable(false);
111 
112 		// Change the look of the copy
113 		aNode.setPaint(Color.GREEN);
114 		aNode.setTransparency(1.0f);
115 
116 		// Let's add the copy to the root, and translate it so that it does not
117 		// cover the original node.
118 		layer.addChild(aCopy);
119 		aCopy.setOffset(0, 0);
120 		aCopy.rotate(-45);
121 	}

137 	// Another way to create nodes is to customize other nodes that already
138 	// exist. Here we create an ellipse, except when you press the mouse on
139 	// this ellipse it turns into a square, when you release the mouse it
140 	// goes back to being an ellipse.
141 	public void subclassExistingClasses() {
142 		final PNode n = new PPath(new Ellipse2D.Float(0, 0, 100, 80)) {
143 
144 			public void paint(PPaintContext aPaintContext) {
145 				if (fIsPressed) {
146 					// if mouse is pressed draw self as a square.
147 					Graphics2D g2 = aPaintContext.getGraphics();
148 					g2.setPaint(getPaint());
149 					g2.fill(getBoundsReference());
150 				} else {
151 					// if mouse is not pressed draw self normally.
152 					super.paint(aPaintContext);
153 				}
154 			}
155 		};
156 
157 		n.addInputEventListener(new PBasicInputEventHandler() {
158 			public void mousePressed(PInputEvent aEvent) {
159 				super.mousePressed(aEvent);
160 				fIsPressed = true;
161 				n.invalidatePaint(); // this tells the framework that the node
162 				// needs to be redisplayed.
163 			}
164 
165 			public void mouseReleased(PInputEvent aEvent) {
166 				super.mousePressed(aEvent);
167 				fIsPressed = false;
168 				n.invalidatePaint(); // this tells the framework that the node
169 				// needs to be redisplayed.
170 			}
171 		});
172 
173 		n.setPaint(Color.ORANGE);
174 		getCanvas().getLayer().addChild(n);
175 	}

177 	// Here a new "face" node is created. But instead of drawing the face
178 	// directly
179 	// using Graphics2D we compose the face from other nodes.
180 	public void composeOtherNodes() {
181 		PNode myCompositeFace = PPath.createRectangle(0, 0, 100, 80);
182 
183 		// create parts for the face.
184 		PNode eye1 = PPath.createEllipse(0, 0, 20, 20);
185 		eye1.setPaint(Color.YELLOW);
186 		PNode eye2 = (PNode) eye1.clone();
187 		PNode mouth = PPath.createRectangle(0, 0, 40, 20);
188 		mouth.setPaint(Color.BLACK);
189 
190 		// add the face parts
191 		myCompositeFace.addChild(eye1);
192 		myCompositeFace.addChild(eye2);
193 		myCompositeFace.addChild(mouth);
194 
195 		// don't want anyone grabbing out our eye's.
196 		myCompositeFace.setChildrenPickable(false);
197 
198 		// position the face parts.
199 		eye2.translate(25, 0);
200 		mouth.translate(0, 30);
201 
202 		// set the face bounds so that it neatly contains the face parts.
203 		PBounds b = myCompositeFace.getUnionOfChildrenBounds(null);
204 		myCompositeFace.setBounds(b.inset(-5, -5));
205 
206 		// opps it to small, so scale it up.
207 		myCompositeFace.scale(1.5);
208 
209 		getCanvas().getLayer().addChild(myCompositeFace);
210 	}

212 	// Here a completely new kind of node, a grid node" is created. We do
213 	// all the drawing ourselves here instead of passing the work off to
214 	// other parts of the framework.
215 	public void createCustomNode() {
216 		PNode n = new PNode() {
217 			public void paint(PPaintContext aPaintContext) {
218 				double bx = getX();
219 				double by = getY();
220 				double rightBorder = bx + getWidth();
221 				double bottomBorder = by + getHeight();
222 
223 				Line2D line = new Line2D.Double();
224 				Graphics2D g2 = aPaintContext.getGraphics();
225 
226 				g2.setStroke(new BasicStroke(0));
227 				g2.setPaint(getPaint());
228 
229 				// draw vertical lines
230 				for (double x = bx; x < rightBorder; x += 5) {
231 					line.setLine(x, by, x, bottomBorder);
232 					g2.draw(line);
233 				}
234 
235 				for (double y = by; y < bottomBorder; y += 5) {
236 					line.setLine(bx, y, rightBorder, y);
237 					g2.draw(line);
238 				}
239 			}
240 		};
241 		n.setBounds(0, 0, 100, 80);
242 		n.setPaint(Color.black);
243 		getCanvas().getLayer().addChild(n);
244 	}

254 
255 		/**
256 		 * This is the node that shows the viewed area.
257 		 */
258 		PNode areaVisiblePNode;
259 
260 		/**
261 		 * This is the canvas that is being viewed
262 		 */
263 		PCanvas viewedCanvas;
264 
265 		/**
266 		 * The change listener to know when to update the birds eye view.
267 		 */
268 		PropertyChangeListener changeListener;
269 
270 		int layerCount;
271 
272 		/**
273 		 * Creates a new instance of a BirdsEyeView
274 		 */
275 		public BirdsEyeView() {
276 
277 			// create the PropertyChangeListener for listening to the viewed
278 			// canvas
279 			changeListener = new PropertyChangeListener() {
280 				public void propertyChange(PropertyChangeEvent evt) {
281 					updateFromViewed();
282 				}
283 			};
284 
285 			// create the coverage node
286 			areaVisiblePNode = new P3DRect();
287 			areaVisiblePNode.setPaint(new Color(128, 128, 255));
288 			areaVisiblePNode.setTransparency(.8f);
289 			areaVisiblePNode.setBounds(0, 0, 100, 100);
290 			getCamera().addChild(areaVisiblePNode);
291 
292 			// add the drag event handler
293 			getCamera().addInputEventListener(new PDragSequenceEventHandler() {
294 				protected void startDrag(PInputEvent e) {
295 					if (e.getPickedNode() == areaVisiblePNode)
296 						super.startDrag(e);
297 				}
298 
299 				protected void drag(PInputEvent e) {
300 					PDimension dim = e.getDelta();
301 					viewedCanvas.getCamera().translateView(0 - dim.getWidth(),
302 							0 - dim.getHeight());
303 				}
304 
305 			});
306 
307 			// remove Pan and Zoom
308 			removeInputEventListener(getPanEventHandler());
309 			removeInputEventListener(getZoomEventHandler());
310 
311 			setDefaultRenderQuality(PPaintContext.LOW_QUALITY_RENDERING);
312 
313 		}

315 		public void connect(PCanvas canvas, PLayer[] viewed_layers) {
316 
317 			this.viewedCanvas = canvas;
318 			layerCount = 0;
319 
320 			viewedCanvas.getCamera().addPropertyChangeListener(changeListener);
321 
322 			for (layerCount = 0; layerCount < viewed_layers.length; ++layerCount) {
323 				getCamera().addLayer(layerCount, viewed_layers[layerCount]);
324 			}
325 
326 		}

330 		 */

332 			getCamera().addLayer(new_layer);
333 			layerCount++;
334 		}

365 		/**
366 		 * This method gets the state of the viewed canvas and updates the
367 		 * BirdsEyeViewer This can be called from outside code
368 		 */
369 		public void updateFromViewed() {
370 
371 			double viewedX;
372 			double viewedY;
373 			double viewedHeight;
374 			double viewedWidth;
375 
376 			double ul_camera_x = viewedCanvas.getCamera().getViewBounds()
377 					.getX();
378 			double ul_camera_y = viewedCanvas.getCamera().getViewBounds()
379 					.getY();
380 			double lr_camera_x = ul_camera_x
381 					+ viewedCanvas.getCamera().getViewBounds().getWidth();
382 			double lr_camera_y = ul_camera_y
383 					+ viewedCanvas.getCamera().getViewBounds().getHeight();
384 
385 			Rectangle2D drag_bounds = getCamera().getUnionOfLayerFullBounds();
386 
387 			double ul_layer_x = drag_bounds.getX();
388 			double ul_layer_y = drag_bounds.getY();
389 			double lr_layer_x = drag_bounds.getX() + drag_bounds.getWidth();
390 			double lr_layer_y = drag_bounds.getY() + drag_bounds.getHeight();
391 
392 			// find the upper left corner
393 
394 			// set to the lesser value
395 			if (ul_camera_x < ul_layer_x)
396 				viewedX = ul_layer_x;
397 			else
398 				viewedX = ul_camera_x;
399 
400 			// same for y
401 			if (ul_camera_y < ul_layer_y)
402 				viewedY = ul_layer_y;
403 			else
404 				viewedY = ul_camera_y;
405 
406 			// find the lower right corner
407 
408 			// set to the greater value
409 			if (lr_camera_x < lr_layer_x)
410 				viewedWidth = lr_camera_x - viewedX;
411 			else
412 				viewedWidth = lr_layer_x - viewedX;
413 
414 			// same for height
415 			if (lr_camera_y < lr_layer_y)
416 				viewedHeight = lr_camera_y - viewedY;
417 			else
418 				viewedHeight = lr_layer_y - viewedY;
419 
420 			Rectangle2D bounds = new Rectangle2D.Double(viewedX, viewedY,
421 					viewedWidth, viewedHeight);
422 			bounds = getCamera().viewToLocal(bounds);
423 			areaVisiblePNode.setBounds(bounds);
424 
425 			// keep the birds eye view centered
426 			getCamera().animateViewToCenterBounds(drag_bounds, true, 0);
427 
428 		}
