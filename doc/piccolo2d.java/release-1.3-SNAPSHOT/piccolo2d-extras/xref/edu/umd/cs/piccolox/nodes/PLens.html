32  import java.awt.Color;
33  import java.awt.Paint;
34  import java.beans.PropertyChangeEvent;
35  import java.beans.PropertyChangeListener;
36  
37  import edu.umd.cs.piccolo.PCamera;
38  import edu.umd.cs.piccolo.PLayer;
39  import edu.umd.cs.piccolo.PNode;
40  import edu.umd.cs.piccolo.event.PDragEventHandler;
41  import edu.umd.cs.piccolo.nodes.PPath;

43  /**
44   * <b>PLens</b> is a simple default lens implementation for Piccolo. See
45   * piccolo/examples LensExample for one possible use of this lens. Lens's are
46   * often application specific, it may be easiest to study this code, and then
47   * implement your own custom lens using the general principles illustrated here.
48   * <p>
49   * The basic design here is to add a PCamera as the child of a Pnode (the lens
50   * node). The camera is the viewing part of the lens, and the node is the title
51   * bar that can be used to move the lens around. Users of this lens will
52   * probably want to set up some lens specific event handler and attach it to the
53   * camera.
54   * </p>
55   * <p>
56   * A lens also needs a layer that it will look at (it should not be the same as
57   * the layer that it's added to because then it will draw itself in a recursive
58   * loop. Last of all the PLens will need to be added to the PCanvas layer (so
59   * that it can be seen by the main camera).
60   * </p>
61   * 
62   * @version 1.0
63   * @author Jesse Grosjean
64   */
65  public class PLens extends PNode {
66  
67      public static double LENS_DRAGBAR_HEIGHT = 20;
68      public static Paint DEFAULT_DRAGBAR_PAINT = Color.DARK_GRAY;
69      public static Paint DEFAULT_LENS_PAINT = Color.LIGHT_GRAY;
70  
71      private PPath dragBar;
72      private PCamera camera;
73      private PDragEventHandler lensDragger;
74  
75      public PLens() {
76          // Drag bar gets resized to fit the available space, so any rectangle
77          // will do here
78          dragBar = PPath.createRectangle(0, 0, 100, 100);
79          dragBar.setPaint(DEFAULT_DRAGBAR_PAINT);
80          // This forces drag events to percolate up to PLens object
81          dragBar.setPickable(false);
82          addChild(dragBar);
83  
84          camera = new PCamera();
85          camera.setPaint(DEFAULT_LENS_PAINT);
86          addChild(camera);
87  
88          // create an event handler to drag the lens around. Note that this event
89          // handler consumes events in case another conflicting event handler has
90          // been installed higher up in the heirarchy.
91          lensDragger = new PDragEventHandler();
92          lensDragger.getEventFilter().setMarksAcceptedEventsAsHandled(true);
93          addInputEventListener(lensDragger);
94  
95          // When this PLens is dragged around adjust the cameras view transform.
96          addPropertyChangeListener(PNode.PROPERTY_TRANSFORM, new PropertyChangeListener() {
97              public void propertyChange(PropertyChangeEvent evt) {
98                  camera.setViewTransform(getInverseTransform());
99              }
100         });
101     }
102 
103     public PLens(PLayer layer) {
104         this();
105         addLayer(0, layer);
106     }
107 
108     public PCamera getCamera() {
109         return camera;
110     }
111 
112     public PPath getDragBar() {
113         return dragBar;
114     }
115 
116     public PDragEventHandler getLensDraggerHandler() {
117         return lensDragger;
118     }
119 
120     public void addLayer(int index, PLayer layer) {
121         camera.addLayer(index, layer);
122     }
123 
124     public void removeLayer(PLayer layer) {
125         camera.removeLayer(layer);
126     }
127 
128     // when the lens is resized this method gives us a chance to layout the
129     // lenses camera child appropriately.
130     protected void layoutChildren() {
131         dragBar.setPathToRectangle((float) getX(), (float) getY(), (float) getWidth(), (float) LENS_DRAGBAR_HEIGHT);
132         camera.setBounds(getX(), getY() + LENS_DRAGBAR_HEIGHT, getWidth(), getHeight() - LENS_DRAGBAR_HEIGHT);
133     }
134 }
