17  /**
18   * An example of how to implement decorator groups. Decorator groups are nodes
19   * that base their bounds and rendering on their children. This seems to be a
20   * common type of visual node that requires some potentially non-obvious
21   * subclassing to get right.
22   * 
23   * Both a volatile and a non-volatile implementation are shown. The volatile
24   * implementation might be used in cases where you want to keep a
25   * scale-independent pen width border around a group of objects. The
26   * non-volatile implementation can be used in more standard cases where the
27   * decorator's bounds are stable during zooming.
28   * 
29   * @author Lance Good
30   */
31  public class GroupExample extends PFrame {
32  
33      public GroupExample() {
34          this(null);
35      }
36  
37      public GroupExample(PCanvas aCanvas) {
38          super("GroupExample", false, aCanvas);
39      }
40  
41      public void initialize() {
42          super.initialize();
43  
44          getCanvas().removeInputEventListener(getCanvas().getPanEventHandler());
45  
46          // Create a decorator group that is NOT volatile
47          DecoratorGroup dg = new DecoratorGroup();
48          dg.setPaint(Color.magenta);
49  
50          // Put some nodes under the group for it to decorate
51          PPath p1 = PPath.createEllipse(25, 25, 75, 75);
52          p1.setPaint(Color.red);
53          PPath p2 = PPath.createRectangle(125, 75, 50, 50);
54          p2.setPaint(Color.blue);
55  
56          // Add everything to the Piccolo hierarchy
57          dg.addChild(p1);
58          dg.addChild(p2);
59          getCanvas().getLayer().addChild(dg);
60  
61          // Create a decorator group that IS volatile
62          VolatileDecoratorGroup vdg = new VolatileDecoratorGroup(getCanvas().getCamera());
63          vdg.setPaint(Color.cyan);
64  
65          // Put some nodes under the group for it to decorate
66          PPath p3 = PPath.createEllipse(275, 175, 50, 50);
67          p3.setPaint(Color.blue);
68          PPath p4 = PPath.createRectangle(175, 175, 75, 75);
69          p4.setPaint(Color.green);
70  
71          // Add everything to the Piccolo hierarchy
72          vdg.addChild(p3);
73          vdg.addChild(p4);
74          getCanvas().getLayer().addChild(vdg);
75  
76          // Create a selection handler so we can see that the decorator actually
77          // works
78          ArrayList selectableParents = new ArrayList();
79          selectableParents.add(dg);
80          selectableParents.add(vdg);
81  
82          PSelectionEventHandler ps = new PSelectionEventHandler(getCanvas().getLayer(), selectableParents);
83          getCanvas().addInputEventListener(ps);
84      }
85  
86      public static void main(String[] args) {
87          new GroupExample();
88      }
89  }

91  /**
92   * This is the non-volatile implementation of a decorator group that paints a
93   * background rectangle based on the bounds of its children.
94   */
95  class DecoratorGroup extends PNode {
96      int INDENT = 10;
97  
98      PBounds cachedChildBounds = new PBounds();
99      PBounds comparisonBounds = new PBounds();
100 
101     public DecoratorGroup() {
102         super();
103     }
104 
105     /**
106      * Change the default paint to fill an expanded bounding box based on its
107      * children's bounds
108      */
109     public void paint(PPaintContext ppc) {
110         Paint paint = getPaint();
111         if (paint != null) {
112             Graphics2D g2 = ppc.getGraphics();
113             g2.setPaint(paint);
114 
115             PBounds bounds = getUnionOfChildrenBounds(null);
116             bounds.setRect(bounds.getX() - INDENT, bounds.getY() - INDENT, bounds.getWidth() + 2 * INDENT, bounds
117                     .getHeight()
118                     + 2 * INDENT);
119             g2.fill(bounds);
120         }
121     }
122 
123     /**
124      * Change the full bounds computation to take into account that we are
125      * expanding the children's bounds Do this instead of overriding
126      * getBoundsReference() since the node is not volatile
127      */
128     public PBounds computeFullBounds(PBounds dstBounds) {
129         PBounds result = getUnionOfChildrenBounds(dstBounds);
130 
131         cachedChildBounds.setRect(result);
132         result.setRect(result.getX() - INDENT, result.getY() - INDENT, result.getWidth() + 2 * INDENT, result
133                 .getHeight()
134                 + 2 * INDENT);
135         localToParent(result);
136         return result;
137     }
138 
139     /**
140      * This is a crucial step. We have to override this method to invalidate the
141      * paint each time the bounds are changed so we repaint the correct region
142      */
143     public boolean validateFullBounds() {
144         comparisonBounds = getUnionOfChildrenBounds(comparisonBounds);
145 
146         if (!cachedChildBounds.equals(comparisonBounds)) {
147             setPaintInvalid(true);
148         }
149         return super.validateFullBounds();
150     }
151 }

153 /**
154  * This is the volatile implementation of a decorator group that paints a
155  * background rectangle based on the bounds of its children.
156  */
157 class VolatileDecoratorGroup extends PNode {
158     int INDENT = 10;
159 
160     PBounds cachedChildBounds = new PBounds();
161     PBounds comparisonBounds = new PBounds();
162     PCamera renderCamera;
163 
164     public VolatileDecoratorGroup(PCamera camera) {
165         super();
166         renderCamera = camera;
167     }
168 
169     /**
170      * Indicate that the bounds are volatile for this group
171      */
172     public boolean getBoundsVolatile() {
173         return true;
174     }
175 
176     /**
177      * Since our bounds are volatile, we can override this method to indicate
178      * that we are expanding our bounds beyond our children
179      */
180     public PBounds getBoundsReference() {
181         PBounds bds = super.getBoundsReference();
182         getUnionOfChildrenBounds(bds);
183 
184         cachedChildBounds.setRect(bds);
185         double scaledIndent = INDENT / renderCamera.getViewScale();
186         bds.setRect(bds.getX() - scaledIndent, bds.getY() - scaledIndent, bds.getWidth() + 2 * scaledIndent, bds
187                 .getHeight()
188                 + 2 * scaledIndent);
189 
190         return bds;
191     }
192 
193     /**
194      * This is a crucial step. We have to override this method to invalidate the
195      * paint each time the bounds are changed so we repaint the correct region
196      */
197     public boolean validateFullBounds() {
198         comparisonBounds = getUnionOfChildrenBounds(comparisonBounds);
199 
200         if (!cachedChildBounds.equals(comparisonBounds)) {
201             setPaintInvalid(true);
202         }
203         return super.validateFullBounds();
204     }
