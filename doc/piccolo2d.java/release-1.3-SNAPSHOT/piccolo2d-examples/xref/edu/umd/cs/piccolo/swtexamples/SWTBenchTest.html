296     void runTest(SWTGraphics2D g, int ctx, int testNum) {
297         Random r1 = new Random(1);
298         Random r2 = new Random(1);
299 
300         System.out.println("Test: " + testNames[testNum]);        
301         long t1 = System.currentTimeMillis();
302         int i = 0;
303         while (true) {
304             if (i % 10 == 0) setup(ctx, g, r1);
305             test(testNum, g, r2);
306             i++;
307             long t2 = System.currentTimeMillis();
308             if (t2 - t1 >= 5000) {
309                 break;
310             }
311         }
312         results[ctx][testNum] += i / 5; 
313         System.out.println("Shapes per second: " + (results[ctx][testNum]));
314     }

316     void runAll(SWTGraphics2D g) {
317         System.out.println("BENCHMARKING: " + g);
318 
319         if (antialiased) {
320             System.out.println("ANTIALIASED");
321             g.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
322                     RenderingHints.VALUE_ANTIALIAS_ON);
323             g.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
324                     RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
325         }
326 
327         for (int ctx = 0; ctx < NUM_CONTEXTS; ctx++) {
328             System.out.println("Context: " + contextNames[ctx]);
329             for (int i = 0; i < NUM_TESTS; i++) {
330                 g.setClip(null);
331                 g.setTransform(IDENTITY);
332                 runTest(g, ctx, i);
333             }
334         }
335 
336         if (offscreen) {
337             g.dispose();
338         }
339 
340         String fileName = g.getClass().getName().replace('.', '_');
341         if (offscreen) fileName += "-offscreen";
342         if (antialiased) fileName += "-antialiased";
343         dumpResults(fileName + ".txt");
344         System.exit(0);
345     }

347     void dumpResults(String fileName) {
348         try {
349             FileOutputStream fout = new FileOutputStream(fileName);
350             PrintWriter out = new PrintWriter(fout);
351             out.print('\t');
352             for (int i = 0; i < NUM_TESTS; i++) {
353                 out.print(testNames[i]);
354                 out.print('\t');
355             }
356             out.println("");
357             for (int ctx = 0; ctx < NUM_CONTEXTS; ctx++) {
358                 out.print(contextNames[ctx]);
359                 for (int i = 0; i < NUM_TESTS; i++) {
360                     out.print('\t');
361                     out.print(results[ctx][i]);
362                 }
363                 out.println("");
364             }
365             out.close();
366             results = new int[NUM_CONTEXTS][NUM_TESTS];
367         } catch (IOException e) {
368             e.printStackTrace();
369             System.exit(1);
370         }
371     }

389     public static void main(String args[]) {
390 		// Create frame
391 		Display display = new Display();
392 		Shell shell = new Shell(display);
393 		shell.setLayout(new FillLayout());
394 
395 		// Add bench test
396         SWTBenchTest m = new SWTBenchTest(shell,SWT.NO_REDRAW_RESIZE | SWT.NO_BACKGROUND);
397 		m.setSize(512,512);
398         for (int i = 0; i < args.length; i++) {
399             if (args[i].intern() == "-offscreen")
400                 m.offscreen = true;
401             else if (args[i].intern() == "-anti")
402                 m.antialiased = true;
403             else {
404                 System.out.println("Usage: java BenchTest [-anti] [-offscreen]");
405                 System.exit(1);
406             }
407         }
408 
409 		shell.pack();
410 		shell.open();
411 
412 		while (!shell.isDisposed ()) {
413 			if (!display.readAndDispatch ()) display.sleep ();
414 		}
415 		display.dispose ();
416     }
