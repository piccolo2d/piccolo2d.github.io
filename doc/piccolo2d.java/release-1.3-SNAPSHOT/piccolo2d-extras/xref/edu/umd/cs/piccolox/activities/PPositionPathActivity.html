89      public void setPositions(GeneralPath path) {
90          PathIterator pi = path.getPathIterator(null, 1);
91          ArrayList points = new ArrayList();
92          float point[] = new float[6];
93          float distanceSum = 0;
94          float lastMoveToX = 0;
95          float lastMoveToY = 0;
96  
97          while (!pi.isDone()) {
98              int type = pi.currentSegment(point);
99  
100             switch (type) {
101                 case PathIterator.SEG_MOVETO:
102                     points.add(new Point2D.Float(point[0], point[1]));
103                     lastMoveToX = point[0];
104                     lastMoveToY = point[1];
105                     break;
106 
107                 case PathIterator.SEG_LINETO:
108                     points.add(new Point2D.Float(point[0], point[1]));
109                     break;
110 
111                 case PathIterator.SEG_CLOSE:
112                     points.add(new Point2D.Float(lastMoveToX, lastMoveToY));
113                     break;
114 
115                 case PathIterator.SEG_QUADTO:
116                 case PathIterator.SEG_CUBICTO:
117                     throw new RuntimeException();
118             }
119 
120             if (points.size() > 1) {
121                 Point2D last = (Point2D) points.get(points.size() - 2);
122                 Point2D current = (Point2D) points.get(points.size() - 1);
123                 distanceSum += last.distance(current);
124             }
125 
126             pi.next();
127         }
128 
129         int size = points.size();
130         Point2D newPositions[] = new Point2D[size];
131         float newKnots[] = new float[size];
132 
133         for (int i = 0; i < size; i++) {
134             newPositions[i] = (Point2D) points.get(i);
135             if (i > 0) {
136                 float dist = (float) newPositions[i - 1].distance(newPositions[i]);
137                 newKnots[i] = newKnots[i - 1] + (dist / distanceSum);
138             }
139         }
140 
141         setPositions(newPositions);
142         setKnots(newKnots);
143     }

145     public void setRelativeTargetValue(float zeroToOne, int startKnot, int endKnot) {
146         Point2D start = getPosition(startKnot);
147         Point2D end = getPosition(endKnot);
148         target.setPosition(start.getX() + (zeroToOne * (end.getX() - start.getX())), start.getY()
149                 + (zeroToOne * (end.getY() - start.getY())));
150     }
