34  import edu.umd.cs.piccolo.PCamera;

41  /**
42   * <b>PZoomToEventHandler</b> is used to zoom the camera view to the node
43   * clicked on with button one.
44   * <P>
45   * @version 1.0
46   * @author Jesse Grosjean
47   */
48  public class PZoomToEventHandler extends PBasicInputEventHandler {
49  
50  	public PZoomToEventHandler() {
51  		setEventFilter(new PInputEventFilter(InputEvent.BUTTON1_MASK));
52  	}
53  
54  	public void mousePressed(PInputEvent aEvent) {
55  		zoomTo(aEvent); 	
56  	}
57  
58  	protected void zoomTo(final PInputEvent aEvent) {
59  		PBounds zoomToBounds;
60  		PNode picked = aEvent.getPickedNode();
61  		
62  		if (picked instanceof PCamera) {
63  			PCamera c = (PCamera) picked;
64  			zoomToBounds = c.getUnionOfLayerFullBounds();
65  		} else {
66  			zoomToBounds = picked.getGlobalFullBounds();
67  		}
68  		
69  		aEvent.getCamera().animateViewToCenterBounds(zoomToBounds, true, 500);
70  	}	
71  }
