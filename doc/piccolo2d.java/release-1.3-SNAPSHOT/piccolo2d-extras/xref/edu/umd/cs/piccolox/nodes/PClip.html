42  /**
43   * <b>PClip</b> is a simple node that applies a clip before rendering or picking its
44   * children. PClip is a subclass of PPath, the clip applies is the GeneralPath wrapped
45   * by its super class. See piccolo/examples ClipExample.
46   * <P>
47   * @version 1.0
48   * @author Jesse Grosjean 
49   */
50  public class PClip extends PPath {
51  
52  	public PBounds computeFullBounds(PBounds dstBounds) {
53  		if (dstBounds == null) dstBounds = new PBounds();
54  		dstBounds.reset();
55  		dstBounds.add(getBoundsReference());
56  		localToParent(dstBounds);
57  		return dstBounds;
58  	}
59  
60  	public void repaintFrom(PBounds localBounds, PNode childOrThis) {
61  		if (childOrThis != this) {
62  			Rectangle2D.intersect(getBoundsReference(), localBounds, localBounds);
63  			super.repaintFrom(localBounds, childOrThis);
64  		} else {
65  			super.repaintFrom(localBounds, childOrThis);
66  		}
67  	}
68  
69  	protected void paint(PPaintContext paintContext) {
70  		Paint p = getPaint();			
71  		if (p != null) {
72  			Graphics2D g2 = paintContext.getGraphics();
73  			g2.setPaint(p);
74  			g2.fill(getPathReference());
75  		}
76  		paintContext.pushClip(getPathReference());
77  	}
78  	
79  	protected void paintAfterChildren(PPaintContext paintContext) {
80  		paintContext.popClip(getPathReference());
81  		if (getStroke() != null && getStrokePaint() != null) {
82  			Graphics2D g2 = paintContext.getGraphics();
83  			g2.setPaint(getStrokePaint());
84  			g2.setStroke(getStroke());
85  			g2.draw(getPathReference());
86  		}		
87  	}
88  	
89  	public boolean fullPick(PPickPath pickPath) {
90  		if (getPickable() && fullIntersects(pickPath.getPickBounds())) {
91  			pickPath.pushNode(this);
92  			pickPath.pushTransform(getTransformReference(false));
93  			
94  			if (pick(pickPath)) {
95  				return true;
96  			}
97  			
98  			if (getChildrenPickable() && getPathReference().intersects(pickPath.getPickBounds())) { 		
99  				int count = getChildrenCount();
100 				for (int i = count - 1; i >= 0; i--) {
101 					PNode each = getChild(i);
102 					if (each.fullPick(pickPath))
103 						return true;
104 				}				
105 			}
106 
107 			if (pickAfterChildren(pickPath)) {
108 				return true;
109 			}
110 
111 			pickPath.popTransform(getTransformReference(false));
112 			pickPath.popNode(this);
113 		}
114 
115 		return false;
116 	}	
117 }
