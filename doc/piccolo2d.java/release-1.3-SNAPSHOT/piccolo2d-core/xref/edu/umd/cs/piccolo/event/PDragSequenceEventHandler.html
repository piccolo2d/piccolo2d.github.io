38  /**
39   * <b>PDragSequenceEventHandler</b> is designed to support mouse pressed,
40   * dragged, and released interaction sequences. Support is also provided for
41   * running a continuous activity during the drag sequence.
42   * <P>
43   * PDragSequenceEventHandler should be subclassed by a concrete event handler
44   * that implements a particular interaction. See PPanEventHandler,
45   * PZoomEventHandler, and PDragEventHandler for examples.
46   * <P>
47   * 
48   * @version 1.0
49   * @author Jesse Grosjean
50   */
51  public abstract class PDragSequenceEventHandler extends PBasicInputEventHandler {
52  
53      private double minDragStartDistance = 0;
54      private transient boolean isDragging = false;
55      private transient Point2D mousePressedCanvasPoint;
56      private transient PActivity dragActivity;
57      private transient PInputEvent dragEvent;
58      private transient int sequenceInitiatedButton = MouseEvent.NOBUTTON;
59  
60      public PDragSequenceEventHandler() {
61      }
62  
63      // ****************************************************************
64      // Basics
65      // ****************************************************************
66  
67      public boolean isDragging() {
68          return isDragging;
69      }
70  
71      public void setIsDragging(boolean isDragging) {
72          this.isDragging = isDragging;
73      }
74  
75      public double getMinDragStartDistance() {
76          return minDragStartDistance;
77      }
78  
79      /**
80       * Set the minimum distance that the mouse should be dragged (in screen
81       * coords) before a new drag sequence is initiate.
82       */
83      public void setMinDragStartDistance(double minDistance) {
84          minDragStartDistance = minDistance;
85      }
86  
87      /**
88       * Return the point in canvas coordinates where the mouse was last pressed.
89       */
90      public Point2D getMousePressedCanvasPoint() {
91          if (mousePressedCanvasPoint == null) {
92              mousePressedCanvasPoint = new Point2D.Double();
93          }
94          return mousePressedCanvasPoint;
95      }
96  
97      // ****************************************************************
98      // Dragging - Methods to indicate the stages of the drag sequence.
99      // ****************************************************************
100 
101     /**
102      * Subclasses should override this method to get notified of the start of a
103      * new drag sequence. Note that that overriding methods must still call
104      * super.startDrag() for correct behavior.
105      */
106     protected void startDrag(PInputEvent e) {
107         dragEvent = e;
108         startDragActivity(e);
109         setIsDragging(true);
110         e.getComponent().setInteracting(true);
111     }
112 
113     /**
114      * Subclasses should override this method to get notified of the drag events
115      * in a drag sequence. Note that that overriding methods must still call
116      * super.startDrag() for correct behavior.
117      */
118     protected void drag(PInputEvent e) {
119         dragEvent = e;
120     }
121 
122     /**
123      * Subclasses should override this method to get notified of the end event
124      * in a drag sequence. Note that that overriding methods must still call
125      * super.startDrag() for correct behavior.
126      */
127     protected void endDrag(PInputEvent e) {
128         stopDragActivity(e);
129         dragEvent = null;
130         e.getComponent().setInteracting(false);
131         setIsDragging(false);
132     }
133 
134     protected boolean shouldStartDragInteraction(PInputEvent e) {
135         return getMousePressedCanvasPoint().distance(e.getCanvasPosition()) >= getMinDragStartDistance();
136     }
137 
138     // ****************************************************************
139     // Drag Activity - Used for scheduling an activity during a drag
140     // sequence. For example zooming and auto panning are implemented
141     // using this.
142     // ****************************************************************
143 
144     protected PActivity getDragActivity() {
145         return dragActivity;
146     }
147 
148     protected void startDragActivity(PInputEvent aEvent) {
149         dragActivity = new PActivity(-1, PUtil.DEFAULT_ACTIVITY_STEP_RATE);
150         dragActivity.setDelegate(new PActivity.PActivityDelegate() {
151             public void activityStarted(PActivity activity) {
152                 dragActivityFirstStep(dragEvent);
153             }
154 
155             public void activityStepped(PActivity activity) {
156                 dragActivityStep(dragEvent);
157             }
158 
159             public void activityFinished(PActivity activity) {
160                 dragActivityFinalStep(dragEvent);
161             }
162         });
163 
164         aEvent.getCamera().getRoot().addActivity(dragActivity);
165     }
166 
167     protected void stopDragActivity(PInputEvent aEvent) {
168         dragActivity.terminate();
169         dragActivity = null;
170     }
171 
172     /**
173      * Override this method to get notified when the drag activity starts
174      * stepping.
175      */
176     protected void dragActivityFirstStep(PInputEvent aEvent) {
177     }
178 
179     /**
180      * During a drag sequence an activity is scheduled that runs continuously
181      * while the drag sequence is active. This can be used to support some
182      * additional behavior that is not driven directly by mouse events. For
183      * example PZoomEventHandler uses it for zooming and PPanEventHandler uses
184      * it for auto panning.
185      */
186     protected void dragActivityStep(PInputEvent aEvent) {
187     }
188 
189     /**
190      * Override this method to get notified when the drag activity stops
191      * stepping.
192      */
193     protected void dragActivityFinalStep(PInputEvent aEvent) {
194     }

196     // ****************************************************************
197     // Events - subclasses should not override these methods, instead
198     // override the appropriate drag method.
199     // ****************************************************************
200 
201     public void mousePressed(PInputEvent e) {
202         super.mousePressed(e);
203 
204         if (sequenceInitiatedButton == MouseEvent.NOBUTTON) {
205             sequenceInitiatedButton = e.getButton();
206         }
207         else {
208             return;
209         }
210 
211         getMousePressedCanvasPoint().setLocation(e.getCanvasPosition());
212         if (!isDragging()) {
213             if (shouldStartDragInteraction(e)) {
214                 startDrag(e);
215             }
216         }
217     }
218 
219     public void mouseDragged(PInputEvent e) {
220         super.mouseDragged(e);
221 
222         if (sequenceInitiatedButton != MouseEvent.NOBUTTON) {
223             if (!isDragging()) {
224                 if (shouldStartDragInteraction(e)) {
225                     startDrag(e);
226                 }
227                 return;
228             }
229             drag(e);
230         }
231     }
232 
233     public void mouseReleased(PInputEvent e) {
234         super.mouseReleased(e);
235         if (sequenceInitiatedButton == e.getButton()) {
236             if (isDragging())
237                 endDrag(e);
238             sequenceInitiatedButton = MouseEvent.NOBUTTON;
239         }
240     }

246     /**
247      * Returns a string representing the state of this node. This method is
248      * intended to be used only for debugging purposes, and the content and
249      * format of the returned string may vary between implementations. The
250      * returned string may be empty but may not be <code>null</code>.
251      * 
252      * @return a string representation of this node's state
253      */
254     protected String paramString() {
255         StringBuffer result = new StringBuffer();
256 
257         result.append("minDragStartDistance=" + minDragStartDistance);
258         result.append(",mousePressedCanvasPoint="
259                 + (mousePressedCanvasPoint == null ? "null" : mousePressedCanvasPoint.toString()));
260         result.append(",sequenceInitiatedButton=" + sequenceInitiatedButton);
261         if (isDragging)
262             result.append(",dragging");
263         result.append(',');
264         result.append(super.paramString());
265 
266         return result.toString();
267     }
