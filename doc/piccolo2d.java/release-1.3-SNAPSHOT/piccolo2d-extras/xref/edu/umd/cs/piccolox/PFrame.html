50  /**
51   * <b>PFrame</b> is meant to be subclassed by applications that just need a
52   * PCanvas in a JFrame. It also includes full screen mode functionality when run
53   * in JDK 1.4. These subclasses should override the initialize method and start
54   * adding their own code there. Look in the examples package to see lots of uses
55   * of PFrame.
56   * 
57   * @version 1.0
58   * @author Jesse Grosjean
59   */
60  public class PFrame extends JFrame {
61  
62      private PCanvas canvas;
63      private GraphicsDevice graphicsDevice;
64      private EventListener escapeFullScreenModeListener;
65  
66      public PFrame() {
67          this("", false, null);
68      }
69  
70      public PFrame(String title, boolean fullScreenMode, PCanvas aCanvas) {
71          this(title, GraphicsEnvironment.getLocalGraphicsEnvironment().getDefaultScreenDevice(), fullScreenMode, aCanvas);
72      }
73  
74      public PFrame(String title, GraphicsDevice aDevice, final boolean fullScreenMode, final PCanvas aCanvas) {
75          super(title, aDevice.getDefaultConfiguration());
76  
77          graphicsDevice = aDevice;
78  
79          setBackground(null);
80          setBounds(getDefaultFrameBounds());
81  
82          try {
83              setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
84          }
85          catch (SecurityException e) {
86          } // expected from applets
87  
88          if (aCanvas == null) {
89              canvas = new PCanvas();
90          }
91          else {
92              canvas = aCanvas;
93          }
94  
95          setContentPane(canvas);
96          validate();
97          setFullScreenMode(fullScreenMode);
98          canvas.requestFocus();
99          beforeInitialize();
100 
101         // Manipulation of Piccolo's scene graph should be done from Swings
102         // event dispatch thread since Piccolo is not thread safe. This code
103         // calls initialize() from that thread once the PFrame is initialized,
104         // so you are safe to start working with Piccolo in the initialize()
105         // method.
106         SwingUtilities.invokeLater(new Runnable() {
107             public void run() {
108                 PFrame.this.initialize();
109                 repaint();
110             }
111         });
112     }

126     public boolean isFullScreenMode() {
127         return graphicsDevice.getFullScreenWindow() != null;
128     }
129 
130     public void setFullScreenMode(boolean fullScreenMode) {
131         if (fullScreenMode) {
132             addEscapeFullScreenModeListener();
133 
134             if (isDisplayable()) {
135                 dispose();
136             }
137 
138             setUndecorated(true);
139             setResizable(false);
140             graphicsDevice.setFullScreenWindow(this);
141 
142             if (graphicsDevice.isDisplayChangeSupported()) {
143                 chooseBestDisplayMode(graphicsDevice);
144             }
145             validate();
146         }
147         else {
148             removeEscapeFullScreenModeListener();
149 
150             if (isDisplayable()) {
151                 dispose();
152             }
153 
154             setUndecorated(false);
155             setResizable(true);
156             graphicsDevice.setFullScreenWindow(null);
157             validate();
158             setVisible(true);
159         }
160     }
161 
162     protected void chooseBestDisplayMode(GraphicsDevice device) {
163         DisplayMode best = getBestDisplayMode(device);
164         if (best != null) {
165             device.setDisplayMode(best);
166         }
167     }
168 
169     protected DisplayMode getBestDisplayMode(GraphicsDevice device) {
170         Iterator itr = getPreferredDisplayModes(device).iterator();
171         while (itr.hasNext()) {
172             DisplayMode each = (DisplayMode) itr.next();
173             DisplayMode[] modes = device.getDisplayModes();
174             for (int i = 0; i < modes.length; i++) {
175                 if (modes[i].getWidth() == each.getWidth() && modes[i].getHeight() == each.getHeight()
176                         && modes[i].getBitDepth() == each.getBitDepth()) {
177                     return each;
178                 }
179             }
180         }
181 
182         return null;
183     }
184 
185     /**
186      * By default return the current display mode. Subclasses may override this
187      * method to return other modes in the collection.
188      */
189     protected Collection getPreferredDisplayModes(GraphicsDevice device) {
190         ArrayList result = new ArrayList();
191 
192         result.add(device.getDisplayMode());
193         /*
194          * result.add(new DisplayMode(640, 480, 32, 0)); result.add(new
195          * DisplayMode(640, 480, 16, 0)); result.add(new DisplayMode(640, 480,
196          * 8, 0));
197          */
198 
199         return result;
200     }

202     /**
203      * This method adds a key listener that will take this PFrame out of full
204      * screen mode when the escape key is pressed. This is called for you
205      * automatically when the frame enters full screen mode.
206      */
207     public void addEscapeFullScreenModeListener() {
208         removeEscapeFullScreenModeListener();
209         escapeFullScreenModeListener = new KeyAdapter() {
210             public void keyPressed(KeyEvent aEvent) {
211                 if (aEvent.getKeyCode() == KeyEvent.VK_ESCAPE) {
212                     setFullScreenMode(false);
213                 }
214             }
215         };
216         canvas.addKeyListener((KeyListener) escapeFullScreenModeListener);
217     }
218 
219     /**
220      * This method removes the escape full screen mode key listener. It will be
221      * called for you automatically when full screen mode exits, but the method
222      * has been made public for applications that wish to use other methods for
223      * exiting full screen mode.
224      */
225     public void removeEscapeFullScreenModeListener() {
226         if (escapeFullScreenModeListener != null) {
227             canvas.removeKeyListener((KeyListener) escapeFullScreenModeListener);
228             escapeFullScreenModeListener = null;
229         }
230     }
