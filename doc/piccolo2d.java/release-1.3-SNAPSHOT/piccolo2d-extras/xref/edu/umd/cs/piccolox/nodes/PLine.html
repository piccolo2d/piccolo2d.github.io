84  	public boolean setBounds(double x, double y, double width, double height) {
85  		if (line == null || !super.setBounds(x, y, width, height)) {
86  			return false;
87  		}
88  
89  		Rectangle2D lineBounds = line.getBounds2D();
90  		Rectangle2D lineStrokeBounds = getLineBoundsWithStroke();
91  		double strokeOutset = Math.max(lineStrokeBounds.getWidth() - lineBounds.getWidth(), 
92                                         lineStrokeBounds.getHeight() - lineBounds.getHeight());
93  		
94  		x += strokeOutset / 2;
95  		y += strokeOutset / 2;
96  		width -= strokeOutset;
97  		height -= strokeOutset;
98  		
99  		TEMP_TRANSFORM.setToIdentity();
100 		TEMP_TRANSFORM.translate(x, y);
101 		TEMP_TRANSFORM.scale(width / lineBounds.getWidth(), height / lineBounds.getHeight());
102 		TEMP_TRANSFORM.translate(-lineBounds.getX(), -lineBounds.getY());		
103         line.transformPoints(TEMP_TRANSFORM);
104 		
105 		return true;
106 	}

108 	public boolean intersects(Rectangle2D aBounds) {
109 		if (super.intersects(aBounds)) {
110 			if (line.intersects(aBounds)) {
111 				return true;
112 			} else if (stroke != null && strokePaint != null) {
113 				return stroke.createStrokedShape(line).intersects(aBounds);
114 			}
115 		}
116 		return false;
117 	}
