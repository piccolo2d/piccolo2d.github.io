22  /**
23   * <b>PSWTText</b> creates a visual component to support text. Multiple lines can
24   * be entered, and basic editing is supported. A caret is drawn,
25   * and can be repositioned with mouse clicks.  The text object is positioned
26   * so that its upper-left corner is at the origin, though this can be changed
27   * with the translate methods.
28   * <P>
29   * <b>Warning:</b> Serialized and ZSerialized objects of this class will not be
30   * compatible with future Jazz releases. The current serialization support is
31   * appropriate for short term storage or RMI between applications running the
32   * same version of Jazz. A future release of Jazz will provide support for long
33   * term persistence.

224      * Multline text is returned as a single string
225      * where each line is separated by a newline character.
226      * Single line text does not have any newline characters.
227      */
228     public String getText() {
229         String line;
230         String result = new String();
231         int lineNum = 0;
232 
233         for (Iterator i = lines.iterator() ; i.hasNext() ; ) {
234             if (lineNum > 0) {
235                 result += '\n';
236             }
237             line = (String)i.next();
238             result += line;
239             lineNum++;
240         }
241 
242         return result;
243     }

262     /**
263      * Sets the text of this visual component to str. Multiple lines
264      * of text are separated by a newline character.
265      * @param str use this string.
266      */
267     public void setText(String str) {
268         int pos = 0;
269         int index;
270         boolean done = false;
271         lines = new ArrayList();
272         do {
273             index = str.indexOf('\n', pos);
274             if (index == -1) {
275                 lines.add(str);
276                 done = true;
277             } else {
278                 lines.add(str.substring(0, index));
279                 str = str.substring(index + 1);
280             }
281         } while (!done);
282 
283 		recomputeBounds();
284     }

347     /**
348      * Renders the text object.
349      * <p>
350      * The transform, clip, and composite will be set appropriately when this object
351      * is rendered.  It is up to this object to restore the transform, clip, and composite of
352      * the Graphics2D if this node changes any of them. However, the color, font, and stroke are
353      * unspecified by Jazz.  This object should set those things if they are used, but
354      * they do not need to be restored.
355      *
356      * @param ppc Contains information about current render.
357      */
358     public void paint(PPaintContext ppc) {
359         Graphics2D g2 = ppc.getGraphics();
360         AffineTransform at = null;
361         boolean translated = false;
362         if (!lines.isEmpty()) {
363 
364             if ((translateX != 0.0) || (translateY != 0.0)) {
365                 at = g2.getTransform(); // save transform
366                 g2.translate(translateX, translateY);
367                 translated = true;
368             }
369 
370                                 // If font too small and not antialiased, then greek
371             double renderedFontSize = font.getSize() * ppc.getScale();
372                                 // BBB: HACK ALERT - July 30, 1999
373                                 // This is a workaround for a bug in Sun JDK 1.2.2 where
374                                 // fonts that are rendered at very small magnifications show up big!
375                                 // So, we render as greek if requested (that's normal)
376                                 // OR if the font is very small (that's the workaround)
377             if ((renderedFontSize < 0.5) ||
378                 (renderedFontSize < greekThreshold)) {
379                 paintAsGreek(ppc);
380             } else {
381                 paintAsText(ppc);
382             }
383             if (translated) {
384                 g2.setTransform(at); // restore transform
385             }
386         }
387     }

402     /**
403      * Paints this object normally (show it's text).
404      * Note that the entire text gets rendered so that it's upper
405      * left corner appears at the origin of this local object.
406      * @param ppc The graphics context to paint into.
407      */
408     public void paintAsText(PPaintContext ppc) {
409 		SWTGraphics2D sg2 = (SWTGraphics2D)ppc.getGraphics();
410 		
411 		if (getPaint() != null) {
412             sg2.setBackground((Color)getPaint());
413             Rectangle2D rect = new Rectangle2D.Double(0.0, 0.0, getWidth(), getHeight());
414             sg2.fillRect(rect.getX(),rect.getY(),rect.getWidth(),rect.getHeight());
415         }
416 
417 
418         sg2.translate(padding,padding);            
419 
420         double scale = Math.min(sg2.getTransform().getScaleX(),sg2.getTransform().getScaleY());
421         double dSize = scale*font.getSize();
422         double fixupScale = Math.floor(dSize)/dSize;
423 		
424         // This moves the text size down to the next closest integer size - to help it stay in
425         // it's alloted bounds.  This is because SWT only supports integer font metrics
426         sg2.scale(fixupScale,fixupScale);
427         
428         
429 								// Render each line of text
430                                 // Note that the entire text gets rendered so that it's upper left corner
431                                 // appears at the origin of this local object.
432         sg2.setColor(penColor);
433         sg2.setFont(font);
434 
435         int lineNum = 0;
436         String line;
437         double y;
438 
439 		FontMetrics metrics = sg2.getSWTFontMetrics();
440 
441         for (Iterator i = lines.iterator() ; i.hasNext() ; ) {
442             line = (String)i.next();
443 
444 		    // ADDED BY LEG ON 2/25/03 - BUG CAUSING PROBLEMS AT CERTAIN
445 		    // SCALES WHEN LINE WAS EMPTY
446 		    line = (line.equals("")) ? " " : line;
447 
448             y = (lineNum * metrics.getHeight());
449 
450             sg2.drawString(line, (double)0, (double)y);
451 
452             lineNum++;
453         }
454         
455         sg2.scale(1/fixupScale,1/fixupScale);
456         
457         sg2.translate(-padding,-padding);            
458     }

460     /**
461      * Notifies this object that it has changed and that it
462      * should update its notion of its bounding box.
463      */
464     protected void recomputeBounds() {
465 		Point bds;
466         double lineWidth;
467         double maxWidth = 0.0;
468         double height;
469 
470        	height = 0.0;
471 
472         boolean hasText = true;
473         if ((lines.size() == 1) && (((String)lines.get(0)).equals(""))) {
474             hasText = false;
475         }
476 
477 		GC gc = new GC(Display.getDefault());
478 		SWTGraphics2D g2 = new SWTGraphics2D(gc,Display.getDefault());
479 		g2.setFont(font);		
480 		FontMetrics fm = g2.getSWTFontMetrics();
481 		
482         if (!lines.isEmpty() && hasText) {
483             String line;
484             int lineNum = 0;
485             for (Iterator i = lines.iterator() ; i.hasNext() ; ) {
486                 line = (String)i.next();
487 
488 
489                             // Find the longest line in the text
490 				bds = gc.stringExtent(line);
491 				lineWidth = bds.x;
492 
493                 if (lineWidth > maxWidth) {
494                     maxWidth = lineWidth;
495                 }
496                             // Find the heighest line in the text
497                 if (lineNum == 0) {
498                     height += fm.getAscent()+fm.getDescent()+fm.getLeading();
499                 } else {
500                     height += fm.getHeight();
501                 }
502 
503                 lineNum++;
504             }
505         } else {
506                             // If no text, then we want to have the bounds of a space character,
507                             // so get those bounds here
508 			bds = gc.stringExtent(" ");
509             maxWidth = bds.x;
510             height = bds.y;
511         }
512 
513 		gc.dispose();
514 
515                                 // Finally, set the bounds of this text
516         setBounds(translateX,translateY,maxWidth+2*DEFAULT_PADDING,height+2*DEFAULT_PADDING);
517     }
